551 bool Foam::expressions::exprResult::writeEntryChecked
552 (
553  const word& keyword,
554  Ostream& os
555 ) const
556 {
557  if (!isType<Type>())

559  return false;

561

564  if (isUniform_ && is_contiguous<Type>::value)
