107  bool fieldFound
108  (

110  (
111  fieldPtr,
112  mesh,

114  solverName,
115  useSolverNameForFields

117  );
118 
119  // No base or custom field found.
120  // Construct field based on linear interpolation
121  if (!fieldFound)
122  {
123  word phiName(baseName);
124  if (useSolverNameForFields)

128  IOobject header

134  IOobject::AUTO_WRITE
135  );
136  fieldPtr.reset
