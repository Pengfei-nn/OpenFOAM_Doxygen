416 void alphatWallBoilingWallFunctionFvPatchScalarField::updateCoeffs()
417 {
418 
419  if (updated())
420  {
421  return;
422  }
423 
424  // Check that partitioningModel has been constructed
425  if (!partitioningModel_.valid())
426  {
427  FatalErrorInFunction
428  << "partitioningModel has not been constructed!"
429  << abort(FatalError);
430  }
431 
432  // Lookup the fluid model
433  const phaseSystem& fluid =
434  refCast<const phaseSystem>
435  (
436  db().lookupObject<phaseSystem>("phaseProperties")
437  );
438 
439  const saturationModel& satModel =
440  db().lookupObject<saturationModel>("saturationModel");
441 
442  const label patchi = patch().index();
443 
444  const scalar t = this->db().time().timeOutputValue();
445  scalar relax = relax_->value(t);
446 
447  switch (phaseType_)
448  {
449  case vaporPhase:
450  {
451  const phaseModel& vapor
452  (
453  fluid.phases()[internalField().group()]
454  );
455 
456  const fvPatchScalarField& hewv =
457  vapor.thermo().he().boundaryField()[patchi];
458 
459  // Vapor Liquid phase fraction at the wall
460  const scalarField vaporw(vapor.boundaryField()[patchi]);
461 
462  // NOTE! Assumes 1-thisPhase for liquid fraction in
463  // multiphase simulations
464  const scalarField fLiquid
465  (
466  partitioningModel_->fLiquid(1-vaporw)
467  );
468 
469  const tmp<scalarField> talphaw = vapor.thermo().alpha(patchi);
470  const scalarField& alphaw = talphaw();
471 
472  const scalarField heSnGrad(max(hewv.snGrad(), scalar(1e-16)));
473 
474  // Convective thermal diffusivity for single phase
475  const scalarField alphatv(calcAlphat(*this));
476 
477  forAll (*this, i)
478  {
479  this->operator[](i) =
480  (
481  (1 - fLiquid[i])*(alphatv[i])
482  /max(vaporw[i], scalar(1e-8))
483  );
484  }
485 
486  if (debug)
487  {
488  Info<< "alphat for vapour : " << nl << endl;
489 
490  Info<< " alphatEffv: " << gMin(vaporw*(*this + alphaw))
491  << " - " << gMax(vaporw*(*this + alphaw)) << endl;
492 
493  const scalarField qEff(vaporw*(*this + alphaw)*hewv.snGrad());
494 
495  scalar Qeff = gSum(qEff*patch().magSf());
496  Info<< " Effective heat transfer rate to vapor:" << Qeff
497  << nl << endl;
498  }
499  break;
500  }
501  case liquidPhase:
502  {
503  // Check that nucleationSiteModel has been constructed
504  if (!nucleationSiteModel_.valid())
505  {
506  FatalErrorInFunction
507  << "nucleationSiteModel has not been constructed!"
508  << abort(FatalError);
509  }
510 
511  // Check that departureDiameterModel has been constructed
512  if (!departureDiamModel_.valid())
513  {
514  FatalErrorInFunction
515  << "departureDiameterModel has not been constructed!"
516  << abort(FatalError);
517  }
518 
519  // Check that nucleationSiteModel has been constructed
520  if (!departureFreqModel_.valid())
521  {
522  FatalErrorInFunction
523  << "departureFrequencyModel has not been constructed!"
524  << abort(FatalError);
525  }
526 
527  const phaseModel& liquid
528  (
529  fluid.phases()[internalField().group()]
530  );
531 
532  const phaseModel& vapor(fluid.phases()[otherPhaseName_]);
533 
534  // Retrieve turbulence properties from models
535  const phaseCompressibleTurbulenceModel& turbModel =
536  db().lookupObject<phaseCompressibleTurbulenceModel>
537  (
538  IOobject::groupName
539  (
540  turbulenceModel::propertiesName,
541  liquid.name()
542  )
543  );
544  const phaseCompressibleTurbulenceModel& vaporTurbModel =
545  db().lookupObject<phaseCompressibleTurbulenceModel>
546  (
547  IOobject::groupName
548  (
549  turbulenceModel::propertiesName,
550  vapor.name()
551  )
552  );
553 
554  const tmp<scalarField> tnutw = turbModel.nut(patchi);
555 
556  const scalar Cmu25(pow025(Cmu_));
557 
558  const scalarField& y = turbModel.y()[patchi];
559 
560  const tmp<scalarField> tmuw = turbModel.mu(patchi);
561  const scalarField& muw = tmuw();
562 
563  const tmp<scalarField> talphaw = liquid.thermo().alphahe(patchi);
564  const scalarField& alphaw = talphaw();
565 
566  const tmp<volScalarField> tk = turbModel.k();
567  const volScalarField& k = tk();
568  const fvPatchScalarField& kw = k.boundaryField()[patchi];
569 
570  const fvPatchVectorField& Uw =
571  turbModel.U().boundaryField()[patchi];
572  const scalarField magUp(mag(Uw.patchInternalField() - Uw));
573  const scalarField magGradUw(mag(Uw.snGrad()));
574 
575  const fvPatchScalarField& rhow =
576  turbModel.rho().boundaryField()[patchi];
577 
578 
579  const fvPatchScalarField& Tw =
580  liquid.thermo().T().boundaryField()[patchi];
581  const scalarField Tc(Tw.patchInternalField());
582 
583  const scalarField uTau(Cmu25*sqrt(kw));
584 
585  const scalarField yPlus(uTau*y/(muw/rhow));
586 
587  const scalarField Pr(muw/alphaw);
588 
589  // Molecular-to-turbulent Prandtl number ratio
590  const scalarField Prat(Pr/Prt_);
591 
592  // Thermal sublayer thickness
593  const scalarField P(this->Psmooth(Prat));
594 
595  const scalarField yPlusTherm(this->yPlusTherm(P, Prat));
596 
597  const fvPatchScalarField& rhoVaporw =
598  vaporTurbModel.rho().boundaryField()[patchi];
599 
600  tmp<volScalarField> tCp = liquid.thermo().Cp();
601  const volScalarField& Cp = tCp();
602  const fvPatchScalarField& Cpw = Cp.boundaryField()[patchi];
603 
604  // Saturation temperature
605  const tmp<volScalarField> tTsat =
606  satModel.Tsat(liquid.thermo().p());
607 
608  const volScalarField& Tsat = tTsat();
609  const fvPatchScalarField& Tsatw(Tsat.boundaryField()[patchi]);
610  const scalarField Tsatc(Tsatw.patchInternalField());
611 
612  const fvPatchScalarField& pw =
613  liquid.thermo().p().boundaryField()[patchi];
614 
615  const fvPatchScalarField& hew =
616  liquid.thermo().he().boundaryField()[patchi];
617 
618  const scalarField hw
619  (
620  liquid.thermo().he().member() == "e"
621  ? hew.patchInternalField() + pw/rhow.patchInternalField()
622  : hew.patchInternalField()
623  );
624 
625  const scalarField L
626  (
627  vapor.thermo().he().member() == "e"
628  ? vapor.thermo().he(pw, Tsatc, patchi) + pw/rhoVaporw - hw
629  : vapor.thermo().he(pw, Tsatc, patchi) - hw
630  );
631 
632  // Liquid phase fraction at the wall
633  const scalarField liquidw(liquid.boundaryField()[patchi]);
634 
635  const scalarField fLiquid(partitioningModel_->fLiquid(liquidw));
636 
637  // Liquid temperature at y+=250 is estimated from logarithmic
638  // thermal wall function (Koncar, Krepper & Egorov, 2005)
639  const scalarField Tplus_y250
640  (
641  Prt_*(log(E_*250)/kappa_ + P)
642  );
643  const scalarField Tplus(Prt_*(log(E_*yPlus)/kappa_ + P));
644  scalarField Tl(Tw - (Tplus_y250/Tplus)*(Tw - Tc));
645  Tl = max(Tc - 40, Tl);
646 
647  // Film, transient boiling regimes
648  scalarField Qtb(this->size(), 0);
649  scalarField tDNB(this->size(), GREAT);
650  scalarField TLeiden(this->size(), GREAT);
651  scalarField htcFilmBoiling(this->size(), 0);
652 
653  if
654  (
655  CHFModel_.valid()
656  && CHFSoobModel_.valid()
657  && TDNBModel_.valid()
658  && MHFModel_.valid()
659  && LeidenfrostModel_.valid()
660  && filmBoilingModel_.valid()
661  )
662  {
663 
664  const scalarField CHF
665  (
666  CHFModel_->CHF
667  (
668  liquid,
669  vapor,
670  patchi,
671  Tl,
672  Tsatw,
673  L
674  )
675  );
676 
677  // Effect of sub-cooling to the CHF in saturated conditions
678  const scalarField CHFSubCool
679  (
680  CHFSoobModel_->CHFSubCool
681  (
682  liquid,
683  vapor,
684  patchi,
685  Tl,
686  Tsatw,
687  L
688  )
689  );
690 
691  const scalarField CHFtotal(CHF*CHFSubCool);
692 
693  tDNB =
694  TDNBModel_->TDNB
695  (
696  liquid,
697  vapor,
698  patchi,
699  Tl,
700  Tsatw,
701  L
702  );
703 
704  const scalarField MHF
705  (
706  MHFModel_->MHF
707  (
708  liquid,
709  vapor,
710  patchi,
711  Tl,
712  Tsatw,
713  L
714  )
715  );
716 
717  TLeiden =
718  LeidenfrostModel_->TLeid
719  (
720  liquid,
721  vapor,
722  patchi,
723  Tl,
724  Tsatw,
725  L
726  );
727 
728  // htc for film boiling
729  htcFilmBoiling =
730  filmBoilingModel_->htcFilmBoil
731  (
732  liquid,
733  vapor,
734  patchi,
735  Tl,
736  Tsatw,
737  L
738  );
739 
740  // htc for film transition boiling
741 
742  // Indicator between CHF (phi = 0) and MHF (phi = 1)
743  const scalarField phi
744  (
745  min
746  (
747  max
748  (
749  wp_*(Tw - tDNB)/(TLeiden - tDNB),
750  scalar(0)
751  ),
752  scalar(1)
753  )
754  );
755 
756  Qtb = CHFtotal*(1 - phi) + phi*MHF;
757 
758  }
759 
760 
761  // Sub-cool boiling Nucleation
762  const scalarField N
763  (
764  nucleationSiteModel_->N
765  (
766  liquid,
767  vapor,
768  patchi,
