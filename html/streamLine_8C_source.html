116  particles,

161 bool Foam::functionObjects::streamLine::read(const dictionary& dict)
162 {
163  if (streamLineBase::read(dict))
164  {
165  bool subCycling = dict.found("nSubCycle");
166  bool fixedLength = dict.found("trackLength");
167 
168  if (subCycling && fixedLength)
169  {
170  FatalIOErrorInFunction(dict)
171  << "Cannot both specify automatic time stepping (through '"
172  << "nSubCycle' specification) and fixed track length (through '"
173  << "trackLength')"
174  << exit(FatalIOError);
175  }
176 
177  nSubCycle_ = 1;
178  if (dict.readIfPresent("nSubCycle", nSubCycle_))
179  {
180  trackLength_ = VGREAT;
181  nSubCycle_ = max(nSubCycle_, 1);
182 
183 
184  Info<< " automatic track length specified through"
185  << " number of sub cycles : " << nSubCycle_ << nl
186  << endl;
187  }
188  }
189  return true;
190 }
