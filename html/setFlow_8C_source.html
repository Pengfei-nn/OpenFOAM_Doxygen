68 void Foam::functionObjects::setFlow::setPhi(const volVectorField& U)
69 {
70  surfaceScalarField* phiptr =
71  mesh_.getObjectPtr<surfaceScalarField>(phiName_);
72 
73  if (!phiptr)
74  {
75  return;
76  }
77 
78  if (rhoName_ != "none")
79  {
80  const volScalarField* rhoptr =
81  mesh_.findObject<volScalarField>(rhoName_);
82 
83  if (rhoptr)
84  {
85  const volScalarField& rho = *rhoptr;
86  *phiptr = fvc::flux(rho*U);
87  }
88  else
89  {
90  FatalErrorInFunction
91  << "Unable to find rho field'" << rhoName_
92  << "' in the mesh database. Available fields are:"
93  << mesh_.names<volScalarField>()
94  << exit(FatalError);
95  }
96  }
97  else
98  {
99  *phiptr = fvc::flux(U);
100  }
101 }

137 bool Foam::functionObjects::setFlow::read(const dictionary& dict)
138 {
139  if (fvMeshFunctionObject::read(dict))
140  {
141  Info<< name() << ":" << endl;
142 
143  modeTypeNames.readEntry("mode", dict, mode_);
144 
145  Info<< " operating mode: " << modeTypeNames[mode_] << endl;
146 
147  if (dict.readIfPresent("U", UName_))
148  {
149  Info<< " U field name: " << UName_ << endl;
150  }
151 
152  if (dict.readIfPresent("rho", rhoName_))
153  {
154  Info<< " rho field name: " << rhoName_ << endl;
155  }
156 
157  if (dict.readIfPresent("phi", phiName_))
158  {
159  Info<< " phi field name: " << phiName_ << endl;
160  }
161 
162  if (dict.readIfPresent("reverseTime", reverseTime_))
163  {
164  Info<< " reverse flow direction at time: " << reverseTime_
165  << endl;
166  reverseTime_ = mesh_.time().userTimeToTime(reverseTime_);
167  }
168 
169  // Scaling is applied across all modes
170  scalePtr_ = Function1<scalar>::New("scale", dict);
171 
172  switch (mode_)
173  {
174  case modeType::FUNCTION:
175  {
176  velocityPtr_ = Function1<vector>::New("velocity", dict);
177  break;
178  }
179  case modeType::ROTATION:
180  {
181  omegaPtr_ = Function1<scalar>::New("omega", dict);
182 
183  dict.readEntry("origin", origin_);
184  const vector refDir(dict.get<vector>("refDir").normalise());
185  const vector axis(dict.get<vector>("axis").normalise());
186 
187  R_ = tensor(refDir, axis, refDir^axis);
188  break;
189  }
190  case modeType::VORTEX2D:
191  case modeType::VORTEX3D:
192  {
193  dict.readEntry("origin", origin_);
194  const vector refDir(dict.get<vector>("refDir").normalise());
195  const vector axis(dict.get<vector>("axis").normalise());
196 
197  R_ = tensor(refDir, axis, refDir^axis);
198  break;
199  }
200  }
201 
202  Info<< endl;
203 
204  return true;
205  }
206 
207  return false;
208 }

211 bool Foam::functionObjects::setFlow::execute()
212 {
213  volVectorField* Uptr =
214  mesh_.getObjectPtr<volVectorField>(UName_);
215 
216  surfaceScalarField* phiptr =
217  mesh_.getObjectPtr<surfaceScalarField>(phiName_);
218 
219  Log << nl << name() << ":" << nl;
220 
221  if (!Uptr || !phiptr)
222  {
223  Info<< " Either field " << UName_ << " or " << phiName_
224  << " not found in the mesh database" << nl;
225 
226  return true;
227  }
228 
229  const scalar t = mesh_.time().timeOutputValue();
230 
231  Log << " setting " << UName_ << " and " << phiName_ << nl;
232 
233  volVectorField& U = *Uptr;
234  surfaceScalarField& phi = *phiptr;
235 
236  switch (mode_)
237  {
238  case modeType::FUNCTION:
239  {
240  const vector Uc = velocityPtr_->value(t);
241  U == dimensionedVector("Uc", dimVelocity, Uc);
242  U.correctBoundaryConditions();
243  setPhi(U);
244 
245  break;
246  }
247  case modeType::ROTATION:
248  {
249  const volVectorField& C = mesh_.C();
250  const volVectorField d
251  (
252  typeName + ":d",
253  C - dimensionedVector("origin", dimLength, origin_)
254  );
255  const scalarField x(d.component(vector::X));
256  const scalarField z(d.component(vector::Z));
257 
258  const scalar omega = omegaPtr_->value(t);
259  vectorField& Uc = U.primitiveFieldRef();
260  Uc.replace(vector::X, -omega*z);
261  Uc.replace(vector::Y, scalar(0));
262  Uc.replace(vector::Z, omega*x);
263 
264  volVectorField::Boundary& Ubf = U.boundaryFieldRef();
265  forAll(Ubf, patchi)
266  {
267  vectorField& Uf = Ubf[patchi];
268  if (Uf.size())
269  {
270  const vectorField& Cp = C.boundaryField()[patchi];
271  const vectorField dp(Cp - origin_);
272  const scalarField xp(dp.component(vector::X));
273  const scalarField zp(dp.component(vector::Z));
274  Uf.replace(vector::X, -omega*zp);
275  Uf.replace(vector::Y, scalar(0));
276  Uf.replace(vector::Z, omega*xp);
277  }
278  }
279 
280  U = U & R_;
281  U.correctBoundaryConditions();
282  setPhi(U);
283 
284  break;
285  }
286  case modeType::VORTEX2D:
287  {
288  const scalar pi = Foam::constant::mathematical::pi;
289 
290  const volVectorField& C = mesh_.C();
291 
292  const volVectorField d
293  (
294  typeName + ":d",
295  C - dimensionedVector("origin", dimLength, origin_)
296  );
297  const scalarField x(d.component(vector::X));
298  const scalarField z(d.component(vector::Z));
299 
300  vectorField& Uc = U.primitiveFieldRef();
301  Uc.replace(vector::X, -sin(2*pi*z)*sqr(sin(pi*x)));
302  Uc.replace(vector::Y, scalar(0));
303  Uc.replace(vector::Z, sin(2*pi*x)*sqr(sin(pi*z)));
304 
305  U = U & R_;
306 
307  // Calculating incompressible flux based on stream function
308  // Note: R_ rotation not implemented in phi calculation
309  const scalarField xp(mesh_.points().component(0) - origin_[0]);
310  const scalarField yp(mesh_.points().component(1) - origin_[1]);
311  const scalarField zp(mesh_.points().component(2) - origin_[2]);
312  const scalarField psi((1.0/pi)*sqr(sin(pi*xp))*sqr(sin(pi*zp)));
313 
314  scalarField& phic = phi.primitiveFieldRef();
315  forAll(phic, fi)
316  {
317  phic[fi] = 0;
318  const face& f = mesh_.faces()[fi];
319  const label nPoints = f.size();
320 
321  forAll(f, fpi)
322  {
323  const label p1 = f[fpi];
324  const label p2 = f[(fpi + 1) % nPoints];
325  phic[fi] += 0.5*(psi[p1] + psi[p2])*(yp[p2] - yp[p1]);
326  }
327  }
328 
329  surfaceScalarField::Boundary& phibf = phi.boundaryFieldRef();
330  forAll(phibf, patchi)
331  {
332  scalarField& phif = phibf[patchi];
333  const label start = mesh_.boundaryMesh()[patchi].start();
334 
335  forAll(phif, fi)
336  {
337  phif[fi] = 0;
338  const face& f = mesh_.faces()[start + fi];
339  const label nPoints = f.size();
340 
341  forAll(f, fpi)
342  {
343  const label p1 = f[fpi];
344  const label p2 = f[(fpi + 1) % nPoints];
345  phif[fi] += 0.5*(psi[p1] + psi[p2])*(yp[p2] - yp[p1]);
346  }
347  }
348  }
349 
350  break;
351  }
352  case modeType::VORTEX3D:
353  {
354  const scalar pi = Foam::constant::mathematical::pi;
355  const volVectorField& C = mesh_.C();
356 
357  const volVectorField d
358  (
359  typeName + ":d",
360  C - dimensionedVector("origin", dimLength, origin_)
361  );
362  const scalarField x(d.component(vector::X));
363  const scalarField y(d.component(vector::Y));
364  const scalarField z(d.component(vector::Z));
365 
366  vectorField& Uc = U.primitiveFieldRef();
367  Uc.replace(vector::X, 2*sqr(sin(pi*x))*sin(2*pi*y)*sin(2*pi*z));
368  Uc.replace(vector::Y, -sin(2*pi*x)*sqr(sin(pi*y))*sin(2*pi*z));
369  Uc.replace(vector::Z, -sin(2*pi*x)*sin(2*pi*y)*sqr(sin(pi*z)));
370 
371  U = U & R_;
372  U.correctBoundaryConditions();
373 
374  // Calculating phi
375  // Note: R_ rotation not implemented in phi calculation
376  const vectorField Cf(mesh_.Cf().primitiveField() - origin_);
377  const scalarField Xf(Cf.component(vector::X));
378  const scalarField Yf(Cf.component(vector::Y));
379  const scalarField Zf(Cf.component(vector::Z));
380  vectorField Uf(Xf.size());
381  Uf.replace(0, 2*sqr(sin(pi*Xf))*sin(2*pi*Yf)*sin(2*pi*Zf));
382  Uf.replace(1, -sin(2*pi*Xf)*sqr(sin(pi*Yf))*sin(2*pi*Zf));
383  Uf.replace(2, -sin(2*pi*Xf)*sin(2*pi*Yf)*sqr(sin(pi*Zf)));
384 
385  scalarField& phic = phi.primitiveFieldRef();
386  const vectorField& Sfc = mesh_.Sf().primitiveField();
387  phic = Uf & Sfc;
388 
389  surfaceScalarField::Boundary& phibf = phi.boundaryFieldRef();
390  const surfaceVectorField::Boundary& Sfbf =
391  mesh_.Sf().boundaryField();
392  const surfaceVectorField::Boundary& Cfbf =
393  mesh_.Cf().boundaryField();
394 
395  forAll(phibf, patchi)
396  {
397  scalarField& phif = phibf[patchi];
398  const vectorField& Sff = Sfbf[patchi];
399  const vectorField& Cff = Cfbf[patchi];
400  const scalarField xf(Cff.component(vector::X));
401  const scalarField yf(Cff.component(vector::Y));
402  const scalarField zf(Cff.component(vector::Z));
403  vectorField Uf(xf.size());
404  Uf.replace(0, 2*sqr(sin(pi*xf))*sin(2*pi*yf)*sin(2*pi*zf));
405  Uf.replace(1, -sin(2*pi*xf)*sqr(sin(pi*yf))*sin(2*pi*zf));
406  Uf.replace(2, -sin(2*pi*xf)*sin(2*pi*yf)*sqr(sin(pi*zf)));
407  phif = Uf & Sff;
408  }
409 
410  break;
411  }
412  }
413 
414  if (t > reverseTime_)
415  {
416  Log << " flow direction: reverse" << nl;
417  U.negate();
418  phi.negate();
419  }
420 
421  // Apply scaling
422  const scalar s = scalePtr_->value(t);
423  U *= s;
424  phi *= s;
425 
426  U.correctBoundaryConditions();
427 
428  const scalarField sumPhi(fvc::surfaceIntegrate(phi));
429  Log << " Continuity error: max(mag(sum(phi))) = "
430  << gMax(mag(sumPhi)) << nl << endl;
431 
432  return true;
433 }
