120 void adjointMeshMovementSolver::solve()
121 {
122  read();
123 
124  // Add source from the adjoint eikonal equation
125  if (adjointEikonalSolverPtr_.valid())
126  {
127  source_ -=
128  fvc::div(adjointEikonalSolverPtr_().getFISensitivityTerm()().T());
129  }
130 
131  // Iterate the adjoint to the mesh movement equation
132  for (label iter = 0; iter < nLaplaceIters_; iter++)
133  {
134  Info<< "Adjoint Mesh Movement Iteration: " << iter << endl;
135 
136  fvVectorMatrix maEqn
137  (
138  fvm::laplacian(ma_)
139  + source_
140  );
141 
142  maEqn.boundaryManipulate(ma_.boundaryFieldRef());
143 
144  //scalar residual = max(maEqn.solve().initialResidual());
145  scalar residual = mag(maEqn.solve().initialResidual());
146 
147  Info<< "Max ma " << gMax(mag(ma_)()) << endl;
148 
149  Info<< "ExecutionTime = " << mesh_.time().elapsedCpuTime() << " s"
150  << " ClockTime = " << mesh_.time().elapsedClockTime() << " s"
151  << nl << endl;
152 
153  // Check convergence
154  if (residual < tolerance_)
155  {
156  Info<< "\n***Reached adjoint mesh movement convergence limit, "
157  "iteration " << iter << "***\n\n";
158  break;
159  }
160  }
161  ma_.write();
162 }
