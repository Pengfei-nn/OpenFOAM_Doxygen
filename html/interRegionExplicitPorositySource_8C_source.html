52 void Foam::fv::interRegionExplicitPorositySource::initialise()
53 {
54  if (!firstIter_)
55  {
56  return;
57  }
58 
59  const word zoneName(name_ + ":porous");
60 
61  const fvMesh& nbrMesh = mesh_.time().lookupObject<fvMesh>(nbrRegionName_);
62  const cellZoneMesh& cellZones = nbrMesh.cellZones();
63  label zoneID = cellZones.findZoneID(zoneName);
64 
65  if (zoneID == -1)
66  {
67  cellZoneMesh& cz = const_cast<cellZoneMesh&>(cellZones);
68 
69  zoneID = cz.size();
70 
71  cz.setSize(zoneID + 1);
72 
73  cz.set
74  (
75  zoneID,
76  new cellZone
77  (
78  zoneName,
79  nbrMesh.faceNeighbour(), // Neighbour internal cells
80  zoneID,
81  cellZones
82  )
83  );
84 
85  cz.clearAddressing();
86  }
87  else
88  {
89  FatalErrorInFunction
90  << "Unable to create porous cellZone " << zoneName
91  << ": zone already exists"
92  << abort(FatalError);
93  }
94 
95  porosityPtr_.reset
96  (
97  porosityModel::New
98  (
99  name_,
100  nbrMesh,
101  coeffs_,
102  zoneName
103  ).ptr()
104  ),
105 
106  firstIter_ = false;
107 }

136 void Foam::fv::interRegionExplicitPorositySource::addSup
137 (
138  fvMatrix<vector>& eqn,
139  const label fieldi
140 )
141 {
142  initialise();
143 
144  const fvMesh& nbrMesh = mesh_.time().lookupObject<fvMesh>(nbrRegionName_);
145 
146  const volVectorField& U = eqn.psi();
147 
148  volVectorField UNbr
149  (
150  IOobject
151  (
152  name_ + ":UNbr",
153  nbrMesh.time().timeName(),
154  nbrMesh,
155  IOobject::NO_READ,
156  IOobject::NO_WRITE
157  ),
158  nbrMesh,
159  dimensionedVector(U.dimensions(), Zero)
160  );
161 
162  // Map local velocity onto neighbour region
163  meshInterp().mapSrcToTgt
164  (
165  U.primitiveField(),
166  plusEqOp<vector>(),
167  UNbr.primitiveFieldRef()
168  );
169 
170  fvMatrix<vector> nbrEqn(UNbr, eqn.dimensions());
171 
172  porosityPtr_->addResistance(nbrEqn);
173 
174  // Convert source from neighbour to local region
175  fvMatrix<vector> porosityEqn(U, eqn.dimensions());
176  scalarField& Udiag = porosityEqn.diag();
177  vectorField& Usource = porosityEqn.source();
178 
179  Udiag.setSize(eqn.diag().size(), 0.0);
180  Usource.setSize(eqn.source().size(), Zero);
181 
182  meshInterp().mapTgtToSrc(nbrEqn.diag(), plusEqOp<scalar>(), Udiag);
183  meshInterp().mapTgtToSrc(nbrEqn.source(), plusEqOp<vector>(), Usource);
184 
185  eqn -= porosityEqn;
186 }
