214 void Foam::radiation::P1::calculate()
215 {
216  a_ = absorptionEmission_->a();
217  e_ = absorptionEmission_->e();
218  E_ = absorptionEmission_->E();
219  const volScalarField sigmaEff(scatter_->sigmaEff());
220 
221  const dimensionedScalar a0("a0", a_.dimensions(), ROOTVSMALL);
222 
223  // Construct diffusion
224  const volScalarField gamma
225  (
226  IOobject
227  (
228  "gammaRad",
229  G_.mesh().time().timeName(),
230  G_.mesh(),
231  IOobject::NO_READ,
232  IOobject::NO_WRITE
233  ),
234  1.0/(3.0*a_ + sigmaEff + a0)
235  );
236 
237  // Solve G transport equation
238  solve
239  (
240  fvm::laplacian(gamma, G_)
241  - fvm::Sp(a_, G_)
242  ==
243  - 4.0*(e_*physicoChemical::sigma*pow4(T_) ) - E_
244  );
245 
246  volScalarField::Boundary& qrBf = qr_.boundaryFieldRef();
247 
248  // Calculate radiative heat flux on boundaries.
249  forAll(mesh_.boundaryMesh(), patchi)
250  {
251  if (!G_.boundaryField()[patchi].coupled())
252  {
253  qrBf[patchi] =
254  -gamma.boundaryField()[patchi]
255  *G_.boundaryField()[patchi].snGrad();
256  }
257  }
258 }
