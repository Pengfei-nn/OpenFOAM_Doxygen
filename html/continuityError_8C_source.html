102 bool Foam::functionObjects::continuityError::write()
103 {
104  const auto phiPtr = mesh_.lookupObjectPtr<surfaceScalarField>(phiName_);
105 
106  if (!phiPtr)
107  {
108  WarningInFunction
109  << "Unable to find flux field " << phiName_
110  << endl;
111 
112  return false;
113  }
114 
115  const volScalarField error(fvc::div(*phiPtr));
116  const scalar deltaT = mesh_.time().deltaTValue();
117 
118  scalar local = deltaT*mag(error)().weightedAverage(mesh_.V()).value();
119  scalar global = deltaT*error.weightedAverage(mesh_.V()).value();
120  cumulative_ += global;
121 
122  Ostream& os = file();
123 
124  writeCurrentTime(os);
125 
126  os << local << tab
127  << global << tab
128  << cumulative_ << endl;
129 
130  Log << type() << " " << name() << " write:" << nl
131  << " local = " << local << nl
132  << " global = " << global << nl
133  << " cumulative = " << cumulative_ << nl
134  << endl;
135 
136  setResult("local", local);
137  setResult("global", global);
138  setResult("cumulative", cumulative_);
139 
140  setProperty<scalar>("cumulative", cumulative_);
141 
142  return true;
143 }
