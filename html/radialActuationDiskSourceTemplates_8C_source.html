37 void Foam::fv::radialActuationDiskSource::
38 addRadialActuationDiskAxialInertialResistance
39 (
40  vectorField& Usource,
41  const labelList& cells,
42  const scalarField& Vcells,
43  const RhoFieldType& rho,
44  const vectorField& U
45 ) const
46 {
47  scalar a = 1.0 - Cp_/Ct_;
48  scalarField Tr(cells.size());
49  const vector uniDiskDir = diskDir_/mag(diskDir_);
50 
51  tensor E(Zero);
52  E.xx() = uniDiskDir.x();
53  E.yy() = uniDiskDir.y();
54  E.zz() = uniDiskDir.z();
55 
56  const Field<vector> zoneCellCentres(mesh().cellCentres(), cells);
57  const Field<scalar> zoneCellVolumes(mesh().cellVolumes(), cells);
58 
59  const vector avgCentre = gSum(zoneCellVolumes*zoneCellCentres)/V();
60  const scalar maxR = gMax(mag(zoneCellCentres - avgCentre));
61 
62  scalar intCoeffs =
63  radialCoeffs_[0]
64  + radialCoeffs_[1]*sqr(maxR)/2.0
65  + radialCoeffs_[2]*pow4(maxR)/3.0;
66 
67  vector upU = vector(VGREAT, VGREAT, VGREAT);
68  scalar upRho = VGREAT;
69  if (upstreamCellId_ != -1)
70  {
71  upU = U[upstreamCellId_];
72  upRho = rho[upstreamCellId_];
73  }
74  reduce(upU, minOp<vector>());
75  reduce(upRho, minOp<scalar>());
76 
77  scalar T = 2.0*upRho*diskArea_*mag(upU)*a*(1.0 - a);
78  forAll(cells, i)
79  {
80  scalar r2 = magSqr(mesh().cellCentres()[cells[i]] - avgCentre);
81 
82  Tr[i] =
83  T
84  *(radialCoeffs_[0] + radialCoeffs_[1]*r2 + radialCoeffs_[2]*sqr(r2))
85  /intCoeffs;
86 
87  Usource[cells[i]] += ((Vcells[cells[i]]/V_)*Tr[i]*E) & upU;
88  }
89 
90  if (debug)
91  {
92  Info<< "Source name: " << name() << nl
93  << "Average centre: " << avgCentre << nl
94  << "Maximum radius: " << maxR << endl;
95  }
96 }
