32 Description
33  Converts from one surface mesh format to another.
34 
35 Usage
36  \b surfaceConvert inputFile outputFile [OPTION]
37 
38  Options:
39  - \par -clean
40  Perform some surface checking/cleanup on the input surface
41 
42  - \par -scale <scale>
43  Specify a scaling factor for writing the files
44 
45  - \par -group
46  Orders faces by region
47 
48 Note
49  The filename extensions are used to determine the file format type.

65 int main(int argc, char *argv[])
66 {

74  argList::addArgument("output", "The output surface file");
75 
76  argList::addBoolOption
77  (
78  "clean",
79  "Perform some surface checking/cleanup on the input surface"
80  );
81  argList::addBoolOption
82  (
83  "group",
84  "Reorder faces into groups; one per region"
85  );
86  argList::addOption
87  (
88  "scale",
89  "factor",
90  "Input geometry scaling factor"
91  );
92  argList::addOption
93  (
94  "precision",
95  "int",
96  "Write to output with the specified precision"
97  );
98  argList::addOptionCompat("precision", {"writePrecision", 1812});
99 
100  argList args(argc, argv);
101 
102  {
103  const unsigned prec = args.lookupOrDefault<unsigned>("precision", 0u);
104  if (prec)
105  {
106  Info<< "Output write precision set to " << prec << endl;
107 
108  IOstream::defaultPrecision(prec);
109  Sout.precision(prec);
110  }
111  }
112 
113  const fileName importName = args[1];
114  const fileName exportName = args[2];
115 
116  if (importName == exportName)
117  {
118  FatalErrorInFunction
119  << "Output file " << exportName << " would overwrite input file."
120  << exit(FatalError);
121  }
122 
123  // Check that reading/writing is supported
124  if
125  (
126  !triSurface::canRead(importName, true)
127  || !triSurface::canWriteType(exportName.ext(), true)
128  )
129  {
130  return 1;
131  }
132 
133  const scalar scaleFactor = args.get<scalar>("scale", -1);
134 
135  Info<< "Reading : " << importName << endl;
136  triSurface surf(importName, scaleFactor);
137 
138  Info<< "Read surface:" << endl;
139  surf.writeStats(Info);
140  Info<< endl;
141 
142  if (args.found("clean"))
143  {
144  Info<< "Cleaning up surface" << endl;
145  surf.cleanup(true);
146 
147  Info<< "After cleaning up surface:" << endl;
148  surf.writeStats(Info);
149  Info<< endl;
150  }
151 
152  const bool sortByRegion = args.found("group");
153  if (sortByRegion)
154  {
155  Info<< "Reordering faces into groups; one per region." << endl;
156  }
157  else
158  {
159  Info<< "Maintaining face ordering" << endl;
160  }
161 
162  Info<< "writing " << exportName;
163  Info<< endl;
164 
165  surf.write(exportName, sortByRegion);
166 
167  Info<< "\nEnd\n" << endl;
168 
169  return 0;
170 }
