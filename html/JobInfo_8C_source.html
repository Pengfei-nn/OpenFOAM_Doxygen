105  if (writeJobInfo && Pstream::master())
106  {
107  string jobDir = Foam::getEnv("FOAM_JOB_DIR");
108  if (jobDir.empty())

110  jobDir = home()/FOAM_RESOURCE_USER_CONFIG_DIRNAME/"jobControl";

117  if (!isDir(jobDir) && !mkDir(jobDir))

131  FatalErrorInFunction
