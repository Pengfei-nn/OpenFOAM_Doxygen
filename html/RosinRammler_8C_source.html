78 Foam::scalar Foam::distributionModels::RosinRammler::sample() const
79 {
80  scalar K = 1.0 - exp(-pow((maxValue_ - minValue_)/d_, n_));
81  scalar y = rndGen_.sample01<scalar>();
82  scalar x = minValue_ + d_*::pow(-log(1.0 - y*K), 1.0/n_);
83  return x;
84 }
