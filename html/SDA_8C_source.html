69 Foam::septernion Foam::solidBodyMotionFunctions::SDA::transformation() const
70 {
71  scalar time = time_.value();
72 
73  scalar Tpi = Tp_ + dTp_*(time/dTi_); // Current roll period [sec]
74  scalar wr = twoPi/Tpi; // Current Freq [/sec]
75 
76  // Current Phase for roll [rad]
77  scalar r = dTp_/dTi_;
78  scalar u = Tp_ + r*time;
79  scalar phr = twoPi*((Tp_/u - 1) + log(mag(u)) - log(Tp_))/r;
80 
81  // Current Phase for Sway [rad]
82  scalar phs = phr + pi;
83 
84  // Current Phase for Heave [rad]
85  scalar phh = phr + piByTwo;
86 
87  scalar rollA = max(rollAmax_*exp(-sqr(Tpi - Tpn_)/(2*Q_)), rollAmin_);
88 
89  vector T
90  (
91  0,
92  swayA_*(sin(wr*time + phs) - sin(phs)),
93  heaveA_*(sin(wr*time + phh) - sin(phh))
94  );
95  quaternion R(quaternion::XYZ, vector(rollA*sin(wr*time + phr), 0, 0));
96  septernion TR(septernion(-CofG_ - T)*R*septernion(CofG_));
97 
98  DebugInFunction << "Time = " << time << " transformation: " << TR << endl;
99 
100  return TR;
101 }

104 bool Foam::solidBodyMotionFunctions::SDA::read(const dictionary& SBMFCoeffs)
105 {
106  solidBodyMotionFunction::read(SBMFCoeffs);
107 
108  SBMFCoeffs_.readEntry("CofG", CofG_);
109  SBMFCoeffs_.readEntry("lamda", lamda_);
110  SBMFCoeffs_.readEntry("rollAmax", rollAmax_);
111  SBMFCoeffs_.readEntry("rollAmin", rollAmin_);
112  SBMFCoeffs_.readEntry("heaveA", heaveA_);
113  SBMFCoeffs_.readEntry("swayA", swayA_);
114  SBMFCoeffs_.readEntry("Q", Q_);
115  SBMFCoeffs_.readEntry("Tp", Tp_);
116  SBMFCoeffs_.readEntry("Tpn", Tpn_);
117  SBMFCoeffs_.readEntry("dTi", dTi_);
118  SBMFCoeffs_.readEntry("dTp", dTp_);
119 
120  // Rescale parameters according to the given scale parameter
121  if (lamda_ > 1 + SMALL)
122  {
123  heaveA_ /= lamda_;
124  swayA_ /= lamda_;
125  Tp_ /= sqrt(lamda_);
126  Tpn_ /= sqrt(lamda_);
127  dTi_ /= sqrt(lamda_);
128  dTp_ /= sqrt(lamda_);
129  }
130 
131  return true;
132 }
