166 template<class ReactionThermo, class ThermoType>
167 Foam::scalar Foam::StandardChemistryModel<ReactionThermo, ThermoType>::omega
168 (
169  const Reaction<ThermoType>& R,
170  const scalarField& c,
171  const scalar T,
172  const scalar p,
173  scalar& pf,
174  scalar& cf,
175  label& lRef,
176  scalar& pr,
177  scalar& cr,
178  label& rRef
179 ) const
180 {
181  const scalar kf = R.kf(p, T, c);
182  const scalar kr = R.kr(kf, p, T, c);
183 
184  pf = 1.0;
185  pr = 1.0;
186 
187  const label Nl = R.lhs().size();
188  const label Nr = R.rhs().size();
189 
190  label slRef = 0;
191  lRef = R.lhs()[slRef].index;
192 
193  pf = kf;
194  for (label s = 1; s < Nl; s++)
195  {
196  const label si = R.lhs()[s].index;
197 
198  if (c[si] < c[lRef])
199  {
200  const scalar exp = R.lhs()[slRef].exponent;
201  pf *= pow(max(c[lRef], 0.0), exp);
202  lRef = si;
203  slRef = s;
204  }
205  else
206  {
207  const scalar exp = R.lhs()[s].exponent;
208  pf *= pow(max(c[si], 0.0), exp);
209  }
210  }
211  cf = max(c[lRef], 0.0);
212 
213  {
214  const scalar exp = R.lhs()[slRef].exponent;
215  if (exp < 1.0)
216  {
217  if (cf > SMALL)
218  {
219  pf *= pow(cf, exp - 1.0);
220  }
221  else
222  {
223  pf = 0.0;
224  }
225  }
226  else
227  {
228  pf *= pow(cf, exp - 1.0);
229  }
230  }
231 
232  label srRef = 0;
233  rRef = R.rhs()[srRef].index;
234 
235  // Find the matrix element and element position for the rhs
236  pr = kr;
237  for (label s = 1; s < Nr; s++)
238  {
239  const label si = R.rhs()[s].index;
240  if (c[si] < c[rRef])
241  {
242  const scalar exp = R.rhs()[srRef].exponent;
243  pr *= pow(max(c[rRef], 0.0), exp);
244  rRef = si;
245  srRef = s;
246  }
247  else
248  {
249  const scalar exp = R.rhs()[s].exponent;
250  pr *= pow(max(c[si], 0.0), exp);
251  }
252  }
253  cr = max(c[rRef], 0.0);
254 
255  {
256  const scalar exp = R.rhs()[srRef].exponent;
257  if (exp < 1.0)
258  {
259  if (cr>SMALL)
260  {
261  pr *= pow(cr, exp - 1.0);
262  }
263  else
264  {
265  pr = 0.0;
266  }
267  }
268  else
269  {
270  pr *= pow(cr, exp - 1.0);
271  }
272  }
273 
274  return pf*cf - pr*cr;
275 }

278 template<class ReactionThermo, class ThermoType>
279 void Foam::StandardChemistryModel<ReactionThermo, ThermoType>::derivatives
280 (
281  const scalar time,
282  const scalarField& c,
283  scalarField& dcdt
284 ) const
285 {
286  const scalar T = c[nSpecie_];
287  const scalar p = c[nSpecie_ + 1];
288 
289  forAll(c_, i)
290  {
291  c_[i] = max(c[i], 0.0);
292  }
293 
294  omega(c_, T, p, dcdt);
295 
296  // Constant pressure
297  // dT/dt = ...
298  scalar rho = 0.0;
299  scalar cSum = 0.0;
300  for (label i = 0; i < nSpecie_; i++)
301  {
302  const scalar W = specieThermo_[i].W();
303  cSum += c_[i];
304  rho += W*c_[i];
305  }
306  scalar cp = 0.0;
307  for (label i=0; i<nSpecie_; i++)
308  {
309  cp += c_[i]*specieThermo_[i].cp(p, T);
310  }
311  cp /= rho;
312 
313  scalar dT = 0.0;
314  for (label i = 0; i < nSpecie_; i++)
315  {
316  const scalar hi = specieThermo_[i].ha(p, T);
317  dT += hi*dcdt[i];
318  }
319  dT /= rho*cp;
320 
321  dcdt[nSpecie_] = -dT;
322 
323  // dp/dt = ...
324  dcdt[nSpecie_ + 1] = 0.0;
325 }

328 template<class ReactionThermo, class ThermoType>
329 void Foam::StandardChemistryModel<ReactionThermo, ThermoType>::jacobian
330 (
331  const scalar t,
332  const scalarField& c,
333  scalarField& dcdt,
334  scalarSquareMatrix& dfdc
335 ) const
336 {
337  const scalar T = c[nSpecie_];
338  const scalar p = c[nSpecie_ + 1];
339 
340  forAll(c_, i)
341  {
342  c_[i] = max(c[i], 0.0);
343  }
344 
345  dfdc = Zero;
346 
347  // Length of the first argument must be nSpecie_
348  omega(c_, T, p, dcdt);
349 
350  forAll(reactions_, ri)
351  {
352  const Reaction<ThermoType>& R = reactions_[ri];
353 
354  const scalar kf0 = R.kf(p, T, c_);
355  const scalar kr0 = R.kr(kf0, p, T, c_);
356 
357  forAll(R.lhs(), j)
358  {
359  const label sj = R.lhs()[j].index;
360  scalar kf = kf0;
361  forAll(R.lhs(), i)
362  {
363  const label si = R.lhs()[i].index;
364  const scalar el = R.lhs()[i].exponent;
365  if (i == j)
366  {
367  if (el < 1.0)
368  {
369  if (c_[si] > SMALL)
370  {
371  kf *= el*pow(c_[si], el - 1.0);
372  }
373  else
374  {
375  kf = 0.0;
376  }
377  }
378  else
379  {
380  kf *= el*pow(c_[si], el - 1.0);
381  }
382  }
383  else
384  {
385  kf *= pow(c_[si], el);
386  }
387  }
388 
389  forAll(R.lhs(), i)
390  {
391  const label si = R.lhs()[i].index;
392  const scalar sl = R.lhs()[i].stoichCoeff;
393  dfdc(si, sj) -= sl*kf;
394  }
395  forAll(R.rhs(), i)
396  {
397  const label si = R.rhs()[i].index;
398  const scalar sr = R.rhs()[i].stoichCoeff;
399  dfdc(si, sj) += sr*kf;
400  }
401  }
402 
403  forAll(R.rhs(), j)
404  {
405  const label sj = R.rhs()[j].index;
406  scalar kr = kr0;
407  forAll(R.rhs(), i)
408  {
409  const label si = R.rhs()[i].index;
410  const scalar er = R.rhs()[i].exponent;
411  if (i == j)
412  {
413  if (er < 1.0)
414  {
415  if (c_[si] > SMALL)
416  {
417  kr *= er*pow(c_[si], er - 1.0);
418  }
419  else
420  {
421  kr = 0.0;
422  }
423  }
424  else
425  {
426  kr *= er*pow(c_[si], er - 1.0);
427  }
428  }
429  else
430  {
431  kr *= pow(c_[si], er);
432  }
433  }
434 
435  forAll(R.lhs(), i)
436  {
437  const label si = R.lhs()[i].index;
438  const scalar sl = R.lhs()[i].stoichCoeff;
439  dfdc(si, sj) += sl*kr;
440  }
441  forAll(R.rhs(), i)
442  {
443  const label si = R.rhs()[i].index;
444  const scalar sr = R.rhs()[i].stoichCoeff;
445  dfdc(si, sj) -= sr*kr;
446  }
447  }
448  }
449 
450  // Calculate the dcdT elements numerically
451  const scalar delta = 1.0e-3;
452 
453  omega(c_, T + delta, p, dcdt_);
454  for (label i=0; i<nSpecie_; i++)
455  {
456  dfdc(i, nSpecie_) = dcdt_[i];
457  }
458 
459  omega(c_, T - delta, p, dcdt_);
460  for (label i=0; i<nSpecie_; i++)
461  {
462  dfdc(i, nSpecie_) = 0.5*(dfdc(i, nSpecie_) - dcdt_[i])/delta;
463  }
464 
465  dfdc(nSpecie_, nSpecie_) = 0;
466  dfdc(nSpecie_ + 1, nSpecie_) = 0;
467 }

689 template<class ReactionThermo, class ThermoType>
690 template<class DeltaTType>
691 Foam::scalar Foam::StandardChemistryModel<ReactionThermo, ThermoType>::solve
692 (
693  const DeltaTType& deltaT
694 )
695 {
696  BasicChemistryModel<ReactionThermo>::correct();
697 
698  scalar deltaTMin = GREAT;
699 
700  if (!this->chemistry_)
701  {
702  return deltaTMin;
703  }
704 
705  tmp<volScalarField> trho(this->thermo().rho());
706  const scalarField& rho = trho();
707 
708  const scalarField& T = this->thermo().T();
709  const scalarField& p = this->thermo().p();
710 
711  scalarField c0(nSpecie_);
712 
713  forAll(rho, celli)
714  {
715  scalar Ti = T[celli];
716 
717  if (Ti > Treact_)
718  {
719  const scalar rhoi = rho[celli];
720  scalar pi = p[celli];
721 
722  for (label i=0; i<nSpecie_; i++)
723  {
724  c_[i] = rhoi*Y_[i][celli]/specieThermo_[i].W();
725  c0[i] = c_[i];
726  }
727 
728  // Initialise time progress
729  scalar timeLeft = deltaT[celli];
730 
731  // Calculate the chemical source terms
732  while (timeLeft > SMALL)
733  {
734  scalar dt = timeLeft;
735  this->solve(c_, Ti, pi, dt, this->deltaTChem_[celli]);
736  timeLeft -= dt;
737  }
738 
739  deltaTMin = min(this->deltaTChem_[celli], deltaTMin);
740 
741  this->deltaTChem_[celli] =
742  min(this->deltaTChem_[celli], this->deltaTChemMax_);
743 
744  for (label i=0; i<nSpecie_; i++)
745  {
746  RR_[i][celli] =
747  (c_[i] - c0[i])*specieThermo_[i].W()/deltaT[celli];
748  }
749  }
750  else
751  {
752  for (label i=0; i<nSpecie_; i++)
753  {
754  RR_[i][celli] = 0;
755  }
756  }
757  }
758 
759  return deltaTMin;
760 }
