52 {

54  << " points: "
55  << returnReduce(mesh.points().size(), sumOp<label>()) << nl;
56 
57  label nInternalPoints = returnReduce
58  (
59  mesh.nInternalPoints(),
60  sumOp<label>()
61  );
62 
63  if (nInternalPoints != -Pstream::nProcs())
64  {
65  Info<< " internal points: " << nInternalPoints << nl;
66 
67  if (returnReduce(mesh.nInternalPoints(), minOp<label>()) == -1)
68  {
69  WarningInFunction
70  << "Some processors have their points sorted into internal"
71  << " and external and some do not." << endl
72  << "This can cause problems later on." << endl;
73  }
74  }
75 
76  if (allTopology && nInternalPoints != -Pstream::nProcs())
77  {
78  label nEdges = returnReduce(mesh.nEdges(), sumOp<label>());
79  label nInternalEdges = returnReduce
80  (
81  mesh.nInternalEdges(),
82  sumOp<label>()
83  );
84  label nInternal1Edges = returnReduce
85  (
86  mesh.nInternal1Edges(),
87  sumOp<label>()
88  );
89  label nInternal0Edges = returnReduce
90  (
91  mesh.nInternal0Edges(),
92  sumOp<label>()
93  );
94 
95  Info<< " edges: " << nEdges << nl
96  << " internal edges: " << nInternalEdges << nl
97  << " internal edges using one boundary point: "
98  << nInternal1Edges-nInternal0Edges << nl
99  << " internal edges using two boundary points: "
100  << nInternalEdges-nInternal1Edges << nl;
101  }
102 
103  label nFaces = returnReduce(mesh.faces().size(), sumOp<label>());
104  label nIntFaces = returnReduce(mesh.faceNeighbour().size(), sumOp<label>());
105  label nCells = returnReduce(mesh.cells().size(), sumOp<label>());
106  label nPatches = mesh.boundaryMesh().size();
107 
108  Info<< " faces: " << nFaces << nl
109  << " internal faces: " << nIntFaces << nl
110  << " cells: " << nCells << nl
111  << " faces per cell: "
112  << (scalar(nFaces) + scalar(nIntFaces))/max(1, nCells) << nl
113  << " boundary patches: ";
114 
115  if (Pstream::parRun())
116  {
117  // Number of global patches and min-max range of total patches
118  Info<< mesh.boundaryMesh().nNonProcessor() << ' '
119  << returnReduce(labelMinMax(nPatches), minMaxOp<label>()) << nl;
120  }
121  else
122  {
123  Info<< nPatches << nl;
124  }
125 
126  Info<< " point zones: " << mesh.pointZones().size() << nl
127  << " face zones: " << mesh.faceZones().size() << nl
128  << " cell zones: " << mesh.cellZones().size() << nl
129  << endl;
130 
131  // Construct shape recognizers
132  hexMatcher hex;
133  prismMatcher prism;
134  wedgeMatcher wedge;
135  pyrMatcher pyr;
136  tetWedgeMatcher tetWedge;
137  tetMatcher tet;
138 
139  // Counters for different cell types
140  label nHex = 0;
141  label nWedge = 0;
142  label nPrism = 0;
143  label nPyr = 0;
144  label nTet = 0;
145  label nTetWedge = 0;
146  label nUnknown = 0;
147 
148  Map<label> polyhedralFaces;
149

151  {
152  if (hex.isA(mesh, celli))
153  {
154  nHex++;
155  }
156  else if (tet.isA(mesh, celli))
157  {
158  nTet++;
159  }
160  else if (pyr.isA(mesh, celli))
161  {
162  nPyr++;
163  }
164  else if (prism.isA(mesh, celli))
165  {
166  nPrism++;
167  }
168  else if (wedge.isA(mesh, celli))
169  {
170  nWedge++;
171  }
172  else if (tetWedge.isA(mesh, celli))
173  {
174  nTetWedge++;
175  }
176  else
177  {
178  nUnknown++;
179  polyhedralFaces(mesh.cells()[celli].size())++;
180  }
181  }
182 
183  reduce(nHex,sumOp<label>());
184  reduce(nPrism,sumOp<label>());
185  reduce(nWedge,sumOp<label>());
186  reduce(nPyr,sumOp<label>());
187  reduce(nTetWedge,sumOp<label>());
188  reduce(nTet,sumOp<label>());
189  reduce(nUnknown,sumOp<label>());
190 
191  Info<< "Overall number of cells of each type:" << nl
192  << " hexahedra: " << nHex << nl
193  << " prisms: " << nPrism << nl
194  << " wedges: " << nWedge << nl
195  << " pyramids: " << nPyr << nl
196  << " tet wedges: " << nTetWedge << nl
197  << " tetrahedra: " << nTet << nl
198  << " polyhedra: " << nUnknown
199  << endl;
200 
201  if (nUnknown > 0)
202  {
203  Pstream::mapCombineGather(polyhedralFaces, plusEqOp<label>());
204 
205  Info<< " Breakdown of polyhedra by number of faces:" << nl
206  << " faces" << " number of cells" << endl;
207 
208  const labelList sortedKeys = polyhedralFaces.sortedToc();
209 
210  forAll(sortedKeys, keyi)
211  {
212  const label nFaces = sortedKeys[keyi];
213 
214  Info<< setf(std::ios::right) << setw(13)
215  << nFaces << " " << polyhedralFaces[nFaces] << nl;
216  }
217  }
218 
219  Info<< endl;
220 }

314 void Foam::mergeAndWrite
315 (
316  surfaceWriter& writer,
317  const cellSet& set
318 )
319 {
320  const polyMesh& mesh = refCast<const polyMesh>(set.db());
321  const polyBoundaryMesh& pbm = mesh.boundaryMesh();
322 
323 
324  // Determine faces on outside of cellSet
325  bitSet isInSet(mesh.nCells());
326  for (const label celli : set)
327  {
328  isInSet.set(celli);
329  }
330 
331 
332  boolList bndInSet(mesh.nBoundaryFaces());
333  forAll(pbm, patchi)
334  {
335  const polyPatch& pp = pbm[patchi];
336  const labelList& fc = pp.faceCells();
337  forAll(fc, i)
338  {
339  bndInSet[pp.start()+i-mesh.nInternalFaces()] = isInSet[fc[i]];
340  }
341  }
342  syncTools::swapBoundaryFaceList(mesh, bndInSet);
343 
344 
345  DynamicList<label> outsideFaces(3*set.size());
346  for (label facei = 0; facei < mesh.nInternalFaces(); facei++)
347  {
348  const bool ownVal = isInSet[mesh.faceOwner()[facei]];
349  const bool neiVal = isInSet[mesh.faceNeighbour()[facei]];
350 
351  if (ownVal != neiVal)
352  {
353  outsideFaces.append(facei);
354  }
355  }
356 
357 
358  forAll(pbm, patchi)
359  {
360  const polyPatch& pp = pbm[patchi];
361  const labelList& fc = pp.faceCells();
362  if (pp.coupled())
363  {
364  forAll(fc, i)
365  {
366  label facei = pp.start()+i;
367 
368  const bool neiVal = bndInSet[facei-mesh.nInternalFaces()];
369  if (isInSet[fc[i]] && !neiVal)
370  {
371  outsideFaces.append(facei);
372  }
373  }
374  }
375  else
376  {
377  forAll(fc, i)
378  {
379  if (isInSet[fc[i]])
380  {
381  outsideFaces.append(pp.start()+i);
382  }
383  }
384  }
385  }
386 
387 
388  const indirectPrimitivePatch setPatch
389  (
390  IndirectList<face>(mesh.faces(), outsideFaces),
391  mesh.points()
392  );
393 
394  fileName outputDir
395  (
396  set.time().globalPath()
397  / functionObject::outputPrefix
398  / mesh.pointsInstance()
399  / set.name()
400  );
401  outputDir.clean();
402 
403  mergeAndWrite(mesh, writer, set.name(), setPatch, outputDir);
404 }

407 void Foam::mergeAndWrite
408 (
409  const writer<scalar>& writer,
410  const pointSet& set
411 )
412 {
413  const polyMesh& mesh = refCast<const polyMesh>(set.db());
414 
415  pointField mergedPts;
416  labelList mergedIDs;
417 
418  if (Pstream::parRun())
419  {
420  // Note: we explicitly do not merge the points
421  // (mesh.globalData().mergePoints etc) since this might
422  // hide any synchronisation problem
423 
424  globalIndex globalNumbering(mesh.nPoints());
425 
426  mergedPts.setSize(returnReduce(set.size(), sumOp<label>()));
427  mergedIDs.setSize(mergedPts.size());
428 
429  labelList setPointIDs(set.sortedToc());
430 
431  // Get renumbered local data
432  pointField myPoints(mesh.points(), setPointIDs);
433  labelList myIDs(globalNumbering.toGlobal(setPointIDs));
434 
435  if (Pstream::master())
436  {
437  // Insert master data first
438  label pOffset = 0;
439  SubList<point>(mergedPts, myPoints.size(), pOffset) = myPoints;
440  SubList<label>(mergedIDs, myIDs.size(), pOffset) = myIDs;
441  pOffset += myPoints.size();
442 
443  // Receive slave ones
444  for (int slave=1; slave<Pstream::nProcs(); slave++)
445  {
446  IPstream fromSlave(Pstream::commsTypes::scheduled, slave);
447 
448  pointField slavePts(fromSlave);
449  labelList slaveIDs(fromSlave);
450 
451  SubList<point>(mergedPts, slavePts.size(), pOffset) = slavePts;
452  SubList<label>(mergedIDs, slaveIDs.size(), pOffset) = slaveIDs;
453  pOffset += slaveIDs.size();
454  }
455  }
456  else
457  {
458  // Construct processor stream with estimate of size. Could
459  // be improved.
460  OPstream toMaster
461  (
462  Pstream::commsTypes::scheduled,
463  Pstream::masterNo(),
464  myPoints.byteSize() + myIDs.byteSize()
465  );
466  toMaster << myPoints << myIDs;
467  }
468  }
469  else
470  {
471  mergedIDs = set.sortedToc();
472  mergedPts = pointField(mesh.points(), mergedIDs);
473  }
474 
475 
476  // Write with scalar pointID
477  if (Pstream::master())
478  {
479  scalarField scalarPointIDs(mergedIDs.size());
480  forAll(mergedIDs, i)
481  {
482  scalarPointIDs[i] = 1.0*mergedIDs[i];
483  }
484 
485  coordSet points(set.name(), "distance", mergedPts, mag(mergedPts));
486 
487  List<const scalarField*> flds(1, &scalarPointIDs);
488 
489  wordList fldNames(1, "pointID");
490 
491  // Output e.g. pointSet p0 to
492  // postProcessing/<time>/p0.vtk
493  fileName outputDir
494  (
495  set.time().globalPath()
496  / functionObject::outputPrefix
497  / mesh.pointsInstance()
498  // set.name()
499  );
500  outputDir.clean();
501  mkDir(outputDir);
502 
503  fileName outputFile(outputDir/writer.getFileName(points, wordList()));
504  //fileName outputFile(outputDir/set.name());
505 
506  OFstream os(outputFile);
507 
508  writer.write(points, fldNames, flds, os);
509  }
510 }
