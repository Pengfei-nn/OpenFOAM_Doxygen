51 // Return the file location mode as a string.
52 //
53 // - u : location mask 0700
54 // - g : location mask 0070
55 // - o : location mask 0007
56 static inline std::string locationToString(unsigned short location)
57 {
58  std::string mode;
59 
60  if (location & 0700) { mode += 'u'; } // User
61  if (location & 0070) { mode += 'g'; } // Group
62  if (location & 0007) { mode += 'o'; } // Other
63  if (mode.empty()) { mode = "???"; }
64 
65  return mode;
66 }

95 // - ~/.OpenFOAM
96 static inline bool userResourceDir(Foam::fileName& queried)
97 {

105  return true;

107  #elif defined FULLDEBUG
108  #warning FOAM_RESOURCE_USER_CONFIG_DIRNAME \

110  #endif
111 
112  return false;

116 // Assign 'queried' parameter to the group resource directory.
117 // Return true if this directory exists.
118 // Otherwise clears the parameter and returns false.
119 //
120 // Corresponds to foamEtcFile -mode=g
121 // Looks for
122 // - ${WM_PROJECT_SITE}/etc
123 // - ${WM_PROJECT_DIR}/site/etc
124 //
125 // Optionally (compile-time defined):
126 // - FOAM_CONFIGURED_PROJECT_DIR/site/etc
127 static inline bool groupResourceDir(Foam::fileName& queried)
128 {
129  #ifdef FOAM_RESOURCE_SITE_ENVNAME
130  queried = Foam::getEnv(FOAM_RESOURCE_SITE_ENVNAME)/"etc";
131  if (queried.size() > 3)
132  {
133  return Foam::isDir(queried);
134  }
135  #elif defined FULLDEBUG
136  #warning FOAM_RESOURCE_SITE_ENVNAME \
137  is undefined (was this intentional?)
138  #endif
139 
140  // Fallback when WM_PROJECT_SITE is unset
141 
142  #ifdef FOAM_RESOURCE_SITE_FALLBACK_ENVNAME
143  queried = Foam::getEnv(FOAM_RESOURCE_SITE_FALLBACK_ENVNAME)/"site/etc";
144  if (queried.size() > 8 && Foam::isDir(queried))
145  {
146  return true;
147  }
148  #elif defined FULLDEBUG
149  #warning FOAM_RESOURCE_SITE_FALLBACK_ENVNAME \
150  is undefined (was this intentional?)
151  #endif
152 
153  // Compile-time paths
154 
155  #ifdef FOAM_CONFIGURED_PROJECT_DIR
156  queried = FOAM_CONFIGURED_PROJECT_DIR/Foam::string("site/etc");
157  if (queried.size() > 8 && Foam::isDir(queried))
158  {
159  return true;
160  }
161  #endif
162

210 // Check if the named file/directory matches the type required.
211 //
212 // - typeRequired (UNDEFINED) => accept either FILE or DIRECTORY
213 // - typeRequired (FILE | DIRECTORY) => accept only that type
214 static inline bool accept
215 (
216  const Foam::fileName& name,
217  const Foam::fileName::Type typeRequired
218 )
219 {
220  // followLink(true), checkGzip(true)
221  // -> returns (UNDEFINED | FILE | DIRECTORY), no need to check for (LINK)
222  const auto t = name.type(true, true);
223 
224  return
225  (
226  // Found something?
227  Foam::fileName::Type::UNDEFINED != t
228  &&
229  (
230  // Any particular type required?
231  Foam::fileName::Type::UNDEFINED == typeRequired
232  ? (Foam::fileName::Type::UNDEFINED != t)
233  : (typeRequired == t)
234  )
235  );
236 }

243 Foam::fileNameList Foam::etcDirs(bool test)
244 {
245  // Use foamVersion::api (instead of the OPENFOAM define) to ensure this
246  // stays properly synchronized with the build information
247  const Foam::fileName version(std::to_string(Foam::foamVersion::api));
248 
249  Foam::fileNameList list(5);
250  Foam::fileName queried;
251  label nDirs = 0;
252 
253  // User resource directories
254  if (userResourceDir(queried) || (!test && queried.size()))
255  {
256  list[nDirs++] = queried/version;
257  list[nDirs++] = queried;
258  }
259 
260  // Group (site) resource directories
261  if (groupResourceDir(queried) || (!test && queried.size()))
262  {
263  list[nDirs++] = queried/version;
264  list[nDirs++] = queried;
265  }
266 
267  // Other (project) resource directory
268  if (projectResourceDir(queried) || (!test && queried.size()))
269  {
270  list[nDirs++] = queried;
271  }
272 
273  list.resize(nDirs);
274 
275  return list;
276 }
