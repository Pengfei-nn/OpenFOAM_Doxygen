96 turbulentTemperatureRadCoupledMixedFvPatchScalarField::
97 turbulentTemperatureRadCoupledMixedFvPatchScalarField
98 (
99  const fvPatch& p,
100  const DimensionedField<scalar, volMesh>& iF,
101  const dictionary& dict
102 )
103 :
104  mixedFvPatchScalarField(p, iF),
105  temperatureCoupledBase(patch(), dict),
106  TnbrName_(dict.lookupOrDefault<word>("Tnbr", "T")),
107  qrNbrName_(dict.lookupOrDefault<word>("qrNbr", "none")),
108  qrName_(dict.lookupOrDefault<word>("qr", "none")),
109  thicknessLayers_(0),
110  kappaLayers_(0),
111  contactRes_(0.0),
112  thermalInertia_(dict.lookupOrDefault<Switch>("thermalInertia", false))
113 {
114  if (!isA<mappedPatchBase>(this->patch().patch()))
115  {
116  FatalErrorInFunction
117  << "' not type '" << mappedPatchBase::typeName << "'"
118  << "\n for patch " << p.name()
119  << " of field " << internalField().name()
120  << " in file " << internalField().objectPath()
121  << exit(FatalError);
122  }
123 
124  if (dict.readIfPresent("thicknessLayers", thicknessLayers_))
125  {
126  dict.readEntry("kappaLayers", kappaLayers_);
127 
128  if (thicknessLayers_.size() > 0)
129  {
130  // Calculate effective thermal resistance by harmonic averaging
131  forAll(thicknessLayers_, iLayer)
132  {
133  contactRes_ += thicknessLayers_[iLayer]/kappaLayers_[iLayer];
134  }
135  contactRes_ = 1.0/contactRes_;
136  }
137  }
138 
139  fvPatchScalarField::operator=(scalarField("value", dict, p.size()));
140 
141  if (dict.found("refValue"))
142  {
143  // Full restart
144  refValue() = scalarField("refValue", dict, p.size());
145  refGrad() = scalarField("refGradient", dict, p.size());
146  valueFraction() = scalarField("valueFraction", dict, p.size());
147  }
148  else
149  {
150  // Start from user entered data. Assume fixedValue.
151  refValue() = *this;
152  refGrad() = 0.0;
153  valueFraction() = 1.0;
154  }
155 }

179 void turbulentTemperatureRadCoupledMixedFvPatchScalarField::updateCoeffs()
180 {
181  if (updated())
182  {
183  return;
184  }
185 
186  const polyMesh& mesh = patch().boundaryMesh().mesh();
187 
188  // Since we're inside initEvaluate/evaluate there might be processor
189  // comms underway. Change the tag we use.
190  int oldTag = UPstream::msgType();
191  UPstream::msgType() = oldTag+1;
192 
193  // Get the coupling information from the mappedPatchBase
194  const label patchi = patch().index();
195  const mappedPatchBase& mpp =
196  refCast<const mappedPatchBase>(patch().patch());
197  const polyMesh& nbrMesh = mpp.sampleMesh();
198  const label samplePatchi = mpp.samplePolyPatch().index();
199  const fvPatch& nbrPatch =
200  refCast<const fvMesh>(nbrMesh).boundary()[samplePatchi];
201 
202 
203  scalarField Tc(patchInternalField());
204  scalarField& Tp = *this;
205 
206  const turbulentTemperatureRadCoupledMixedFvPatchScalarField&
207  nbrField = refCast
208  <const turbulentTemperatureRadCoupledMixedFvPatchScalarField>
209  (
210  nbrPatch.lookupPatchField<volScalarField, scalar>(TnbrName_)
211  );
212 
213  // Swap to obtain full local values of neighbour internal field
214  scalarField TcNbr(nbrField.patchInternalField());
215  mpp.distribute(TcNbr);
216 
217 
218  // Swap to obtain full local values of neighbour K*delta
219  scalarField KDeltaNbr;
220  if (contactRes_ == 0.0)
221  {
222  KDeltaNbr = nbrField.kappa(nbrField)*nbrPatch.deltaCoeffs();
223  }
224  else
225  {
226  KDeltaNbr.setSize(nbrField.size(), contactRes_);
227  }
228  mpp.distribute(KDeltaNbr);
229 
230  scalarField KDelta(kappa(Tp)*patch().deltaCoeffs());
231 
232  scalarField qr(Tp.size(), Zero);
233  if (qrName_ != "none")
234  {
235  qr = patch().lookupPatchField<volScalarField, scalar>(qrName_);
236  }
237 
238  scalarField qrNbr(Tp.size(), Zero);
239  if (qrNbrName_ != "none")
240  {
241  qrNbr = nbrPatch.lookupPatchField<volScalarField, scalar>(qrNbrName_);
242  mpp.distribute(qrNbr);
243  }
244 
245  // inertia therm
246  if (thermalInertia_)
247  {
248  const scalar dt = mesh.time().deltaTValue();
249  scalarField mCpDtNbr;
250 
251  {
252  const basicThermo* thermo =
253  nbrMesh.findObject<basicThermo>(basicThermo::dictName);
254 
255  if (thermo)
256  {
257  const scalarField& ppn =
258  thermo->p().boundaryField()[samplePatchi];
259  const scalarField& Tpn =
260  thermo->T().boundaryField()[samplePatchi];
261 
262  mCpDtNbr =
263  (
264  thermo->Cp(ppn, Tpn, samplePatchi)
265  * thermo->rho(samplePatchi)
266  / nbrPatch.deltaCoeffs()/dt
267  );
268 
269  mpp.distribute(mCpDtNbr);
270  }
271  else
272  {
273  mCpDtNbr.setSize(Tp.size(), Zero);
274  }
275  }
276 
277  scalarField mCpDt;
278 
279  // Local inertia therm
280  {
281  const basicThermo* thermo =
282  mesh.findObject<basicThermo>(basicThermo::dictName);
283 
284  if (thermo)
285  {
286  const scalarField& pp = thermo->p().boundaryField()[patchi];
287 
288  mCpDt =
289  (
290  thermo->Cp(pp, Tp, patchi)
291  * thermo->rho(patchi)
292  / patch().deltaCoeffs()/dt
293  );
294  }
295  else
296  {
297  // Issue warning?
298  mCpDt.setSize(Tp.size(), Zero);
299  }
300  }
301 
302  const volScalarField& T =
303  this->db().lookupObject<volScalarField>
304  (
305  this->internalField().name()
306  );
307 
308  const fvPatchField<scalar>& TpOld = T.oldTime().boundaryField()[patchi];
309 
310  scalarField alpha(KDeltaNbr + mCpDt + mCpDtNbr);
311 
312  valueFraction() = alpha/(alpha + KDelta);
313  scalarField c(KDeltaNbr*TcNbr + (mCpDt + mCpDtNbr)*TpOld);
314  refValue() = c/alpha;
315  refGrad() = (qr + qrNbr)/kappa(Tp);
316  }
317  else
318  {
319  valueFraction() = KDeltaNbr/(KDeltaNbr + KDelta);
320  refValue() = TcNbr;
321  refGrad() = (qr + qrNbr)/kappa(Tp);
322  }
323 
324  mixedFvPatchScalarField::updateCoeffs();
325 
326  if (debug)
327  {
328  scalar Q = gSum(kappa(Tp)*patch().magSf()*snGrad());
329 
330  Info<< patch().boundaryMesh().mesh().name() << ':'
331  << patch().name() << ':'
332  << this->internalField().name() << " <- "
333  << nbrMesh.name() << ':'
334  << nbrPatch.name() << ':'
335  << this->internalField().name() << " :"
336  << " heat transfer rate:" << Q
337  << " walltemperature "
338  << " min:" << gMin(Tp)
339  << " max:" << gMax(Tp)
340  << " avg:" << gAverage(Tp)
341  << endl;
342  }
343 
344  // Restore tag
345  UPstream::msgType() = oldTag;
346 }
