33 Description
34  Creates a zero directory with fields appropriate for the chosen solver and
35  turbulence model. Operates on both single and multi-region cases.
36 
37 Usage
38  The set-up is configured using a 'caseProperties' dictionary, located under
39  the $FOAM_CASE/system (or system/regionName if multi-region) directory.
40  This consists of a lists of initial and boundary conditions, e.g.
41 
42  \verbatim
43  initialConditions
44  {
45  U uniform (0 0 0);
46  p uniform 0;
47  }
48 
49  boundaryConditions
50  {
51  topWall
52  {
53  category wall;
54  patches (movingWall);
55  type noSlip;
56  options
57  {
58  wallFunction highReynolds;
59  motion moving;
60  };
61  values
62  {
63  U uniform (1 0 0);
64  }
65  }
66 
67  walls
68  {
69  category wall;
70  patches (fixedWalls);
71  type noSlip;
72  options
73  {
74  wallFunction highReynolds;
75  motion stationary;
76  };
77  }
78  }
79  \endverbatim

94 const word getClassType(const word& pType)
95 {
96  if (pType == pTraits<scalar>::typeName)
97  {
98  return GeometricField<scalar, fvPatchField, volMesh>::typeName;
99  }
100  else if (pType == pTraits<vector>::typeName)
101  {
102  return GeometricField<vector, fvPatchField, volMesh>::typeName;
103  }
104  else if (pType == pTraits<sphericalTensor>::typeName)
105  {
106  return GeometricField<sphericalTensor, fvPatchField, volMesh>::typeName;
107  }
108  else if (pType == pTraits<symmTensor>::typeName)
109  {
110  return GeometricField<symmTensor, fvPatchField, volMesh>::typeName;
111  }
112  else if (pType == pTraits<tensor>::typeName)
113  {
114  return GeometricField<tensor, fvPatchField, volMesh>::typeName;
115  }
116  else
117  {
118  // Error
119  return word::null;
120  }
121 }

124 void createFieldFiles
125 (
126  const Time& runTime,
127  const word& regionName,
128  const word& regionPrefix,
129  const wordList& fieldNames,
130  const wordList& fieldTypes,
131  const PtrList<dimensionSet>& fieldDimensions,
132  const caseInfo& cInfo,
133  const boundaryTemplates& bcTemplates
134 )
135 {
136  Info<< " Generating field files" << nl << endl;
137 
138  // Create files
139  forAll(fieldNames, i)
140  {
141  const_cast<word&>(IOdictionary::typeName) =
142  getClassType(fieldTypes[i]);
143 
144  dictionary field;
145 
146  fileName regionPath = "/";
147 
148  if (regionName != word::null)
149  {
150  regionPath += regionName + '/';
151  }
152 
153  field.add
154  (
155  "#include",
156  "$FOAM_CASE/system" + regionPath + "caseProperties"
157  );
158 
159  field.add("dimensions", fieldDimensions[i]);
160 
161  string iField("${:initialConditions." + fieldNames[i] + '}');
162  field.add("internalField", iField.c_str());
163 
164  dictionary boundaryField =
165  cInfo.generateBoundaryField
166  (
167  regionPrefix,
168  fieldNames[i],
169  bcTemplates
170  );
171 
172  field.add("boundaryField", boundaryField);
173 
174  // Expand all of the dictionary redirections and remove unnecessary
175  // entries
176  OStringStream os;
177  os << field;
178 
179  entry::disableFunctionEntries = 0;
180  dictionary field2(IStringStream(os.str())());
181  entry::disableFunctionEntries = 1;
182  field2.remove("#include");
183  field2.remove("initialConditions");
184  field2.remove("boundaryConditions");
185 
186  // Construct and write field dictionary. Note use of localIOdictionary
187  localIOdictionary fieldOut
188  (
189  IOobject
190  (
191  fieldNames[i],
192  "0",
193  regionName,
194  runTime,
195  IOobject::NO_READ
196  ),
197  field2
198  );
199 
200  fieldOut.regIOobject::writeObject
201  (
202  IOstream::ASCII,
203  IOstream::currentVersion,
204  IOstream::UNCOMPRESSED,
205  true
206  );
207  }
208 }

212 int main(int argc, char *argv[])
213 {
214  argList::addNote
215  (
216  "Create a 0/ directory with fields appropriate for the chosen"
217  " solver and turbulence model."
218  );
219 
220  argList::addOption
221  (
222  "templateDir",
223  "dir",
224  "Read case set-up templates from specified location"
225  );
226 
227  #include "setRootCase.H"
228  #include "createTime.H"
229 
230  Info<< "Reading controlDict" << nl << endl;
231 
232  IOdictionary controlDict
233  (
234  IOobject
235  (
236  "controlDict",
237  runTime.system(),
238  runTime,
239  IOobject::MUST_READ,
240  IOobject::NO_WRITE
241  )
242  );
243 
244  // Template directory: default is from PROJECT/etc directory
245  //
246  // Can use
247  // - foamEtcDir("caseDicts/createZeroDirectory", 0007);
248  // - expand "<etc:o>/caseDicts/createZeroDirectory"
249  // - expand "${WM_PROJECT_DIR}/etc/caseDicts/createZeroDirectory"
250  //
251  // Use "${WM_PROJECT_DIR}/" version for nicer error message
252 
253  fileName baseDir
254  (
255  args.get<fileName>
256  (
257  "templateDir",
258  "${WM_PROJECT_DIR}/etc/caseDicts/createZeroDirectoryTemplates"
259  )
260  );
261 
262  baseDir.expand();
263  baseDir.toAbsolute();
264 
265  if (!Foam::isDir(baseDir))
266  {
267  FatalErrorInFunction
268  << "templateDir " << baseDir << nl
269  << "Does not point to a folder containing case set-up templates"
270  << nl
271  << exit(FatalError);
272  }
273 
274  // Keep variable substitutions - delay until after creation of controlDict
275  // to allow #include files to be processed
276  entry::disableFunctionEntries = 1;
277 
278  // Read the solver
279  const word solverName(controlDict.get<word>("application"));
280 
281  // Generate solver template
282  const solverTemplate solver(baseDir, runTime, solverName);
283 
284  // Read the boundary condition templates
285  const boundaryTemplates bcTemplates
286  (
287  baseDir,
288  runTime,
289  solver.type()
290  );
291 
292  Info<< endl;
293 
294  const label nRegion = solver.nRegion();
295  for (label regionI = 0; regionI < nRegion; regionI++)
296  {
297  const word& regionName = solver.regionName(regionI);
298 
299  if (regionName == word::null)
300  {
301  Info<< "Region: " << polyMesh::defaultRegion << " (default)"
302  << endl;
303  }
304  else
305  {
306  Info<< "Region: " << regionName << endl;
307  }
308 
309  // Read the case set-up info for the current region
310  const caseInfo cInfo(runTime, regionName);
311 
312  cInfo.checkPatches(solver.regionType(regionI), bcTemplates);
313 
314  createFieldFiles
315  (
316  runTime,
317  regionName,
318  solver.regionType(regionI),
319  solver.fieldNames(regionI),
320  solver.fieldTypes(regionI),
321  solver.fieldDimensions(regionI),
322  cInfo,
323  bcTemplates
324  );
325  }
326 
327  Info<< "End\n" << endl;
328 
329  return 0;
330 }
