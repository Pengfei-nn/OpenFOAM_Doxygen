26 Description
27  This function calculates the list of patch edges, defined on the list of
28  points supporting the patch. The edges are ordered:
29  - 0..nInternalEdges-1 : upper triangular order
30  - nInternalEdges.. : boundary edges (no particular order)
31 
32  Other patch addressing information is also calculated:
33  - faceFaces with neighbour faces in ascending order
34  - edgeFaces with ascending face order
35  - faceEdges sorted according to edges of a face
36

52 void
53 Foam::PrimitivePatch<Face, FaceList, PointField, PointType>::
54 calcAddressing() const
55 {
56  if (debug)
57  {
58  Info<< "PrimitivePatch<Face, FaceList, PointField, PointType>::"
59  << "calcAddressing() : calculating patch addressing"
60  << endl;
61  }
62 
63  if (edgesPtr_ || faceFacesPtr_ || edgeFacesPtr_ || faceEdgesPtr_)
64  {
65  // An error to recalculate if already allocated
66  FatalErrorInFunction
67  << "addressing already calculated"
68  << abort(FatalError);
69  }
70 
71  // get reference to localFaces
72  const List<Face>& locFcs = localFaces();
73 
74  // get reference to pointFaces
75  const labelListList& pf = pointFaces();
76 
77  // Guess the max number of edges and neighbours for a face
78  label maxEdges = 0;
79  forAll(locFcs, facei)
80  {
81  maxEdges += locFcs[facei].size();
82  }
83 
84  // create the lists for the various results. (resized on completion)
85  edgesPtr_ = new edgeList(maxEdges);
86  edgeList& edges = *edgesPtr_;
87 
88  edgeFacesPtr_ = new labelListList(maxEdges);
89  labelListList& edgeFaces = *edgeFacesPtr_;
90 
91  // faceFaces created using a dynamic list. Cannot guess size because
92  // of multiple connections
93  List<DynamicList<label>> ff(locFcs.size());
94 
95  faceEdgesPtr_ = new labelListList(locFcs.size());
96  labelListList& faceEdges = *faceEdgesPtr_;
97 
98  // count the number of face neighbours
99  labelList noFaceFaces(locFcs.size());
100 
101  // initialise the lists of subshapes for each face to avoid duplication
102  edgeListList faceIntoEdges(locFcs.size());
103 
104  forAll(locFcs, facei)
105  {
106  faceIntoEdges[facei] = locFcs[facei].edges();
107 
108  labelList& curFaceEdges = faceEdges[facei];
109  curFaceEdges.setSize(faceIntoEdges[facei].size());
110 
111  forAll(curFaceEdges, faceEdgeI)
112  {
113  curFaceEdges[faceEdgeI] = -1;
114  }
115  }
116 
117  // This algorithm will produce a separated list of edges, internal edges
118  // starting from 0 and boundary edges starting from the top and
119  // growing down.
120 
121  label nEdges = 0;
122 
123  bool found = false;
124 
125  // Note that faceIntoEdges is sorted acc. to local vertex numbering
126  // in face (i.e. curEdges[0] is edge between f[0] and f[1])
127 
128  // For all local faces ...
129  forAll(locFcs, facei)
130  {
131  // Get reference to vertices of current face and corresponding edges.
132  const Face& curF = locFcs[facei];
133  const edgeList& curEdges = faceIntoEdges[facei];
134 
135  // Record the neighbour face. Multiple connectivity allowed
136  List<DynamicList<label>> neiFaces(curF.size());
137  List<DynamicList<label>> edgeOfNeiFace(curF.size());
138 
139  label nNeighbours = 0;
140 
141  // For all edges ...
142  forAll(curEdges, edgeI)
143  {
144  // If the edge is already detected, skip
145  if (faceEdges[facei][edgeI] >= 0) continue;
146 
147  found = false;
148 
149  // Set reference to the current edge
150  const edge& e = curEdges[edgeI];
151 
152  // Collect neighbours for the current face vertex.
153 
154  const labelList& nbrFaces = pf[e.start()];
155 
156  forAll(nbrFaces, nbrFacei)
157  {
158  // set reference to the current neighbour
159  label curNei = nbrFaces[nbrFacei];
160 
161  // Reject neighbours with the lower label
162  if (curNei > facei)
163  {
164  // get the reference to subshapes of the neighbour
165  const edgeList& searchEdges = faceIntoEdges[curNei];
166 
167  forAll(searchEdges, neiEdgeI)
168  {
169  if (searchEdges[neiEdgeI] == e)
170  {
171  // Match
172  found = true;
173 
174  neiFaces[edgeI].append(curNei);
175  edgeOfNeiFace[edgeI].append(neiEdgeI);
176 
177  // Record faceFaces both ways
178  ff[facei].append(curNei);
179  ff[curNei].append(facei);
180 
181  // Keep searching due to multiple connectivity
182  }
183  }
184  }
185  } // End of neighbouring faces
186 
187  if (found)
188  {
189  // Register another detected internal edge
190  nNeighbours++;
191  }
192  } // End of current edges
193 
194  // Add the edges in increasing number of neighbours.
195  // Note: for multiply connected surfaces, the lower index neighbour for
196  // an edge will come first.
197 
198  // Add the faces in the increasing order of neighbours
199  for (label neiSearch = 0; neiSearch < nNeighbours; neiSearch++)
200  {
201  // Find the lowest neighbour which is still valid
202  label nextNei = -1;
203  label minNei = locFcs.size();
204 
205  forAll(neiFaces, nfI)
206  {
207  if (neiFaces[nfI].size() && neiFaces[nfI][0] < minNei)
208  {
209  nextNei = nfI;
210  minNei = neiFaces[nfI][0];
211  }
212  }
213 
214  if (nextNei > -1)
215  {
216  // Add the face to the list of faces
217  edges[nEdges] = curEdges[nextNei];
218 
219  // Set face-edge and face-neighbour-edge to current face label
220  faceEdges[facei][nextNei] = nEdges;
221 
222  DynamicList<label>& cnf = neiFaces[nextNei];
223  DynamicList<label>& eonf = edgeOfNeiFace[nextNei];
224 
225  // Set edge-face addressing
226  labelList& curEf = edgeFaces[nEdges];
227  curEf.setSize(cnf.size() + 1);
228  curEf[0] = facei;
229 
230  forAll(cnf, cnfI)
231  {
232  faceEdges[cnf[cnfI]][eonf[cnfI]] = nEdges;
233 
234  curEf[cnfI + 1] = cnf[cnfI];
235  }
236 
237  // Stop the neighbour from being used again
238  cnf.clear();
239  eonf.clear();
240 
241  // Increment number of faces counter
242  nEdges++;
243  }
244  else
245  {
246  FatalErrorInFunction
247  << "Error in internal edge insertion"
248  << abort(FatalError);
249  }
250  }
251  }
252 
253  nInternalEdges_ = nEdges;
254 
255  // Do boundary faces
256 
257  forAll(faceEdges, facei)
258  {
259  labelList& curEdges = faceEdges[facei];
260 
261  forAll(curEdges, edgeI)
262  {
263  if (curEdges[edgeI] < 0)
264  {
265  // Grab edge and faceEdge
266  edges[nEdges] = faceIntoEdges[facei][edgeI];
267  curEdges[edgeI] = nEdges;
268 
269  // Add edgeFace
270  labelList& curEf = edgeFaces[nEdges];
271  curEf.setSize(1);
272  curEf[0] = facei;
273 
274  nEdges++;
275  }
276  }
277  }
278 
279  // edges
280  edges.setSize(nEdges);
281 
282  // edgeFaces list
283  edgeFaces.setSize(nEdges);
284 
285  // faceFaces list
286  faceFacesPtr_ = new labelListList(locFcs.size());
287  labelListList& faceFaces = *faceFacesPtr_;
288 
289  forAll(faceFaces, facei)
290  {
291  faceFaces[facei].transfer(ff[facei]);
292  }
293 
294 
295  if (debug)
296  {
297  Info<< "PrimitivePatch<Face, FaceList, PointField, PointType>::"
298  << "calcAddressing() : finished calculating patch addressing"
299  << endl;
300  }
301 }
