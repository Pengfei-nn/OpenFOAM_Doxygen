64 Foam::Istream& Foam::operator>>(Istream& is, keyType& val)
65 {
66  token t(is);
67 
68  if (!t.good())
69  {
70  FatalIOErrorInFunction(is)
71  << "Bad token - could not get a word/regex"
72  << exit(FatalIOError);
73  is.setBad();
74  return is;
75  }
76 
77  if (t.isWord())
78  {
79  val = t.wordToken();
80  val.setType(keyType::LITERAL);
81  }
82  else if (t.isString())
83  {
84  // Assign from string, treat as regular expression
85  val = t.stringToken();
86  val.setType(keyType::REGEX);
87 
88  // Flag empty strings as an error
89  if (val.empty())
90  {
91  FatalIOErrorInFunction(is)
92  << "Empty word/expression"
93  << exit(FatalIOError);
94  is.setBad();
95  return is;
96  }
97  }
98  else
99  {
100  FatalIOErrorInFunction(is)
101  << "Wrong token type - expected word or string, found "
102  << t.info()
103  << exit(FatalIOError);
104  is.setBad();
105  return is;
106  }
107 
108  is.check(FUNCTION_NAME);
109  return is;
110 }
