48 int main(int argc, char *argv[])
49 {

81

84

86  {

88  if (dist < minDist)
89  {

92  }
93  }
94 
95  Info<< "Nearest vertex:" << nl

100  << endl;
101 
102  //
103  // Nearest face
104  //
105 
106  const pointField& points = surf1.points();
107 
108  minIndex = -1;
109  minDist = GREAT;
110 
111  forAll(surf1, facei)
112  {
113  const point centre = surf1[facei].centre(points);
114 
115  const scalar dist = mag(centre - samplePt);
116  if (dist < minDist)
117  {
118  minDist = dist;
119  minIndex = facei;
120  }
121  }
122 
123  const face& f = surf1[minIndex];
124 
125  Info<< "Face with nearest centre:" << nl
126  << " index : " << minIndex << nl
127  << " centre : " << f.centre(points) << nl
128  << " face : " << f << nl
129  << " vertex coords:" << endl;
130 
131  forAll(f, fp)
132  {
133  Info<< " " << points[f[fp]] << nl;
134  }
135 
136  const List<surfZone>& surfZones = surf1.surfZones();
137  label surfZone = -1;
138  forAll(surfZones, zonei)
139  {
140  if (minIndex >= surfZones[zonei].start())
141  {
142  surfZone = zonei;
143  }
144  }
145  Info<< " zone/region : " << surfZone << endl;
146 
147  Info<< nl
148  << "End\n" << endl;
149 
150  return 0;
151 }
