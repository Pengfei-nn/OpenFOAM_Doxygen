117 template<class CloudType>
118 bool Foam::SHF<CloudType>::update
119 (
120  const scalar dt,
121  const vector& g,
122  scalar& d,
123  scalar& tc,
124  scalar& ms,
125  scalar& nParticle,
126  scalar& KHindex,
127  scalar& y,
128  scalar& yDot,
129  const scalar d0,
130  const scalar rho,
131  const scalar mu,
132  const scalar sigma,
133  const vector& U,
134  const scalar rhoc,
135  const scalar muc,
136  const vector& Urel,
137  const scalar Urmag,
138  const scalar tMom,
139  scalar& dChild,
140  scalar& massChild
141 )
142 {
143  Random& rndGen = this->owner().rndGen();
144 
145  bool addChild = false;
146 
147  scalar d03 = pow3(d);
148  scalar rhopi6 = rho*constant::mathematical::pi/6.0;
149  scalar mass0 = nParticle*rhopi6*d03;
150  scalar mass = mass0;
151 
152  scalar weGas = 0.5*rhoc*sqr(Urmag)*d/sigma;
153  scalar weLiquid = 0.5*rho*sqr(Urmag)*d/sigma;
154 
155  // correct the Reynolds number. Reitz is using radius instead of diameter
156  scalar reLiquid = 0.5*Urmag*d/mu;
157  scalar ohnesorge = sqrt(weLiquid)/(reLiquid + VSMALL);
158 
159  scalar weGasCorr = weGas/(1.0 + weCorrCoeff_*ohnesorge);
160 
161  // droplet deformation characteristic time
162 
163  scalar tChar = d/Urmag*sqrt(rho/rhoc);
164 
165  scalar tFirst = cInit_*tChar;
166 
167  scalar tSecond = 0;
168  scalar tCharSecond = 0;
169 
170  bool bag = false;
171  bool multimode = false;
172  bool shear = false;
173  bool success = false;
174 
175 
176  // update the droplet characteristic time
177  tc += dt;
178 
179  if (weGas > weConst_)
180  {
181  if (weGas < weCrit1_)
182  {
183  tCharSecond = c1_*pow((weGas - weConst_), cExp1_);
184  }
185  else if (weGas >= weCrit1_ && weGas <= weCrit2_)
186  {
187  tCharSecond = c2_*pow((weGas - weConst_), cExp2_);
188  }
189  else
190  {
191  tCharSecond = c3_*pow((weGas - weConst_), cExp3_);
192  }
193  }
194 
195  scalar weC = weBuCrit_*(1.0 + ohnCoeffCrit_*pow(ohnesorge, ohnExpCrit_));
196  scalar weB = weBuBag_*(1.0 + ohnCoeffBag_*pow(ohnesorge, ohnExpBag_));
197  scalar weMM = weBuMM_*(1.0 + ohnCoeffMM_*pow(ohnesorge, ohnExpMM_));
198 
199  if (weGas > weC && weGas < weB)
200  {
201  bag = true;
202  }
203 
204  if (weGas >= weB && weGas <= weMM)
205  {
206  multimode = true;
207  }
208 
209  if (weGas > weMM)
210  {
211  shear = true;
212  }
213 
214  tSecond = tCharSecond*tChar;
215 
216  scalar tBreakUP = tFirst + tSecond;
217  if (tc > tBreakUP)
218  {
219  scalar d32 = coeffD_*d*pow(ohnesorge, onExpD_)*pow(weGasCorr, weExpD_);
220 
221  if (bag || multimode)
222  {
223  scalar d05 = d32Coeff_ * d32;
224 
225  scalar x = 0.0;
226  scalar yGuess = 0.0;
227  scalar dGuess = 0.0;
228 
229  while(!success)
230  {
231  x = cDmaxBM_*rndGen.sample01<scalar>();
232  dGuess = sqr(x)*d05;
233  yGuess = rndGen.sample01<scalar>();
234 
235  scalar p =
236  x
237  /(2.0*sqrt(constant::mathematical::twoPi)*sigma_)
238  *exp(-0.5*sqr((x - mu_)/sigma_));
239 
240  if (yGuess < p)
241  {
242  success = true;
243  }
244  }
245 
246  d = dGuess;
247  tc = 0.0;
248  }
249 
250  if (shear)
251  {
252  scalar dC = weConst_*sigma/(rhoc*sqr(Urmag));
253  scalar d32Red = 4.0*(d32*dC)/(5.0*dC - d32);
254 
255  scalar d05 = d32Coeff_ * d32Red;
256 
257  scalar x = 0.0;
258  scalar yGuess = 0.0;
259  scalar dGuess = 0.0;
260 
261  while(!success)
262  {
263  x = cDmaxS_*rndGen.sample01<scalar>();
264  dGuess = sqr(x)*d05;
265  yGuess = rndGen.sample01<scalar>();
266 
267  scalar p =
268  x
269  /(2.0*sqrt(constant::mathematical::twoPi)*sigma_)
270  *exp(-0.5*sqr((x - mu_)/sigma_));
271 
272  if (yGuess<p)
273  {
274  success = true;
275  }
276  }
277 
278  d = dC;
279  dChild = dGuess;
280  massChild = corePerc_*mass0;
281  mass -= massChild;
282 
283  addChild = true;
284  // reset timer
285  tc = 0.0;
286  }
287 
288  // correct nParticle to conserve mass
289  nParticle = mass/(rhopi6*pow3(d));
290  }
291 
292  return addChild;
293 }
