111 void writeWeights(const polyMesh& mesh)
112 {
113  const fileName outputDir
114  (
115  mesh.time().globalPath()/functionObject::outputPrefix/"checkAMI"
116  );
117 
118  for (const polyPatch& pp : mesh.boundaryMesh())
119  {
120  if (isA<cyclicAMIPolyPatch>(pp))
121  {
122  const cyclicAMIPolyPatch& cpp =
123  refCast<const cyclicAMIPolyPatch>(pp);
124 
125  if (cpp.owner())
126  {
127  Info<< "Calculating AMI weights between owner patch: "
128  << cpp.name() << " and neighbour patch: "
129  << cpp.neighbPatch().name() << endl;
130 
131  const AMIPatchToPatchInterpolation& ami =
132  cpp.AMI();
133 
134  writeWeights
135  (
136  mesh,
137  ami.tgtWeightsSum(),
138  cpp.neighbPatch(),
139  outputDir,
140  "patch" + Foam::name(pp.index()) + "-tgt",
141  mesh.time()
142  );
143  writeWeights
144  (
145  mesh,
146  ami.srcWeightsSum(),
147  cpp,
148  outputDir,
149  "patch" + Foam::name(pp.index()) + "-src",

152  }
153  }

183  }

187 
188 
189  pimpleControl pimple(mesh);

191  bool moveMeshOuterCorrectors

194  );
195 
196  while (runTime.loop())
197  {
198  Info<< "Time = " << runTime.timeName() << endl;
199 
200  while (pimple.loop())
201  {
202  if (pimple.firstIter() || moveMeshOuterCorrectors)
203  {
204  mesh.update();
205  }
206  }
207 
208  if (overwrite)
209  {
210  mesh.setInstance(oldInstance);
211  runTime.write();
212  runTime.printExecutionTime(Info);
213  break;
214  }
215 
216 
217  mesh.checkMesh(true);
218 
219  if (checkAMI)
220  {
221  writeWeights(mesh);
222  }
223 
224  runTime.write();
225 
226  runTime.printExecutionTime(Info);
227  }
228 
229  Info<< "End\n" << endl;
230 
231  return 0;
