53 void Foam::fv::VoFSolidificationMeltingSource::update()
54 {
55  if (curTimeIndex_ == mesh_.time().timeIndex())
56  {
57  return;
58  }
59 
60  if (debug)
61  {
62  Info<< type() << ": " << name_
63  << " - updating solid phase fraction" << endl;
64  }
65 
66  alphaSolid_.oldTime();
67 
68  const twoPhaseMixtureThermo& thermo
69  (
70  mesh_.lookupObject<twoPhaseMixtureThermo>
71  (
72  twoPhaseMixtureThermo::dictName
73  )
74  );
75 
76  const volScalarField& TVoF = thermo.thermo1().T();
77  const volScalarField CpVoF(thermo.thermo1().Cp());
78  const volScalarField& alphaVoF = thermo.alpha1();
79 
80  forAll(cells_, i)
81  {
82  const label celli = cells_[i];
83 
84  alphaSolid_[celli] = min
85  (
86  relax_*alphaVoF[celli]*alphaSolidT_->value(TVoF[celli])
87  + (1 - relax_)*alphaSolid_[celli],
88  alphaVoF[celli]
89  );
90  }
91 
92  alphaSolid_.correctBoundaryConditions();
93 
94  curTimeIndex_ = mesh_.time().timeIndex();
95 }

176 void Foam::fv::VoFSolidificationMeltingSource::addSup
177 (
178  fvMatrix<vector>& eqn,
179  const label fieldi
180 )
181 {
182  if (debug)
183  {
184  Info<< type() << ": applying source to " << eqn.psi().name() << endl;
185  }
186 
187  update();
188 
189  scalarField& Sp = eqn.diag();
190  const scalarField& V = mesh_.V();
191 
192  forAll(cells_, i)
193  {
194  const label celli = cells_[i];
195  const scalar Vc = V[celli];
196  const scalar alphaFluid = 1 - alphaSolid_[celli];
197 
198  const scalar S = Cu_*sqr(1 - alphaFluid)/(pow3(alphaFluid) + q_);
199 
200  Sp[celli] -= Vc*S;
201  }
202 }
