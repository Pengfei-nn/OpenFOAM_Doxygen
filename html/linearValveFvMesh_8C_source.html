48 void Foam::linearValveFvMesh::addZonesAndModifiers()
49 {
50  // Add zones and modifiers for motion action
51 
52  if
53  (
54  pointZones().size()
55  || faceZones().size()
56  || cellZones().size()
57  || topoChanger_.size()
58  )
59  {
60  InfoInFunction
61  << "Zones and modifiers already present. Skipping."
62  << endl;
63 
64  return;
65  }
66 
67  Info<< "Time = " << time().timeName() << endl
68  << "Adding zones and modifiers to the mesh" << endl;
69 
70  // Add zones
71  List<pointZone*> pz(1);
72 
73  // An empty zone for cut points
74  pz[0] = new pointZone("cutPointZone", 0, pointZones());
75 
76 
77  // Do face zones for slider
78 
79  List<faceZone*> fz(3);
80 
81  // Inner slider
82  const word innerSliderName
83  (
84  motionDict_.subDict("slider").get<word>("inside")
85  );
86  const polyPatch& innerSlider = boundaryMesh()[innerSliderName];
87 
88  fz[0] = new faceZone
89  (
90  "insideSliderZone",
91  identity(innerSlider.size(), innerSlider.start()),
92  false, // none are flipped
93  0,
94  faceZones()
95  );
96 
97  // Outer slider
98  const word outerSliderName
99  (
100  motionDict_.subDict("slider").get<word>("outside")
101  );
102  const polyPatch& outerSlider = boundaryMesh()[outerSliderName];
103 
104  fz[1] = new faceZone
105  (
106  "outsideSliderZone",
107  identity(outerSlider.size(), outerSlider.start()),
108  false, // none are flipped
109  1,
110  faceZones()
111  );
112 
113  // An empty zone for cut faces
114  fz[2] = new faceZone("cutFaceZone", 2, faceZones());
115 
116  List<cellZone*> cz(0);
117 
118  Info<< "Adding point, face and cell zones" << endl;
119  addZones(pz, fz, cz);
120 
121  // Add a topology modifier
122  Info<< "Adding topology modifiers" << endl;
123  topoChanger_.setSize(1);
124  topoChanger_.set
125  (
126  0,
127  new slidingInterface
128  (
129  "mixerSlider",
130  0,
131  topoChanger_,
132  outerSliderName + "Zone",
133  innerSliderName + "Zone",
134  "cutPointZone",
135  "cutFaceZone",
136  outerSliderName,
137  innerSliderName,
138  slidingInterface::INTEGRAL,
139  true // Attach-detach action
140  )
141  );
142  topoChanger_.writeOpt() = IOobject::AUTO_WRITE;
143 
144  // Write mesh
145  write();
146 }

193 bool Foam::linearValveFvMesh::attached() const
194 {
195  const polyTopoChanger& topoChanges = topoChanger_;
196 
197  bool result = false;
198 
199  forAll(topoChanges, modI)
200  {
201  if (isA<slidingInterface>(topoChanges[modI]))
202  {
203  result =
204  result
205  || refCast<const slidingInterface>(topoChanges[modI]).attached();
206  }
207  }
208 
209  // Check thal all sliders are in sync (debug only)
210  forAll(topoChanges, modI)
211  {
212  if (isA<slidingInterface>(topoChanges[modI]))
213  {
214  if
215  (
216  result
217  != refCast<const slidingInterface>(topoChanges[modI]).attached()
218  )
219  {
220  FatalErrorInFunction
221  << "Slider " << modI
222  << " named " << topoChanges[modI].name()
223  << " out of sync: Should be" << result
224  << abort(FatalError);
225  }
226  }
227  }
228 
229  if (result)
230  {
231  Info<< "linearValveFvMesh: attached!" << endl;
232  }
233  else
234  {
235  Info<< "linearValveFvMesh: detached!" << endl;
236  }
237 
238  return result;
239 }

277 void Foam::linearValveFvMesh::update()
278 {
279  // Detaching the interface
280  if (attached())
281  {
282  Info<< "Decoupling sliding interfaces" << endl;
283  makeSlidersLive();
284 
285  // Changing topology by hand
286  resetMorph();
287  setMorphTimeIndex(3*time().timeIndex());
288  updateMesh();
289 
290  msPtr_->updateMesh();
291  }
292  else
293  {
294  Info<< "Sliding interfaces decoupled" << endl;
295  }
296 
297  // Perform mesh motion
298  makeSlidersDead();
299 
300  // Changing topology by hand
301  resetMorph();
302  setMorphTimeIndex(3*time().timeIndex() + 1);
303  updateMesh();
304 
305  msPtr_->updateMesh();
306 
307  if (topoChangeMap.valid())
308  {
309  if (topoChangeMap().hasMotionPoints())
310  {
311  Info<< "Topology change; executing pre-motion" << endl;
312  movePoints(topoChangeMap().preMotionPoints());
313  }
314  }
315 
316  // Solve for motion
317  msPtr_->solve();
318 
319  movePoints(msPtr_->curPoints());
320 
321  // Attach the interface
322  Info<< "Coupling sliding interfaces" << endl;
323  makeSlidersLive();
324  resetMorph();
325  setMorphTimeIndex(3*time().timeIndex() + 2);
326  updateMesh();
327 
328  Info<< "Moving points post slider attach" << endl;
329 
330  msPtr_->updateMesh();
331 
332  Info<< "Sliding interfaces coupled: " << attached() << endl;
333 }
