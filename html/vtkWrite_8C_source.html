164 bool Foam::functionObjects::vtkWrite::read(const dictionary& dict)
165 {
166  timeFunctionObject::read(dict);
167 
168  readSelection(dict);
169 
170  // We probably cannot trust old information after a reread
171  series_.clear();
172 
173  // verbose_ = dict.lookupOrDefault<bool>("verbose", true);
174  doInternal_ = dict.lookupOrDefault<bool>("internal", true);
175  doBoundary_ = dict.lookupOrDefault<bool>("boundary", true);
176  oneBoundary_ = dict.lookupOrDefault<bool>("single", false);
177  interpolate_ = dict.lookupOrDefault<bool>("interpolate", false);
178 
179  //
180  // Writer options - default is xml base64
181  //
182  writeOpts_ = vtk::formatType::INLINE_BASE64;
183 
184  writeOpts_.ascii
185  (
186  dict.found("format")
187  && (IOstream::formatEnum(dict.get<word>("format")) == IOstream::ASCII)
188  );
189 
190  if (dict.lookupOrDefault("legacy", false))
191  {
192  writeOpts_.legacy(true);
193  }
194 
195  writeOpts_.precision
196  (
197  dict.lookupOrDefault
198  (
199  "precision",
200  IOstream::defaultPrecision()
201  )
202  );
203 
204  // Info<< type() << " " << name() << " output-format: "
205  // << writeOpts_.description() << nl;
206 
207  const int padWidth = dict.lookupOrDefault<int>("width", 8);
208 
209  // Appropriate printf format - Enforce min/max sanity limits
210  if (padWidth < 1 || padWidth > 31)
211  {
212  printf_.clear();
213  }
214  else
215  {
216  printf_ = "%0" + std::to_string(padWidth) + "d";
217  }
218 
219  //
220  // Other options
221  //
222 
223  decompose_ = dict.lookupOrDefault("decompose", false);
224  writeIds_ = dict.lookupOrDefault("writeIds", false);
225 
226 
227  // Output directory
228 
229  outputDir_.clear();
230  dict.readIfPresent("directory", outputDir_);
231 
232  if (outputDir_.size())
233  {
234  // User-defined output directory
235  outputDir_.expand();
236  if (!outputDir_.isAbsolute())
237  {
238  outputDir_ = time_.globalPath()/outputDir_;
239  }
240  }
241  else
242  {
243  // Standard postProcessing/ naming
244  outputDir_ = time_.globalPath()/functionObject::outputPrefix/name();
245  }
246  outputDir_.clean();
247 
248  return true;
249 }

261  // useTimeName ? time_.timeName() : Foam::name(time_.timeIndex());

268  );

276  // Skip

278  }
279 
280 
281  fileName vtkName = time_.globalCaseName();

283  vtk::vtmWriter vtmMultiRegion;
284 
285  Info<< name() << " output Time: " << time_.timeName() << nl;
286 
287  label regioni = 0;
288  for (const word& regionName : meshes_.sortedToc())
289  {

308  true // prune
309  );
310 
311  const label nVolFields =
312  (
313  (doInternal_ || doBoundary_)
314  ? baseMesh.count
315  (
316  stringListOps::foundOp<word>(fieldTypes::volume),
317  acceptField
318  )
319  : 0
320  );
321 
322  // Undecided if we want to automatically support DimensionedFields
323  // or only on demand:
324  const label nDimFields = 0;
325  // (
326  // (doInternal_ || doBoundary_)
327  // ? baseMesh.count
328  // (
329  // stringListOps::foundOp<word>(fieldTypes::internal),
330  // acceptField
331  // )
332  // : 0
333  // );

335 
336  // Setup for the vtm writer.
337  // For legacy format, the information added is simply ignored.
338

346 
347  // Collect individual boundaries into a vtm file
348  vtk::vtmWriter vtmBoundaries;
349 
350  // Setup the internal writer
351  autoPtr<vtk::internalWriter> internalWriter;
352 
353  // Interpolator for volume and dimensioned fields
354  autoPtr<volPointInterpolation> pInterp;
355 
356  if (doInternal_)
357  {
358  if (interpolate_)
359  {
360  pInterp.reset(new volPointInterpolation(meshProxy.mesh()));
361  }
362 
363  if (vtuMeshCells.empty())
364  {
365  // Use the appropriate mesh (baseMesh or subMesh)
366  vtuMeshCells.reset(meshProxy.mesh());
367  }
368 
369  internalWriter = autoPtr<vtk::internalWriter>::New
370  (
371  meshProxy.mesh(),
372  vtuMeshCells,
373  writeOpts_,
374  // Output name for internal
375  (
376  writeOpts_.legacy()
377  ? vtmOutputBase
378  : (vtmOutputBase / "internal")
379  ),
380  Pstream::parRun()
381  );
382 
383  Info<< " Internal : "
384  << time_.relativePath(internalWriter->output())
385  << endl;
386 
387  // No sub-block for internal
388  vtmWriter.append_vtu
389  (
390  "internal",
391  vtmOutputBase.name()/"internal"
392  );
393 
394  internalWriter->writeTimeValue(timeValue);
395  internalWriter->writeGeometry();
396  }
397 
398 
399  // Setup the patch writers
400 
401  const polyBoundaryMesh& patches = meshProxy.mesh().boundaryMesh();
402 
403  PtrList<vtk::patchWriter> patchWriters;
404  PtrList<PrimitivePatchInterpolation<primitivePatch>> patchInterps;
405 
406  labelList patchIds;
407  if (doBoundary_)
408  {
409  patchIds = getSelectedPatches(patches);
410  }
411 
412  if (oneBoundary_ && patchIds.size())
413  {
414  auto writer = autoPtr<vtk::patchWriter>::New
415  (
416  meshProxy.mesh(),
417  patchIds,
418  writeOpts_,
419  // Output name for one patch: "boundary"
420  (
421  writeOpts_.legacy()
422  ? (outputDir_/regionPrefix/"boundary"/"boundary" + timeDesc)
423  : (vtmOutputBase / "boundary")
424  ),
425  Pstream::parRun()
426  );
427 
428  // No sub-block for one-patch
429  vtmWriter.append_vtp
430  (
431  "boundary",
432  vtmOutputBase.name()/"boundary"
433  );
434 
435  Info<< " Boundaries: "
436  << time_.relativePath(writer->output()) << nl;
437 
438 
439  writer->writeTimeValue(timeValue);
440  writer->writeGeometry();
441 
442  // Transfer writer to list for later use
443  patchWriters.resize(1);
444  patchWriters.set(0, writer);
445 
446  // Avoid patchInterpolation for each sub-patch
447  patchInterps.resize(1); // == nullptr
448  }
449  else if (patchIds.size())
450  {
451  patchWriters.resize(patchIds.size());
452  if (interpolate_)
453  {
454  patchInterps.resize(patchIds.size());
455  }
456 
457  label nPatchWriters = 0;
458  label nPatchInterps = 0;
459 
460  for (const label patchId : patchIds)
461  {
462  const polyPatch& pp = patches[patchId];
463 
464  auto writer = autoPtr<vtk::patchWriter>::New
465  (
466  meshProxy.mesh(),
467  labelList(one(), pp.index()),
468  writeOpts_,
469  // Output name for patch: "boundary"/name
470  (
471  writeOpts_.legacy()
472  ?
473  (
474  outputDir_/regionPrefix/pp.name()
475  / (pp.name()) + timeDesc
476  )
477  : (vtmOutputBase / "boundary" / pp.name())
478  ),
479  Pstream::parRun()
480  );
481 
482  if (!nPatchWriters)
483  {
484  vtmWriter.beginBlock("boundary");
485  vtmBoundaries.beginBlock("boundary");
486  }
487 
488  vtmWriter.append_vtp
489  (
490  pp.name(),
491  vtmOutputBase.name()/"boundary"/pp.name()
492  );
493 
494  vtmBoundaries.append_vtp
495  (
496  pp.name(),
497  "boundary"/pp.name()
498  );
499 
500  Info<< " Boundary : "
501  << time_.relativePath(writer->output()) << nl;
502 
503  writer->writeTimeValue(timeValue);
504  writer->writeGeometry();
505 
506  // Transfer writer to list for later use
507  patchWriters.set(nPatchWriters++, writer);
508 
509  if (patchInterps.size())
510  {
511  patchInterps.set
512  (
513  nPatchInterps++,
514  new PrimitivePatchInterpolation<primitivePatch>(pp)
515  );
516  }
517  }
518 
519  if (nPatchWriters)
520  {
521  vtmWriter.endBlock("boundary");
522  vtmBoundaries.endBlock("boundary");
523  }
524 
525  patchWriters.resize(nPatchWriters);
526  patchInterps.resize(nPatchInterps);
527  }
528 
529  // CellData
530  {
531  if (internalWriter.valid())
532  {
533  // Optionally with cellID and procID fields
534  internalWriter->beginCellData
535  (
536  (writeIds_ ? 1 + (internalWriter->parallel() ? 1 : 0) : 0)
537  + nVolFields + nDimFields
538  );
539 
540  if (writeIds_)
541  {
542  internalWriter->writeCellIDs();
543  internalWriter->writeProcIDs(); // parallel only
544  }
545  }
546 
547  if (nVolFields)
548  {
549  for (vtk::patchWriter& writer : patchWriters)
550  {
551  // Optionally with patchID field
552  writer.beginCellData
553  (
554  (writeIds_ ? 1 : 0)
555  + nVolFields
556  );
557 
558  if (writeIds_)
559  {
560  writer.writePatchIDs();
561  }
562  }
563  }
564 
565  writeAllVolFields
566  (
567  internalWriter,
568  patchWriters,
569  meshProxy,
570  acceptField
571  );
572 
573  // writeAllDimFields
574  // (
575  // internalWriter,
576  // meshProxy,
577  // acceptField
578  // );
579 
580  // End CellData is implicit
581  }
582 
583 
584  // PointData
585  // - only construct pointMesh on request since it constructs
586  // edge addressing
587  if (interpolate_)
588  {
589  // Begin PointData
590  if (internalWriter.valid())
591  {
592  internalWriter->beginPointData
593  (
594  nVolFields + nDimFields
595  );
596  }
597 
598  forAll(patchWriters, writeri)
599  {
600  const label nPatchFields =
601  (
602  writeri < patchInterps.size() && patchInterps.set(writeri)

604  : 0
605  );
606 
607  if (nPatchFields)
608  {
609  patchWriters[writeri].beginPointData(nPatchFields);
610  }
611  }
612 
613  writeAllVolFields
614  (
615  internalWriter, pInterp,
616  patchWriters, patchInterps,
617  meshProxy,
618  acceptField
619  );
620 
621  // writeAllDimFields
622  // (
623  // internalWriter, pInterp,
624  // meshProxy,
625  // acceptField
626  // );
627 
628  // writeAllPointFields
629  // (
630  // internalWriter,
631  // patchWriters,
632  // meshProxy,
633  // acceptField
634  // );
635 
636  // End PointData is implicit
637  }
638 
639 
640  // Finish writers
641  if (internalWriter.valid())
642  {
643  internalWriter->close();
644  }
645 
646  for (vtk::patchWriter& writer : patchWriters)
647  {
648  writer.close();
649  }
650 
651  pInterp.clear();
652  patchWriters.clear();
653  patchInterps.clear();
654

657 
658  if (Pstream::master())
659  {
660  // Naming for vtm, file series etc.
661  fileName outputName(vtmOutputBase);
662 
663  if (writeOpts_.legacy())
664  {
665  if (doInternal_)
