98  )

110 template<class BasePhaseModel, class phaseThermo>
111 void Foam::MultiComponentPhaseModel<BasePhaseModel, phaseThermo>
112 ::calculateVolumeFractions()
113 {
114  volScalarField Xtotal(0.0*X_[0]);
115  const volScalarField W(thermo().W());
116 
117  forAll(X_, i)
118  {
119  const dimensionedScalar Wi
120  (
121  "W",
122  dimMass/dimMoles,
123  thermo().composition().W(i)
124  );
125 
126  if (i != inertIndex_)
127  {
128  X_[i] = W*Y()[i]/Wi;
129  Xtotal += X_[i];
130  X_[i].correctBoundaryConditions();
131  }
132  }
133  X_[inertIndex_] = 1.0 - Xtotal;
134  X_[inertIndex_].correctBoundaryConditions();
135 }

138 template<class BasePhaseModel, class phaseThermo>
139 void Foam::MultiComponentPhaseModel<BasePhaseModel, phaseThermo>::
140 calculateMassFractions()
141 {
142  volScalarField W(X_[0]*thermo().composition().W(0));
143  for(label i=1; i< species_.size(); i++)
144  {
145  W += X_[i]*thermo().composition().W(i);
146  }
147 
148  forAll(Y(), i)
149  {
150  Y()[i] = X_[i]*thermo().composition().W(i)/W;
151 
152  Info<< Y()[i].name() << " mass fraction = "
153  << " Min(Y) = " << min(Y()[i]).value()
154  << " Max(Y) = " << max(Y()[i]).value()
155  << endl;
156  }
157 }

183 template<class BasePhaseModel, class phaseThermo>
184 void Foam::MultiComponentPhaseModel<BasePhaseModel, phaseThermo>::solveYi
185 (
186  PtrList<volScalarField::Internal>& Su,
187  PtrList<volScalarField::Internal>& Sp
188 )
189 {
190  const volScalarField& alpha1 = *this;
191 
192  const fvMesh& mesh = alpha1.mesh();
193 
194  const dictionary& MULEScontrols = mesh.solverDict(alpha1.name());
195 
196  scalar cAlpha(readScalar(MULEScontrols.lookup("cYi")));
197 
198  PtrList<surfaceScalarField> phiYiCorrs(species_.size());
199  const surfaceScalarField& phi = this->fluid().phi();
200 
201  surfaceScalarField phic(mag((phi)/mesh.magSf()));
202 
203  phic = min(cAlpha*phic, max(phic));
204 
205  surfaceScalarField phir(0.0*phi);
206 
207  forAllConstIter(phaseSystem::phaseModelTable,this->fluid().phases(),iter2)
208  {
209  const volScalarField& alpha2 = iter2()();
210  if (&alpha2 == &alpha1)
211  {
212  continue;
213  }
214 
215  phir += phic*this->fluid().nHatf(alpha1, alpha2);
216  }
217 
218  // Do not compress interface at non-coupled boundary faces
219  // (inlets, outlets etc.)
220  surfaceScalarField::Boundary& phirBf = phir.boundaryFieldRef();
221  forAll(phir.boundaryField(), patchi)
222  {
223  fvsPatchScalarField& phirp = phirBf[patchi];
224 
225  if (!phirp.coupled())
226  {
227  phirp == 0;
228  }
229  }
230 
231  word phirScheme("div(Yiphir," + alpha1.name() + ')');
232 
233  forAll(X_, i)
234  {
235  if (inertIndex_ != i)
236  {
237  volScalarField& Yi = X_[i];
238 
239  phiYiCorrs.set
240  (
241  i,
242  new surfaceScalarField
243  (
244  "phi" + Yi.name() + "Corr",
245  fvc::flux
246  (
247  phi,
248  Yi,
249  "div(phi," + Yi.name() + ')'
250  )
251  )
252  );
253 
254  surfaceScalarField& phiYiCorr = phiYiCorrs[i];
255 
256  forAllConstIter
257  (
258  phaseSystem::phaseModelTable, this->fluid().phases(), iter2
259  )
260  {
261  //const volScalarField& alpha2 = iter2()().oldTime();
262  const volScalarField& alpha2 = iter2()();
263 
264  if (&alpha2 == &alpha1)
265  {
266  continue;
267  }
268 
269  phiYiCorr += fvc::flux
270  (
271  -fvc::flux(-phir, alpha2, phirScheme),
272  Yi,
273  phirScheme
274  );
275  }
276 
277  // Ensure that the flux at inflow BCs is preserved
278  forAll(phiYiCorr.boundaryField(), patchi)
279  {
280  fvsPatchScalarField& phiYiCorrp =
281  phiYiCorr.boundaryFieldRef()[patchi];
282 
283  if (!phiYiCorrp.coupled())
284  {
285  const scalarField& phi1p = phi.boundaryField()[patchi];
286  const scalarField& Yip = Yi.boundaryField()[patchi];
287 
288  forAll(phiYiCorrp, facei)
289  {
290  if (phi1p[facei] < 0)
291  {
292  phiYiCorrp[facei] = Yip[facei]*phi1p[facei];
293  }
294  }
295  }
296  }
297 
298  MULES::limit
299  (
300  1.0/mesh.time().deltaT().value(),
301  geometricOneField(),
302  Yi,
303  phi,
304  phiYiCorr,
305  Sp[i],
306  Su[i],
307  oneField(),
308  zeroField(),
309  true
310  );
311  }
312  }
313 
314  volScalarField Yt(0.0*X_[0]);
315 
316  scalar nYiSubCycles
317  (
318  MULEScontrols.lookupOrDefault<scalar>("nYiSubCycles", 1)
319  );
320 
321  forAll(X_, i)
322  {
323  if (inertIndex_ != i)
324  {
325  volScalarField& Yi = X_[i];
326 
327  fvScalarMatrix YiEqn
328  (
329  fv::EulerDdtScheme<scalar>(mesh).fvmDdt(Yi)
330  + fv::gaussConvectionScheme<scalar>
331  (
332  mesh,
333  phi,
334  upwind<scalar>(mesh, phi)
335  ).fvmDiv(phi, Yi)
336  ==
337  Su[i] + fvm::Sp(Sp[i], Yi)
338  );
339 
340  YiEqn.solve();
341 
342  surfaceScalarField& phiYiCorr = phiYiCorrs[i];
343 
344  // Add a bounded upwind U-mean flux
345  phiYiCorr += YiEqn.flux();
346 
347  if (nYiSubCycles > 1)
348  {
349  for
350  (
351  subCycle<volScalarField> YiSubCycle(Yi, nYiSubCycles);
352  !(++YiSubCycle).end();
353  )
354  {
355  MULES::explicitSolve
356  (
357  geometricOneField(),
358  Yi,
359  phi,
360  phiYiCorr,
361  Sp[i],
362  Su[i],
363  oneField(),
364  zeroField()
365  );
366  }
367  }
368  else
369  {
370  MULES::explicitSolve
371  (
372  geometricOneField(),
373  Yi,
374  phi,
375  phiYiCorr,
376  Sp[i],
377  Su[i],
378  oneField(),
379  zeroField()
380  );
381  }
382  Yt += Yi;
383  }
384  }
385 
386  X_[inertIndex_] = scalar(1) - Yt;
387  X_[inertIndex_].max(0.0);
388 
389  calculateMassFractions();
390 }
