69 Foam::ParticleErosion<CloudType>::ParticleErosion
70 (

103  // trigger creation of the Q field

156 template<class CloudType>
157 void Foam::ParticleErosion<CloudType>::postPatch
158 (
159  const parcelType& p,
160  const polyPatch& pp,
161  bool&
162 )
163 {
164  const label patchi = pp.index();
165 
166  const label localPatchi = applyToPatch(patchi);
167 
168  if (localPatchi != -1)
169  {
170  vector nw;
171  vector Up;
172 
173  // patch-normal direction
174  this->owner().patchData(p, pp, nw, Up);
175 
176  // particle velocity relative to patch
177  const vector& U = p.U() - Up;
178 
179  // quick reject if particle travelling away from the patch
180  if ((nw & U) < 0)
181  {
182  return;
183  }
184 
185  const scalar magU = mag(U);
186  const vector Udir = U/magU;
187 
188  // determine impact angle, alpha
189  const scalar alpha = mathematical::piByTwo - acos(nw & Udir);
190 
191  const scalar coeff = p.nParticle()*p.mass()*sqr(magU)/(p_*psi_*K_);
192 
193  const label patchFacei = pp.whichFace(p.face());
194  scalar& Q = QPtr_->boundaryFieldRef()[patchi][patchFacei];
195  if (tan(alpha) < K_/6.0)
196  {
197  Q += coeff*(sin(2.0*alpha) - 6.0/K_*sqr(sin(alpha)));
198  }
199  else
200  {
201  Q += coeff*(K_*sqr(cos(alpha))/6.0);
202  }
203  }
204 }
