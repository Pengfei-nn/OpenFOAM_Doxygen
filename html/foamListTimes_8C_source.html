33 Description
34  List times using the timeSelector, or use to remove selected time
35  directories.
36 
37 Usage
38  \b foamListTimes [OPTION]
39 
40  Options:
41  - \par -processor
42  List times from processor0/ directory

44  - \par -rm
45  Remove selected time directories
46 
47  - \par -verbose

117  }
118 
119  timePaths = autoPtr<TimePaths>::New
120  (

122  args.caseName()/"processor0"
123  );
124  }
125  else
126  {
127  timePaths = autoPtr<TimePaths>::New
128  (
129  args.rootPath(),
130  args.caseName()
131  );
132  }
133 
134 
135  const instantList timeDirs(timeSelector::select(timePaths->times(), args));
136 
137  const label nTimes = timeDirs.size();
138 
139  if (removeFiles)
140  {
141  if (nProcs)
142  {
143  if (verbose)
144  {
145  Info<< "Removing " << nTimes

147  }
148

150  {

152 
153  if (verbose)

157  << endl;
158  }
159

161 
162  rmDir(path, true);
163

165  {
166  path =
167  (

170  / timeName
171  );
172 
173  rmDir(path, true);
174  }
175  }
176  }
177  else
178  {
179  if (verbose)
180  {
181  Info<< "Removing " << nTimes
182  << " time directories" << endl;
183  }
184 
185  forAllReverse(timeDirs, timei)
186  {
187  const word& timeName = timeDirs[timei].name();
188 
189  if (verbose)
190  {
191  Info<< " rm " << timeName
192  << " [" << (nTimes - timei) << '/' << nTimes << ']'
193  << endl;
194  }
195 
196  rmDir(args.path()/timeName, true);
197  }
198  }
199  }
200  else
201  {
202  for (const instant& t : timeDirs)
203  {
204  Info<< t.name() << nl;
