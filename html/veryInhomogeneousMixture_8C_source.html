65 template<class ThermoType>
66 const ThermoType& Foam::veryInhomogeneousMixture<ThermoType>::mixture
67 (
68  const scalar ft,
69  const scalar fu
70 ) const
71 {
72  if (ft < 0.0001)
73  {
74  return oxidant_;
75  }
76  else
77  {
78  scalar ox = 1 - ft - (ft - fu)*stoicRatio().value();
79  scalar pr = 1 - fu - ox;
80 
81  mixture_ = fu*fuel_;
82  mixture_ += ox*oxidant_;
83  mixture_ += pr*products_;
84 
85  return mixture_;
86  }
87 }
