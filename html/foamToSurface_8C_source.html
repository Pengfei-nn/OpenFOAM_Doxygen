60 {

82  #include "setRootCase.H"
83 
84  fileName exportName = args[1];
85 
86  const scalar scaleFactor = args.get<scalar>("scale", 0);
87  const bool doTriangulate = args.found("tri");
88 
89  fileName exportBase = exportName.lessExt();
90  word exportExt = exportName.ext();
91 
92  if (!meshedSurface::canWriteType(exportExt, true))
93  {
94  return 1;
95  }
96 
97  #include "createTime.H"
98  instantList timeDirs = timeSelector::select0(runTime, args);
99  #include "createPolyMesh.H"
100 
101  forAll(timeDirs, timeI)
102  {
103  runTime.setTime(timeDirs[timeI], timeI);
104  #include "getTimeIndex.H"
105 
106  polyMesh::readUpdateState state = mesh.readUpdate();
107 
108  if (timeI == 0 || state != polyMesh::UNCHANGED)
109  {
110  if (state == polyMesh::UNCHANGED)
111  {
112  exportName = exportBase + "." + exportExt;
113  }
114  else
115  {
116  exportName =
117  exportBase + '_' + runTime.timeName() + "." + exportExt;
118  }
119 
120  meshedSurface surf(mesh.boundaryMesh());
121  surf.scalePoints(scaleFactor);
122 
123  Info<< "writing " << exportName;
124  if (doTriangulate)
125  {
126  Info<< " triangulated";
127  surf.triangulate();
128  }
129 
130  if (scaleFactor <= 0)
131  {
132  Info<< " without scaling" << endl;
133  }
134  else
135  {
136  Info<< " with scaling " << scaleFactor << endl;
137  }
138  surf.write(exportName);
139  }
140 
141  Info<< nl << endl;
142  }
143 
144  Info<< "End\n" << endl;
145 
146  return 0;
147 }
