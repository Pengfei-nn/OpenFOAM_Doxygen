196 Foam::solveScalarField::cmptType Foam::lduMatrix::solver::normFactor
197 (
198  const solveScalarField& psi,
199  const solveScalarField& source,
200  const solveScalarField& Apsi,
201  solveScalarField& tmpField
202 ) const
203 {
204  // --- Calculate A dot reference value of psi
205  matrix_.sumA(tmpField, interfaceBouCoeffs_, interfaces_);
206 
207  tmpField *= gAverage(psi, matrix_.lduMesh_.comm());
208 
209  return
210  gSum
211  (
212  (mag(Apsi - tmpField) + mag(source - tmpField))(),
213  matrix_.lduMesh_.comm()
214  )
215  + solverPerformance::small_;
216 
217  // At convergence this simpler method is equivalent to the above
218  // return 2*gSumMag(source) + solverPerformance::small_;
