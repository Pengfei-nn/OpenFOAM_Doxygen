55  const edgeScalarField gammaMagSf(gamma*this->mesh().magLe());
56 
57  tmp<faMatrix<Type>> tfam
58  (
59  new faMatrix<Type>

63  )
64  );

77  fam.boundaryCoeffs()[patchI] =

79  }
