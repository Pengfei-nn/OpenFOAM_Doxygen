133  const HashTable<Container, string>& cache,
134  const arrayRange& selector,
135  const bool singleDataset
136 ) const
137 {
138  const auto blockNo = output->GetNumberOfBlocks();
139  vtkSmartPointer<vtkMultiBlockDataSet> block;

141 
142  const List<label> partIds = selector.intersection(selectedPartIds_);
143 
144  for (const auto partId : partIds)
145  {
146  const auto& longName = selectedPartIds_[partId];
147  const word shortName = getFoamName(longName);
148 
149  auto iter = cache.find(longName);
150  if (iter.found() && iter.val().dataset)
151  {
152  auto dataset = iter.val().dataset;

156  output->SetBlock(blockNo, dataset);
157  output->GetMetaData(blockNo)->Set
158  (

160  shortName.c_str()
161  );
162 
163  ++datasetNo;
164  break;
165  }
166  else if (datasetNo == 0)
167  {
168  block = vtkSmartPointer<vtkMultiBlockDataSet>::New();
169  output->SetBlock(blockNo, block);
170  output->GetMetaData(blockNo)->Set
171  (
172  vtkCompositeDataSet::NAME(),
173  selector.name()
174  );
175  }
176 
177  block->SetBlock(datasetNo, dataset);
178  block->GetMetaData(datasetNo)->Set
179  (
180  vtkCompositeDataSet::NAME(),
181  shortName.c_str()
182  );
183 
184  ++datasetNo;
185  }
186  }

192 int Foam::vtkPVFoam::setTime(const std::vector<double>& requestTimes)
193 {
194  if (requestTimes.empty())
195  {
196  return 0;
197  }
198 
199  Time& runTime = dbPtr_();
200 
201  // Get times list. Flush first to force refresh.
202  fileHandler().flush();
203  instantList Times = runTime.times();
204 
205  int nearestIndex = timeIndex_;
206  for (const double& timeValue : requestTimes)
207  {
208  const int index = Time::findClosestTimeIndex(Times, timeValue);
209  if (index >= 0 && index != timeIndex_)
210  {
211  nearestIndex = index;
212  break;
213  }
214  }
215 
216  if (nearestIndex < 0)
217  {
218  nearestIndex = 0;
219  }
220 
221  if (debug)
222  {
223  Info<< "<beg> setTime(";
224  unsigned reqi = 0;
225  for (const double& timeValue : requestTimes)
226  {
227  if (reqi) Info<< ", ";
228  Info<< timeValue;
229  ++reqi;
230  }
231  Info<< ") - previousIndex = " << timeIndex_
232  << ", nearestIndex = " << nearestIndex << nl;
233  }
234 
235  // See what has changed
236  if (timeIndex_ != nearestIndex)
237  {
238  timeIndex_ = nearestIndex;
239  runTime.setTime(Times[nearestIndex], nearestIndex);
240 
241  // When mesh changes, so do fields
242  meshState_ =
243  (
244  volMeshPtr_
245  ? volMeshPtr_->readUpdate()
246  : polyMesh::TOPO_CHANGE
247  );
248 
249  reader_->UpdateProgress(0.05);
250 
251  // this seems to be needed for catching Lagrangian fields
252  updateInfo();
253  }
254 
255  DebugInfo
256  << "<end> setTime() - selectedTime="
257  << Times[nearestIndex].name() << " index=" << timeIndex_
258  << "/" << Times.size()
259  << " meshUpdateState=" << updateStateName(meshState_) << nl;
260 
261  return nearestIndex;
262 }

273 Foam::vtkPVFoam::vtkPVFoam
274 (
275  const char* const vtkFileName,
276  vtkPVFoamReader* reader
277 )
278 :
279  reader_(reader),
280  dbPtr_(nullptr),
281  volMeshPtr_(nullptr),
282  areaMeshPtr_(nullptr),
283  meshRegion_(polyMesh::defaultRegion),
284  meshDir_(polyMesh::meshSubDir),
285  timeIndex_(-1),
286  decomposePoly_(false),
287  meshState_(polyMesh::TOPO_CHANGE),
288  rangeVolume_("volMesh"),
289  rangeArea_("areaMesh"),
290  rangePatches_("patch"),
291  rangeClouds_("lagrangian"),
292  rangeCellZones_("cellZone"),
293  rangeFaceZones_("faceZone"),
294  rangePointZones_("pointZone"),
295  rangeCellSets_("cellSet"),
296  rangeFaceSets_("faceSet"),
297  rangePointSets_("pointSet")
298 {
299  if (debug)
300  {
301  Info<< "vtkPVFoam - " << vtkFileName << nl;
302  printMemory();
303  }
304 
305  fileName FileName(vtkFileName);
306 
307  // avoid argList and get rootPath/caseName directly from the file
308  fileName fullCasePath(FileName.path());
309 
310  if (!isDir(fullCasePath))
311  {
312  return;
313  }
314  if (fullCasePath == ".")
315  {
316  fullCasePath = cwd();
317  }
318 
319  // OPENFOAM API
320  setEnv("FOAM_API", std::to_string(foamVersion::api), true);
321 
322  // The name of the executable, unless already present in the environment
323  setEnv("FOAM_EXECUTABLE", "paraview", false);
324 
325  // Set the case as an environment variable - some BCs might use this
326  if (fullCasePath.name().find("processors", 0) == 0)
327  {
328  // FileName e.g. "cavity/processors256/processor1.OpenFOAM
329  // Remove the processors section so it goes into processorDDD
330  // checking below.
331  fullCasePath = fullCasePath.path()/fileName(FileName.name()).lessExt();
332  }
333 
334 
335  if (fullCasePath.name().find("processor", 0) == 0)
336  {
337  // Give filehandler opportunity to analyse number of processors
338  (void)fileHandler().filePath(fullCasePath);
339 
340  const fileName globalCase = fullCasePath.path();
341 
342  setEnv("FOAM_CASE", globalCase, true);
343  setEnv("FOAM_CASENAME", globalCase.name(), true);
344  }
345  else
346  {
347  setEnv("FOAM_CASE", fullCasePath, true);
348  setEnv("FOAM_CASENAME", fullCasePath.name(), true);
349  }
350 
351  // look for 'case{region}.OpenFOAM'
352  // could be stringent and insist the prefix match the directory name...
353  // Note: cannot use fileName::name() due to the embedded '{}'
354  string caseName(fileName(FileName).lessExt());
355  const auto beg = caseName.find_last_of("/{");
356  const auto end = caseName.find('}', beg);
357 
358  if
359  (
360  beg != std::string::npos && caseName[beg] == '{'
361  && end != std::string::npos && end == caseName.size()-1
362  )
363  {
364  meshRegion_ = caseName.substr(beg+1, end-beg-1);
365 
366  // some safety
367  if (meshRegion_.empty())
368  {
369  meshRegion_ = polyMesh::defaultRegion;
370  }
371 
372  if (meshRegion_ != polyMesh::defaultRegion)
373  {
374  meshDir_ = meshRegion_/polyMesh::meshSubDir;
375  }
376  }
377 
378  DebugInfo
379  << "fullCasePath=" << fullCasePath << nl
380  << "FOAM_CASE=" << getEnv("FOAM_CASE") << nl
381  << "FOAM_CASENAME=" << getEnv("FOAM_CASENAME") << nl
382  << "region=" << meshRegion_ << nl;
383 
384  // Create time object
385  dbPtr_.reset
386  (
387  new Time
388  (
389  Time::controlDictName,
390  fileName(fullCasePath.path()),
391  fileName(fullCasePath.name())
392  )
393  );
394 
395  dbPtr_().functionObjects().off();
396 
397  updateInfo();
398 }

414 void Foam::vtkPVFoam::updateInfo()
415 {
416  DebugInfo
417  << "<beg> updateInfo"
418  << " [volMeshPtr=" << (volMeshPtr_ ? "set" : "nullptr")
419  << "] timeIndex="
420  << timeIndex_ << nl;
421 
422  resetCounters();
423 
424  // Part selection
425  {
426  vtkDataArraySelection* select = reader_->GetPartSelection();
427 
428  // There are two ways to ensure we have the correct list of parts:
429  // 1. remove everything and then set particular entries 'on'
430  // 2. build a 'char **' list and call SetArraysWithDefault()
431  //
432  // Nr. 2 has the potential advantage of not touching the modification
433  // time of the vtkDataArraySelection, but the qt/paraview proxy
434  // layer doesn't care about that anyhow.
435 
436  HashSet<string> enabled;
437  if (!select->GetNumberOfArrays() && !volMeshPtr_)
438  {
439  // Fake enable 'internalMesh' on the first call
440  enabled = { "internalMesh" };
441  }
442  else
443  {
444  // Preserve the enabled selections
445  enabled = getSelectedArraySet(select);
446  }
447 
448  select->RemoveAllArrays(); // Clear existing list
449 
450  // Update mesh parts list - add Lagrangian at the bottom
451  updateInfoInternalMesh(select);
452  updateInfoPatches(select, enabled);
453  updateInfoSets(select);
454  updateInfoZones(select);
455  updateInfoAreaMesh(select);
456  updateInfoLagrangian(select);
457 
458  setSelectedArrayEntries(select, enabled); // Adjust/restore selected
459  }
460 
461  // Volume and area fields - includes save/restore of selected
462  updateInfoContinuumFields(reader_->GetVolFieldSelection());
463 
464  // Point fields - includes save/restore of selected
465  updateInfoPointFields(reader_->GetPointFieldSelection());
466 
467  // Lagrangian fields - includes save/restore of selected
468  updateInfoLagrangianFields(reader_->GetLagrangianFieldSelection());
469 
470  DebugInfo << "<end> updateInfo" << nl;
471 }

474 void Foam::vtkPVFoam::Update
475 (
476  vtkMultiBlockDataSet* output,
477  vtkMultiBlockDataSet* outputLagrangian
478 )
479 {
480  if (debug)
481  {
482  cout<< "<beg> Foam::vtkPVFoam::Update\n";
483  output->Print(cout);
484  if (outputLagrangian) outputLagrangian->Print(cout);
485  printMemory();
486  }
487  reader_->UpdateProgress(0.1);
488 
489  const int caching = reader_->GetMeshCaching();
490  const bool oldDecomp = decomposePoly_;
491  decomposePoly_ = !reader_->GetUseVTKPolyhedron();
492 
493  // Set up mesh parts selection(s)
494  // Update cached, saved, unneed values.
495  {
496  vtkDataArraySelection* selection = reader_->GetPartSelection();
497  const int n = selection->GetNumberOfArrays();
498 
499  selectedPartIds_.clear();
500  HashSet<string> nowActive;
501 
502  for (int id=0; id < n; ++id)
503  {
504  const string str(selection->GetArrayName(id));
505  const bool status = selection->GetArraySetting(id);
506 
507  if (status)
508  {
509  selectedPartIds_.set(id, str); // id -> name
510  nowActive.set(str);
511  }
512 
513  if (debug > 1)
514  {
515  Info<< " part[" << id << "] = " << status
516  << " : " << str << nl;
517  }
518  }
519 
520  // Dispose of unneeded components
521  cachedVtp_.retain(nowActive);
522  cachedVtu_.retain(nowActive);
523 
524  if
525  (
526  !caching
527  || meshState_ == polyMesh::TOPO_CHANGE
528  || meshState_ == polyMesh::TOPO_PATCH_CHANGE
529  )
530  {
531  // Eliminate cached values that would be unreliable
532  forAllIters(cachedVtp_, iter)
533  {
534  iter.val().clearGeom();
535  iter.val().clear();
536  }
537  forAllIters(cachedVtu_, iter)
538  {
539  iter.val().clearGeom();
540  iter.val().clear();
541  }
542  }
543  else if (oldDecomp != decomposePoly_)
544  {
545  // poly-decompose changed - dispose of cached values
546  forAllIters(cachedVtu_, iter)
547  {
548  iter.val().clearGeom();
549  iter.val().clear();
550  }
551  }
552  }
553 
554  reader_->UpdateProgress(0.15);
555 
556  // Update the OpenFOAM mesh
557  {
558  if (debug)
559  {
560  Info<< "<beg> updateFoamMesh" << nl;
561  printMemory();
562  }
563 
564  if (!caching)
565  {
566  delete volMeshPtr_;
567  delete areaMeshPtr_;
568 
569  volMeshPtr_ = nullptr;
570  areaMeshPtr_ = nullptr;
571  }
572 
573  // Check to see if the OpenFOAM mesh has been created
574  if (!volMeshPtr_)
575  {
576  DebugInfo
577  << "Creating OpenFOAM mesh for region " << meshRegion_
578  << " at time=" << dbPtr_().timeName() << nl;
579 
580  volMeshPtr_ = new fvMesh
581  (
582  IOobject
583  (
584  meshRegion_,
585  dbPtr_().timeName(),
586  dbPtr_(),
587  IOobject::MUST_READ
588  )
589  );
590 
591  meshState_ = polyMesh::TOPO_CHANGE; // New mesh
592  }
593  else
594  {
595  DebugInfo << "Using existing OpenFOAM mesh" << nl;
596  }
597 
598  if (rangeArea_.intersects(selectedPartIds_))
599  {
600  if (!areaMeshPtr_)
601  {
602  areaMeshPtr_ = new faMesh(*volMeshPtr_);
603  }
604  }
605  else
606  {
607  delete areaMeshPtr_;
608 
609  areaMeshPtr_ = nullptr;
610  }
611 
612  if (debug)
613  {
614  Info<< "<end> updateFoamMesh" << nl;
615  printMemory();
616  }
617  }
618 
619  reader_->UpdateProgress(0.4);
620 
621  convertMeshVolume();
622  convertMeshPatches();
623  reader_->UpdateProgress(0.6);
624 
625  if (reader_->GetIncludeZones())
626  {
627  convertMeshCellZones();
628  convertMeshFaceZones();
629  convertMeshPointZones();
630  reader_->UpdateProgress(0.65);
631  }
632 
633  if (reader_->GetIncludeSets())
634  {
635  convertMeshCellSets();
636  convertMeshFaceSets();
637  convertMeshPointSets();
638  reader_->UpdateProgress(0.7);
639  }
640 
641  convertMeshArea();
642 
643  convertMeshLagrangian();
644 
645  reader_->UpdateProgress(0.8);
646 
647  // Update fields
648  convertVolFields();
649  convertPointFields();
650  convertAreaFields();
651 
652  convertLagrangianFields();
653 
654  // Assemble multiblock output
655  addOutputBlock(output, cachedVtu_, rangeVolume_, true); // One dataset
656  addOutputBlock(output, cachedVtp_, rangePatches_);
657  addOutputBlock(output, cachedVtu_, rangeCellZones_);
658  addOutputBlock(output, cachedVtp_, rangeFaceZones_);
659  addOutputBlock(output, cachedVtp_, rangePointZones_);
660  addOutputBlock(output, cachedVtu_, rangeCellSets_);
661  addOutputBlock(output, cachedVtp_, rangeFaceSets_);
662  addOutputBlock(output, cachedVtp_, rangePointSets_);
663  addOutputBlock(output, cachedVtp_, rangeArea_);
664  addOutputBlock
665  (
666  (outputLagrangian ? outputLagrangian : output),
667  cachedVtp_,
668  rangeClouds_
669  );
670 
671  DebugInfo << "done reader part" << nl << nl;
672  reader_->UpdateProgress(0.95);
673 
674  meshState_ = polyMesh::UNCHANGED;
675 
676  if (caching & 2)
677  {
678  // Suppress caching of Lagrangian since it normally always changes.
679  cachedVtp_.filterKeys
680  (
681  [](const word& k){ return k.starts_with("lagrangian/"); },
682  true // prune
683  );
684  }
685  else
686  {
687  cachedVtp_.clear();
688  cachedVtu_.clear();
689  }
690 }

708 std::vector<double> Foam::vtkPVFoam::findTimes(const bool skipZero) const
709 {
710  std::vector<double> times;
711 
712  if (dbPtr_.valid())
713  {
714  const Time& runTime = dbPtr_();
715  // Get times list. Flush first to force refresh.
716  fileHandler().flush();
717  instantList timeLst = runTime.times();
718 
719  // find the first time for which this mesh appears to exist
720  label begIndex = timeLst.size();
721  forAll(timeLst, timei)
722  {
723  if
724  (
725  IOobject
726  (
727  "points",
728  timeLst[timei].name(),
729  meshDir_,
730  runTime
731  ).typeHeaderOk<pointIOField>(false, false)
732  )
733  {
734  begIndex = timei;
735  break;
736  }
737  }
738 
739  label nTimes = timeLst.size() - begIndex;
740 
741  // skip "constant" time whenever possible
742  if (begIndex == 0 && nTimes > 1)
743  {
744  if (timeLst[begIndex].name() == runTime.constant())
745  {
746  ++begIndex;
747  --nTimes;
748  }
749  }
750 
751  // skip "0/" time if requested and possible
752  if (skipZero && nTimes > 1 && timeLst[begIndex].name() == "0")
753  {
754  ++begIndex;
755  --nTimes;
756  }
757 
758  times.reserve(nTimes);
759  while (nTimes-- > 0)
760  {
761  times.push_back(timeLst[begIndex++].value());
762  }
763  }
764  else
765  {
766  if (debug)
767  {
768  cout<< "no valid dbPtr:\n";
769  }
770  }
771 
772  return times;
773 }

776 void Foam::vtkPVFoam::renderPatchNames
777 (
778  vtkRenderer* renderer,
779  const bool show
780 )
781 {
782  // Always remove old actors first
783 
784  for (auto& actor : patchTextActors_)
785  {
786  renderer->RemoveViewProp(actor);
787  }
788  patchTextActors_.clear();
789 
790  if (show && volMeshPtr_)
791  {
792  // Get the display patches, strip off any prefix/suffix
793  wordHashSet selectedPatches
794  (
795  getSelected(reader_->GetPartSelection(), rangePatches_)
796  );
797 
798  if (selectedPatches.empty())
799  {
800  return;
801  }
802 
803  const polyBoundaryMesh& pbMesh = volMeshPtr_->boundaryMesh();
804 
805  // Find the total number of zones
806  // Each zone will take the patch name
807  // Number of zones per patch ... zero zones should be skipped
808  labelList nZones(pbMesh.size(), Zero);
809 
810  // Per global zone number the average face centre position
811  List<DynamicList<point>> zoneCentre(pbMesh.size());
812 
813 
814  // Loop through all patches to determine zones, and centre of each zone
815  forAll(pbMesh, patchi)
816  {
817  const polyPatch& pp = pbMesh[patchi];
818 
819  // Only include the patch if it is selected
820  if (!selectedPatches.found(pp.name()))
821  {
822  continue;
823  }
824 
825  const labelListList& edgeFaces = pp.edgeFaces();
826  const vectorField& n = pp.faceNormals();
827 
828  boolList featEdge(pp.nEdges(), false);
829 
830  forAll(edgeFaces, edgei)
831  {
832  const labelList& eFaces = edgeFaces[edgei];
833 
834  if (eFaces.size() == 1)
835  {
836  // Note: could also do ones with > 2 faces but this gives
837  // too many zones for baffles
838  featEdge[edgei] = true;
839  }
840  else if (mag(n[eFaces[0]] & n[eFaces[1]]) < 0.5)
841  {
842  featEdge[edgei] = true;
843  }
844  }
845 
846  // Do topological analysis of patch, find disconnected regions
847  patchZones pZones(pp, featEdge);
848 
849  nZones[patchi] = pZones.nZones();
850 
851  labelList zoneNFaces(pZones.nZones(), Zero);
852 
853  // Create storage for additional zone centres
854  forAll(zoneNFaces, zonei)
855  {
856  zoneCentre[patchi].append(Zero);
857  }
858 
859  // Do averaging per individual zone
860  forAll(pp, facei)
861  {
862  const label zonei = pZones[facei];
863  zoneCentre[patchi][zonei] += pp[facei].centre(pp.points());
864  zoneNFaces[zonei]++;
865  }
866 
867  forAll(zoneCentre[patchi], zonei)
868  {
869  zoneCentre[patchi][zonei] /= zoneNFaces[zonei];
870  }
871  }
872 
873  // Count number of zones we're actually going to display.
874  // This is truncated to a max per patch
875 
876  const label MAXPATCHZONES = 20;
877 
878  label displayZoneI = 0;
879 
880  forAll(pbMesh, patchi)
881  {
882  displayZoneI += min(MAXPATCHZONES, nZones[patchi]);
883  }
884 
885  DebugInfo
886  << "displayed zone centres = " << displayZoneI << nl
887  << "zones per patch = " << nZones << nl;
888 
889  // Set the size of the patch labels to max number of zones
890  patchTextActors_.setSize(displayZoneI);
891 
892  DebugInfo << "constructing patch labels" << nl;
893 
894  // Actor index
895  displayZoneI = 0;
896 
897  forAll(pbMesh, patchi)
898  {
899  const polyPatch& pp = pbMesh[patchi];
900 
901  // Only selected patches will have a non-zero number of zones
902  const label nDisplayZones = min(MAXPATCHZONES, nZones[patchi]);
903  label increment = 1;
904  if (nZones[patchi] >= MAXPATCHZONES)
905  {
906  increment = nZones[patchi]/MAXPATCHZONES;
907  }
908 
909  label globalZoneI = 0;
910  for (label i = 0; i < nDisplayZones; ++i, globalZoneI += increment)
911  {
912  DebugInfo
913  << "patch name = " << pp.name() << nl
914  << "anchor = " << zoneCentre[patchi][globalZoneI] << nl
915  << "globalZoneI = " << globalZoneI << nl;
916 
917  // Into a list for later removal
918  patchTextActors_[displayZoneI++] = createTextActor
919  (
920  pp.name(),
921  zoneCentre[patchi][globalZoneI]
922  );
923  }
924  }
925 
926  // Resize the patch names list to the actual number of patch names added
927  patchTextActors_.setSize(displayZoneI);
928  }
929 
930  // Add text to each renderer
931  for (auto& actor : patchTextActors_)
932  {
933  renderer->AddViewProp(actor);
934  }
935 }
