50 // Cell blockage.
51 //
52 // Use simple sum, because this will eventually be divided by a notional
53 // number of rows to give a per-row blockage ratio
54 //
55 // b is the (fractional) area blockage. f is 1 or 0.5 to split between ends
56 // Thus if b isclose to 1, the obstacle is totally blocking the cell in this direction,
57 // and we could modify the behaviour if we wish.
58 inline static void add_blockage_c
59 (
60  scalar& a,
61  bool& blocked,
62  const scalar b,
63  const scalar f = 1.0
64 )
65 {
66  a += b * f;
67  if (b > pars.blockageNoCT)
68  {
69  blocked = true;
70  }
71 }
72 
73 
74 // Face blockage
75 //
76 // Adds more area blockage to existing amount by assuming partial overlap,
77 // i.e. multiplying porosities.
78 //
79 // Simple addition if the existing amount is negative, because negative
80 // blocks (summed first) should just cancel out part of positive blocks.
81 inline static void add_blockage_f
82 (
83  scalar& a,
84  const scalar b,
85  bool isHole
86 )
87 {
88  if (a > 0.0)
89  {
90  // Both positive
91  a = 1.0 - (1.0 - a) * (1.0 - b);
92  }
93  else if (b < pars.blockedFacePar || isHole)
94  {
95  // Add until it eventually becomes positive
96  a += b;
97  }
98  else
99  {
100  // If one obstacle blocks face, face is blocked, regardless of
101  // overlap calculations, unless an input negative obstacle makes a
102  // hole in it
103  a = b;
104  }
105 }

536  betai_inv1(ix,iy,iz).x() += vol_block / (1.0 - area_block.x() + floatSMALL);
537  betai_inv1(ix,iy,iz).y() += vol_block / (1.0 - aboverlap(iz,ix) + floatSMALL);
538 
539  // The off-diagonal elements of drag are stored in "drag" for BOTH round and sharp obstacles
540  drag_s(ix,iy,iz).xz() += c_drag(iz,ix).xy() * yoverlap[iy] / zgrid.width(iz) / xgrid.width(ix);
541 
542  add_blockage_f(face_block(ix,iy,iz).z(), a_lblock(iz,ix) * yoverlap[iy], hole_in_face(ix,iy,iz).z());
543  add_blockage_f(face_block(ix,iy,iz).x(), b_lblock(iz,ix) * yoverlap[iy], hole_in_face(ix,iy,iz).x());
544  }
545  if (cymin == cymax)
546  {
547  // Does not span cell. Block nearest face.
548  add_blockage_f(face_block(ix,cfymax,iz).y(), aboverlap(iz,ix), hole_in_face(ix,cfymax,iz).y());
549  }
550  else
551  {
552  // In at least two cells.
553  // Block first and last faces overlapped
554  add_blockage_f(face_block(ix,cymin+1,iz).y(), aboverlap(iz,ix), hole_in_face(ix,cymin+1,iz).y());
555  if (cymax > cymin+1)
556  {
557  add_blockage_f(face_block(ix,cymax,iz).y(), aboverlap(iz,ix), hole_in_face(ix,cymax,iz).y());
558  }
559  }
560 
561  for (label iy = cymin+1; iy < cymax; ++iy)
562  {
563  // Internal only
564  along_block(ix,iy,iz).y() += aboverlap(iz,ix);
565  }
566 
567  drag_s(ix,cymin,iz).yy() += aboverlap(iz,ix) * zgrid.width(iz) * xgrid.width(ix) / 2.0;
568  drag_s(ix,cymax,iz).yy() += aboverlap(iz,ix) * zgrid.width(iz) * xgrid.width(ix) / 2.0;
569 
570  add_blockage_c(area_block_s(ix,cymin,iz).y(), dirn_block(ix,cymin,iz).y(), aboverlap(iz,ix), 0.5);
571  add_blockage_c(area_block_s(ix,cymax,iz).y(), dirn_block(ix,cymax,iz).y(), aboverlap(iz,ix), 0.5);
572  }
573  }
574 
575  break;
576  }
577 
578  default: // orientation
579  {
580  Info<< "Unexpected orientation " << int(obs.orient) << nl;
581  break;
582  }
583  }
584 }

587 void Foam::PDRarrays::addBlockage
588 (

590  DynamicList<PDRpatchDef>& patches,
591  const int volumeSign
592 )
593 {
594  // The volumeSign indicates whether this pass is for negative or positive
595  // obstacles. Both if 0.
596  if
597  (
598  equal(obs.vbkge, 0)
599  || (volumeSign < 0 && obs.vbkge >= 0)
600  || (volumeSign > 0 && obs.vbkge < 0)
601  )
602  {
603  return;
604  }
605 
606  if (isNull(block()))
607  {
608  FatalErrorInFunction
609  << "No PDRblock set" << nl
610  << exit(FatalError);
611  }
612 
613  const PDRblock& pdrBlock = block();
614  const PDRblock::location& xgrid = pdrBlock.grid().x();
615  const PDRblock::location& ygrid = pdrBlock.grid().y();
616  const PDRblock::location& zgrid = pdrBlock.grid().z();

618  scalarList& xoverlap = overlap_1d.x();
619  scalarList& yoverlap = overlap_1d.y();
620  scalarList& zoverlap = overlap_1d.z();
621 
622 
623  // 0 will be used later for faces found to be blocked.
624  // 2 is used for wall-function faces.
625  label patchNum = PDRpatchDef::LAST_PREDEFINED;
626 
627  // Only the part of the panel that covers full cell faces will be used
628  // so later should keep the panel in the list plus a hole (-ve obstacle) for
629  // part that will become blowoff b.c.
630 
631  int indir = 0;
632 
633  // Panel or patch
634  const bool isPatch =
635  (
636  (obs.typeId == PDRobstacle::LOUVRE_BLOWOFF && obs.blowoff_type > 0)
637  || (obs.typeId == PDRobstacle::RECT_PATCH)
638  );
639 
640  if (isPatch)
641  {
642  const auto& identifier = obs.identifier;
643 
644  const auto spc = identifier.find_first_of(" \t\n\v\f\r");
645 
646  const word patchName = word::validate(identifier.substr(0, spc));
647 
648  patchNum = ListOps::find
649  (
650  patches,
651  [=](const PDRpatchDef& p){ return patchName == p.patchName; },
652  1 // skip 0 (blocked face)
653  );
654 
655  if (patchNum < 1)
656  {
657  // The patch name was not already in the list
658  patchNum = patches.size();
659 
660  patches.append(PDRpatchDef(patchName));
661  }
662 
663 
664  PDRpatchDef& p = patches[patchNum];
665 
666  if (obs.typeId == PDRobstacle::RECT_PATCH)
667  {
668  indir = obs.inlet_dirn;
669  p.patchType = 0;
670  }
671  else
672  {
673  p.patchType = obs.blowoff_type;
674  p.blowoffPress = obs.blowoff_press;
675  p.blowoffTime = obs.blowoff_time;
676  if (obs.span.x() < 1e-5)
677  {
678  indir = 1;
679  }
680  else if (obs.span.y() < 1e-5)
681  {
682  indir = 2;
683  }
684  else if (obs.span.z() < 1e-5)
685  {
686  indir = 3;
687  }
688  else
689  {
690  FatalErrorInFunction
691  << "Blowoff panel should have zero thickness" << nl
692  << exit(FatalError);
693  }
694  }
695  }
696 
697  int cxmin, cxmax, cfxmin, cfxmax;
698  one_d_overlap
699  (
700  obs.x(), obs.x() + obs.span.x(),
701  pdrBlock.grid().x(),
702  xoverlap, &cxmin, &cxmax, &cfxmin, &cfxmax
703  ); assert(cxmax >=0);
704 
705  int cymin, cymax, cfymin, cfymax;
706  one_d_overlap
707  (
708  obs.y(), obs.y() + obs.span.y(),
709  pdrBlock.grid().y(),
710  yoverlap, &cymin, &cymax, &cfymin, &cfymax
711  ); assert(cymax >=0);
712 
713  int czmin, czmax, cfzmin, cfzmax;
714  one_d_overlap
715  (
716  obs.z(), obs.z() + obs.span.z(),
717  pdrBlock.grid().z(),
718  zoverlap, &czmin, &czmax, &cfzmin, &cfzmax
719  ); assert(czmax >=0);
720 
721  two_d_overlap(xoverlap, cxmin, cxmax, yoverlap, cymin, cymax, aboverlap);
722 
723 
724  const scalar vbkge = obs.vbkge;
725  const scalar xbkge = obs.xbkge;
726  const scalar ybkge = obs.ybkge;
727  const scalar zbkge = obs.zbkge;
728 
729  // Compensate for double-counting of drag if two edges in same cell
730  const vector double_f
731  (
732  ((cxmin == cxmax) ? 0.5 : 1.0),
733  ((cymin == cymax) ? 0.5 : 1.0),
734  ((czmin == czmax) ? 0.5 : 1.0)
735  );
736 
737  for (label ix = cxmin; ix <= cfxmax; ix++)
738  {
739  const scalar xov = xoverlap[ix];
740 
741  scalar area, cell_area, temp;
742 
743  for (label iy = cymin; iy <= cfymax; iy++)
744  {
745  const scalar yov = yoverlap[iy];
746 
747  for (label iz = czmin; iz <= cfzmax; iz++)
748  {
749  const scalar zov = zoverlap[iz];
750 
751  if
752  (
753  isPatch
754  &&
755  (
756  (indir == -1 && ix == cfxmin)
757  || (indir == 1 && ix == cfxmax)
758  || (indir == -2 && iy == cfymin)
759  || (indir == 2 && iy == cfymax)
760  || (indir == -3 && iz == cfzmin)
761  || (indir == 3 && iz == cfzmax)
762  )
763  )
764  {
765  /* Type RECT_PATCH (16) exists to set all faces it covers to be in a particular patch
766  usually an inlet or outlet.
767  ?? If the face not on a cell boundary, this will move it to the lower-cordinate
768  face of the relevant cell. It should be at the face of teh volume blocked by
769  the obstacle. But, if that is not at a cell boundary, the obstacle will be
770  putting blockage in front of teh vent, so we should be checking that it is
771  at a cell boundary. */
772 
773  switch (indir) // Face orientation
774  {
775  // X
776  case -1:
777  case 1:
778  if (yov * zov > pars.blockedFacePar)
779  {
780  face_patch(ix,iy,iz).x() = patchNum;
781  }
782  break;
783 
784  // Y
785  case -2:
786  case 2:
787  if (zov * xov > pars.blockedFacePar)
788  {
789  face_patch(ix,iy,iz).y() = patchNum;
790  }
791  break;
792 
793  // Z
794  case -3:
795  case 3:
796  if (xov * yov > pars.blockedFacePar)
797  {
798  face_patch(ix,iy,iz).z() = patchNum;
799  }
800  break;
801  }
802  } // End of code for patch
803 
804 
805  const scalar vol_block = aboverlap(ix,iy) * zov * vbkge;
806  v_block(ix,iy,iz) += vol_block;
807 
808  // These are the face blockages
809  if ((ix > cxmin && ix <= cxmax) || (cxmin == cxmax && ix == cfxmax))
810  {
811  temp = yov * zov * xbkge;
812 
813  // Has -ve volumeSign only when processing user-defined
814  // -ve blocks
815  if (volumeSign < 0)
816  {
817  hole_in_face(ix,iy,iz).x() = true;
818  }
819  add_blockage_f(face_block(ix,iy,iz).x(), temp, hole_in_face(ix,iy,iz).x());
820  if (temp > pars.blockedFacePar && ! hole_in_face(ix,iy,iz).x() && !isPatch)
821  {
822  // Put faces of block in wall patch
823  face_patch(ix,iy,iz).x() = PDRpatchDef::WALL_PATCH;
824  }
825  }
826  if ((iy > cymin && iy <= cymax) || (cymin == cymax && iy == cfymax))
827  {
828  temp = zov * xov * ybkge;
829  if (volumeSign < 0)
830  {
831  hole_in_face(ix,iy,iz).y() = true;
832  }
833  add_blockage_f(face_block(ix,iy,iz).y(), temp, hole_in_face(ix,iy,iz).y());
834  if (temp > pars.blockedFacePar && ! hole_in_face(ix,iy,iz).y() && !isPatch)
835  {
836  face_patch(ix,iy,iz).y() = PDRpatchDef::WALL_PATCH;
837  }
838  }
839  if ((iz > czmin && iz <= czmax) || (czmin == czmax && iz == cfzmax))
840  {
841  temp = xov * yov * zbkge;
842  if (volumeSign < 0)
843  {
844  hole_in_face(ix,iy,iz).z() = true;
845  }
846  add_blockage_f(face_block(ix,iy,iz).z(), temp, hole_in_face(ix,iy,iz).z());
847  if (temp > pars.blockedFacePar && ! hole_in_face(ix,iy,iz).z() && !isPatch)
848  {
849  face_patch(ix,iy,iz).z() = PDRpatchDef::WALL_PATCH;
850  }
851  }
852 
853  // These are the interior blockages
854  /* As for cylinders, blockage that extends longitudinally all the way through the cell
855  should not be in x_block etc., but it does go into new arrays along_block etc. */
856  area = yov * zov * xbkge; // Note this is fraction of cell c-s area
857  if (ix < cxmin || ix > cxmax)
858  {}
859  else if (ix > cxmin && ix < cxmax && xbkge >= 1.0)
860  {
861  along_block(ix,iy,iz).x() += area;
862  betai_inv1(ix,iy,iz).x() += vol_block / (1.0 - area + floatSMALL);
863  }
864  else if (ix == cxmin || ix == cxmax)
865  {
866  // If front and back of the obstacle are not in the
867  // same cell, put half in each
868  const scalar double_f = (cxmin == cxmax ? 1.0 : 0.5);
869 
870  add_blockage_c(area_block_s(ix,iy,iz).x(), dirn_block(ix,iy,iz).x(), area, double_f);
871  cell_area = (ygrid.width(iy) * zgrid.width(iz));
872  surf(ix,iy,iz) += double_f * area * cell_area;
873  betai_inv1(ix,iy,iz).x() += vol_block / (1.0 - area + floatSMALL);
874 
875  // To get Lobs right for a grating, allow for the fact that it is series of bars by having additional count
876  if (obs.typeId == PDRobstacle::GRATING && obs.orient == vector::X)
877  {
878  // * cell_area to make dimensional, then / std::sqrt(cell_area) to get width
879  temp = area * std::sqrt(cell_area) / obs.slat_width - 1.0;
880  if (temp > 0.0) { grating_count(ix,iy,iz).x() += temp; }
881  }
882  }
883 
884  /* As for cylinders, blockage that extends longitudinally all the way through the cell
885  should not be in x_block etc., but it does go into new arrays along_block etc. */
886  area = zov * xov * ybkge;
887  if (iy < cymin || iy > cymax)
888  {}
889  else if (iy > cymin && iy < cymax && ybkge >= 1.0)
890  {
891  along_block(ix,iy,iz).y() += area;
892  betai_inv1(ix,iy,iz).y() += vol_block / (1.0 - area + floatSMALL);
893  }
894  else if (iy == cymin || iy == cymax)
895  {
896  // If front and back of the obstacle are not in the
897  // same cell, put half in each
898  const scalar double_f = (cymin == cymax ? 1.0 : 0.5);
899 
900  add_blockage_c(area_block_s(ix,iy,iz).y(), dirn_block(ix,iy,iz).y(), area, double_f);
901  cell_area = (zgrid.width(iz) * xgrid.width(ix));
902  surf(ix,iy,iz) += double_f * area * cell_area;
903  betai_inv1(ix,iy,iz).y() += vol_block / (1.0 - area + floatSMALL);
904 
905  if (obs.typeId == PDRobstacle::GRATING && obs.orient == vector::Y)
906  {
907  // * cell_area to make dimensional, then / std::sqrt(cell_area) to get width
908  temp = area * std::sqrt(cell_area) / obs.slat_width - 1.0;
909  if (temp > 0.0) { grating_count(ix,iy,iz).y() += temp; }
910  }
911  }
912 
913  area = xov * yov * zbkge;
914  if (iz < czmin || iz > czmax)
915  {}
916  else if (iz > czmin && iz < czmax && zbkge >= 1.0)
917  {
918  along_block(ix,iy,iz).z() += area;
919  betai_inv1(ix,iy,iz).z() += vol_block / (1.0 - area + floatSMALL);
920  }
921  else if (iz == czmin || iz == czmax)
922  {
923  // If front and back of the obstacle are not in the
924  // same cell, put half in each
925  const scalar double_f = (czmin == czmax ? 1.0 : 0.5);
926 
927  add_blockage_c(area_block_s(ix,iy,iz).z(), dirn_block(ix,iy,iz).z(), area, double_f);
928  cell_area = (xgrid.width(ix) * ygrid.width(iy));
929  surf(ix,iy,iz) += double_f * area * cell_area;
930  betai_inv1(ix,iy,iz).z() += vol_block / (1.0 - area + floatSMALL);
931 
932  if (obs.typeId == PDRobstacle::GRATING && obs.orient == vector::Z)
933  {
934  // * cell_area to make dimensional, then / std::sqrt(cell_area) to get width
935  temp = area * std::sqrt(cell_area) / obs.slat_width - 1.0;
936  if (temp > 0.0) { grating_count(ix,iy,iz).z() += temp; }
937  }
938  }
939  }
940  }
941  }
942 
943  if (obs.typeId == PDRobstacle::RECT_PATCH)
944  {
945  // Was only needed to set face_patch values
946  return;
947  }
948 
949 
950  /* A narrow obstacle completely crossing the cell adds 1 to the count for the transverse directions
951  If all four edges are not in the relevant cell, we take 1/4 for each edge that is in.
952  If it does not totally cross the cell, then the count is reduced proportionately
953  If it is porous, then the count is reduced proportionately.
954  ?? Should it be? At least this is consistent with effect going smoothly to zero as porosity approaches 1 ??
955 
956  Note that more than 1 can be added for one obstacle, if sides and ends are in the cell.
957 
958  We do all the x-aligned edges first, both for y-blockage and z-blockage. */
959 
960  /* Intersection obstacles can have more edges than the intersecting obstacles that
961  generated them, so not good to incorporate these in N and drag. */
962 
963  for (label ix = cxmin; ix <= cxmax; ix++)
964  {
965  // Factor of 0.25 because 4 edges to be done
966  scalar olap25 = 0.25 * xoverlap[ix];
967 
968  const scalar temp =
969  ((pars.noIntersectN && vbkge < 0.334) ? 0 : (olap25 * vbkge));
970 
971  obs_count(ix,cymin,czmin) += temp;
972  obs_count(ix,cymax,czmin) += temp;
973  obs_count(ix,cymin,czmax) += temp;
974  obs_count(ix,cymax,czmax) += temp;
975 
976  sub_count(ix,cymin,czmin).x() += temp;
977  sub_count(ix,cymax,czmin).x() += temp;
978  sub_count(ix,cymin,czmax).x() += temp;
979  sub_count(ix,cymax,czmax).x() += temp;
980 
981  // The 0.25 becomes 0.5 to allow for front/back faces in drag direction
982  olap25 *= 2.0;
983 
984  drag_s(ix,cymin,czmin).yy() += zoverlap[czmin] * double_f.z() * olap25 * ybkge / ygrid.width(cymin);
985  drag_s(ix,cymax,czmin).yy() += zoverlap[czmin] * double_f.z() * olap25 * ybkge / ygrid.width(cymax);
986  drag_s(ix,cymin,czmax).yy() += zoverlap[czmax] * double_f.z() * olap25 * ybkge / ygrid.width(cymin);
987  drag_s(ix,cymax,czmax).yy() += zoverlap[czmax] * double_f.z() * olap25 * ybkge / ygrid.width(cymax);
988 
989  drag_s(ix,cymin,czmin).zz() += yoverlap[cymin] * double_f.y() * olap25 * zbkge / zgrid.width(czmin);
990  drag_s(ix,cymax,czmin).zz() += yoverlap[cymax] * double_f.y() * olap25 * zbkge / zgrid.width(czmin);
991  drag_s(ix,cymin,czmax).zz() += yoverlap[cymin] * double_f.y() * olap25 * zbkge / zgrid.width(czmax);
992  drag_s(ix,cymax,czmax).zz() += yoverlap[cymax] * double_f.y() * olap25 * zbkge / zgrid.width(czmax);
993 
994  // Porous obstacles do not only have drag at edges
995  if (xbkge < 1.0)
996  {
997  for (label iy = cymin+1; iy < cymax; iy++)
998  {
999  for (label iz = czmin+1; iz < czmax; iz++)
1000  {
1001  // Internal only
1002  drag_s(ix,iy,iz).xx() = xbkge / xgrid.width(ix);
1003  }
1004  }
1005  }
1006  }
1007 
1008  for (label iy = cymin; iy <= cymax; iy++)
1009  {
1010  scalar olap25 = 0.25 * yoverlap[iy];
1011  const scalar temp =
1012  ((pars.noIntersectN && vbkge < 0.334) ? 0 : (olap25 * vbkge));
1013 
1014  obs_count(cxmin,iy,czmin) += temp;
1015  obs_count(cxmax,iy,czmin) += temp;
1016  obs_count(cxmin,iy,czmax) += temp;
1017  obs_count(cxmax,iy,czmax) += temp;
1018 
1019  sub_count(cxmin,iy,czmin).y() += temp;
1020  sub_count(cxmax,iy,czmin).y() += temp;
1021  sub_count(cxmin,iy,czmax).y() += temp;
1022  sub_count(cxmax,iy,czmax).y() += temp;
1023 
1024  olap25 *= 2.0;
1025 
1026  if (iy > cymin && iy < cymax) // Avoid re-doing corners already done above
1027  {
1028  drag_s(cxmin,iy,czmin).zz() += xoverlap[cxmin] * double_f.x() * olap25 * zbkge / zgrid.width(czmin);
1029  drag_s(cxmax,iy,czmin).zz() += xoverlap[cxmin] * double_f.x() * olap25 * zbkge / zgrid.width(czmin);
1030  drag_s(cxmin,iy,czmax).zz() += xoverlap[cxmax] * double_f.x() * olap25 * zbkge / zgrid.width(czmax);
1031  drag_s(cxmax,iy,czmax).zz() += xoverlap[cxmax] * double_f.x() * olap25 * zbkge / zgrid.width(czmax);
1032  }
1033  drag_s(cxmin,iy,czmin).xx() += zoverlap[czmin] * double_f.z() * olap25 * xbkge / xgrid.width(cxmin);
1034  drag_s(cxmax,iy,czmin).xx() += zoverlap[czmax] * double_f.z() * olap25 * xbkge / xgrid.width(cxmax);
1035  drag_s(cxmin,iy,czmax).xx() += zoverlap[czmin] * double_f.z() * olap25 * xbkge / xgrid.width(cxmin);
1036  drag_s(cxmax,iy,czmax).xx() += zoverlap[czmax] * double_f.z() * olap25 * xbkge / xgrid.width(cxmax);
1037 
1038  // Porous obstacles do not only have drag at edges
1039  if (ybkge < 1.0)
1040  {
1041  for (label iz = czmin+1; iz < czmax; iz++)
1042  {
1043  for (label ix = cxmin+1; ix < cxmax; ix++)
1044  {
1045  // Internal only
1046  drag_s(ix,iy,iz).yy() = ybkge / ygrid.width(iy);
1047  }
1048  }
1049  }
1050  }
1051 
1052  for (label iz = czmin; iz <= czmax; iz++)
1053  {
1054  scalar olap25 = 0.25 * zoverlap[iz];
1055  const scalar temp =
1056  ((pars.noIntersectN && vbkge < 0.334) ? 0 : (olap25 * vbkge));
1057 
1058  obs_count(cxmin,cymin,iz) += temp;
1059  obs_count(cxmin,cymax,iz) += temp;
1060  obs_count(cxmax,cymin,iz) += temp;
1061  obs_count(cxmax,cymax,iz) += temp;
1062 
1063  sub_count(cxmin,cymin,iz).z() += temp;
1064  sub_count(cxmin,cymax,iz).z() += temp;
1065  sub_count(cxmax,cymin,iz).z() += temp;
1066  sub_count(cxmax,cymax,iz).z() += temp;
1067 
1068  olap25 *= 2.0;
1069 
1070  if (iz > czmin && iz < czmax) // Avoid re-doing corners already done above
1071  {
