110  ) const
111  {
112  List<word> matched(SLList<T>::size());
113 
114  label matchi = 0;
115  forAllConstIters(*this, iter)
116  {
117  const word& name = iter().name();
118 
119  if (white.match(name) && !black.match(name))
120  {
121  matched[matchi++] = name;
122  }
123  }
124  matched.setSize(matchi);
125 
126  return matched;
127  }
