114 template<class SourcePatch, class TargetPatch>
115 bool Foam::faceAreaWeightAMI<SourcePatch, TargetPatch>::processSourceFace
116 (
117  const label srcFacei,
118  const label tgtStartFacei,
119 
120  // list of tgt face neighbour faces
121  DynamicList<label>& nbrFaces,
122  // list of faces currently visited for srcFacei to avoid multiple hits
123  DynamicList<label>& visitedFaces,
124 
125  // temporary storage for addressing and weights
126  List<DynamicList<label>>& srcAddr,
127  List<DynamicList<scalar>>& srcWght,
128  List<DynamicList<label>>& tgtAddr,
129  List<DynamicList<scalar>>& tgtWght
130 )
131 {
132  addProfiling(ami, "faceAreaWeightAMI::processSourceFace");
133 
134  if (tgtStartFacei == -1)
135  {
136  return false;
137  }
138 
139  nbrFaces.clear();
140  visitedFaces.clear();
141 
142  // append initial target face and neighbours
143  nbrFaces.append(tgtStartFacei);
144  this->appendNbrFaces
145  (
146  tgtStartFacei,
147  this->tgtPatch_,
148  visitedFaces,
149  nbrFaces
150  );
151 
152  bool faceProcessed = false;
153 
154  label maxNeighbourFaces = nbrFaces.size();
155 
156  do
157  {
158  // process new target face
159  label tgtFacei = nbrFaces.remove();
160  visitedFaces.append(tgtFacei);
161  scalar area = interArea(srcFacei, tgtFacei);
162 
163  // store when intersection fractional area > tolerance
164  if (area/this->srcMagSf_[srcFacei] > faceAreaIntersect::tolerance())
165  {
166  srcAddr[srcFacei].append(tgtFacei);
167  srcWght[srcFacei].append(area);
168 
169  tgtAddr[tgtFacei].append(srcFacei);
170  tgtWght[tgtFacei].append(area);
171 
172  this->appendNbrFaces
173  (
174  tgtFacei,
175  this->tgtPatch_,
176  visitedFaces,
177  nbrFaces
178  );
179 
180  faceProcessed = true;
181 
182  maxNeighbourFaces = max(maxNeighbourFaces, nbrFaces.size());
183  }
184 
185  } while (nbrFaces.size() > 0);
186 
187  if (debug > 1)
188  {
189  DebugVar(maxNeighbourFaces);
190  }
191 
192  return faceProcessed;
193 }

196 template<class SourcePatch, class TargetPatch>
197 void Foam::faceAreaWeightAMI<SourcePatch, TargetPatch>::setNextFaces
198 (
199  label& startSeedi,
200  label& srcFacei,
201  label& tgtFacei,
202  const boolList& mapFlag,
203  labelList& seedFaces,
204  const DynamicList<label>& visitedFaces,
205  bool errorOnNotFound
206 ) const
207 {
208  addProfiling(ami, "faceAreaWeightAMI::setNextFaces");
209 
210  const labelList& srcNbrFaces = this->srcPatch_.faceFaces()[srcFacei];
211 
212  // initialise tgtFacei
213  tgtFacei = -1;
214 
215  // set possible seeds for later use
216  bool valuesSet = false;
217  for (label faceS: srcNbrFaces)
218  {
219  if (mapFlag[faceS] && seedFaces[faceS] == -1)
220  {
221  for (label faceT : visitedFaces)
222  {
223  const scalar threshold =
224  this->srcMagSf_[faceS]*faceAreaIntersect::tolerance();
225 
226  // store when intersection fractional area > tolerance
227  // Check that faces have enough overlap for robust walking
228  if (overlaps(faceS, faceT, threshold))
229  {
230  seedFaces[faceS] = faceT;
231 
232  if (!valuesSet)
233  {
234  srcFacei = faceS;
235  tgtFacei = faceT;
236  valuesSet = true;
237  }
238  }
239  }
240  }
241  }
242 
243  // set next src and tgt faces if not set above
244  if (valuesSet)
245  {
246  return;
247  }
248  else
249  {
250  // try to use existing seed
251  bool foundNextSeed = false;
252  for (label facei = startSeedi; facei < mapFlag.size(); ++facei)
253  {
254  if (mapFlag[facei])
255  {
256  if (!foundNextSeed)
257  {
258  startSeedi = facei;
259  foundNextSeed = true;
260  }
261 
262  if (seedFaces[facei] != -1)
263  {
264  srcFacei = facei;
265  tgtFacei = seedFaces[facei];
266 
267  return;
268  }
269  }
270  }
271 
272  // perform new search to find match
273  if (debug)
274  {
275  Pout<< "Advancing front stalled: searching for new "
276  << "target face" << endl;
277  }
278 
279  foundNextSeed = false;
280  for (label facei = startSeedi; facei < mapFlag.size(); ++facei)
281  {
282  if (mapFlag[facei])
283  {
284  if (!foundNextSeed)
285  {
286  startSeedi = facei + 1;
287  foundNextSeed = true;
288  }
289 
290  srcFacei = facei;
291  tgtFacei = this->findTargetFace(srcFacei);
292 
293  if (tgtFacei >= 0)
294  {
295  return;
296  }
297  }
298  }
299 
300  if (errorOnNotFound)
301  {
302  FatalErrorInFunction
303  << "Unable to set source and target faces" << abort(FatalError);
304  }
305  }
306 }

309 template<class SourcePatch, class TargetPatch>
310 Foam::scalar Foam::faceAreaWeightAMI<SourcePatch, TargetPatch>::interArea
311 (
312  const label srcFacei,
313  const label tgtFacei
314 ) const
315 {
316  addProfiling(ami, "faceAreaWeightAMI::interArea");
317 
318  scalar area = 0;
319 
320  const pointField& srcPoints = this->srcPatch_.points();
321  const pointField& tgtPoints = this->tgtPatch_.points();
322 
323  // references to candidate faces
324  const face& src = this->srcPatch_[srcFacei];
325  const face& tgt = this->tgtPatch_[tgtFacei];
326 
327  // quick reject if either face has zero area
328  if
329  (
330  (this->srcMagSf_[srcFacei] < ROOTVSMALL)
331  || (this->tgtMagSf_[tgtFacei] < ROOTVSMALL)
332  )
333  {
334  return area;
335  }
336 
337  // create intersection object
338  faceAreaIntersect inter
339  (
340  srcPoints,
341  tgtPoints,
342  this->srcTris_[srcFacei],
343  this->tgtTris_[tgtFacei],
344  this->reverseTarget_,
345  AMIInterpolation<SourcePatch, TargetPatch>::cacheIntersections_
346  );
347 
348  // crude resultant norm
349  vector n(-this->srcPatch_.faceNormals()[srcFacei]);
350  if (this->reverseTarget_)
351  {
352  n -= this->tgtPatch_.faceNormals()[tgtFacei];
353  }
354  else
355  {
356  n += this->tgtPatch_.faceNormals()[tgtFacei];
357  }
358  scalar magN = mag(n);
359 
360  if (magN > ROOTVSMALL)
361  {
362  area = inter.calc(src, tgt, n/magN);
363  }
364  else
365  {
366  WarningInFunction
367  << "Invalid normal for source face " << srcFacei
368  << " points " << UIndirectList<point>(srcPoints, src)
369  << " target face " << tgtFacei
370  << " points " << UIndirectList<point>(tgtPoints, tgt)
371  << endl;
372  }
373 
374 
375  if ((debug > 1) && (area > 0))
376  {
377  this->writeIntersectionOBJ(area, src, tgt, srcPoints, tgtPoints);
378  }
379 
380  return area;
381 }

449 template<class SourcePatch, class TargetPatch>
450 void Foam::faceAreaWeightAMI<SourcePatch, TargetPatch>::
451 restartUncoveredSourceFace
452 (
453  List<DynamicList<label>>& srcAddr,
454  List<DynamicList<scalar>>& srcWght,
455  List<DynamicList<label>>& tgtAddr,
456  List<DynamicList<scalar>>& tgtWght
457 )
458 {
459  addProfiling(ami, "faceAreaWeightAMI::restartUncoveredSourceFace");
460 
461  // Collect all src faces with a low weight
462  // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
463 
464  labelHashSet lowWeightFaces(100);
465  forAll(srcWght, srcFacei)
466  {
467  scalar s = sum(srcWght[srcFacei]);
468  scalar t = s/this->srcMagSf_[srcFacei];
469 
470  if (t < 0.5)
471  {
472  lowWeightFaces.insert(srcFacei);
473  }
474  }
475 
476  if (debug)
477  {
478  Pout<< "faceAreaWeightAMI: restarting search on "
479  << lowWeightFaces.size() << " faces since sum of weights < 0.5"
480  << endl;
481  }
482 
483  if (lowWeightFaces.size() > 0)
484  {
485  // Erase all the lowWeight source faces from the target
486  // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
487 
488  DynamicList<label> okSrcFaces(10);
489  DynamicList<scalar> okSrcWeights(10);
490  forAll(tgtAddr, tgtFacei)
491  {
492  okSrcFaces.clear();
493  okSrcWeights.clear();
494  DynamicList<label>& srcFaces = tgtAddr[tgtFacei];
495  DynamicList<scalar>& srcWeights = tgtWght[tgtFacei];
496  forAll(srcFaces, i)
497  {
498  if (!lowWeightFaces.found(srcFaces[i]))
499  {
500  okSrcFaces.append(srcFaces[i]);
501  okSrcWeights.append(srcWeights[i]);
502  }
503  }
504  if (okSrcFaces.size() < srcFaces.size())
505  {
506  srcFaces.transfer(okSrcFaces);
507  srcWeights.transfer(okSrcWeights);
508  }
509  }
510 
511 
512 
513  // Restart search from best hit
514  // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~
515 
516  // list of tgt face neighbour faces
517  DynamicList<label> nbrFaces(10);
518 
519  // list of faces currently visited for srcFacei to avoid multiple hits
520  DynamicList<label> visitedFaces(10);
521 
522  for (const label srcFacei : lowWeightFaces)
523  {
524  const label tgtFacei = this->findTargetFace(srcFacei);
525  if (tgtFacei != -1)
526  {
527  //bool faceProcessed =
528  processSourceFace
529  (
530  srcFacei,
531  tgtFacei,
532 
533  nbrFaces,
534  visitedFaces,
535 
536  srcAddr,
537  srcWght,
538  tgtAddr,
539  tgtWght
540  );
541  // ? Check faceProcessed to see if restarting has worked.
542  }
543  }
544  }
545 }
