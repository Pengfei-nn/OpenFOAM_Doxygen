63  // No profiling since there is no time loop

66  argList::noMandatoryArgs();
67  argList::addArgument("regionType ... regionType");
68 
69  #include "setRootCase.H"
70 
71  // As per "createTime.H", but quieter.
72  Time runTime(Time::controlDictName, args);
73 
74  regionProperties rp(runTime);
75 
76  // We now handle checking args and general sanity etc.
77  wordList regionTypes;
78 
79  if (args.size() > 1)
80  {
81  regionTypes.setSize(args.size()-1);
82 
83  label nTypes = 0;
84  for (label argi = 1; argi < args.size(); ++argi)
85  {
86  regionTypes[nTypes] = args[argi];
87 
88  if (rp.found(regionTypes[nTypes]))
89  {
90  ++nTypes;
91  }
92  else
93  {
94  std::cerr<< "No region-type: " << regionTypes[nTypes] << nl;
95  }
96  }
97 
98  regionTypes.setSize(nTypes);
99  }
100  else
101  {
102  regionTypes = rp.sortedToc();
103  }
104 
105 
106  for (const word& regionType : regionTypes)
107  {
108  const wordList& regionNames = rp[regionType];
109 
110  for (const word& regionName : regionNames)
111  {
112  Info<< regionName << nl;
113  }
114  }
115 
116  return 0;
117 }
