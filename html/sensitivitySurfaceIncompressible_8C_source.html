55 void sensitivitySurface::addGeometricSens()
56 {
57  if (includeObjective_)
58  {
59  // Grab objective refs
60  PtrList<objective>& functions
61  (objectiveManager_.getObjectiveFunctions());
62  // Compute sens for all points in parameterized patches.
63  // Interfacing points will be accumulated later
64  autoPtr<pointBoundaryVectorField> pointSensdSdb
65  (
66  createZeroBoundaryPointFieldPtr<vector>(mesh_)
67  );
68  autoPtr<pointBoundaryVectorField> pointSensdndb
69  (
70  createZeroBoundaryPointFieldPtr<vector>(mesh_)
71  );
72  // Geometric (or "direct") sensitivities are better computed directly
73  // on the points
74  for (const label patchI : sensitivityPatchIDs_)
75  {
76  const fvPatch& patch = mesh_.boundary()[patchI];
77  vectorField nf(patch.nf());
78 
79  // point sens result for patch
80  vectorField& patchdSdb = pointSensdSdb()[patchI];
81  vectorField& patchdndb = pointSensdndb()[patchI];
82 
83  vectorField dSdbMultiplierTot(patch.size(), Zero);
84  vectorField dndbMultiplierTot(patch.size(), Zero);
85  forAll(functions, funcI)
86  {
87  dSdbMultiplierTot +=
88  functions[funcI].weight()
89  *functions[funcI].dSdbMultiplier(patchI);
90  dndbMultiplierTot +=
91  functions[funcI].weight()
92  *functions[funcI].dndbMultiplier(patchI);
93  }
94  // Correspondance of local point addressing to global point
95  // addressing
96  const labelList& meshPoints = patch.patch().meshPoints();
97  // List with mesh faces. Global addressing
98  const faceList& faces = mesh_.faces();
99  // Each local patch point belongs to these local patch faces
100  // (local numbering)
101  const labelListList& patchPointFaces = patch.patch().pointFaces();
102  // index of first face in patch
103  const label patchStartIndex = patch.start();
104  // geometry differentiation engine
105  deltaBoundary dBoundary(mesh_);
106  // Loop over patch points.
107  // Collect contributions from each boundary face this point
108  // belongs to
109  forAll(meshPoints, ppI)
110  {
111  const labelList& pointFaces = patchPointFaces[ppI];
112  forAll(pointFaces, pfI)
113  {
114  label localFaceIndex = pointFaces[pfI];
115  label globalFaceIndex = patchStartIndex + localFaceIndex;
116  const face& faceI = faces[globalFaceIndex];
117  // Point coordinates. All indices in global numbering
118  pointField p(faceI.points(mesh_.points()));
119  tensorField p_d(faceI.size(), Zero);
120  forAll(faceI, facePointI)
121  {
122  if (faceI[facePointI] == meshPoints[ppI])
123  {
124  p_d[facePointI] = tensor::I;
125  }
126  }
127  tensorField deltaNormals =
128  dBoundary.makeFaceCentresAndAreas_d(p, p_d);
129 
130  // Element [1] is the variation in the (dimensional) normal
131  const tensor& deltaSf = deltaNormals[1];
132  patchdSdb[ppI] +=
133  dSdbMultiplierTot[localFaceIndex] & deltaSf;
134 
135  // Element [2] is the variation in the unit normal
136  const tensor& deltaNf = deltaNormals[2];
137  patchdndb[ppI] +=
138  dndbMultiplierTot[localFaceIndex] & deltaNf;
139  }
140  }
141  }
142  // Do parallel communications to avoid wrong values at processor
143  // boundaries
144  vectorField dSdbGlobal(mesh_.nPoints(), Zero);
145  vectorField dndbGlobal(mesh_.nPoints(), Zero);
146  for (const label patchI : sensitivityPatchIDs_)
147  {
148  const labelList& meshPoints =
149  mesh_.boundaryMesh()[patchI].meshPoints();
150  forAll(meshPoints, ppI)
151  {
152  const label globaPointI = meshPoints[ppI];
153  dSdbGlobal[globaPointI] += pointSensdSdb()[patchI][ppI];
154  dndbGlobal[globaPointI] += pointSensdndb()[patchI][ppI];
155  }
156  }
157  // Accumulate over processors
158  syncTools::syncPointList
159  (
160  mesh_, dSdbGlobal, plusEqOp<vector>(), vector::zero
161  );
162  syncTools::syncPointList
163  (
164  mesh_, dndbGlobal, plusEqOp<vector>(), vector::zero
165  );
166  // Transfer back to local fields and map to faces
167  for (const label patchI : sensitivityPatchIDs_)
168  {
169  const fvPatch& patch = mesh_.boundary()[patchI];
170  const labelList& meshPoints = patch.patch().meshPoints();
171  const scalarField& magSf = patch.magSf();
172  pointSensdSdb()[patchI].map(dSdbGlobal, meshPoints);
173  pointSensdndb()[patchI].map(dndbGlobal, meshPoints);
174  // Map dSf/dx and dnf/dx term from points to faces. Ideally, all
175  // sensitivities should be computed at points rather than faces.
176  PrimitivePatchInterpolation<polyPatch> patchInter(patch.patch());
177  vectorField dSdbFace
178  (
179  patchInter.pointToFaceInterpolate(pointSensdSdb()[patchI])
180  );
181  // dSdb already contains the face area. Divide with it to make it
182  // compatible with the rest of the terms
183  dSdbFace /= magSf;
184 
185  tmp<vectorField> tdndbFace =
186  patchInter.pointToFaceInterpolate(pointSensdndb()[patchI]);
187  const vectorField& dndbFace = tdndbFace();
188 
189  // Add to sensitivity fields
190  wallFaceSensVecPtr_()[patchI] += dSdbFace + dndbFace;
191  }
192  }
193 }

406 void sensitivitySurface::accumulateIntegrand(const scalar dt)
407 {
408  // Grab references
409  const volScalarField& p = primalVars_.p();
410  const volVectorField& U = primalVars_.U();
411 
412  const volScalarField& pa = adjointVars_.pa();
413  const volVectorField& Ua = adjointVars_.Ua();
414  autoPtr<incompressibleAdjoint::adjointRASModel>& adjointTurbulence =
415  adjointVars_.adjointTurbulence();
416 
417  Info<< " Calculating auxilary quantities " << endl;
418  // Fields needed to calculate adjoint sensitivities
419  const autoPtr<incompressible::RASModelVariables>&
420  turbVars = primalVars_.RASModelVariables();
421  const singlePhaseTransportModel& lamTransp = primalVars_.laminarTransport();
422  volScalarField nuEff(lamTransp.nu() + turbVars->nutRef());
423  volTensorField gradUa(fvc::grad(Ua));
424  volTensorField gradU(fvc::grad(U));
425 
426  // Explicitly correct the boundary gradient to get rid of the tangential
427  // component
428  forAll(mesh_.boundary(), patchI)
429  {
430  const fvPatch& patch = mesh_.boundary()[patchI];
431  if (isA<wallFvPatch>(patch))
432  {
433  tmp<vectorField> tnf = mesh_.boundary()[patchI].nf();
434  const vectorField& nf = tnf();
435  gradU.boundaryFieldRef()[patchI] =
436  nf*U.boundaryField()[patchI].snGrad();
437  }
438  }
439 
440  // Auxiliary terms
441  volVectorField gradp(fvc::grad(p));
442  volTensorField stress(nuEff*(gradU + T(gradU)));
443  autoPtr<volVectorField> stressXPtr
444  (
445  createZeroFieldPtr<vector>(mesh_, "stressX", stress.dimensions())
446  );
447  autoPtr<volVectorField> stressYPtr
448  (
449  createZeroFieldPtr<vector>(mesh_, "stressY", stress.dimensions())
450  );
451  autoPtr<volVectorField> stressZPtr
452  (
453  createZeroFieldPtr<vector>(mesh_, "stressZ", stress.dimensions())
454  );
455 
456  stressXPtr().replace(0, stress.component(0));
457  stressXPtr().replace(1, stress.component(1));
458  stressXPtr().replace(2, stress.component(2));
459 
460  stressYPtr().replace(0, stress.component(3));
461  stressYPtr().replace(1, stress.component(4));
462  stressYPtr().replace(2, stress.component(5));
463 
464  stressZPtr().replace(0, stress.component(6));
465  stressZPtr().replace(1, stress.component(7));
466  stressZPtr().replace(2, stress.component(8));
467 
468  volTensorField gradStressX(fvc::grad(stressXPtr()));
469  volTensorField gradStressY(fvc::grad(stressYPtr()));
470  volTensorField gradStressZ(fvc::grad(stressZPtr()));
471 
472  // Accumulate source for additional post-processing PDEs, if necessary
473  if (includeDistance_)
474  {
475  eikonalSolver_->accumulateIntegrand(dt);
476  }
477 
478  if (includeMeshMovement_)
479  {
480  meshMovementSolver_->accumulateIntegrand(dt);
481  }
482 
483  // Terms from the adjoint turbulence model
484  const boundaryVectorField& adjointTMsensitivities =
485  adjointTurbulence->wallShapeSensitivities();
486 
487  DebugInfo
488  << " Calculating adjoint sensitivity. " << endl;
489 
490  // Sensitivities do not include locale surface area by default.
491  // Part of the sensitivities that multiplies dxFace/db
492  for (const label patchI : sensitivityPatchIDs_)
493  {
494  const fvPatch& patch = mesh_.boundary()[patchI];
495  tmp<vectorField> tnf = patch.nf();
496  const vectorField& nf = tnf();
497 
498  // Includes spurious tangential gradU part. Deprecated
499  /*
500  vectorField stressAndPressureTerm =
501  (
502  - (
503  Ua.boundaryField()[patchI].snGrad()
504  + (gradUa.boundaryField()[patchI] & nf)
505  ) * nuEff.boundaryField()[patchI]
506  + pa.boundaryField()[patchI] *nf
507  ) & gradU.boundaryField()[patchI].T();
508  */
509 
510  // Adjoint stress term
511  vectorField stressTerm
512  (
513  - (
514  Ua.boundaryField()[patchI].snGrad()
515  & U.boundaryField()[patchI].snGrad()
516  )
517  * nuEff.boundaryField()[patchI]
518  * nf
519  );
520 
521 
522  if (includeTransposeStresses_)
523  {
524  stressTerm -=
525  nuEff.boundaryField()[patchI]
526  * (
527  // Note: in case of laminar or low-Re flows,
528  // includes a spurious tangential gradUa component
529  // (gradUa.boundaryField()[patchI] & nf)
530  ((Ua.boundaryField()[patchI].snGrad() &nf)*nf)
531  & U.boundaryField()[patchI].snGrad()
532  )
533  * nf;
534  }
535 
536  if (includeDivTerm_)
537  {
538  stressTerm +=
539  scalar(1./3.)*nuEff.boundaryField()[patchI]
540  * (
541  ((Ua.boundaryField()[patchI].snGrad() &nf)*nf)
542  & U.boundaryField()[patchI].snGrad()
543  )
544  * nf;
545  }
546 
547  vectorField gradStressTerm(patch.size(), Zero);
548  if (includeGradStressTerm_)
549  {
550  // Terms corresponding to contributions from converting delta to
551  // thetas are added through the corresponding adjoint boundary
552  // conditions instead of grabing contributions from the objective
553  // function. Useful to have a unified formulation for low- and
554  // high-re meshes
555  const fvPatchVectorField& Uab = Ua.boundaryField()[patchI];
556  gradStressTerm = - ((Uab & nf)*gradp.boundaryField()[patchI]);
557  gradStressTerm +=
558  (
559  Uab.component(0) * gradStressX.boundaryField()[patchI]
560  + Uab.component(1) * gradStressY.boundaryField()[patchI]
561  + Uab.component(2) * gradStressZ.boundaryField()[patchI]
562  ) & nf;
563  }
564 
565  // Adjoint pressure terms
566  vectorField pressureTerm(patch.size(), Zero);
567  if (includePressureTerm_)
568  {
569  pressureTerm =
570  (
571  (nf*pa.boundaryField()[patchI])
572  & U.boundaryField()[patchI].snGrad()
573  )* nf;
574  }
575 
576  PtrList<objective>& functions
577  (objectiveManager_.getObjectiveFunctions());
578 
579  // Term from objectives including x directly (e.g. moments)
580  vectorField dxdbMultiplierTot(pressureTerm.size(), Zero);
581  if (includeObjective_)
582  {
583  forAll(functions, funcI)
584  {
585  dxdbMultiplierTot +=
586  functions[funcI].weight()
587  * (
588  functions[funcI].dxdbDirectMultiplier(patchI)
589  );
590  }
591  }
592 
593  // Fill in sensitivity fields
594  wallFaceSensVecPtr_()[patchI] +=
595  (
596  stressTerm
597  + gradStressTerm
598  + pressureTerm
599  + adjointTMsensitivities[patchI]
600  + dxdbMultiplierTot
601  )*dt;
602  }
603 
604  // Add the sensitivity part corresponding to changes of the normal vector
605  // Computed at points and mapped to faces
606  addGeometricSens();
607 }

610 void sensitivitySurface::assembleSensitivities()
611 {
612  // Update geometric fields for use by external users
613  if (writeGeometricInfo_)
614  {
615  for (const label patchI : sensitivityPatchIDs_)
616  {
617  const fvPatch& patch = mesh_.boundary()[patchI];
618  tmp<vectorField> tnf = patch.nf();
619  const vectorField& nf = tnf();
620  const vectorField& Sf = patch.Sf();
621  const vectorField& Cf = patch.Cf();
622 
623  nfOnPatchPtr_().boundaryFieldRef()[patchI] = nf;
624  SfOnPatchPtr_().boundaryFieldRef()[patchI] = Sf;
625  CfOnPatchPtr_().boundaryFieldRef()[patchI] = Cf;
626  }
627  }
628 
629  // Solve extra equations if necessary
630  // Solved using accumulated sources over time
631  autoPtr<boundaryVectorField> distanceSensPtr(nullptr);
632  if (includeDistance_)
633  {
634  eikonalSolver_->solve();
635  distanceSensPtr.reset(createZeroBoundaryPtr<vector>(mesh_));
636  const boundaryVectorField& sens =
637  eikonalSolver_->distanceSensitivities();
638  for (const label patchI : sensitivityPatchIDs_)
639  {
640  distanceSensPtr()[patchI] = sens[patchI];
641  }
642  }
643 
644  autoPtr<boundaryVectorField> meshMovementSensPtr(nullptr);
645  if (includeMeshMovement_)
646  {
647  meshMovementSolver_->solve();
648  meshMovementSensPtr.reset(createZeroBoundaryPtr<vector>(mesh_));
649  const boundaryVectorField& sens =
650  meshMovementSolver_->meshMovementSensitivities();
651  for (const label patchI : sensitivityPatchIDs_)
652  {
653  meshMovementSensPtr()[patchI] = sens[patchI];
654  }
655  }
656 
657 
658  // Project to normal face vector
659  label nPassedFaces(0);
660  for (const label patchI : sensitivityPatchIDs_)
661  {
662  const fvPatch& patch = mesh_.boundary()[patchI];
663  tmp<vectorField> tnf(patch.nf());
664  const vectorField& nf = tnf();
665 
666  // Distance related terms
667  if (includeDistance_)
668  {
669  wallFaceSensVecPtr_()[patchI] += distanceSensPtr()[patchI];
670  }
671 
672  // Mesh movement related terms
673  if (includeMeshMovement_)
674  {
675  wallFaceSensVecPtr_()[patchI] += meshMovementSensPtr()[patchI];
676  }
677 
678  if (includeSurfaceArea_)
679  {
680  wallFaceSensVecPtr_()[patchI] *= patch.magSf();
681  }
682 
683  wallFaceSensNormalPtr_()[patchI] = wallFaceSensVecPtr_()[patchI] & nf;
684  wallFaceSensNormalVecPtr_()[patchI] =
685  wallFaceSensNormalPtr_()[patchI] * nf;
686 
687  forAll(patch, fI)
688  {
689  derivatives_[nPassedFaces + fI]
690  = wallFaceSensNormalPtr_()[patchI][fI];
691  }
692  nPassedFaces += patch.size();
693  }
694 }
