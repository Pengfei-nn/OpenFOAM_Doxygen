60 void Foam::tabulatedWallFunctions::general::invertTable()
61 {
62  scalarList Rey(uPlus_.size(), Zero);
63 
64  // Calculate Reynolds number
65  forAll(uPlus_, i)
66  {
67  Rey[i] = yPlus_[i]*uPlus_[i];
68  if (invertedTable_.log10())
69  {
70  Rey[i] = ::log10(max(ROOTVSMALL, Rey[i]));
71  }
72  }
73 
74  // Populate the U+ vs Re table
75  forAll(invertedTable_, i)
76  {
77  scalar Re = i*invertedTable_.dx() + invertedTable_.x0();
78  invertedTable_[i] = max(0, interpolate(Re, Rey, uPlus_));
79  }
80 }

83 Foam::scalar Foam::tabulatedWallFunctions::general::interpolate
84 (
85  const scalar xi,
86  const scalarList& x,
87  const scalarList& fx
88 ) const
89 {
90  switch (interpType_)
91  {
92  case itLinear:
93  {
94  if (xi <= x[0])
95  {
96  return fx[0];
97  }
98  else if (xi >= x.last())
99  {
100  return fx.last();
101  }
102  else
103  {
104  label i2 = 0;
105  while (x[i2] < xi)
106  {
107  i2++;
108  }
109  label i1 = i2 - 1;
110 
111  return (xi - x[i1])/(x[i2] - x[i1])*(fx[i2] - fx[i1]) + fx[i1];
112  }
113 
114  break;
115  }
116  default:
117  {
118  FatalErrorInFunction
119  << "Unknown interpolation method" << nl
120  << abort(FatalError);
121  }
122  }
123 
124  return 0.0;
125 }
