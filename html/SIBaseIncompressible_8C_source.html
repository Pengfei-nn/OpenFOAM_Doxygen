48 void SIBase::read()
49 {
50  surfaceSensitivity_.read();
51  includeObjective_ =
52  dict().lookupOrDefault<bool>("includeObjectiveContribution", true);
53 
54  // If includeObjective is set to true both here and in the surface
55  // sensitivities, set the one in the latter to false to avoid double
56  // contributions
57  bool surfSensIncludeObjective(surfaceSensitivity_.getIncludeObjective());
58  if (includeObjective_ && surfSensIncludeObjective)
59  {
60  WarningInFunction
61  << "includeObjectiveContribution set to true in both "
62  << "surfaceSensitivities and the parameterization options" << nl
63  << "This will lead to double contributions " << nl
64  << "Disabling the former"
65  << endl;
66  surfaceSensitivity_.setIncludeObjective(false);
67  }
68 
69  // Make sure surface area is included in the sensitivity map
70  surfaceSensitivity_.setIncludeSurfaceArea(true);
71 }

136 void SIBase::accumulateIntegrand(const scalar dt)
137 {
138  // Accumulate multiplier of dxFace/db
139  surfaceSensitivity_.accumulateIntegrand(dt);
140 
141  // Accumulate direct sensitivities
142  if (includeObjective_)
143  {
144  PtrList<objective>& functions
145  (
146  objectiveManager_.getObjectiveFunctions()
147  );
148  for (const label patchI : sensitivityPatchIDs_)
149  {
150  const scalarField magSfDt(mesh_.boundary()[patchI].magSf()*dt);
151  for (objective& func : functions)
152  {
153  const scalar wei = func.weight();
154  dSfdbMult_()[patchI] += wei*func.dSdbMultiplier(patchI)*dt;
155  dnfdbMult_()[patchI] += wei*func.dndbMultiplier(patchI)*magSfDt;
156  dxdbDirectMult_()[patchI] +=
157  wei*func.dxdbDirectMultiplier(patchI)*magSfDt;
158  }
159  }
160  }
161 }
