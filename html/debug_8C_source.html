97 // Like dictionary getOrAdd with LITERAL, but circumventing
98 // writeOptionalEntries to avoid extremely noisy output
99 template<class T>
100 static inline T getOrAdd
101 (
102  dictionary& dict,
103  const char* name,
104  const T deflt
105 )
106 {
107  const entry* eptr = dict.findEntry(name, keyType::LITERAL);
108 
109  if (eptr)
110  {
111  return eptr->get<T>();
112  }
113 
114  dict.add(new primitiveEntry(name, deflt));
115  return deflt;
116 }

143 Foam::dictionary& Foam::debug::controlDict()
144 {
145  if (!controlDictPtr_)
146  {
147  string controlDictString(Foam::getEnv("FOAM_CONTROLDICT"));
148  if (!controlDictString.empty())
149  {
150  // Read from environment
151  IStringStream is(controlDictString);
152  controlDictPtr_ = new dictionary(is);
153  }
154  else
155  {
156  fileNameList controlDictFiles = findEtcFiles("controlDict", true);
157  controlDictPtr_ = new dictionary();
158  forAllReverse(controlDictFiles, i)
159  {
160  IFstream is(controlDictFiles[i]);
161 
162  if (!is.good())
163  {
164  SafeFatalIOErrorInFunction
165  (
166  is,
167  "Cannot open controlDict"
168  );
169  }
170  controlDictPtr_->merge(dictionary(is));
171  }
172  }
173  }
174 
175  return *controlDictPtr_;
176 }

197  std::exit(1);

199 
200  subDictPtr = &(eptr->dict());
201  }
202

351 // Write the switch names.
352 //
353 // Use flatOutput with -1 for the length to ensure we always have newlines,
354 // even if the lists are short
355 static void listSwitches
356 (
357  const wordList& debugSwitches,
358  const wordList& infoSwitches,
359  const wordList& optSwitches,
360  const bool unset
361 )
362 {
363  IOobject::writeDivider(Info);
364 
365  if (unset)
366  {
367  fileNameList controlDictFiles = findEtcFiles("controlDict", true);
368  dictionary controlDict;
369  forAllReverse(controlDictFiles, i)
370  {
371  IFstream is(controlDictFiles[i]);
372 
373  controlDict.merge(dictionary(is));
374  }
375 
376  // Use a HashSet to track switches that have not been set
377  wordHashSet hashed;
378 
379  // DebugSwitches
380  if (notNull(debugSwitches))
381  {
382  hashed = debugSwitches;
383  hashed.unset(controlDict.subDict("DebugSwitches").toc());
384 
385  Info<< "Unset DebugSwitches"
386  << flatOutput(hashed.sortedToc(), -1) << nl;
387  }
388 
389  // InfoSwitches
390  if (notNull(infoSwitches))
391  {
392  hashed = infoSwitches;
393  hashed.unset(controlDict.subDict("InfoSwitches").toc());
394 
395  Info<< "Unset InfoSwitches"
396  << flatOutput(hashed.sortedToc(), -1) << nl;
397  }
398 
399  // OptimisationSwitches
400  if (notNull(optSwitches))
401  {
402  hashed = optSwitches;
403  hashed.unset(controlDict.subDict("OptimisationSwitches").toc());
404 
405  Info<< "Unset OptimisationSwitches"
406  << flatOutput(hashed.sortedToc(), -1) << nl;
407  }
408  }
409  else
410  {
411  // DebugSwitches
412  if (notNull(debugSwitches))
413  {
414  Info<< "DebugSwitches"
415  << flatOutput(debugSwitches, -1) << nl;
416  }
417 
418  // InfoSwitches
419  if (notNull(infoSwitches))
420  {
421  Info<< "InfoSwitches"
422  << flatOutput(infoSwitches, -1) << nl;
423  }
424 
425  // OptimisationSwitches
426  if (notNull(optSwitches))
427  {
428  Info<< "OptimisationSwitches"
429  << flatOutput(optSwitches, -1) << nl;
430  }
431  }
432 }
