36 template<class SourcePatch, class TargetPatch>
37 void Foam::AMIMethod<SourcePatch, TargetPatch>::checkPatches() const
38 {
39  if (debug && (!srcPatch_.size() || !tgtPatch_.size()))
40  {
41  Pout<< "AMI: Patches not on processor: Source faces = "
42  << srcPatch_.size() << ", target faces = " << tgtPatch_.size()
43  << endl;
44  }
45 
46 
47  if (conformal())
48  {
49  const scalar maxBoundsError = 0.05;
50 
51  // check bounds of source and target
52  boundBox bbSrc(srcPatch_.points(), srcPatch_.meshPoints(), true);
53  boundBox bbTgt(tgtPatch_.points(), tgtPatch_.meshPoints(), true);
54 
55  boundBox bbTgtInf(bbTgt);
56  bbTgtInf.inflate(maxBoundsError);
57 
58  if (!bbTgtInf.contains(bbSrc))
59  {
60  WarningInFunction
61  << "Source and target patch bounding boxes are not similar"
62  << nl
63  << " source box span : " << bbSrc.span() << nl
64  << " target box span : " << bbTgt.span() << nl
65  << " source box : " << bbSrc << nl
66  << " target box : " << bbTgt << nl
67  << " inflated target box : " << bbTgtInf << endl;
68  }
69  }
70 }

73 template<class SourcePatch, class TargetPatch>
74 bool Foam::AMIMethod<SourcePatch, TargetPatch>::initialise
75 (
76  labelListList& srcAddress,
77  scalarListList& srcWeights,
78  labelListList& tgtAddress,
79  scalarListList& tgtWeights,
80  label& srcFacei,
81  label& tgtFacei
82 )
83 {
84  checkPatches();
85 
86  // set initial sizes for weights and addressing - must be done even if
87  // returns false below
88  srcAddress.setSize(srcPatch_.size());
89  srcWeights.setSize(srcPatch_.size());
90  tgtAddress.setSize(tgtPatch_.size());
91  tgtWeights.setSize(tgtPatch_.size());
92 
93  // check that patch sizes are valid
94  if (!srcPatch_.size())
95  {
96  return false;
97  }
98  else if (!tgtPatch_.size())
99  {
100  WarningInFunction
101  << srcPatch_.size() << " source faces but no target faces" << endl;
102 
103  return false;
104  }
105 
106  // reset the octree
107  resetTree();
108 
109  // find initial face match using brute force/octree search
110  if ((srcFacei == -1) || (tgtFacei == -1))
111  {
112  srcFacei = 0;
113  tgtFacei = 0;
114  bool foundFace = false;
115  forAll(srcPatch_, facei)
116  {
117  tgtFacei = findTargetFace(facei);
118  if (tgtFacei >= 0)
119  {
120  srcFacei = facei;
121  foundFace = true;
122  break;
123  }
124  }
125 
126  if (!foundFace)
127  {
128  if (requireMatch_)
129  {
130  FatalErrorInFunction
131  << "Unable to find initial target face"
132  << abort(FatalError);
133  }
134 
135  return false;
136  }
137  }
138 
139  if (debug)
140  {
141  Pout<< "AMI: initial target face = " << tgtFacei << endl;
142  }
143 
144  return true;
145 }

200 template<class SourcePatch, class TargetPatch>
201 void Foam::AMIMethod<SourcePatch, TargetPatch>::resetTree()
202 {
203  // Clear the old octree
204  treePtr_.clear();
205 
206  treeBoundBox bb(tgtPatch_.points(), tgtPatch_.meshPoints());
207  bb.inflate(0.01);
208 
209  if (!treePtr_.valid())
210  {
211  treePtr_.reset
212  (
213  new indexedOctree<treeType>
214  (
215  treeType
216  (
217  false,
218  tgtPatch_,
219  indexedOctree<treeType>::perturbTol()
220  ),
221  bb, // overall search domain
222  8, // maxLevel
223  10, // leaf size
224  3.0 // duplicity
225  )
226  );
227  }

231 template<class SourcePatch, class TargetPatch>
232 Foam::label Foam::AMIMethod<SourcePatch, TargetPatch>::findTargetFace
233 (
234  const label srcFacei
235 ) const
236 {
237  label targetFacei = -1;
238 
239  const pointField& srcPts = srcPatch_.points();
240  const face& srcFace = srcPatch_[srcFacei];
241  const point srcPt = srcFace.centre(srcPts);
242  const scalar srcFaceArea = srcMagSf_[srcFacei];
243 
244  pointIndexHit sample = treePtr_->findNearest(srcPt, 10.0*srcFaceArea);
245 
246  if (sample.hit())
247  {
248  targetFacei = sample.index();
249 
250  if (debug)
251  {
252  Pout<< "Source point = " << srcPt << ", Sample point = "
253  << sample.hitPoint() << ", Sample index = " << sample.index()
254  << endl;
255  }
256  }
257 
258  return targetFacei;
259 }

262 template<class SourcePatch, class TargetPatch>
263 void Foam::AMIMethod<SourcePatch, TargetPatch>::appendNbrFaces
264 (
265  const label facei,
266  const TargetPatch& patch,
267  const DynamicList<label>& visitedFaces,
268  DynamicList<label>& faceIDs
269 ) const
270 {
271  static const scalar thetaCos = Foam::cos(degToRad(89.0));
272 
273  const labelList& nbrFaces = patch.faceFaces()[facei];
274 
275  // filter out faces already visited from face neighbours
276  for (const label nbrFacei : nbrFaces)
277  {
278  bool valid = true;
279  for (const label visitedFacei : visitedFaces)
280  {
281  if (nbrFacei == visitedFacei)
282  {
283  valid = false;
284  break;
285  }
286  }
287 
288  if (valid)
289  {
290  for (const label testFacei : faceIDs)
291  {
292  if (nbrFacei == testFacei)
293  {
294  valid = false;
295  break;
296  }
297  }
298  }
299 
300  // prevent addition of face if it is not on the same plane-ish
301  if (valid)
302  {
303  const vector& n1 = patch.faceNormals()[facei];
304  const vector& n2 = patch.faceNormals()[nbrFacei];
305 
306  const scalar cosI = n1 & n2;
307 
308  if (cosI > thetaCos)
309  {
310  faceIDs.append(nbrFacei);
311  }
312  }
313  }
314 }

317 template<class SourcePatch, class TargetPatch>
318 template<class PatchType>
319 void Foam::AMIMethod<SourcePatch, TargetPatch>::triangulatePatch
320 (
321  const PatchType& patch,
322  List<DynamicList<face>>& tris,
323  List<scalar>& magSf
324 ) const
325 {
326  const pointField& points = patch.points();
327  tris.setSize(patch.size());
328  magSf.setSize(patch.size());
329 
330  // Using methods that index into existing points
331  forAll(patch, facei)
332  {
333  switch (triMode_)
334  {
335  case faceAreaIntersect::tmFan:
336  {
337  faceAreaIntersect::triangleFan(patch[facei], tris[facei]);
338  break;
339  }
340  case faceAreaIntersect::tmMesh:
341  {
342  patch[facei].triangles(points, tris[facei]);
343  break;
344  }
345  }
346 
347  const DynamicList<face>& triFaces = tris[facei];
348  magSf[facei] = 0;
349  for (const face& f : triFaces)
350  {
351  magSf[facei] +=
352  triPointRef
353  (
354  points[f[0]],
355  points[f[1]],
356  points[f[2]]
357  ).mag();
358  }
359  }
360 }
