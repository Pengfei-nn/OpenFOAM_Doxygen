76 Foam::vector Foam::eigenValues(const tensor& t)
77 {
78  // Coefficients of the characteristic cubic polynomial (a = 1)
79  const scalar b =
80  - t.xx() - t.yy() - t.zz();
81  const scalar c =
82  t.xx()*t.yy() + t.xx()*t.zz() + t.yy()*t.zz()
83  - t.xy()*t.yx() - t.yz()*t.zy() - t.zx()*t.xz();
84  const scalar d =
85  - t.xx()*t.yy()*t.zz()
86  - t.xy()*t.yz()*t.zx() - t.xz()*t.zy()*t.yx()
87  + t.xx()*t.yz()*t.zy() + t.yy()*t.zx()*t.xz() + t.zz()*t.xy()*t.yx();
88 
89  // Solve
90  Roots<3> roots = cubicEqn(1, b, c, d).roots();
91 
92  // Check the root types
93  vector lambda = vector::zero;
94  forAll(roots, i)
95  {
96  switch (roots.type(i))
97  {
98  case roots::real:
99  lambda[i] = roots[i];
100  break;
101  case roots::complex:
102  WarningInFunction
103  << "Complex eigenvalues detected for tensor: " << t
104  << endl;
105  lambda[i] = 0;
106  break;
107  case roots::posInf:
108  lambda[i] = VGREAT;
109  break;
110  case roots::negInf:
111  lambda[i] = - VGREAT;
112  break;
113  case roots::nan:
114  FatalErrorInFunction
115  << "Eigenvalue calculation failed for tensor: " << t
116  << exit(FatalError);
117  }
118  }
119 
120  // Sort the eigenvalues into ascending order
121  if (lambda.x() > lambda.y())
122  {
123  Swap(lambda.x(), lambda.y());
124  }
125  if (lambda.y() > lambda.z())
126  {
127  Swap(lambda.y(), lambda.z());
128  }
129  if (lambda.x() > lambda.y())
130  {
131  Swap(lambda.x(), lambda.y());
132  }
133 
134  return lambda;
135 }

138 Foam::vector Foam::eigenVector
139 (
140  const tensor& T,
141  const scalar lambda,
142  const vector& direction1,
143  const vector& direction2
144 )
145 {
146  // Construct the linear system for this eigenvalue
147  tensor A(T - lambda*I);
148 
149  // Determinants of the 2x2 sub-matrices used to find the eigenvectors
150  scalar sd0, sd1, sd2;
151  scalar magSd0, magSd1, magSd2;
152 
153  // Sub-determinants for a unique eivenvalue
154  sd0 = A.yy()*A.zz() - A.yz()*A.zy();
155  sd1 = A.zz()*A.xx() - A.zx()*A.xz();
156  sd2 = A.xx()*A.yy() - A.xy()*A.yx();
157  magSd0 = mag(sd0);
158  magSd1 = mag(sd1);
159  magSd2 = mag(sd2);
160 
161  // Evaluate the eigenvector using the largest sub-determinant
162  if (magSd0 >= magSd1 && magSd0 >= magSd2 && magSd0 > SMALL)
163  {
164  vector ev
165  (
166  1,
167  (A.yz()*A.zx() - A.zz()*A.yx())/sd0,
168  (A.zy()*A.yx() - A.yy()*A.zx())/sd0
169  );
170 
171  return ev/mag(ev);
172  }
173  else if (magSd1 >= magSd2 && magSd1 > SMALL)
174  {
175  vector ev
176  (
177  (A.xz()*A.zy() - A.zz()*A.xy())/sd1,
178  1,
179  (A.zx()*A.xy() - A.xx()*A.zy())/sd1
180  );
181 
182  return ev/mag(ev);
183  }
184  else if (magSd2 > SMALL)
185  {
186  vector ev
187  (
188  (A.xy()*A.yz() - A.yy()*A.xz())/sd2,
189  (A.yx()*A.xz() - A.xx()*A.yz())/sd2,
190  1
191  );
192 
193  return ev/mag(ev);
194  }
195 
196  // Sub-determinants for a repeated eigenvalue
197  sd0 = A.yy()*direction1.z() - A.yz()*direction1.y();
198  sd1 = A.zz()*direction1.x() - A.zx()*direction1.z();
199  sd2 = A.xx()*direction1.y() - A.xy()*direction1.x();
200  magSd0 = mag(sd0);
201  magSd1 = mag(sd1);
202  magSd2 = mag(sd2);
203 
204  // Evaluate the eigenvector using the largest sub-determinant
205  if (magSd0 >= magSd1 && magSd0 >= magSd2 && magSd0 > SMALL)
206  {
207  vector ev
208  (
209  1,
210  (A.yz()*direction1.x() - direction1.z()*A.yx())/sd0,
211  (direction1.y()*A.yx() - A.yy()*direction1.x())/sd0
212  );
213 
214  return ev/mag(ev);
215  }
216  else if (magSd1 >= magSd2 && magSd1 > SMALL)
217  {
218  vector ev
219  (
220  (direction1.z()*A.zy() - A.zz()*direction1.y())/sd1,
221  1,
222  (A.zx()*direction1.y() - direction1.x()*A.zy())/sd1
223  );
224 
225  return ev/mag(ev);
226  }
227  else if (magSd2 > SMALL)
228  {
229  vector ev
230  (
231  (A.xy()*direction1.z() - direction1.y()*A.xz())/sd2,
232  (direction1.x()*A.xz() - A.xx()*direction1.z())/sd2,
233  1
234  );
235 
236  return ev/mag(ev);
237  }
238 
239  // Triple eigenvalue
240  return direction1^direction2;
241 }

245 {
246  vector Ux(1, 0, 0), Uy(0, 1, 0), Uz(0, 0, 1);
247 
248  Ux = eigenVector(T, lambdas.x(), Uy, Uz);
249  Uy = eigenVector(T, lambdas.y(), Uz, Ux);
