101 template<class Type>
102 void Foam::wedgePointPatchField<Type>::evaluate(const Pstream::commsTypes)
103 {
104  // In order to ensure that the wedge patch is always flat, take the
105  // normal vector from the first point
106  const vector& nHat = this->patch().pointNormals()[0];
107 
108  tmp<Field<Type>> tvalues =
109  transform(I - nHat*nHat, this->patchInternalField());
110 
111  // Get internal field to insert values into
112  Field<Type>& iF = const_cast<Field<Type>&>(this->primitiveField());
113 
114  this->setInInternalField(iF, tvalues());
115 }
