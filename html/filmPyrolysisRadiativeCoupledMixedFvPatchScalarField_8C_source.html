228 void filmPyrolysisRadiativeCoupledMixedFvPatchScalarField::updateCoeffs()
229 {
230  if (updated())
231  {
232  return;
233  }
234 
235  // Get the coupling information from the mappedPatchBase
236  const mappedPatchBase& mpp =
237  refCast<const mappedPatchBase>(patch().patch());
238 
239  const label patchi = patch().index();
240  const label nbrPatchi = mpp.samplePolyPatch().index();
241  const polyMesh& mesh = patch().boundaryMesh().mesh();
242  const polyMesh& nbrMesh = mpp.sampleMesh();
243  const fvPatch& nbrPatch =
244  refCast<const fvMesh>(nbrMesh).boundary()[nbrPatchi];
245 
246  scalarField intFld(patchInternalField());
247 
248  const filmPyrolysisRadiativeCoupledMixedFvPatchScalarField&
249  nbrField =
250  refCast
251  <
252  const filmPyrolysisRadiativeCoupledMixedFvPatchScalarField
253  >
254  (
255  nbrPatch.lookupPatchField<volScalarField, scalar>(TnbrName_)
256  );
257 
258  // Swap to obtain full local values of neighbour internal field
259  scalarField nbrIntFld(nbrField.patchInternalField());
260  mpp.distribute(nbrIntFld);
261 
262  scalarField& Tp = *this;
263 
264  const scalarField K(this->kappa(*this));
265  const scalarField nbrK(nbrField.kappa(*this));
266 
267  // Swap to obtain full local values of neighbour K*delta
268  scalarField KDeltaNbr(nbrK*nbrPatch.deltaCoeffs());
269  mpp.distribute(KDeltaNbr);
270 
271  scalarField myKDelta(K*patch().deltaCoeffs());
272 
273  scalarList Tfilm(patch().size(), Zero);
274  scalarList htcwfilm(patch().size(), Zero);
275  scalarList filmDelta(patch().size(), Zero);
276 
277  const pyrolysisModelType& pyrolysis = pyrModel();
278  const filmModelType& film = filmModel();
279 
280  // Obtain Rad heat (qr)
281  scalarField qr(patch().size(), Zero);
282 
283  label coupledPatchi = -1;
284  if (pyrolysisRegionName_ == mesh.name())
285  {
286  coupledPatchi = patchi;
287  if (qrName_ != "none")
288  {
289  qr = nbrPatch.lookupPatchField<volScalarField, scalar>(qrName_);
290  mpp.distribute(qr);
291  }
292  }
293  else if (pyrolysis.primaryMesh().name() == mesh.name())
294  {
295  coupledPatchi = nbrPatch.index();
296  if (qrName_ != "none")
297  {
298  qr = patch().lookupPatchField<volScalarField, scalar>(qrName_);
299  }
300  }
301  else
302  {
303  FatalErrorInFunction
304  << type() << " condition is intended to be applied to either the "
305  << "primary or pyrolysis regions only"
306  << exit(FatalError);
307  }
308 
309  const label filmPatchi = pyrolysis.nbrCoupledPatchID(film, coupledPatchi);
310 
311  const scalarField htcw(film.htcw().h()().boundaryField()[filmPatchi]);
312 
313  // Obtain htcw
314  htcwfilm =
315  pyrolysis.mapRegionPatchField
316  (
317  film,
318  coupledPatchi,
319  filmPatchi,
320  htcw,
321  true
322  );
323 
324 
325  // Obtain Tfilm at the boundary through Ts.
326  // NOTE: Tf is not good as at the boundary it will retrieve Tp
327  Tfilm = film.Ts().boundaryField()[filmPatchi];
328  film.toPrimary(filmPatchi, Tfilm);
329 
330  // Obtain delta
331  filmDelta =
332  pyrolysis.mapRegionPatchField<scalar>
333  (
334  film,
335  "deltaf",
336  coupledPatchi,
337  true
338  );
339 
340  // Estimate wetness of the film (1: wet , 0: dry)
341  scalarField ratio
342  (
343  min
344  (
345  max
346  (
347  (filmDelta - filmDeltaDry_)/(filmDeltaWet_ - filmDeltaDry_),
348  scalar(0)
349  ),
350  scalar(1)
351  )
352  );
353 
354  scalarField qConv(ratio*htcwfilm*(Tfilm - Tp)*convectiveScaling_);
355 
356  scalarField qRad((1.0 - ratio)*qr);
357 
358  scalarField alpha(KDeltaNbr - (qRad + qConv)/Tp);
359 
360  valueFraction() = alpha/(alpha + (1.0 - ratio)*myKDelta);
361 
362  refValue() = ratio*Tfilm + (1.0 - ratio)*(KDeltaNbr*nbrIntFld)/alpha;
363 
364  mixedFvPatchScalarField::updateCoeffs();
365 
366  if (debug)
367  {
368  scalar Qc = gSum(qConv*patch().magSf());
369  scalar qr = gSum(qRad*patch().magSf());
370  scalar Qt = gSum((qConv + qRad)*patch().magSf());
371 
372  Info<< mesh.name() << ':'
373  << patch().name() << ':'
374  << this->internalField().name() << " <- "
375  << nbrMesh.name() << ':'
376  << nbrPatch.name() << ':'
377  << this->internalField().name() << " :" << nl
378  << " convective heat[W] : " << Qc << nl
379  << " radiative heat [W] : " << qr << nl
380  << " total heat [W] : " << Qt << nl
381  << " wall temperature "
382  << " min:" << gMin(*this)
383  << " max:" << gMax(*this)
384  << " avg:" << gAverage(*this)
385  << endl;
386  }
387 }
