62 void Foam::isoCutFace::calcSubFaceCentreAndArea()
63 {
64  const label nPoints = subFacePoints_.size();
65 
66  // If the face is a triangle, do a direct calculation for efficiency
67  // and to avoid round-off error-related problems
68  if (nPoints == 3)
69  {
70  subFaceCentre_ = sum(subFacePoints_)/scalar(3);
71  subFaceArea_ =
72  0.5
73  *(
74  (subFacePoints_[1] - subFacePoints_[0])
75  ^(subFacePoints_[2] - subFacePoints_[0])
76  );
77  }
78  else if (nPoints > 0)
79  {
80  vector sumN(Zero);
81  scalar sumA(0.0);
82  vector sumAc(Zero);
83  const point fCentre = sum(subFacePoints_)/scalar(nPoints);
84 
85  for (label pi = 0; pi < nPoints; pi++)
86  {
87  const point& nextPoint = subFacePoints_[subFacePoints_.fcIndex(pi)];
88 
89  vector c = subFacePoints_[pi] + nextPoint + fCentre;
90  vector n =
91  (nextPoint - subFacePoints_[pi])^(fCentre - subFacePoints_[pi]);
92  scalar a = magSqr(n);
93 
94  sumN += n;
95  sumA += a;
96  sumAc += a*c;
97  }
98 
99  // This is to deal with zero-area faces. Mark very small faces
100  // to be detected in e.g., processorPolyPatch.
101  if (sumA < ROOTVSMALL)
102  {
103  subFaceCentre_ = fCentre;
104  subFaceArea_ = vector::zero;
105  }
106  else
107  {
108  subFaceCentre_ = (1.0/3.0)*sumAc/sumA;
109  subFaceArea_ = 0.5*sumN;
110  }
111  }
112 
113  subFaceCentreAndAreaIsCalculated_ = true;
114 }

117 Foam::label Foam::isoCutFace::calcSubFace
118 (
119  const scalar isoValue,
120  const pointField& points,
121  const scalarField& f,
122  const labelList& pLabels
123 )
124 {
125  // Face status set to one of the values:
126  // -1: face is fully below the isosurface
127  // 0: face is cut, i.e. has values larger and smaller than isoValue
128  // +1: face is fully above the isosurface
129  label faceStatus;
130 
131  scalar f1 = f[pLabels[0]];
132 
133  // If vertex values are very close to isoValue lift them slightly to avoid
134  // dealing with the many special cases of a face being touched either at a
135  // single point, along an edge, or the entire face being on the surface.
136  if (mag(f1 - isoValue) < 10*SMALL)
137  {
138  f1 += sign(f1 - isoValue)*10*SMALL;
139  }
140 
141  // Finding cut edges, the point along them where they are cut, and all fully
142  // submerged face points.
143  forAll(pLabels, pi)
144  {
145  label pl2 = pLabels[pLabels.fcIndex(pi)];
146  scalar f2 = f[pl2];
147  if (mag(f2 - isoValue) < 10*SMALL)
148  {
149  f2 += sign(f2 - isoValue)*10*SMALL;
150  }
151 
152  if (f1 > isoValue)
153  {
154  nFullySubmergedPoints_ += 1;
155 
156  if (f2 < isoValue)
157  {
158  lastEdgeCut_ = (isoValue - f1)/(f2 - f1);
159  }
160  }
161  else if (f1 < isoValue && f2 > isoValue)
162  {
163  if (firstFullySubmergedPoint_ == -1)
164  {
165  firstFullySubmergedPoint_ = pLabels.fcIndex(pi);
166 
167  firstEdgeCut_ = (isoValue - f1)/(f2 - f1);
168  }
169  else
170  {
171  if (debug)
172  {
173  const face fl(pLabels);
174 
175  WarningInFunction
176  << "More than two face cuts for face " << fl
177  << endl;
178 
179  Pout<< "Face values: f-isoValue = " << endl;
180  forAll(fl, fpi)
181  {
182  Pout<< f[fl[fpi]] - isoValue << " ";
183  }
184  Pout<< " " << endl;
185  }
186  }
187  }
188  f1 = f2;
189  }
190 
191  if (firstFullySubmergedPoint_ != -1)
192  {
193  // Face is cut
194  faceStatus = 0;
195  subFacePoints(points, pLabels);
196  }
197  else if (f1 < isoValue)
198  {
199  // Face entirely above isosurface
200  faceStatus = 1;
201  }
202  else
203  {
204  // Face entirely below isosurface
205  faceStatus = -1;
206  }
207 
208  return faceStatus;
209 }

340 Foam::scalar Foam::isoCutFace::timeIntegratedFaceFlux
341 (
342  const label facei,
343  const vector& x0,
344  const vector& n0,
345  const scalar Un0,
346  const scalar f0,
347  const scalar dt,
348  const scalar phi,
349  const scalar magSf
350 )
351 {
352 /* Temporarily taken out
353  // Treating rare cases where isoface normal is not calculated properly
354  if (mag(n0) < 0.5)
355  {
356  scalar alphaf = 0;
357  scalar waterInUpwindCell = 0;
358 
359  if (phi > 10*SMALL || !mesh_.isInternalFace(facei))
360  {
361  const label upwindCell = mesh_.faceOwner()[facei];
362  alphaf = alpha1In_[upwindCell];
363  waterInUpwindCell = alphaf*mesh_.cellVolumes()[upwindCell];
364  }
365  else
366  {
367  const label upwindCell = mesh_.faceNeighbour()[facei];
368  alphaf = alpha1In_[upwindCell];
369  waterInUpwindCell = alphaf*mesh_.cellVolumes()[upwindCell];
370  }
371 
372  if (debug)
373  {
374  WarningInFunction
375  << "mag(n0) = " << mag(n0)
376  << " so timeIntegratedFlux calculates dVf from upwind"
377  << " cell alpha value: " << alphaf << endl;
378  }
379 
380  return min(alphaf*phi*dt, waterInUpwindCell);
381  }
382 */
383 
384  // Find sorted list of times where the isoFace will arrive at face points
385  // given initial position x0 and velocity Un0*n0
386 
387  // Get points for this face
388  const face& f = mesh_.faces()[facei];
389  const pointField fPts(f.points(mesh_.points()));
390  const label nPoints = fPts.size();
391 
392  scalarField pTimes(fPts.size());
393  if (mag(Un0) > 10*SMALL) // Note: tolerances
394  {
395  // Here we estimate time of arrival to the face points from their normal
396  // distance to the initial surface and the surface normal velocity
397 
398  pTimes = ((fPts - x0) & n0)/Un0;
399 
400  scalar dVf = 0;
401 
402  // Check if pTimes changes direction more than twice when looping face
403  label nShifts = 0;
404  forAll(pTimes, pi)
405  {
406  const label oldEdgeSign =
407  sign(pTimes[(pi + 1) % nPoints] - pTimes[pi]);
408  const label newEdgeSign =
409  sign(pTimes[(pi + 2) % nPoints] - pTimes[(pi + 1) % nPoints]);
410 
411  if (newEdgeSign != oldEdgeSign)
412  {
413  nShifts++;
414  }
415  }
416 
417  if (nShifts == 2)
418  {
419  dVf = phi/magSf*timeIntegratedArea(fPts, pTimes, dt, magSf, Un0);
420  }
421  else if (nShifts > 2)
422  {
423  // Triangle decompose the face
424  pointField fPts_tri(3);
425  scalarField pTimes_tri(3);
426  fPts_tri[0] = mesh_.faceCentres()[facei];
427  pTimes_tri[0] = ((fPts_tri[0] - x0) & n0)/Un0;
428  for (label pi = 0; pi < nPoints; pi++)
429  {
430  fPts_tri[1] = fPts[pi];
431  pTimes_tri[1] = pTimes[pi];
432  fPts_tri[2] = fPts[(pi + 1) % nPoints];
433  pTimes_tri[2] = pTimes[(pi + 1) % nPoints];
434  const scalar magSf_tri =
435  mag
436  (
437  0.5
438  *(fPts_tri[2] - fPts_tri[0])
439  ^(fPts_tri[1] - fPts_tri[0])
440  );
441  const scalar phi_tri = phi*magSf_tri/magSf;
442  dVf += phi_tri/magSf_tri
443  *timeIntegratedArea
444  (
445  fPts_tri,
446  pTimes_tri,
447  dt,
448  magSf_tri,
449  Un0
450  );
451  }
452  }
453  else
454  {
455  if (debug)
456  {
457  WarningInFunction
458  << "Warning: nShifts = " << nShifts << " on face " << facei
459  << " with pTimes = " << pTimes << " owned by cell "
460  << mesh_.faceOwner()[facei] << endl;
461  }
462  }
463 
464  return dVf;
465  }
466  else
467  {
468  // Un0 is almost zero and isoFace is treated as stationary
469  calcSubFace(facei, f0);
470  const scalar alphaf = mag(subFaceArea()/magSf);
471 
472  if (debug)
473  {
474  WarningInFunction
475  << "Un0 is almost zero (" << Un0
476  << ") - calculating dVf on face " << facei
477  << " using subFaceFraction giving alphaf = " << alphaf
478  << endl;
479  }
480 
481  return phi*dt*alphaf;
482  }

486 Foam::scalar Foam::isoCutFace::timeIntegratedArea
487 (
488  const pointField& fPts,
489  const scalarField& pTimes,
490  const scalar dt,
491  const scalar magSf,
492  const scalar Un0
493 )
494 {
495  // Initialise time integrated area returned by this function
496  scalar tIntArea = 0.0;
497 
498  // Finding ordering of vertex points
499  labelList order(sortedOrder(pTimes));
500  const scalar firstTime = pTimes[order.first()];
501  const scalar lastTime = pTimes[order.last()];
502 
503  // Dealing with case where face is not cut by surface during time interval
504  // [0,dt] because face was already passed by surface
505  if (lastTime <= 0)
506  {
507  // If all face cuttings were in the past and cell is filling up (Un0>0)
508  // then face must be full during whole time interval
509  tIntArea = magSf*dt*pos0(Un0);
510  return tIntArea;
511  }
512 
513  // Dealing with case where face is not cut by surface during time interval
514  // [0, dt] because dt is too small for surface to reach closest face point
515  if (firstTime >= dt)
516  {
517  // If all cuttings are in the future but non of them within [0,dt] then
518  // if cell is filling up (Un0 > 0) face must be empty during whole time
519  // interval
520  tIntArea = magSf*dt*(1 - pos0(Un0));
521  return tIntArea;
522  }
523 
524  // If we reach this point in the code some part of the face will be swept
525  // during [tSmall, dt-tSmall]. However, it may be the case that there are no
526  // vertex times within the interval. This will happen sometimes for small
527  // time steps where both the initial and the final face-interface
528  // intersection line (FIIL) will be along the same two edges.
529 
530  // Face-interface intersection line (FIIL) to be swept across face
531  DynamicList<point> FIIL(3);
532  // Submerged area at beginning of each sub time interval time
533  scalar initialArea = 0.0;
534  //Running time keeper variable for the integration process
535  scalar time = 0.0;
536 
537  // Special treatment of first sub time interval
538  if (firstTime > 0)
539  {
540  // If firstTime > 0 the face is uncut in the time interval
541  // [0, firstTime] and hence fully submerged in fluid A or B.
542  // If Un0 > 0 cell is filling up and it must initially be empty.
543  // If Un0 < 0 cell must initially be full(y immersed in fluid A).
544  time = firstTime;
545  initialArea = magSf*(1.0 - pos0(Un0));
546  tIntArea = initialArea*time;
547  cutPoints(fPts, pTimes, time, FIIL);
548  }
549  else
550  {
551  // If firstTime <= 0 then face is initially cut and we must
552  // calculate the initial submerged area and FIIL:
553  time = 0.0;
554  // Note: calcSubFace assumes well-defined 2-point FIIL!!!!
555  calcSubFace(fPts, -sign(Un0)*pTimes, time);
556  initialArea = mag(subFaceArea());
557  cutPoints(fPts, pTimes, time, FIIL);
558  }
559 
560  // Making sorted array of all vertex times that are between max(0,firstTime)
561  // and dt and further than tSmall from the previous time.
562  DynamicList<scalar> sortedTimes(pTimes.size());
563  {
564  scalar prevTime = time;
565  const scalar tSmall = max(1e-6*dt, 10*SMALL);
566  forAll(order, ti)
567  {
568  const scalar timeI = pTimes[order[ti]];
569  if ( timeI > prevTime + tSmall && timeI <= dt)
570  {
571  sortedTimes.append(timeI);
572  prevTime = timeI;
573  }
574  }
575  }
576 
577  // Sweeping all quadrilaterals corresponding to the intervals defined above
578  forAll(sortedTimes, ti)
579  {
580  const scalar newTime = sortedTimes[ti];
581  // New face-interface intersection line
582  DynamicList<point> newFIIL(3);
583  cutPoints(fPts, pTimes, newTime, newFIIL);
584 
585  // quadrilateral area coefficients
586  scalar alpha = 0, beta = 0;
587  quadAreaCoeffs(FIIL, newFIIL, alpha, beta);
588  // Integration of area(t) = A*t^2+B*t from t = 0 to 1
589  tIntArea += (newTime - time)*
590  (initialArea + sign(Un0)*(alpha/3.0 + 0.5*beta));
591  // Adding quad area to submerged area
592  initialArea += sign(Un0)*(alpha + beta);
593 
594  FIIL = newFIIL;
595  time = newTime;
596  }
597 
598  // Special treatment of last time interval
599  if (lastTime > dt)
600  {
601  // FIIL will end up cutting the face at dt
602  // New face-interface intersection line
603  DynamicList<point> newFIIL(3);
604  cutPoints(fPts, pTimes, dt, newFIIL);
605 
606  // quadrilateral area coefficients
607  scalar alpha = 0, beta = 0;
608  quadAreaCoeffs(FIIL, newFIIL, alpha, beta);
609  // Integration of area(t) = A*t^2+B*t from t = 0 to 1
610  tIntArea += (dt - time)*
611  (initialArea + sign(Un0)*(alpha/3.0 + 0.5*beta));
612  }
613  else
614  {
615  // FIIL will leave the face at lastTime and face will be fully in fluid
616  // A or fluid B in the time interval from lastTime to dt.
617  tIntArea += magSf*(dt - lastTime)*pos0(Un0);
618  }
619 
620  return tIntArea;
621 }

624 void Foam::isoCutFace::cutPoints
625 (
626  const pointField& pts,
627  const scalarField& f,
628  const scalar f0,
629  DynamicList<point>& cutPoints
630 )
631 {
632  const label nPoints = pts.size();
633  scalar f1(f[0]);
634 
635  // Snapping vertex value to f0 if very close (needed for 2D cases)
636  if (mag(f1 - f0) < 10*SMALL)
637  {
638  f1 = f0;
639  }
640 
641  forAll(pts, pi)
642  {
643  label pi2 = (pi + 1) % nPoints;
644  scalar f2 = f[pi2];
645 
646  // Snapping vertex value
647  if (mag(f2 - f0) < 10*SMALL)
648  {
649  f2 = f0;
650  }
651 
652  if ((f1 < f0 && f2 > f0) || (f1 > f0 && f2 < f0))
653  {
654  const scalar s = (f0 - f1)/(f2 - f1);
655  cutPoints.append(pts[pi] + s*(pts[pi2] - pts[pi]));
656  }
657  else if (f1 == f0)
658  {
659  cutPoints.append(pts[pi]);
660  }
661  f1 = f2;
662  }
663 
664  if (cutPoints.size() > 2)
665  {
666  WarningInFunction << "cutPoints = " << cutPoints << " for pts = " << pts
667  << ", f - f0 = " << f - f0 << " and f0 = " << f0 << endl;
668  }
669 }

672 void Foam::isoCutFace::quadAreaCoeffs
673 (
674  const DynamicList<point>& pf0,
675  const DynamicList<point>& pf1,
676  scalar& alpha,
677  scalar& beta
678 ) const
679 {
680  // Number of points in provided face-interface intersection lines
681  const label np0 = pf0.size();
682  const label np1 = pf1.size();
683 
684  // quad area coeffs such that area(t) = alpha*t^2 + beta*t.
685  // With time interval normalised, we have full quadArea = alpha + beta
686  // and time integrated quad area = alpha/3 + beta/2;
687  alpha = 0.0;
688  beta = 0.0;
689 
690  if (np0 && np1)
691  {
692  // Initialising quadrilateral vertices A, B, C and D
693  vector A(pf0[0]);
694  vector C(pf1[0]);
695  vector B(pf0[0]);
696  vector D(pf1[0]);
697 
698  if (np0 == 2)
699  {
700  B = pf0[1];
701  }
702  else if (np0 > 2)
703  {
704  WarningInFunction << "Vertex face was cut at pf0 = " << pf0 << endl;
705  }
706 
707  if (np1 == 2)
708  {
709  D = pf1[1];
710  }
711  else if (np1 > 2)
712  {
713  WarningInFunction << "Vertex face was cut at pf1 = " << pf1 << endl;
714  }
715 
716  // Swapping pf1 points if pf0 and pf1 point in same general direction
717  // (because we want a quadrilateral ABCD where pf0 = AB and pf1 = CD)
718  if (((B - A) & (D - C)) > 0)
719  {
720  vector tmp = D;
721  D = C;
722  C = tmp;
723  }
724 
725  // Defining local coordinates (xhat, yhat) for area integration of swept
726  // quadrilateral ABCD such that A = (0,0), B = (Bx,0), C = (Cx,Cy) and
727  // D = (Dx,Dy) with Cy = 0 and Dy > 0.
728 
729  const scalar Bx = mag(B - A);
730 
731  vector xhat(Zero);
732  if (Bx > 10*SMALL)
733  {
734  // If |AB| > 0 ABCD we use AB to define xhat
735  xhat = normalised(B - A);
736  }
737  else if (mag(C - D) > 10*SMALL)
738  {
