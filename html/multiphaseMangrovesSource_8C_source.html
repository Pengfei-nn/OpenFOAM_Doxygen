56 Foam::tmp<Foam::volScalarField>
57 Foam::fv::multiphaseMangrovesSource::dragCoeff(const volVectorField& U) const
58 {
59  auto tdragCoeff = tmp<volScalarField>::New
60  (
61  IOobject
62  (
63  typeName + ":dragCoeff",
64  mesh_.time().timeName(),
65  mesh_.time(),
66  IOobject::NO_READ,
67  IOobject::NO_WRITE
68  ),
69  mesh_,
70  dimensionedScalar(dimless/dimTime, Zero)
71  );
72  auto& dragCoeff = tdragCoeff.ref();
73 
74  forAll(zoneIDs_, i)
75  {
76  const scalar a = aZone_[i];
77  const scalar N = NZone_[i];
78  const scalar Cd = CdZone_[i];
79 
80  const labelList& zones = zoneIDs_[i];
81 
82  for (label zonei : zones)
83  {
84  const cellZone& cz = mesh_.cellZones()[zonei];
85 
86  for (label celli : cz)
87  {
88  const vector& Uc = U[celli];
89 
90  dragCoeff[celli] = 0.5*Cd*a*N*mag(Uc);
91  }
92  }
93  }
94 
95  dragCoeff.correctBoundaryConditions();
96 
97  return tdragCoeff;
98 }

102 Foam::fv::multiphaseMangrovesSource::inertiaCoeff() const
103 {
104  auto tinertiaCoeff = tmp<volScalarField>::New
105  (
106  IOobject
107  (
108  typeName + ":inertiaCoeff",
109  mesh_.time().timeName(),
110  mesh_.time(),
111  IOobject::NO_READ,
112  IOobject::NO_WRITE
113  ),
114  mesh_,
115  dimensionedScalar(dimless, Zero)
116  );

118

120 
121  forAll(zoneIDs_, i)
122  {
123  const scalar a = aZone_[i];
124  const scalar N = NZone_[i];
125  const scalar Cm = CmZone_[i];
126 
127  const labelList& zones = zoneIDs_[i];
128 
129  for (label zonei : zones)
130  {
131  const cellZone& cz = mesh_.cellZones()[zonei];
132 
133  for (label celli : cz)
134  {
135  inertiaCoeff[celli] = 0.25*(Cm+1)*pi*a*a*N;
136  }
137  }
138  }

168 void Foam::fv::multiphaseMangrovesSource::addSup
169 (
170  fvMatrix<vector>& eqn,
171  const label fieldi
172 )
173 {
174  const volVectorField& U = eqn.psi();
175 
176  fvMatrix<vector> mangrovesEqn
177  (
178  - fvm::Sp(dragCoeff(U), U)
179  - inertiaCoeff()*fvm::ddt(U)
180  );
181 
182  // Contributions are added to RHS of momentum equation
183  eqn += mangrovesEqn;
184 }

207 bool Foam::fv::multiphaseMangrovesSource::read(const dictionary& dict)
208 {
209  if (option::read(dict))
210  {
211  if (!coeffs_.readIfPresent("UNames", fieldNames_))
212  {
213  fieldNames_.resize(1);
214  fieldNames_.first() = coeffs_.lookupOrDefault<word>("U", "U");
215  }
216 
217  applied_.setSize(fieldNames_.size(), false);
218 
219  // Create the Mangroves models - 1 per region
220  const dictionary& regionsDict(coeffs_.subDict("regions"));
221  const wordList regionNames(regionsDict.toc());
222  aZone_.setSize(regionNames.size(), 1);
223  NZone_.setSize(regionNames.size(), 1);
224  CmZone_.setSize(regionNames.size(), 1);
225  CdZone_.setSize(regionNames.size(), 1);
226  zoneIDs_.setSize(regionNames.size());
227 
228  forAll(zoneIDs_, i)
229  {
230  const word& regionName = regionNames[i];
231  const dictionary& modelDict = regionsDict.subDict(regionName);
232 
233  const word zoneName(modelDict.get<word>("cellZone"));
234 
235  zoneIDs_[i] = mesh_.cellZones().indices(zoneName);
236  if (zoneIDs_[i].empty())
237  {
238  FatalErrorInFunction
239  << "Unable to find cellZone " << zoneName << nl
240  << "Valid cellZones are:" << mesh_.cellZones().names()
241  << exit(FatalError);
242  }
243 
244  modelDict.readEntry("a", aZone_[i]);
245  modelDict.readEntry("N", NZone_[i]);
246  modelDict.readEntry("Cm", CmZone_[i]);
247  modelDict.readEntry("Cd", CdZone_[i]);
248  }
249 
250  return true;
251  }
252 
253  return false;
254 }
