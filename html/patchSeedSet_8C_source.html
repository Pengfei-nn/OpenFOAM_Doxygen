50 void Foam::patchSeedSet::calcSamples
51 (
52  DynamicList<point>& samplingPts,
53  DynamicList<label>& samplingCells,
54  DynamicList<label>& samplingFaces,
55  DynamicList<label>& samplingSegments,
56  DynamicList<scalar>& samplingCurveDist
57 )
58 {
59  if (debug)
60  {
61  Info<< "patchSeedSet : sampling on patches :" << endl;
62  }
63 
64  // Construct search tree for all patch faces.
65  label sz = 0;
66  for (const label patchi : patchSet_)
67  {
68  const polyPatch& pp = mesh().boundaryMesh()[patchi];
69 
70  sz += pp.size();
71 
72  if (debug)
73  {
74  Info<< " " << pp.name() << " size " << pp.size() << endl;
75  }
76  }
77 
78  labelList patchFaces(sz);
79  sz = 0;
80  for (const label patchi : patchSet_)
81  {
82  const polyPatch& pp = mesh().boundaryMesh()[patchi];
83  forAll(pp, i)
84  {
85  patchFaces[sz++] = pp.start()+i;
86  }
87  }
88 
89 
90  if (!rndGenPtr_.valid())
91  {
92  rndGenPtr_.reset(new Random(0));
93  }
94  Random& rndGen = rndGenPtr_();
95 
96 
97  if (selectedLocations_.size())
98  {
99  DynamicList<label> newPatchFaces(patchFaces.size());
100 
101  // Find the nearest patch face
102  {
103  // 1. All processors find nearest local patch face for all
104  // selectedLocations
105 
106  // All the info for nearest. Construct to miss
107  List<mappedPatchBase::nearInfo> nearest(selectedLocations_.size());
108 
109  const indirectPrimitivePatch pp
110  (
111  IndirectList<face>(mesh().faces(), patchFaces),
112  mesh().points()
113  );
114 
115  treeBoundBox patchBb
116  (
117  treeBoundBox(pp.points(), pp.meshPoints()).extend
118  (
119  rndGen,
120  1e-4
121  )
122  );
123  patchBb.min() -= point(ROOTVSMALL, ROOTVSMALL, ROOTVSMALL);
124  patchBb.max() += point(ROOTVSMALL, ROOTVSMALL, ROOTVSMALL);
125 
126  indexedOctree<treeDataFace> boundaryTree
127  (
128  treeDataFace // all information needed to search faces
129  (
130  false, // do not cache bb
131  mesh(),
132  patchFaces // boundary faces only
133  ),
134  patchBb, // overall search domain
135  8, // maxLevel
136  10, // leafsize
137  3.0 // duplicity
138  );
139 
140  // Get some global dimension so all points are equally likely
141  // to be found
142  const scalar globalDistSqr
143  (
144  //magSqr
145  //(
146  // boundBox
147  // (
148  // pp.points(),
149  // pp.meshPoints(),
150  // true
151  // ).span()
152  //)
153  GREAT
154  );
155 
156  forAll(selectedLocations_, sampleI)
157  {
158  const point& sample = selectedLocations_[sampleI];
159 
160  pointIndexHit& nearInfo = nearest[sampleI].first();
161  nearInfo = boundaryTree.findNearest
162  (
163  sample,
164  globalDistSqr
165  );
166 
167  if (!nearInfo.hit())
168  {
169  nearest[sampleI].second().first() = Foam::sqr(GREAT);
170  nearest[sampleI].second().second() =
171  Pstream::myProcNo();
172  }
173  else
174  {
175  point fc(pp[nearInfo.index()].centre(pp.points()));
176  nearInfo.setPoint(fc);
177  nearest[sampleI].second().first() = magSqr(fc-sample);
178  nearest[sampleI].second().second() =
179  Pstream::myProcNo();
180  }
181  }
182 
183 
184  // 2. Reduce on master. Select nearest processor.
185 
186  // Find nearest. Combine on master.
187  Pstream::listCombineGather(nearest, mappedPatchBase::nearestEqOp());
188  Pstream::listCombineScatter(nearest);
189 
190 
191  // 3. Pick up my local faces that have won
192 
193  forAll(nearest, sampleI)
194  {
195  if (nearest[sampleI].first().hit())
196  {
197  label procI = nearest[sampleI].second().second();
198  label index = nearest[sampleI].first().index();
199 
200  if (procI == Pstream::myProcNo())
201  {
202  newPatchFaces.append(pp.addressing()[index]);
203  }
204  }
205  }
206  }
207 
208  if (debug)
209  {
210  Pout<< "Found " << newPatchFaces.size()
211  << " out of " << selectedLocations_.size()
212  << " on local processor" << endl;
213  }
214 
215  patchFaces.transfer(newPatchFaces);
216  }
217 
218 
219  // Shuffle and truncate if in random mode
220  label totalSize = returnReduce(patchFaces.size(), sumOp<label>());
221 
222  if (maxPoints_ < totalSize)
223  {
224  // Check what fraction of maxPoints_ I need to generate locally.
225  label myMaxPoints =
226  label(scalar(patchFaces.size())/totalSize*maxPoints_);
227 
228  labelList subset = identity(patchFaces.size());
229  for (label iter = 0; iter < 4; ++iter)
230  {
231  forAll(subset, i)
232  {
233  label j = rndGen.position<label>(0, subset.size()-1);
234  Swap(subset[i], subset[j]);
235  }
236  }
237  // Truncate
238  subset.setSize(myMaxPoints);
239 
240  // Subset patchFaces
241  patchFaces = labelUIndList(patchFaces, subset)();
242 
243  if (debug)
244  {
245  Pout<< "In random mode : selected " << patchFaces.size()
246  << " faces out of " << patchFaces.size() << endl;
247  }
248  }
249 
250 
251  // Get points on patchFaces.
252  globalIndex globalSampleNumbers(patchFaces.size());
253 
254  samplingPts.setCapacity(patchFaces.size());
255  samplingCells.setCapacity(patchFaces.size());
256  samplingFaces.setCapacity(patchFaces.size());
257  samplingSegments.setCapacity(patchFaces.size());
258  samplingCurveDist.setCapacity(patchFaces.size());
259 
260  // For calculation of min-decomp tet base points
261  (void)mesh().tetBasePtIs();
262 
263  forAll(patchFaces, i)
264  {
265  label facei = patchFaces[i];
266 
267  // Slightly shift point in since on warped face face-diagonal
268  // decomposition might be outside cell for face-centre decomposition!
269  pointIndexHit info = mappedPatchBase::facePoint
270  (
271  mesh(),
272  facei,
273  polyMesh::FACE_DIAG_TRIS
274  );
275  label celli = mesh().faceOwner()[facei];
276 
277  if (info.hit())
278  {
279  // Move the point into the cell
280  const point& cc = mesh().cellCentres()[celli];
281  samplingPts.append
282  (
283  info.hitPoint() + 1e-1*(cc-info.hitPoint())
284  );
285  }
286  else
287  {
288  samplingPts.append(info.rawPoint());
289  }
290  samplingCells.append(celli);
291  samplingFaces.append(facei);
292  samplingSegments.append(0);
293  samplingCurveDist.append(globalSampleNumbers.toGlobal(i));
294  }
295 }
