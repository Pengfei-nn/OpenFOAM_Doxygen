74 {

77 
78  volVectorField ny
79  (
80  IOobject
81  (
82  "ny",
83  mesh_.time().timeName(),

85  IOobject::NO_READ,
86  IOobject::NO_WRITE,

180 void adjointEikonalSolver::solve()
181 {
182  read();
183 
184  // Primal distance field
185  const volScalarField& d = RASModelVars_().d();
186 
187  // Convecting flux
188  tmp<surfaceScalarField> tyPhi = computeYPhi();
189  const surfaceScalarField& yPhi = tyPhi();
190 
191  // Iterate the adjoint to the eikonal equation
192  for (label iter = 0; iter < nEikonalIters_; ++iter)
193  {
194  read();
195 
196  Info<< "Adjoint Eikonal Iteration : " << iter << endl;
197 
198  fvScalarMatrix daEqn
199  (
200  2*fvm::div(-yPhi, da_)
201  + fvm::SuSp(-epsilon_*fvc::laplacian(d), da_)
202  - epsilon_*fvm::laplacian(d, da_)
203  + source_
204  );
205 
206  daEqn.relax();
207  scalar residual = daEqn.solve().initialResidual();
208  Info<< "Max da " << gMax(mag(da_)()) << endl;
209 
210  Info<< "ExecutionTime = " << mesh_.time().elapsedCpuTime() << " s"
211  << " ClockTime = " << mesh_.time().elapsedClockTime() << " s"
212  << nl << endl;
213 
214  // Check convergence
215  if (residual < tolerance_)
216  {
217  Info<< "\n***Reached adjoint eikonal convergence limit, iteration "
218  << iter << "***\n\n";
219  break;
220  }
221  }
222  da_.write();
223 }

233 boundaryVectorField& adjointEikonalSolver::distanceSensitivities()
234 {
235  Info<< "Calculating distance sensitivities " << endl;
236 
237  boundaryVectorField& distanceSens = distanceSensPtr_();
238 
239  const volScalarField& d = RASModelVars_().d();
240  for (const label patchi : sensitivityPatchIDs_)
241  {
242  vectorField nf(mesh_.boundary()[patchi].nf());
243 
244  // No surface area included. Will be done by the actual sensitivity tool
245  distanceSens[patchi] =
246  -2.*da_.boundaryField()[patchi]
247  *d.boundaryField()[patchi].snGrad()
248  *d.boundaryField()[patchi].snGrad()*nf;
249  }
250  return distanceSens;
251 }

254 tmp<volTensorField> adjointEikonalSolver::getFISensitivityTerm() const
255 {
256  Info<< "Calculating distance sensitivities " << endl;
257 
258  const volScalarField& d = RASModelVars_().d();
259  const volVectorField gradD(fvc::grad(d));
260 
261  volVectorField gradDDa
262  (
263  IOobject
264  (
265  "gradDDa",
266  mesh_.time().timeName(),
267  mesh_,
268  IOobject::NO_READ,
269  IOobject::AUTO_WRITE
270  ),
271  mesh_,
272  dimensionedVector(d.dimensions()*da_.dimensions()/dimLength, Zero),
273  patchDistMethod::patchTypes<vector>(mesh_, wallPatchIDs_)
274  );
275  gradDDa = fvc::grad(d*da_);
276 
277  tmp<volTensorField> tdistanceSens
278  (
279  new volTensorField
280  (
281  IOobject
282  (
283  "distanceSensFI",
284  mesh_.time().timeName(),
285  mesh_,
286  IOobject::NO_READ,
287  IOobject::AUTO_WRITE
288  ),
289  mesh_,
290  dimensionedTensor(da_.dimensions(), Zero)
291  )
292  );
293  volTensorField& distanceSens = tdistanceSens.ref();
294 
295  distanceSens =
296  - 2.*da_*gradD*gradD
297  - epsilon_*gradD*gradDDa
298  + epsilon_*da_*d*fvc::grad(gradD);
299 
300  return tdistanceSens;
301 }
