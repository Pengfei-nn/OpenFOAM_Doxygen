236  const bool useGroups // ignored
237 ) const
238 {

296 Foam::label Foam::faBoundaryMesh::whichPatch(const label edgeIndex) const
297 {
298  // Find out which patch the current face belongs to by comparing label
299  // with patch start labels.
300  // If the face is internal, return -1;
301  // if it is off the end of the list, abort
302  if (edgeIndex < mesh().nInternalEdges())
303  {
304  return -1;
305  }
306  else if (edgeIndex >= mesh().nEdges())
307  {
308  FatalErrorInFunction
309  << "Edge " << edgeIndex
310  << " out of bounds. Number of geometric edges " << mesh().nEdges()
311  << abort(FatalError);
312  }
313 
314  forAll(*this, patchi)
315  {
316  label start = mesh_.patchStarts()[patchi];
317  label size = operator[](patchi).faPatch::size();
318 
319  if
320  (
321  edgeIndex >= start
322  && edgeIndex < start + size
323  )
324  {
325  return patchi;
326  }
327  }
328 
329  // If not in any of above, it's trouble!
330  FatalErrorInFunction
331  << "error in patch search algorithm"
332  << abort(FatalError);
333 
334  return -1;
335 }

338 bool Foam::faBoundaryMesh::checkDefinition(const bool report) const
339 {
340  label nextPatchStart = mesh().nInternalEdges();
341  const faBoundaryMesh& bm = *this;
342 
343  bool hasError = false;
344 
345  forAll(bm, patchi)
346  {
347  if (bm[patchi].start() != nextPatchStart && !hasError)
348  {
349  hasError = true;
350 
351  InfoInFunction
352  << " ****Problem with boundary patch " << patchi
353  << " named " << bm[patchi].name()
354  << " of type " << bm[patchi].type()
355  << ". The patch should start on face no " << nextPatchStart
356  << " and the patch specifies " << bm[patchi].start()
357  << "." << endl
358  << "Possibly consecutive patches have this same problem."
359  << " Suppressing future warnings." << endl;
360  }
361 
362  // Warn about duplicate boundary patches?
363 
364  nextPatchStart += bm[patchi].faPatch::size();
365  }
366 
367  if (hasError)
368  {
369  SeriousErrorInFunction
370  << "This mesh is not valid: boundary definition is in error."
371  << endl;
372  }
373  else
374  {
375  if (debug || report)
376  {
377  Info << "Boundary definition OK." << endl;
378  }
379  }
380 
381  return hasError;
382 }
