32 Description
33  Plot3d mesh (ascii/formatted format) converter.
34 
35  Work in progress! Handles ascii multiblock (and optionally singleBlock)
36  format.
37  By default expects blanking. Use -noBlank if none.
38  Use -2D \a thickness if 2D.
39 
40  Niklas Nordin has experienced a problem with lefthandedness of the blocks.
41  The code should detect this automatically - see hexBlock::readPoints but
42  if this goes wrong just set the blockHandedness_ variable to 'right'
43  always.

61 int main(int argc, char *argv[])
62 {
63  argList::addNote
64  (
65  "Plot3d mesh (ascii/formatted format) converter"
66  );
67  argList::noParallel();
68  argList::addArgument("PLOT3D geom file");
69  argList::addOption

71  "scale",
72  "factor",
73  "Geometry scaling factor - default is 1"

83  "Input is a single block"

92  argList args(argc, argv);
93 
94  if (!args.check())
95  {
96  FatalError.exit();
97  }
98 
99  const scalar scaleFactor = args.get<scalar>("scale", 1);
100 
101  const bool readBlank = !args.found("noBlank");
102  const bool singleBlock = args.found("singleBlock");
103  scalar twoDThickness = -1;
104  if (args.readIfPresent("2D", twoDThickness))
105  {
106  Info<< "Reading 2D case by extruding points by " << twoDThickness
107  << " in z direction." << nl << endl;
108  }
109 
110 
111  #include "createTime.H"
112 
113  IFstream plot3dFile(args[1]);
114 
115  // Read the plot3d information using a fixed format reader.
116  // Comments in the file are in C++ style, so the stream parser will remove
117  // them with no intervention
118  label nblock;
119 
120  if (singleBlock)
121  {
122  nblock = 1;
123  }
124  else
125  {
126  plot3dFile >> nblock;
127  }
128 
129  Info<< "Reading " << nblock << " blocks" << endl;
130 
131  PtrList<hexBlock> blocks(nblock);
132 
133  {
134  label nx, ny, nz;
135 
136  forAll(blocks, blockI)
137  {
138  if (twoDThickness > 0)
139  {
140  // Fake second set of points (done in readPoints below)
141  plot3dFile >> nx >> ny;
142  nz = 2;
143  }
144  else
145  {
146  plot3dFile >> nx >> ny >> nz;
147  }
148 
149  Info<< "block " << blockI << " nx:" << nx
150  << " ny:" << ny << " nz:" << nz << endl;
151 
152  blocks.set(blockI, new hexBlock(nx, ny, nz));
153  }
154  }
155 
156  Info<< "Reading block points" << endl;
157  label sumPoints(0);
158  label nMeshCells(0);
159 
160  forAll(blocks, blockI)
161  {
162  Info<< "block " << blockI << ":" << nl;
163  blocks[blockI].readPoints(readBlank, twoDThickness, plot3dFile);
164  sumPoints += blocks[blockI].nBlockPoints();
165  nMeshCells += blocks[blockI].nBlockCells();
166  Info<< nl;
167  }
168 
169  pointField points(sumPoints);
170  labelList blockOffsets(blocks.size());
171  sumPoints = 0;
172  forAll(blocks, blockI)
173  {
174  const pointField& blockPoints = blocks[blockI].points();
175  blockOffsets[blockI] = sumPoints;
176  forAll(blockPoints, i)
177  {
178  points[sumPoints++] = blockPoints[i];
179  }
180  }
181 
182  // From old to new master point
183  labelList oldToNew;
184  pointField newPoints;
185 
186  // Merge points
187  mergePoints
188  (
189  points,
190  SMALL,
191  false,
192  oldToNew,
193  newPoints
194  );
195 
196  Info<< "Merged points within " << SMALL << " distance. Merged from "
197  << oldToNew.size() << " down to " << newPoints.size()
198  << " points." << endl;
199 
200  // Scale the points
201  if (scaleFactor > 1.0 + SMALL || scaleFactor < 1.0 - SMALL)
202  {
203  newPoints *= scaleFactor;
204  }
205 
206  Info<< "Creating cells" << endl;
207 
208  cellShapeList cellShapes(nMeshCells);
209 
210  const cellModel& hex = cellModel::ref(cellModel::HEX);
211 
212  label nCreatedCells = 0;
213 
214  forAll(blocks, blockI)
215  {
216  labelListList curBlockCells = blocks[blockI].blockCells();
217 
218  forAll(curBlockCells, blockCelli)
219  {
220  labelList cellPoints(curBlockCells[blockCelli].size());
221 
222  forAll(cellPoints, pointi)
223  {
224  cellPoints[pointi] =
225  oldToNew
226  [
227  curBlockCells[blockCelli][pointi]
228  + blockOffsets[blockI]
229  ];
230  }
231 
232  // Do automatic collapse from hex.
233  cellShapes[nCreatedCells] = cellShape(hex, cellPoints, true);
234 
235  nCreatedCells++;
236  }
237  }
238 
239  Info<< "Creating boundary patches" << endl;
240 
241  faceListList boundary(0);
242  wordList patchNames(0);
243  wordList patchTypes(0);
244  word defaultFacesName = "defaultFaces";
245  word defaultFacesType = wallPolyPatch::typeName;
246  wordList patchPhysicalTypes(0);
247 
248  polyMesh pShapeMesh
249  (
250  IOobject
251  (
252  polyMesh::defaultRegion,
253  runTime.constant(),
254  runTime
255  ),
256  std::move(newPoints),
257  cellShapes,
258  boundary,
259  patchNames,
260  patchTypes,
261  defaultFacesName,
262  defaultFacesType,
263  patchPhysicalTypes
264  );
265 
266  // Set the precision of the points data to 10
267  IOstream::defaultPrecision(max(10u, IOstream::defaultPrecision()));
268 
269  Info<< "Writing polyMesh" << endl;
270  pShapeMesh.removeFiles();
271  pShapeMesh.write();
272 
273  Info<< "End\n" << endl;
274 
275  return 0;
276 }
