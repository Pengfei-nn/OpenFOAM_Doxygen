44  // A sign-corrected multiply
45  // This is used for porosity of obstacle intersections
46  inline static scalar COMBLK(const scalar a, const scalar b)
47  {
48  if (a < 0)
49  {
50  return -a * b;
51  }
52 
53  return a * b;
54  }

81 void Foam::PDRutils::one_d_overlap
82 (
83  scalar xmin,
84  scalar xmax,
85  const PDRblock::location& grid,
86  List<scalar>& olap,
87  int *cmin, int *cmax,
88  int *cfmin, int *cfmax
89 )
90 {
91  // Looking at one coordinate direction, called x here, for something
92  // that extends from xmin to xmax, calculate how much it overlaps
93  // each cell in this direction. Result returned in 'olap' List is
94  // the proportion of the grid step overlapped, i.e dimensionless.
95  // First and last steps overlapped given by *cmin, *cmax
96  // Ditto for shifted grid given by *cfmin, *cfmax.
97 
98  // Initially zero everwhere
99  olap = Zero;
100 
101  if (olap.size() < grid.nPoints())
102  {
103  FatalErrorInFunction
104  << "The overlap scratch array is too small, has "
105  << olap.size() << " but needs " << grid.nPoints() << nl
106  << exit(FatalError);
107  }
108 
109 
110  // No intersection with the box
111  if (xmax <= grid.first() || grid.last() <= xmin)
112  {
113  // Mark as bad range, cannot iterate
114  *cmin = 0;
115  *cmax = -1;
116 
117  // Another bad range (cannot iterate) but for extra safety ensure
118  // that (cfmin -> cmin) and (cmax -> cfmax) cannot iterate either
119  *cfmin = 1;
120  *cfmax = -2;
121  return;
122  }
123 
124  // Ensure search is within the (point) bounds
125  xmin = grid.clip(xmin);
126  xmax = grid.clip(xmax);
127 
128  // The begin/end of the obstacle
129  *cmin = grid.findCell(xmin);
130  *cmax = grid.findCell(xmax);
131 
132  for (label ix = *cmin; ix <= *cmax; ++ix)
133  {
134  olap[ix] = 1.0;
135  }
136 
137  // Fixup ends
138  if (*cmax == *cmin)
139  {
140  olap[*cmax] = (xmax - xmin) / grid.width(*cmax);
141  }
142  else
143  {
144  if (grid[*cmin] < xmin)
145  {
146  olap[*cmin] = (grid[*cmin+1] - xmin) / grid.width(*cmin);
147  }
148 
149  if (xmax < grid[*cmax+1])
150  {
151  olap[*cmax] = (xmax - grid[*cmax]) / grid.width(*cmax);
152  }
153  }
154  assert(olap[*cmax] >= 0.0);
155 
156 
157  // Is xmin below/above the cell-centre (for virtual staggered-grid) ?
158  *cfmin =
159  (
160  xmin < grid.C(*cmin)
161  ? *cmin
162  : Foam::min(*cmin+1, grid.nCells()-1)
163  );
164 
165  // Is xmax below/above the cell-centre (for virtual staggered-grid) ?
166  *cfmax =
167  (
168  xmax < grid.C(*cmax)
169  ? *cmax
170  : Foam::min(*cmax+1, grid.nCells()-1)
171  );
172 }

206  scalar ac, scalar bc, scalar dia,
207  scalar theta, scalar wa, scalar wb,

209  const PDRblock::location& bgrid, label bmin, label bmax,

215  SquareMatrix<symmTensor2D>& c_drag,
216  SquareMatrix<scalar>& b_lblock,
217  SquareMatrix<scalar>& bc_lblock
218 )
219 {
220  /* This routine calculates the proportion of each (two-dimensional) grid cell
221  overlapped by the circle or angled rectangle. Coordinates are labelled a and b.
222  On entry:
223  ac, bc coordinates of centre of circle or rectangle
224  dia diameter of circle (zero for rectangle)
225  theta, wa, wb parameters for rectangle
226  agrid[] locations of grid lines of a-grid
227  amin, amax first and last cells in a-grid overlapped by object
228  (similarly for b)
229  On exit:
230  abolap 2-D array of (proportionate) area blockage by grid cell
231  a_lblock 2-D array of (proportionate) blockage to a-direction flow
232  (This will be area blockage when extruded in the third coordinate).
233  a_count (2-D array)The contribution of this object to the count of obstacles blocking
234  a-direction flow. This is only non-zero if the object is inside the
235  lateral boundaries of the cell. It is large negative if the cell is
236  totally blocked in this direction.
237  (similarly for b)
238  c_drag 2-D array of tensor that will give tensor drag in each cell (when multiplied
239  Cd, cylinder length, and 0.5 rho*U^2) Dimension: L.
240 
241  Note that this routine does not zero array elements outside the amin to amax, bmin to bmax area.
242  */
243 
244  scalar count, a_lblk, b_lblk, perim, dummy;
245 
246  symmTensor2D vdrag(Zero);
247 
248  // Prevent stepping outside of the array when the obstacle is on the
249  // upper boundary
250 
251  // Upper limit of inclusive range is nCells-1
252  amin = Foam::max(0, amin);
253  bmin = Foam::max(0, bmin);
254  amax = Foam::min(amax, agrid.nCells()-1);
255  bmax = Foam::min(bmax, bgrid.nCells()-1);
256 
257  for (label ia = amin; ia <= amax; ++ia)
258  {
259  // Cell-centred grid
260  const scalar a1 = agrid[ia];
261  const scalar a2 = agrid[ia+1];
262 
263  // Left-shifted staggered face grid (-1 addressing is OK)
264  const scalar af1 = agrid.C(ia-1);
265  const scalar af2 = agrid.C(ia);
266 
267  for (label ib = bmin; ib <= bmax; ++ib)
268  {
269  // Cell-centred grid
270  const scalar b1 = bgrid[ib];
271  const scalar b2 = bgrid[ib+1];
272 
273  // Left-shifted staggered face grid (-1 addressing is OK)
274  const scalar bf1 = bgrid.C(ib-1);
275  const scalar bf2 = bgrid.C(ib);
276 
277  // Do the centred cell
278  if ( dia > 0.0 )
279  {
280  ab_olap(ia,ib) = inters_cy
281  (
282  ac, bc, 0.5*dia, a1, a2, b1, b2, &perim,
283  &dummy, &dummy, &b_lblk, &a_lblk
284  );
285 /* The last two arguments of the above call appear to be reversed, but the inters_cy routine returns
286  the amount of overlap in the a and b direcvtions, which are the blockage to the b and a directions. */
287 
288 /* abolap * cell area is area of cylinder in this cell. Divide by PI%D^2/4 to get proportion of cylinder in cell
289  For whole cylinger c_drag should be = D, so multiply by D. */
290 
291  c_drag(ia,ib).xx() = c_drag(ia,ib).yy() = 4.0 * ab_olap(ia,ib) * (a2 - a1) * (b2 - b1) / dia / mathematical::pi;
292  c_drag(ia,ib).xy() = Zero;
293  c_count(ia,ib) = perim / (mathematical::pi * dia);
294 
295 //******?????
296  scalar area = (a2 - a1) * (b2 - b1);
297  scalar rat = dia * dia / area - 1.5;
298  if (rat > 0.0)
299  {
300  scalar da = ac - 0.5 * (a1 + a2);
301  scalar db = bc - 0.5 * (b1 + b2);
302  scalar dc = std::hypot(da, db);
303  scalar rat1 = min(max((dc / sqrt(area) - 0.3) * 1.4, 0), 1);
304  scalar drg0 = c_drag(ia,ib).xx();
305  scalar drg1 = c_drag(ia,ib).yy();
306  scalar drg = std::hypot(drg0, drg1);
307  c_drag(ia,ib).xx() = drg * ( 1.0 - rat1 ) + drg * da*da/dc/dc * rat1;
308  c_drag(ia,ib).yy() = drg * ( 1.0 - rat1 ) + drg * db*db/dc/dc * rat1;
309  c_drag(ia,ib).xy() = drg * da*db/dc/dc *rat1;
310  }
311  }
312  else
313  {
314  ab_olap(ia,ib) = inters_db( ac, bc, theta, wa, wb, a1, a2, b1, b2, &count, c_drag(ia,ib), &perim, &a_lblk, &b_lblk, &dummy, &dummy );
315  c_count(ia,ib) = perim / ( wa + wb ) * 0.5;
316  }
317  ac_lblock(ia,ib) = a_lblk;
318  bc_lblock(ia,ib) = b_lblk;
319  ab_perim(ia,ib) = perim;
320 
321  // Do the a-shifted cell
322  if ( dia > 0.0 ) // I.e. a cylinder, not a d.b.
323  {
324  if (ac >= af1 && ac < af2)
325  {
326  // Only want to block one layer of faces
327  a_lblock(ia,ib) = l_blockage
328  (
329  ac, bc, 0.5*dia,
330  af1, af2, b1, b2, &count, &dummy, &dummy
331  );
332  }
333  inters_cy
334  (
335  ac, bc, 0.5*dia,
336  af1, af2, b1, b2,
337  &perim, &count, &dummy, &dummy, &dummy
338  );
339  }
340  else
341  {
342  inters_db
343  (
344  ac, bc, theta, wa, wb, af1, af2, b1, b2,
345  &count, vdrag, &dummy, &a_lblk, &b_lblk, &dummy, &dummy
346  );
347  a_lblock(ia,ib) = a_lblk;
348  }
349 
350  // Do the b-shifted cell
351  if ( dia > 0.0 )
352  {
353  if (bc >= bf1 && bc < bf2)
354  {
355  // Only want to block one layer of faces
356  b_lblock(ia,ib) = l_blockage
357  (
358  bc, ac, 0.5*dia, bf1, bf2, a1, a2,
359  &count, &(vdrag.yy()), &dummy
360  );
361  }
362 
363  inters_cy
364  (
365  ac, bc, 0.5*dia,
366  a1, a2, bf1, bf2,
367  &perim, &dummy, &count, &dummy, &dummy
368  );
369  }
370  else
371  {
372  inters_db
373  (
374  ac, bc, theta, wa, wb,
375  a1, a2, bf1, bf2,
376  &count, vdrag, &dummy, &a_lblk, &b_lblk, &dummy, &dummy
377  );
378  b_lblock(ia,ib) = b_lblk;
379  }
380  }
381  }

388 scalar block_overlap
389 (
390  DynamicList<PDRobstacle>& blocks,
391  const labelRange& range,
392  const scalar multiplier
393 )
394 {
395  // Size information
396  const label nBlock = range.size();
397 
398  // The return value
399  scalar totVolume = 0;
400 
401  if (nBlock < 2) return 0;
402 
403 
404  // Sort blocks by their x-position (with sortBias)
405  labelList blkOrder;
406  sortedOrder(blocks[range], blkOrder);
407 
408  DynamicList<PDRobstacle> newBlocks;
409 
410  // Work through the sorted blocks
411  for (label i1 = 0; i1 < nBlock-1; ++i1)
412  {
413  const PDRobstacle& blk1 = blocks[range[blkOrder[i1]]];
414 
415  // Upper coordinates
416  const vector max1 = blk1.pt + blk1.span;
417 
418  // For second block start with the next one on the list, and
419  // stop when we find the first one whose biased x-position
420  // is beyond the end of the block1
421 
422  for (label i2 = i1 + 1; i2 < nBlock; ++i2)
423  {
424  const PDRobstacle& blk2 = blocks[range[blkOrder[i2]]];
425 
426  // Upper coordinates
427  const vector max2 = blk2.pt + blk2.span;
428 
429  if (max1.x() <= blk2.x())
430  {
431  break;
432  }
433 
434  if
435  (
436  max1.y() <= blk2.y()
437  || max1.z() <= blk2.z()
438  || max2.y() <= blk1.y()
439  || max2.z() <= blk1.z()
440  || (blk1.vbkge * blk2.vbkge <= 0)
441  )
442  {
443  continue;
444  }
445 
446 
447  {
448  PDRobstacle over;
449 
450  over.pt = max(blk1.pt, blk2.pt);
451  over.span = min(max1, max2) - over.pt;
452 
453  assert(cmptProduct(over.span) > 0.0);
454 
455  // This routine should only have been called for all +ve o r all -ve obstacles
456  assert(blk1.vbkge * blk2.vbkge > 0);
457  /* At the first level of intersection, we create an obstacle of blockage -1 (if both objects solid)
458  to cancel out the double counting. (multiplier is 1).
459  ?? COMBLK does a (sign corrected) multiply; is this corrrect for porous obstacles?
460  Depends on how blockages were summed in the first place. In fact this -ve obstacle
461  concept only works if the blockages are summed??*/
462  over.vbkge = - COMBLK( blk1.vbkge, blk2.vbkge ) * multiplier;
463  over.xbkge = - COMBLK( blk1.xbkge, blk2.xbkge ) * multiplier;
464  over.ybkge = - COMBLK( blk1.ybkge, blk2.ybkge ) * multiplier;
465  over.zbkge = - COMBLK( blk1.zbkge, blk2.zbkge ) * multiplier;
466  over.typeId = 81 + int(15 * multiplier); // Not subsequently used
467 
468  if (obsHasMinSize(over.span, pars))
469  {
470  // Obstacle satisfies some minimum size checks
471  totVolume -= over.volume();
472 
473  newBlocks.append(over);
474  }
475  }
476  }
477  }
478 
479  blocks.append(std::move(newBlocks));
480 
481  return totVolume;
482 }

527  // need to be considered

531  {
532  ++i2;
533  }
534

536  {

538

576  blk1.y(), max1.y(),
577  &dummy, &dummy, &b_centre
578  );

580  (
581  cyl2.y(), cyl2.x(), 0.5*cyl2.dia(),
582  blk1.y(), max1.y(),
583  blk1.x(), max1.x(),
584  &dummy, &dummy, &a_centre
585  );
586  }

607  over.span.y() = b_lblk;
608  over.span.z() = min(max1.z(), cyl2.z() + cyl2.len()) - over.z();
609  assert(over.x() > -200.0);
610  assert(over.x() < 2000.0);
611  }
612  break;
613

615  {
616  const scalar ym2 = cyl2.y() + cyl2.len();
617  if (blk1.y() > ym2 || cyl2.y() > max1.y()) continue;
618 
619  if ( cyl2.dia() == 0.0 )
620  {
621  area = inters_db
622  (
623  cyl2.z(), cyl2.x(), cyl2.theta(), cyl2.wa, cyl2.wb,
624  blk1.z(), max1.z(),
625  blk1.x(), max1.x(),
626  &dummy, dum2, &dummy, &a_lblk, &b_lblk,
627  &a_centre, &b_centre
628  );
629  }
630  else
631  {
632  area = inters_cy
633  (
634  cyl2.z(), cyl2.x(), 0.5*cyl2.dia(),
635  blk1.z(), max1.z(),
636  blk1.x(), max1.x(),
637  &dummy, &dummy, &dummy, &dummy, &dummy
638  );
639 
640  b_lblk = l_blockage
641  (
642  cyl2.z(), cyl2.x(), 0.5*cyl2.dia(),
643  blk1.z(), max1.z(),
644  blk1.x(), max1.x(),
645  &dummy, &dummy, &b_centre
646  );
647

652  blk1.z(), max1.z(),

654  );

661  // a_lblk and b_lblk are dimensions of enclosing rectangle.
662  // Need to scale to correct area
663  const scalar ratio = std::sqrt( area / a_lblk / b_lblk );
664  a_lblk *= blk1.span.z() * ratio;
665  b_lblk *= blk1.span.x() * ratio;
666 
667  over.z() = a_centre - a_lblk * 0.5;
668  over.x() = b_centre - b_lblk * 0.5;
669  over.y() = max(blk1.y(), cyl2.y());
670 
671  over.span.z() = a_lblk;
672  over.span.x() = b_lblk;
673  over.span.y() = min(max1.y(), cyl2.y() + cyl2.len()) - over.y();
674  }
675  break;
676 
677  case vector::X:
678  {
679  const scalar xm2 = cyl2.x() + cyl2.len();
680  if (blk1.x() > xm2 || cyl2.x() > max1.x()) continue;
681 
682  if ( cyl2.dia() == 0.0 )
683  {
684  area = inters_db
685  (
686  cyl2.y(), cyl2.z(), cyl2.theta(), cyl2.wa, cyl2.wb,
687  blk1.y(), max1.y(),
688  blk1.z(), max1.z(),
689  &dummy, dum2, &dummy, &a_lblk, &b_lblk,
690  &a_centre, &b_centre
691  );
692  }
693  else
694  {
695  area = inters_cy
696  (
697  cyl2.y(), cyl2.z(), 0.5*cyl2.dia(),
698  blk1.y(), max1.y(),
699  blk1.z(), max1.z(),
700  &dummy, &dummy, &dummy, &dummy, &dummy
701  );
702 
703  b_lblk = l_blockage
704  (
705  cyl2.y(), cyl2.z(), 0.5*cyl2.dia(),
706  blk1.y(), max1.y(),
707  blk1.z(), max1.z(),
708  &dummy, &dummy, &b_centre
709  );
710 
711  a_lblk = l_blockage
712  (
713  cyl2.z(), cyl2.y(), 0.5*cyl2.dia(),
714  blk1.z(), max1.z(),
715  blk1.y(), max1.y(),
716  &dummy, &dummy, &a_centre
717  );
718 
719  }
720 
721  if (equal(area, 0)) continue;
722  assert(a_lblk >0.0);
723  assert(b_lblk >0.0);
724 
725  // a_lblk and b_lblk are dimensions of enclosing rectangle.
726  // Need to scale to correct area
727  const scalar ratio = std::sqrt( area / a_lblk / b_lblk );
728  assert(ratio >-10000.0);
729  assert(ratio <10000.0);
730  a_lblk *= blk1.span.y() * ratio;
731  b_lblk *= blk1.span.z() * ratio;
732 
733  over.y() = a_centre - a_lblk * 0.5;
734  over.z() = b_centre - b_lblk * 0.5;
735  over.x() = max(blk1.x(), cyl2.x());
736 
737  over.span.y() = a_lblk;
738  over.span.z() = b_lblk;
739  over.span.x() = min(max1.x(), cyl2.x() + cyl2.len()) - over.x();
740  }
741  break;
742  }
743  over.vbkge = over.xbkge = over.ybkge = over.zbkge = -1.0;
744  over.typeId = PDRobstacle::IGNORE;
745 
746  assert(cmptProduct(over.span) > 0.0);
747  assert(b_lblk >0.0);
748  assert(a_lblk >0.0);
749  assert(over.x() > -10000.0);
750 
751  if (obsHasMinSize(over.span, pars))
752  {
753  // Obstacle satisfies some minimum size checks
754  totVolume -= over.volume();
755 
756  newBlocks.append(over);
757  }
758  }
759  }
760 
761  blocks.append(std::move(newBlocks));
762 
763  return totVolume;
764 }
