252  {

254  const HashTable<labelList>& groups = patches.groupPatchIDs();
255  DynamicList<string> displayNames(groups.size());
256 
257  // Add (non-zero) patch groups to the list of mesh parts
258  // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
259

261  {
262  const auto& groupName = iter.key();
263  const auto& patchIDs = iter.val();
264

266  for (auto patchId : patchIDs)
267  {
268  nFaces += patches[patchId].size();
269  }
270 
271  if (!nFaces)
272  {
273  // Skip if group has no faces
274  continue;
275  }
276 
277  // Valid patch if nFace > 0 - add patch to GUI list
278  const string dpyName = "group/" + groupName;
279  displayNames.append(dpyName);
280 
281  // Optionally replace group with patch name selections
282  // - must remove the group from the select itself, otherwise
283  // it can toggle on, but not toggle off very well
284  if
285  (
286  !reader_->GetShowGroupsOnly()
287  && enabledEntries.erase(dpyName)
288  )
289  {
290  for (auto patchId : patchIDs)
291  {
292  const polyPatch& pp = patches[patchId];
293  if (pp.size())
294  {
295  enabledEntries.insert("patch/" + pp.name());
296  }
297  }
298  }
299  }
300 
301  Foam::sort(displayNames); // Sorted group names
302  rangePatches_ += addToArray(select, displayNames);
303 
304  // Add (non-zero) patches to the list of mesh parts
305  // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
306 
307  if (!reader_->GetShowGroupsOnly())
308  {
309  for (const polyPatch& pp : patches)
310  {
311  if (pp.size())
312  {
313  // Add patch to GUI list
314  select->AddArray(("patch/" + pp.name()).c_str());
315  ++rangePatches_;
316  }
317  }
318  }
319  }
320  else
321  {
322  // mesh not loaded - read from file
323  // but this could fail if we've supplied a bad region name
324  IOobject ioObj
325  (
326  "boundary",
327  dbPtr_().findInstance
328  (
329  meshDir_,
330  "boundary",
331  IOobject::READ_IF_PRESENT
332  ),
333  meshDir_,
334  dbPtr_(),
335  IOobject::READ_IF_PRESENT,
336  IOobject::NO_WRITE,
337  false
338  );
339 
340  // This should only ever fail if the mesh region doesn't exist
341  if (ioObj.typeHeaderOk<polyBoundaryMesh>(true, false))
342  {
343  polyBoundaryMeshEntries patchEntries(ioObj);
344

353  wordList groupNames;
354 
355  sizes[patchi] = patchDict.get<label>("nFaces");
356  names[patchi] = patchEntries[patchi].keyword();
357 
358  if
359  (
360  sizes[patchi] // Valid patch if nFace > 0
361  && patchDict.readIfPresent("inGroups", groupNames)
362  )
363  {
364  for (const auto& groupName : groupNames)
365  {
366  groups(groupName).insert(patchi);
367  }
368  }
369  }
370 
371 
372  // Add (non-zero) patch groups to the list of mesh parts
373  // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
374  DynamicList<string> displayNames(groups.size());
375 
376  forAllConstIters(groups, iter)
377  {
378  const auto& groupName = iter.key();
379  const auto& patchIDs = iter.val();
380 
381  const string dpyName = "group/" + groupName;
382  displayNames.append(dpyName);
383 
384  // Optionally replace group with patch name selections
385  // - must remove the group from the select itself, otherwise
386  // it can toggle on, but not toggle off very well
387  if
388  (
389  !reader_->GetShowGroupsOnly()
390  && enabledEntries.erase(dpyName)
391  )
392  {
393  for (auto patchId : patchIDs)
394  {
395  enabledEntries.insert("patch/" + names[patchId]);
396  }
397  }
398  }
399 
400  Foam::sort(displayNames); // Sorted group names
401  rangePatches_ += addToArray(select, displayNames);
402 
403  // Add (non-zero) patches to the list of mesh parts
404  // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
405 
406  if (!reader_->GetShowGroupsOnly())
407  {
408  forAll(names, patchi)
409  {
410  // Valid patch if nFace > 0 - add patch to GUI list
411  if (sizes[patchi])
412  {
413  select->AddArray(("patch/" + names[patchi]).c_str());
414  ++rangePatches_;
415  }
416  }
417  }
418  }
419  }
