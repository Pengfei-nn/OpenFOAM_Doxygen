385 void extractSurface
386 (
387  const polyMesh& mesh,
388  const Time& runTime,
389  const labelHashSet& includePatches,
390  const fileName& outFileName
391 )
392 {
393  const polyBoundaryMesh& bMesh = mesh.boundaryMesh();
394 
395  // Collect sizes. Hash on names to handle local-only patches (e.g.
396  // processor patches)
397  HashTable<label> patchSize(1024);
398  label nFaces = 0;
399  for (const label patchi : includePatches)
400  {
401  const polyPatch& pp = bMesh[patchi];
402  patchSize.insert(pp.name(), pp.size());
403  nFaces += pp.size();
404  }
405  Pstream::mapCombineGather(patchSize, plusEqOp<label>());
406 
407 
408  // Allocate zone/patch for all patches
409  HashTable<label> compactZoneID(1024);
410  forAllConstIters(patchSize, iter)
411  {
412  label sz = compactZoneID.size();
413  compactZoneID.insert(iter.key(), sz);
414  }
415  Pstream::mapCombineScatter(compactZoneID);
416 
417 
418  // Rework HashTable into labelList just for speed of conversion
419  labelList patchToCompactZone(bMesh.size(), -1);
420  forAllConstIters(compactZoneID, iter)
421  {
422  label patchi = bMesh.findPatchID(iter.key());
423  if (patchi != -1)
424  {
425  patchToCompactZone[patchi] = iter();
426  }
427  }
428 
429  // Collect faces on zones
430  DynamicList<label> faceLabels(nFaces);
431  DynamicList<label> compactZones(nFaces);
432  for (const label patchi : includePatches)
433  {
434  const polyPatch& pp = bMesh[patchi];
435  forAll(pp, i)
436  {
437  faceLabels.append(pp.start()+i);
438  compactZones.append(patchToCompactZone[pp.index()]);
439  }
440  }
441 
442  // Addressing engine for all faces
443  uindirectPrimitivePatch allBoundary
444  (
445  UIndirectList<face>(mesh.faces(), faceLabels),
446  mesh.points()
447  );
448 
449 
450  // Find correspondence to master points
451  labelList pointToGlobal;
452  labelList uniqueMeshPoints;
453  autoPtr<globalIndex> globalNumbers = mesh.globalData().mergePoints
454  (
455  allBoundary.meshPoints(),
456  allBoundary.meshPointMap(),
457  pointToGlobal,
458  uniqueMeshPoints
459  );
460 
461  // Gather all unique points on master
462  List<pointField> gatheredPoints(Pstream::nProcs());
463  gatheredPoints[Pstream::myProcNo()] = pointField
464  (
465  mesh.points(),
466  uniqueMeshPoints
467  );
468  Pstream::gatherList(gatheredPoints);
469 
470  // Gather all faces
471  List<faceList> gatheredFaces(Pstream::nProcs());
472  gatheredFaces[Pstream::myProcNo()] = allBoundary.localFaces();
473  forAll(gatheredFaces[Pstream::myProcNo()], i)
474  {
475  inplaceRenumber(pointToGlobal, gatheredFaces[Pstream::myProcNo()][i]);
476  }
477  Pstream::gatherList(gatheredFaces);
478 
479  // Gather all ZoneIDs
480  List<labelList> gatheredZones(Pstream::nProcs());
481  gatheredZones[Pstream::myProcNo()].transfer(compactZones);
482  Pstream::gatherList(gatheredZones);
483 
484  // On master combine all points, faces, zones
485  if (Pstream::master())
486  {
487  pointField allPoints = ListListOps::combine<pointField>
488  (
489  gatheredPoints,
490  accessOp<pointField>()
491  );
492  gatheredPoints.clear();
493 
494  faceList allFaces = ListListOps::combine<faceList>
495  (
496  gatheredFaces,
497  accessOp<faceList>()
498  );
499  gatheredFaces.clear();
500 
501  labelList allZones = ListListOps::combine<labelList>
502  (
503  gatheredZones,
504  accessOp<labelList>()
505  );
506  gatheredZones.clear();
507 
508 
509  // Zones
510  surfZoneIdentifierList surfZones(compactZoneID.size());
511  forAllConstIters(compactZoneID, iter)
512  {
513  surfZones[iter()] = surfZoneIdentifier(iter.key(), iter());
514  Info<< "surfZone " << iter() << " : " << surfZones[iter()].name()
515  << endl;
516  }
517 
518  UnsortedMeshedSurface<face> unsortedFace
519  (
520  std::move(allPoints),
521  std::move(allFaces),
522  std::move(allZones),
523  surfZones
524  );
525 
526 
527  MeshedSurface<face> sortedFace(unsortedFace);
528 
529  fileName globalCasePath
530  (
531  runTime.processorCase()
532  ? runTime.globalPath()/outFileName
533  : runTime.path()/outFileName
534  );
535  globalCasePath.clean();
536 
537  Info<< "Writing merged surface to " << globalCasePath << endl;
538 
539  sortedFace.write(globalCasePath);
540  }

544 label checkAlignment(const polyMesh& mesh, const scalar tol, Ostream& os)
545 {
546  // Check all edges aligned with one of the coordinate axes
547  const faceList& faces = mesh.faces();
548  const pointField& points = mesh.points();
549 
550  label nUnaligned = 0;
551 
552  forAll(faces, facei)
553  {
554  const face& f = faces[facei];
555  forAll(f, fp)
556  {
557  label fp1 = f.fcIndex(fp);
558  const linePointRef e(edge(f[fp], f[fp1]).line(points));
559  const vector v(e.vec());
560  const scalar magV(mag(v));
561  if (magV > ROOTVSMALL)
562  {
563  for
564  (
565  direction dir = 0;
566  dir < pTraits<vector>::nComponents;
567  ++dir
568  )
569  {
570  const scalar s(mag(v[dir]));
571  if (s > magV*tol && s < magV*(1-tol))
572  {
573  ++nUnaligned;
574  break;
575  }
576  }
577  }
578  }
579  }
580 
581  reduce(nUnaligned, sumOp<label>());
582 
583  if (nUnaligned)
584  {
585  os << "Initial mesh has " << nUnaligned
586  << " edges unaligned with any of the coordinate axes" << nl << endl;
587  }
588  return nUnaligned;
