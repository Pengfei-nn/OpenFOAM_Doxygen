117 inline Foam::vector Foam::extendedEdgeMesh::edgeDirection
118 (
119  label edgeI,
120  label ptI
121 ) const
122 {
123  const edge& e = edges()[edgeI];
124 
125  if (ptI == e.start())
126  {
127  return edgeDirections()[edgeI];
128  }
129  else if (ptI == e.end())
130  {
131  return -edgeDirections()[edgeI];
132  }
133  else
134  {
135  FatalErrorInFunction
136  << "Requested ptI " << ptI << " is not a point on the requested "
137  << "edgeI " << edgeI << ". edgeI start and end: "
138  << e.start() << " " << e.end()
139  << exit(FatalError);
140 
141  return Zero;
142  }
143 }
