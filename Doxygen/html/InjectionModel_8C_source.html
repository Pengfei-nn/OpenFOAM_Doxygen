38 template<class CloudType>
39 bool Foam::InjectionModel<CloudType>::prepareForNextTimeStep
40 (
41  const scalar time,
42  label& newParcels,
43  scalar& newVolumeFraction
44 )
45 {
46  // Initialise values
47  newParcels = 0;
48  newVolumeFraction = 0.0;
49  bool validInjection = false;
50 
51  // Return if not started injection event
52  if (time < SOI_)
53  {
54  timeStep0_ = time;
55  return validInjection;
56  }
57 
58  // Make times relative to SOI
59  scalar t0 = timeStep0_ - SOI_;
60  scalar t1 = time - SOI_;
61 
62  // Number of parcels to inject
63  newParcels = this->parcelsToInject(t0, t1);
64 
65  // Volume of parcels to inject
66  newVolumeFraction =
67  this->volumeToInject(t0, t1)
68  /(volumeTotal_ + ROOTVSMALL);
69 
70  if (newVolumeFraction > 0)
71  {
72  if (newParcels > 0)
73  {
74  timeStep0_ = time;
75  validInjection = true;
76  }
77  else
78  {
79  // Injection should have started, but not sufficient volume to
80  // produce (at least) 1 parcel - hold value of timeStep0_
81  validInjection = false;
82  }
83  }
84  else
85  {
86  timeStep0_ = time;
87  validInjection = false;
88  }
89 
90  return validInjection;
91 }

94 template<class CloudType>
95 bool Foam::InjectionModel<CloudType>::findCellAtPosition
96 (
97  label& celli,
98  label& tetFacei,
99  label& tetPti,
100  vector& position,
101  bool errorOnNotFound
102 )
103 {
104  const volVectorField& cellCentres = this->owner().mesh().C();
105 
106  const vector p0 = position;
107 
108  this->owner().mesh().findCellFacePt
109  (
110  position,
111  celli,
112  tetFacei,
113  tetPti
114  );
115 
116  label proci = -1;
117 
118  if (celli >= 0)
119  {
120  proci = Pstream::myProcNo();
121  }
122 
123  reduce(proci, maxOp<label>());
124 
125  // Ensure that only one processor attempts to insert this Parcel
126 
127  if (proci != Pstream::myProcNo())
128  {
129  celli = -1;
130  tetFacei = -1;
131  tetPti = -1;
132  }
133 
134  // Last chance - find nearest cell and try that one - the point is
135  // probably on an edge
136  if (proci == -1)
137  {
138  celli = this->owner().mesh().findNearestCell(position);
139 
140  if (celli >= 0)
141  {
142  position += SMALL*(cellCentres[celli] - position);
143 
144  this->owner().mesh().findCellFacePt
145  (
146  position,
147  celli,
148  tetFacei,
149  tetPti
150  );
151 
152  if (celli > 0)
153  {
154  proci = Pstream::myProcNo();
155  }
156  }
157 
158  reduce(proci, maxOp<label>());
159 
160  if (proci != Pstream::myProcNo())
161  {
162  celli = -1;
163  tetFacei = -1;
164  tetPti = -1;
165  }
166  }
167 
168  if (proci == -1)
169  {
170  if (errorOnNotFound)
171  {
172  FatalErrorInFunction
173  << "Cannot find parcel injection cell. "
174  << "Parcel position = " << p0 << nl
175  << exit(FatalError);
176  }
177  else
178  {
179  return false;
180  }
181  }
182 
183  return true;
184 }

187 template<class CloudType>
188 Foam::scalar Foam::InjectionModel<CloudType>::setNumberOfParticles
189 (
190  const label parcels,
191  const scalar volumeFraction,
192  const scalar diameter,
193  const scalar rho
194 )
195 {
196  scalar nP = 0.0;
197  switch (parcelBasis_)
198  {
199  case pbMass:
200  {
201  scalar volumep = pi/6.0*pow3(diameter);
202  scalar volumeTot = massTotal_/rho;
203 
204  nP = (volumeFraction*volumeTot + delayedVolume_)/(parcels*volumep);
205  break;
206  }
207  case pbNumber:
208  {
209  nP = massTotal_/(rho*volumeTotal_);
210  break;
211  }
212  case pbFixed:
213  {
214  nP = nParticleFixed_;
215  break;
216  }
217  default:
218  {
219  nP = 0.0;
220  FatalErrorInFunction
221  << "Unknown parcelBasis type" << nl
222  << exit(FatalError);
223  }
224  }
225 
226  return nP;
227 }

246

248  parcelsAddedTotal_ += allParcelsAdded;
249

252 
253  // Update time for start of next injection
254  time0_ = this->owner().db().time().value();
255 
256  // Increment number of injections
257  nInjections_++;
258 }

429 template<class CloudType>
430 template<class TrackCloudType>
431 void Foam::InjectionModel<CloudType>::inject
432 (
433  TrackCloudType& cloud,
434  typename CloudType::parcelType::trackingData& td
435 )
436 {
437  if (!this->active())
438  {
439  return;
440  }
441 
442  const scalar time = this->owner().db().time().value();
443 
444  // Prepare for next time step
445  label parcelsAdded = 0;
446  scalar massAdded = 0.0;
447  label newParcels = 0;
448  scalar newVolumeFraction = 0.0;
449  scalar delayedVolume = 0;
450 
451  if (prepareForNextTimeStep(time, newParcels, newVolumeFraction))
452  {
453  const scalar trackTime = this->owner().solution().trackTime();
454  const polyMesh& mesh = this->owner().mesh();
455 
456  // Duration of injection period during this timestep
457  const scalar deltaT =
458  max(0.0, min(trackTime, min(time - SOI_, timeEnd() - time0_)));
459 
460  // Pad injection time if injection starts during this timestep
461  const scalar padTime = max(0.0, SOI_ - time0_);
462 
463  // Introduce new parcels linearly across carrier phase timestep
464  for (label parcelI = 0; parcelI < newParcels; parcelI++)
465  {
466  if (validInjection(parcelI))
467  {
468  // Calculate the pseudo time of injection for parcel 'parcelI'
469  scalar timeInj = time0_ + padTime + deltaT*parcelI/newParcels;
470 
471  // Determine the injection position and owner cell,
472  // tetFace and tetPt
473  label celli = -1;
474  label tetFacei = -1;
475  label tetPti = -1;
476 
477  vector pos = Zero;
478 
479  setPositionAndCell
480  (
481  parcelI,
482  newParcels,
483  timeInj,
484  pos,
485  celli,
486  tetFacei,
487  tetPti
488  );
489 
490  if (celli > -1)
491  {
492  // Lagrangian timestep
493  const scalar dt = time - timeInj;
494 
495  // Apply corrections to position for 2-D cases
496  meshTools::constrainToMeshCentre(mesh, pos);
497 
498  // Create a new parcel
499  parcelType* pPtr = new parcelType(mesh, pos, celli);
500 
501  // Check/set new parcel thermo properties
502  cloud.setParcelThermoProperties(*pPtr, dt);
503 
504  // Assign new parcel properties in injection model
505  setProperties(parcelI, newParcels, timeInj, *pPtr);
506 
507  // Check/set new parcel injection properties
508  cloud.checkParcelProperties(*pPtr, dt, fullyDescribed());
509 
510  // Apply correction to velocity for 2-D cases
511  meshTools::constrainDirection
512  (
513  mesh,
514  mesh.solutionD(),
515  pPtr->U()
516  );
517 
518  // Number of particles per parcel
519  pPtr->nParticle() =
520  setNumberOfParticles
521  (
522  newParcels,
523  newVolumeFraction,
524  pPtr->d(),
525  pPtr->rho()
526  );
527 
528  if (pPtr->nParticle() >= minParticlesPerParcel_)
529  {
530  parcelsAdded++;
531  massAdded += pPtr->nParticle()*pPtr->mass();
532 
533  if (pPtr->move(cloud, td, dt))
534  {
535  pPtr->typeId() = injectorID_;
536  cloud.addParticle(pPtr);
537  }
538  else
539  {
540  delete pPtr;
541  }
542  }
543  else
544  {
545  delayedVolume += pPtr->nParticle()*pPtr->volume();
546  delete pPtr;
547  }
548  }
549  }
550  }
551  }
552 
553  delayedVolume_ = returnReduce(delayedVolume, sumOp<scalar>());
554 
555  postInjectCheck(parcelsAdded, massAdded);
556 }

559 template<class CloudType>
560 template<class TrackCloudType>
561 void Foam::InjectionModel<CloudType>::injectSteadyState
562 (
563  TrackCloudType& cloud,
564  typename CloudType::parcelType::trackingData& td,
565  const scalar trackTime
566 )
567 {
568  if (!this->active())
569  {
570  return;
571  }
572 
573  const scalar time = this->owner().db().time().value();
574 
575  if (time < SOI_)
576  {
577  return;
578  }
579 
580  const polyMesh& mesh = this->owner().mesh();
581 
582  massTotal_ = massFlowRate_.value(mesh.time().value());
583 
584  // Reset counters
585  time0_ = 0.0;
586  label parcelsAdded = 0;
587  scalar massAdded = 0.0;
588 
589  // Set number of new parcels to inject based on first second of injection
590  label newParcels = parcelsToInject(0.0, 1.0);
591 
592  // Inject new parcels
593  for (label parcelI = 0; parcelI < newParcels; parcelI++)
594  {
595  // Volume to inject is split equally amongst all parcel streams
596  scalar newVolumeFraction = 1.0/scalar(newParcels);
597 
598  // Determine the injection position and owner cell,
599  // tetFace and tetPt
600  label celli = -1;
601  label tetFacei = -1;
602  label tetPti = -1;
603 
604  vector pos = Zero;
605 
606  setPositionAndCell
607  (
608  parcelI,
609  newParcels,
610  0.0,
611  pos,
612  celli,
613  tetFacei,
614  tetPti
615  );
616 
617  if (celli > -1)
618  {
619  // Apply corrections to position for 2-D cases
620  meshTools::constrainToMeshCentre(mesh, pos);
621 
622  // Create a new parcel
623  parcelType* pPtr = new parcelType(mesh, pos, celli);
624 
625  // Check/set new parcel thermo properties
626  cloud.setParcelThermoProperties(*pPtr, 0.0);
627 
628  // Assign new parcel properties in injection model
629  setProperties(parcelI, newParcels, 0.0, *pPtr);
630 
631  // Check/set new parcel injection properties
632  cloud.checkParcelProperties(*pPtr, 0.0, fullyDescribed());
633 
634  // Apply correction to velocity for 2-D cases
635  meshTools::constrainDirection(mesh, mesh.solutionD(), pPtr->U());
636 
637  // Number of particles per parcel
638  pPtr->nParticle() =
639  setNumberOfParticles
640  (
641  1,
642  newVolumeFraction,
643  pPtr->d(),
644  pPtr->rho()
645  );
646 
647  pPtr->typeId() = injectorID_;
648 
649  // Add the new parcel
650  cloud.addParticle(pPtr);
651 
652  massAdded += pPtr->nParticle()*pPtr->mass();
653  parcelsAdded++;
654  }
655  }
656 
657  postInjectCheck(parcelsAdded, massAdded);
658 }
