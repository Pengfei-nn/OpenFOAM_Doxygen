26 Description
27  Calulate the face centres and areas.
28 
29  Calculate the centre by breaking the face into triangles using the face
30  centre and area-weighted averaging their centres. This method copes with
31  small face-concavity.

75 void Foam::primitiveMesh::makeFaceCentresAndAreas
76 (
77  const pointField& p,
78  vectorField& fCtrs,
79  vectorField& fAreas
80 ) const
81 {
82  const faceList& fs = faces();
83 
84  forAll(fs, facei)
85  {
86  const labelList& f = fs[facei];
87  const label nPoints = f.size();
88 
89  // If the face is a triangle, do a direct calculation for efficiency
90  // and to avoid round-off error-related problems
91  if (nPoints == 3)
92  {
93  fCtrs[facei] = (1.0/3.0)*(p[f[0]] + p[f[1]] + p[f[2]]);
94  fAreas[facei] = 0.5*((p[f[1]] - p[f[0]])^(p[f[2]] - p[f[0]]));
95  }
96  else
97  {
98  typedef Vector<solveScalar> solveVector;
99 
100  solveVector sumN = Zero;
101  solveScalar sumA = 0.0;
102  solveVector sumAc = Zero;
103 
104  solveVector fCentre = p[f[0]];
105  for (label pi = 1; pi < nPoints; pi++)
106  {
107  fCentre += solveVector(p[f[pi]]);
108  }
109 
110  fCentre /= nPoints;
111 
112  for (label pi = 0; pi < nPoints; pi++)
113  {
114  const label nextPi(pi == nPoints-1 ? 0 : pi+1);
115  const solveVector nextPoint(p[f[nextPi]]);
116  const solveVector thisPoint(p[f[pi]]);
117 
118  solveVector c = thisPoint + nextPoint + fCentre;
119  solveVector n = (nextPoint - thisPoint)^(fCentre - thisPoint);
120  solveScalar a = mag(n);
121  sumN += n;
122  sumA += a;
123  sumAc += a*c;
124  }
125 
126  // This is to deal with zero-area faces. Mark very small faces
127  // to be detected in e.g., processorPolyPatch.
128  if (sumA < ROOTVSMALL)
129  {
130  fCtrs[facei] = fCentre;
131  fAreas[facei] = Zero;
132  }
133  else
134  {
135  fCtrs[facei] = (1.0/3.0)*sumAc/sumA;
136  fAreas[facei] = 0.5*sumN;
137  }
138  }
139  }
140 }
