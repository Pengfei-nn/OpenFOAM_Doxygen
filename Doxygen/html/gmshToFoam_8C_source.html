32 Description
33  Reads .msh file as written by Gmsh.
34 
35  Needs surface elements on mesh to be present and aligned with outside faces
36  of the mesh. I.e. if the mesh is hexes, the outside faces need to be
37  quads.
38 
39  Note: There is something seriously wrong with the ordering written in the
40  .msh file. Normal operation is to check the ordering and invert prisms
41  and hexes if found to be wrong way round.
42  Use the -keepOrientation to keep the raw information.
43 
44  Note: The code now uses the element (cell,face) physical region id number
45  to create cell zones and faces zones (similar to
46  fluentMeshWithInternalFaces).
47 
48  A use of the cell zone information, is for field initialization with the
49  "setFields" utility. see the classes: topoSetSource, zoneToCell.

113 // Find face in pp which uses all vertices in meshF (in mesh point labels)
114 label findFace(const primitivePatch& pp, const labelList& meshF)
115 {
116  const Map<label>& meshPointMap = pp.meshPointMap();
117 
118  // meshF[0] in pp labels.
119  if (!meshPointMap.found(meshF[0]))
120  {
121  Warning<< "Not using gmsh face " << meshF
122  << " since zero vertex is not on boundary of polyMesh" << endl;
123  return -1;
124  }
125 
126  // Find faces using first point
127  const labelList& pFaces = pp.pointFaces()[meshPointMap[meshF[0]]];
128 
129  // Go through all these faces and check if there is one which uses all of
130  // meshF vertices (in any order ;-)
131  forAll(pFaces, i)
132  {
133  label facei = pFaces[i];
134 
135  const face& f = pp[facei];
136 
137  // Count uses of vertices of meshF for f
138  label nMatched = 0;
139 
140  forAll(f, fp)
141  {
142  if (meshF.found(f[fp]))
143  {
144  nMatched++;
145  }
146  }
147 
148  if (nMatched == meshF.size())
149  {
150  return facei;
151  }
152  }
153 
154  return -1;
155 }

189 // Determine whether cell is inside-out by checking for any wrong-oriented
190 // face.
191 bool correctOrientation(const pointField& points, const cellShape& shape)
192 {
193  // Get centre of shape.
194  const point cc(shape.centre(points));
195 
196  // Get outwards pointing faces.
197  faceList faces(shape.faces());
198 
199  for (const face& f : faces)
200  {
201  const vector areaNorm(f.areaNormal(points));
202 
203  // Check if vector from any point on face to cc points outwards
204  if (((points[f[0]] - cc) & areaNorm) < 0)
205  {
206  // Incorrectly oriented
207  return false;
208  }
209  }
210 
211  return true;
212 }
