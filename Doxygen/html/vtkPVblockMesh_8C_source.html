171 Foam::vtkPVblockMesh::vtkPVblockMesh
172 (
173  const char* const FileName,
174  vtkPVblockMeshReader* reader
175 )
176 :
177  reader_(reader),
178  dbPtr_(nullptr),
179  meshPtr_(nullptr),
180  meshRegion_(polyMesh::defaultRegion),
181  meshDir_(polyMesh::meshSubDir),
182  rangeBlocks_("block"),
183  rangeEdges_("edges"),
184  rangeCorners_("corners")
185 {
186  DebugInfo << "vtkPVblockMesh - " << FileName << endl;
187 
188  // avoid argList and get rootPath/caseName directly from the file
189  fileName fullCasePath(fileName(FileName).path());
190 
191  if (!isDir(fullCasePath))
192  {
193  return;
194  }
195  if (fullCasePath == ".")
196  {
197  fullCasePath = cwd();
198  }
199 
200  // OPENFOAM API
201  setEnv("FOAM_API", std::to_string(foamVersion::api), true);
202 
203  // The name of the executable, unless already present in the environment
204  setEnv("FOAM_EXECUTABLE", "paraview", false);
205 
206  // Set the case as an environment variable - some BCs might use this
207  if (fullCasePath.name().find("processor", 0) == 0)
208  {
209  const fileName globalCase = fullCasePath.path();
210 
211  setEnv("FOAM_CASE", globalCase, true);
212  setEnv("FOAM_CASENAME", globalCase.name(), true);
213  }
214  else
215  {
216  setEnv("FOAM_CASE", fullCasePath, true);
217  setEnv("FOAM_CASENAME", fullCasePath.name(), true);
218  }
219 
220  // look for 'case{region}.OpenFOAM'
221  // could be stringent and insist the prefix match the directory name...
222  // Note: cannot use fileName::name() due to the embedded '{}'
223  string caseName(fileName(FileName).lessExt());
224  string::size_type beg = caseName.find_last_of("/{");
225  string::size_type end = caseName.find('}', beg);
226 
227  if
228  (
229  beg != string::npos && caseName[beg] == '{'
230  && end != string::npos && end == caseName.size()-1
231  )
232  {
233  meshRegion_ = caseName.substr(beg+1, end-beg-1);
234 
235  // some safety
236  if (meshRegion_.empty())
237  {
238  meshRegion_ = polyMesh::defaultRegion;
239  }
240 
241  if (meshRegion_ != polyMesh::defaultRegion)
242  {
243  meshDir_ = meshRegion_/polyMesh::meshSubDir;
244  }
245  }
246 
247  DebugInfo
248  << "fullCasePath=" << fullCasePath << nl
249  << "FOAM_CASE=" << getEnv("FOAM_CASE") << nl
250  << "FOAM_CASENAME=" << getEnv("FOAM_CASENAME") << endl;
251 
252  // Create time object
253  dbPtr_.reset
254  (
255  new Time
256  (
257  Time::controlDictName,
258  fileName(fullCasePath.path()),
259  fileName(fullCasePath.name())
260  )
261  );
262 
263  dbPtr_().functionObjects().off();
264 
265  updateInfo();
266 }

281 void Foam::vtkPVblockMesh::updateInfo()
282 {
283  DebugInfo
284  << "<beg> updateInfo"
285  << " [meshPtr=" << (meshPtr_ ? "set" : "null") << "] " << endl;
286 
287  resetCounters();
288 
289  vtkDataArraySelection* blockSelection = reader_->GetBlockSelection();
290  vtkDataArraySelection* edgeSelection = reader_->GetCurvedEdgesSelection();
291 
292  const bool firstTime = (!blockSelection->GetNumberOfArrays() && !meshPtr_);
293 
294  // Preserve the enabled selections if possible
295  HashSet<string> enabledParts;
296  HashSet<string> enabledEdges;
297  if (!firstTime)
298  {
299  enabledParts = getSelectedArraySet(blockSelection);
300  enabledEdges = getSelectedArraySet(edgeSelection);
301  }
302 
303  // Clear current mesh parts list
304  blockSelection->RemoveAllArrays();
305  edgeSelection->RemoveAllArrays();
306 
307  // need a blockMesh
308  updateFoamMesh();
309 
310  // Update mesh parts list
311  updateInfoBlocks(blockSelection);
312 
313  // Update curved edges list
314  updateInfoEdges(edgeSelection);
315 
316  // Restore the enabled selections
317  if (!firstTime)
318  {
319  setSelectedArrayEntries(blockSelection, enabledParts);
320  setSelectedArrayEntries(edgeSelection, enabledEdges);
321  }
322 
323  DebugInfo << "<end> updateInfo" << endl;
324 }

408 void Foam::vtkPVblockMesh::renderPatchNames
409 (
410  vtkRenderer* renderer,
411  const bool show
412 )
413 {
414  // always remove old actors first
415  forAll(patchTextActors_, actori)
416  {
417  renderer->RemoveViewProp(patchTextActors_[actori]);
418  }
419  patchTextActors_.clear();
420 
421  // the number of text actors
422  label nActors = 0;
423 
424  if (show && meshPtr_)
425  {
426  const blockMesh& blkMesh = *meshPtr_;
427  const dictionary& meshDescription = blkMesh.meshDict();
428  const pointField& cornerPts = blkMesh.vertices();
429  const scalar scaleFactor = blkMesh.scaleFactor();
430 
431  if (!meshDescription.found("boundary"))
432  {
433  return;
434  }
435 
436  // 8 sides per block is plenty
437  patchTextActors_.setSize(8*blkMesh.size());
438 
439  // Collect all variables
440  dictionary varDict(meshDescription.subOrEmptyDict("namedVertices"));
441  varDict.merge(meshDescription.subOrEmptyDict("namedBlocks"));
442 
443  // Read like boundary file
444  const PtrList<entry> patchesInfo(meshDescription.lookup("boundary"));
445 
446  forAll(patchesInfo, patchi)
447  {
448  const entry& patchInfo = patchesInfo[patchi];
449 
450  if (!patchInfo.isDict())
451  {
452  IOWarningInFunction(meshDescription)
453  << "Entry " << patchInfo << " in boundary section is not a"
454  << " valid dictionary."
455  << endl;
456  break;
457  }
458 
459  const word& patchName = patchInfo.keyword();
460 
461  // Read block faces
462  faceList patchFaces = blockMeshTools::read<face>
463  (
464  patchInfo.dict().lookup("faces"),
465  varDict
466  );
467 
468  forAll(patchFaces, facei)
469  {
470  const face& f = patchFaces[facei];
471 
472  // Into a list for later removal
473  patchTextActors_[nActors++] = createTextActor
474  (
475  patchName,
476  f.centre(cornerPts) * scaleFactor
477  );
478 
479  if (nActors == patchTextActors_.size())
480  {
481  // hit max allocated space - bail out
482  break;
483  }
484  }
485 
486  if (nActors == patchTextActors_.size())
487  {
488  // hit max allocated space - bail out
489  break;
490  }
491  }
492 
493  patchTextActors_.setSize(nActors);
494  }
495 
496  // Add text to each renderer
497  forAll(patchTextActors_, actori)
498  {
499  renderer->AddViewProp(patchTextActors_[actori]);
500  }
501 }
