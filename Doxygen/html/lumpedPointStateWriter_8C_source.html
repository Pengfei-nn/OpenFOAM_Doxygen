50 (
51  const fileName& outputFile,
52  const vector& axis
53 ) const
54 {
55  // local-to-global transformation tensors
56  const tensorField& localToGlobal = rotations();
57 
58  OFstream fos(outputFile);
59  std::ostream& os = fos.stdStream();
60 
61  autoPtr<vtk::formatter> format = vtk::newFormatter
62  (
63  os,
64  vtk::formatType::INLINE_ASCII
65  );
66 
67  format().xmlHeader()
68  .beginVTKFile<vtk::fileTag::POLY_DATA>();
69 
70  //
71  // The 'spine' of lumped mass points
72  //
73  {
74  format()
75  .tag
76  (
77  vtk::fileTag::PIECE,
78  vtk::fileAttr::NUMBER_OF_POINTS, points_.size(),
79  vtk::fileAttr::NUMBER_OF_VERTS, points_.size(),
80  vtk::fileAttr::NUMBER_OF_LINES, 1
81  );
82 
83  // 'points'
84  {
85  const uint64_t payLoad = vtk::sizeofData<float, 3>(points_.size());
86 
87  format()
88  .tag(vtk::fileTag::POINTS)
89  .beginDataArray<float, 3>(vtk::dataArrayAttr::POINTS);
90 
91  format().writeSize(payLoad);

93  format().flush();
94 
95  format()
96  .endDataArray()
97  .endTag(vtk::fileTag::POINTS);
98  }
99 
100  // <Verts>
101  format().tag(vtk::fileTag::VERTS);
102 
103  //
104  // 'connectivity'
105  //
106  {
107  const uint64_t payLoad = vtk::sizeofData<label>(points_.size());
108 
109  format().beginDataArray<label>(vtk::dataArrayAttr::CONNECTIVITY);
110  format().writeSize(payLoad);
111 
112  vtk::writeIdentity(format(), points_.size());
113 
114  format().flush();
115 
116  format().endDataArray();
117  }
118 
119  //
120  // 'offsets' (connectivity offsets)
121  // = linear mapping onto points (with 1 offset)
122  //
123  {
124  const uint64_t payLoad = vtk::sizeofData<label>(points_.size());
125 
126  format().beginDataArray<label>(vtk::dataArrayAttr::OFFSETS);
127  format().writeSize(payLoad);
128 
129  vtk::writeIdentity(format(), points_.size(), 1);
130 
131  format().flush();
132 
133  format().endDataArray();
134  }
135 
136  format().endTag(vtk::fileTag::VERTS);
137  // </Verts>
138 
139 
140  // <Lines>
141  format().tag(vtk::fileTag::LINES);
142 
143  //
144  // 'connectivity'
145  //
146  {
147  const uint64_t payLoad = vtk::sizeofData<label>(points_.size());
148 
149  format().beginDataArray<label>(vtk::dataArrayAttr::CONNECTIVITY);
150  format().writeSize(payLoad);
151 
152  vtk::writeIdentity(format(), points_.size());
153 
154  format().flush();
155 
156  format().endDataArray();
157  }
158 
159  //
160  // 'offsets' (connectivity offsets)
161  // = single line
162  //
163  {
164  const uint64_t payLoad = vtk::sizeofData<label>(1);
165 
166  format().beginDataArray<label>(vtk::dataArrayAttr::OFFSETS);
167  format().writeSize(payLoad);
168 
169  format().write(points_.size());
170  format().flush();
171 
172  format().endDataArray();
173  }
174 
175  format().endTag(vtk::fileTag::LINES);
176  format().endPiece();
177  }
178 
179  // Standard corners in local axis
180  FixedList<point, 4> corners;
181 
182  {
183  coordinateRotations::axes orient(axis);
184  coordinateSystem cornerTransform(orient);
185 
186  forAll(standardCorners, corni)
187  {
188  corners[corni] = cornerTransform.transform(standardCorners[corni]);
189  }
190  }
191 
192  //
193  // Planes to visualize location/rotation
194  //
195  {
196  const label nPoints = 4*points_.size(); // 4 points per quad
197  const label nPolys = points_.size();
198 
199  format()
200  .tag
201  (
202  vtk::fileTag::PIECE,
203  vtk::fileAttr::NUMBER_OF_POINTS, nPoints,
204  vtk::fileAttr::NUMBER_OF_POLYS, nPolys
205  );
206 
207  // 'points'
208  {
209  const uint64_t payLoad = vtk::sizeofData<float, 3>(nPoints);
210 
211  format()
212  .tag(vtk::fileTag::POINTS)
213  .beginDataArray<float, 3>(vtk::dataArrayAttr::POINTS);
214 
215  format().writeSize(payLoad);
216 
217  forAll(points_, posI)
218  {
219  const point& origin = points_[posI];
220  const tensor& rotTensor =
221  (
222  posI < localToGlobal.size()
223  ? localToGlobal[posI]
224  : pTraits<tensor>::I
225  );
226 
227 
228  for (const point& cornerPt : corners)
229  {
230  // Local-to-global rotation and translation
231  const point pt = (rotTensor & cornerPt) + origin;
232 
233  vtk::write(format(), pt);
234  }
235  }
236 
237  format().flush();
238 
239  format()
240  .endDataArray()
241  .endTag(vtk::fileTag::POINTS);
242  }
243 
244  // <Polys>
245  format().tag(vtk::fileTag::POLYS);
246 
247  //
248  // 'connectivity' - 4 points (ie, quad)
249  //
250  {
251  const uint64_t payLoad = vtk::sizeofData<label>(4*nPolys);
252 
253  format().beginDataArray<label>(vtk::dataArrayAttr::CONNECTIVITY);
254  format().writeSize(payLoad);
255 
256  vtk::writeIdentity(format(), 4*nPolys);
257 
258  format().flush();
259 
260  format().endDataArray();
261  }
262 
263  //
264  // 'offsets' (connectivity offsets)
265  // = single quad
266  //
267  {
268  const uint64_t payLoad = vtk::sizeofData<label>(nPolys);
269 
270  format().beginDataArray<label>(vtk::dataArrayAttr::OFFSETS);
271  format().writeSize(payLoad);
272 
273  for (const label off : sliceRange(4, nPolys, 4))
274  {
275  format().write(off);
276  }
277  format().flush();
278 
279  format().endDataArray();
280  }
281 
282  format().endTag(vtk::fileTag::POLYS);
283

304  format().endPiece();
305  }

307  // Finally

310 
311  format().endTag(vtk::fileTag::POLY_DATA)
312  .endVTKFile();
