95 (

349 template<class BasicTurbulenceModel>
350 void SpalartAllmaras<BasicTurbulenceModel>::correct()
351 {
352  if (!this->turbulence_)
353  {
354  return;
355  }
356 
357  {
358  // Local references
359  const alphaField& alpha = this->alpha_;
360  const rhoField& rho = this->rho_;
361  const surfaceScalarField& alphaRhoPhi = this->alphaRhoPhi_;
362  fv::options& fvOptions(fv::options::New(this->mesh_));
363 
364  eddyViscosity<RASModel<BasicTurbulenceModel>>::correct();
365 
366  const volScalarField chi(this->chi());
367  const volScalarField fv1(this->fv1(chi));
368 
369  const volScalarField Stilda(this->Stilda(chi, fv1));
370 
371  tmp<fvScalarMatrix> nuTildaEqn
372  (
373  fvm::ddt(alpha, rho, nuTilda_)
374  + fvm::div(alphaRhoPhi, nuTilda_)
375  - fvm::laplacian(alpha*rho*DnuTildaEff(), nuTilda_)
376  - Cb2_/sigmaNut_*alpha*rho*magSqr(fvc::grad(nuTilda_))
377  ==
378  Cb1_*alpha*rho*Stilda*nuTilda_
379  - fvm::Sp(Cw1_*alpha*rho*fw(Stilda)*nuTilda_/sqr(y_), nuTilda_)
380  + fvOptions(alpha, rho, nuTilda_)
381  );
382 
383  nuTildaEqn.ref().relax();
384  fvOptions.constrain(nuTildaEqn.ref());
385  solve(nuTildaEqn);
386  fvOptions.correct(nuTilda_);
387  bound(nuTilda_, dimensionedScalar(nuTilda_.dimensions(), Zero));
388  nuTilda_.correctBoundaryConditions();
389  }
390 
391  // Update nut with latest available k,epsilon
392  correctNut();
393 }
