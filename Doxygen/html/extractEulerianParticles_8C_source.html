97 void Foam::functionObjects::extractEulerianParticles::initialiseBins()
98 {
99  DebugInFunction << endl;
100 
101  if (!nInjectorLocations_)
102  {
103  return;
104  }
105 
106  const faceZone& fz = mesh_.faceZones()[zoneID_];
107 
108  // Agglomerate faceZone faces into nInjectorLocations_ global locations
109  const indirectPrimitivePatch patch
110  (
111  IndirectList<face>(mesh_.faces(), fz),
112  mesh_.points()
113  );
114 
115  const label nFaces = fz.size();
116  label nLocations = nInjectorLocations_;
117 
118  if (Pstream::parRun())
119  {
120  label nGlobalFaces = returnReduce(nFaces, sumOp<label>());
121  scalar fraction = scalar(nFaces)/scalar(nGlobalFaces);
122  nLocations = ceil(fraction*nInjectorLocations_);
123  if (debug)
124  {
125  Pout<< "nFaces:" << nFaces
126  << ", nGlobalFaces:" << nGlobalFaces
127  << ", fraction:" << fraction
128  << ", nLocations:" << nLocations
129  << endl;
130  }
131  }
132 
133  pairPatchAgglomeration ppa
134  (
135  patch.localFaces(),
136  patch.localPoints(),
137  10,
138  50,
139  nLocations,
140  labelMax,
141  180
142  );
143 
144  ppa.agglomerate();
145 
146  label nCoarseFaces = 0;
147  if (nFaces != 0)
148  {
149  fineToCoarseAddr_ = ppa.restrictTopBottomAddressing();
150  nCoarseFaces = max(fineToCoarseAddr_) + 1;
151  }
152 
153  globalCoarseFaces_ = globalIndex(nCoarseFaces);
154 
155  Info<< "Created " << returnReduce(nCoarseFaces, sumOp<label>())

182 void Foam::functionObjects::extractEulerianParticles::setBlockedFaces
183 (
184  const surfaceScalarField& alphaf,
185  const faceZone& fz,
186  boolList& blockedFaces
187 )
188 {
189  DebugInFunction << endl;
190 
191  // Initialise storage for patch and patch-face indices where faceZone
192  // intersects mesh patch(es)
193  patchIDs_.setSize(fz.size(), -1);
194  patchFaceIDs_.setSize(fz.size(), -1);
195 
196  label nBlockedFaces = 0;
197  forAll(fz, localFacei)
198  {
199  const label facei = fz[localFacei];
200 
201  if (mesh_.isInternalFace(facei))
202  {
203  if (alphaf[facei] > alphaThreshold_)
204  {
205  blockedFaces[localFacei] = true;
206  }
207  }
208  else
209  {
210  label patchi = mesh_.boundaryMesh().whichPatch(facei);
211  label patchFacei = -1;
212 
213  const polyPatch& pp = mesh_.boundaryMesh()[patchi];
214  const scalarField& alphafp = alphaf.boundaryField()[patchi];
215 
216  if (isA<coupledPolyPatch>(pp))
217  {
218  const coupledPolyPatch& cpp =
219  refCast<const coupledPolyPatch>(pp);
220 
221  if (cpp.owner())
222  {
223  patchFacei = cpp.whichFace(facei);
224  }
225  }
226  else if (!isA<emptyPolyPatch>(pp))
227  {
228  patchFacei = pp.whichFace(facei);
229  }
230 
231  if (patchFacei == -1)
232  {
233  patchi = -1;
234  }
235  else if (alphafp[patchFacei] > alphaThreshold_)
236  {
237  blockedFaces[localFacei] = true;
238  }
239 
240  patchIDs_[localFacei] = patchi;
241  patchFaceIDs_[localFacei] = patchFacei;
242  }
243  }
244 
245  DebugInFunction << "Number of blocked faces: " << nBlockedFaces << endl;
246 }

249 void Foam::functionObjects::extractEulerianParticles::collectParticle
250 (
251  const scalar time,
252  const label regioni
253 )
254 {
255  DebugInFunction << "collectParticle: " << regioni << endl;
256 
257  const label particlei = regionToParticleMap_[regioni];
258  eulerianParticle p = particles_[particlei];
259 
260  if (p.faceIHit != -1 && nInjectorLocations_)
261  {
262  // Use coarse face index for tag output
263  label coarseFacei = fineToCoarseAddr_[p.faceIHit];
264  p.faceIHit = globalCoarseFaces_.toGlobal(coarseFacei);
265  }
266 
267  reduce(p, sumParticleOp<eulerianParticle>());
268 
269  const scalar pDiameter = cbrt(6.0*p.V/constant::mathematical::pi);
270 
271  if ((pDiameter > minDiameter_) && (pDiameter < maxDiameter_))
272  {
273  if (Pstream::master())
274  {
275  const scalar d = cbrt(6*p.V/constant::mathematical::pi);
276  const point position = p.VC/(p.V + ROOTVSMALL);
277  const vector U = p.VU/(p.V + ROOTVSMALL);
278  label tag = -1;
279  if (nInjectorLocations_)
280  {
281  tag = p.faceIHit;
282  }
283 
284  injectedParticle* ip = new injectedParticle
285  (
286  mesh_,
287  position,
288  tag,
289  time,
290  d,
291  U,
292  false // not looking to set cell owner etc.
293  );
294 
295  cloud_.addParticle(ip);
296 
297  collectedVolume_ += p.V;
298  }
299 
300  ++nCollectedParticles_;
301  }
302  else
303  {
304  // Discard particles over/under diameter threshold
305  ++nDiscardedParticles_;
306  discardedVolume_ += p.V;
307  }
308 }

311 void Foam::functionObjects::extractEulerianParticles::calculateAddressing
312 (
313  const label nNewRegions,
314  const scalar time,
315  labelList& regionFaceIDs
316 )
317 {
318  DebugInFunction << endl;
319 
320  // Determine mapping between old and new regions so that we can
321  // accumulate particle info
322  labelList oldToNewRegion(particles_.size(), -1);
323  labelList newToNewRegion(identity(nNewRegions));
324 
325  forAll(regionFaceIDs, facei)
326  {
327  label newRegioni = regionFaceIDs[facei];
328  label oldRegioni = regions0_[facei];
329 
330  if (newRegioni != -1 && oldRegioni != -1)
331  {
332  // If old region has split into multiple regions we need to
333  // renumber new regions to maintain connectivity with old regions
334  newToNewRegion[newRegioni] =
335  max(newRegioni, oldToNewRegion[oldRegioni]);
336  oldToNewRegion[oldRegioni] = newRegioni;
337  }
338  }
339 
340  // Create map from new regions to slots in particles list
341  // - filter through new-to-new addressing to identify new particles
342  Pstream::listCombineGather(newToNewRegion, maxEqOp<label>());
343  Pstream::listCombineScatter(newToNewRegion);
344 
345  label nParticle = -1;
346  labelHashSet newRegions;
347  Map<label> newRegionToParticleMap;
348  forAll(newToNewRegion, newRegioni0)
349  {
350  label newRegioni = newToNewRegion[newRegioni0];
351  if (newRegions.insert(newRegioni))
352  {
353  ++nParticle;
354  }
355 
356  // New particle slot
357  newRegionToParticleMap.insert(newRegioni0, nParticle);
358  }
359 
360  // Accumulate old region data or create a new particle if there is no
361  // mapping from the old-to-new region
362  Pstream::listCombineGather(oldToNewRegion, maxEqOp<label>());
363  Pstream::listCombineScatter(oldToNewRegion);
364  List<eulerianParticle> newParticles(newRegionToParticleMap.size());
365  forAll(oldToNewRegion, oldRegioni)
366  {
367  label newRegioni = oldToNewRegion[oldRegioni];
368  if (newRegioni == -1)
369  {
370  // No mapping from old-to-new - collect new particle
371  DebugInfo
372  << "Collecting particle from oldRegion:" << oldRegioni
373  << endl;
374 
375  collectParticle(time, oldRegioni);
376  }
377  else
378  {
379  // Combine existing particle into new particle
380  label newParticlei = newRegionToParticleMap[newRegioni];
381  label oldParticlei = regionToParticleMap_[oldRegioni];
382 
383  DebugInfo
384  << "Combining newRegioni: " << newRegioni
385  << "(p:" << newParticlei << ") and "
386  << "oldRegioni: " << oldRegioni
387  << "(p:" << oldParticlei << ")"
388  << endl;
389 
390  newParticles[newParticlei] =
391  sumParticleOp<eulerianParticle>()
392  (
393  newParticles[newParticlei],
394  particles_[oldParticlei]
395  );
396  }
397  }
398 
399  // Reset the particles list and addressing for latest available info
400  particles_.transfer(newParticles);
401  regionToParticleMap_ = newRegionToParticleMap;
402 
403  // Reset the region IDs for the next integration step
404  // - these become the oldRegioni's
405  regions0_ = regionFaceIDs;
406 }

409 void Foam::functionObjects::extractEulerianParticles::accumulateParticleInfo
410 (
411  const surfaceScalarField& alphaf,
412  const surfaceScalarField& phi,
413  const labelList& regionFaceIDs,
414  const faceZone& fz
415 )
416 {
417  DebugInFunction << endl;
418 
419  const volVectorField& U = mesh_.lookupObject<volVectorField>(UName_);
420  const surfaceVectorField Uf(fvc::interpolate(U));
421 
422  const scalar deltaT = mesh_.time().deltaTValue();
423  const pointField& faceCentres = mesh_.faceCentres();
424 
425  forAll(regionFaceIDs, localFacei)
426  {
427  const label newRegioni = regionFaceIDs[localFacei];
428  if (newRegioni != -1)
429  {
430  const label particlei = regionToParticleMap_[newRegioni];
431  const label meshFacei = fz[localFacei];
432  eulerianParticle& p = particles_[particlei];
433 
434  if (p.faceIHit < 0)
435  {
436  // New particle - does not exist in particles_ list
437  p.faceIHit = localFacei;
438  p.V = 0;
439  p.VC = vector::zero;
440  p.VU = vector::zero;
441  }
442 
443  // Accumulate particle properties
444  scalar magPhii = mag(faceValue(phi, localFacei, meshFacei));
445  vector Ufi = faceValue(Uf, localFacei, meshFacei);
446  scalar dV = magPhii*deltaT;
447  p.V += dV;
448  p.VC += dV*faceCentres[meshFacei];
449  p.VU += dV*Ufi;
450  }
451  }
452 }

536 bool Foam::functionObjects::extractEulerianParticles::execute()
537 {
538  DebugInFunction << endl;
539 
540  Log << type() << " " << name() << " output:" << nl;
541 
542  const volScalarField& alpha =
543  mesh_.lookupObject<volScalarField>(alphaName_);
544 
545  const surfaceScalarField alphaf
546  (
547  typeName + ":alphaf",
548  fvc::interpolate(alpha)
549  );
550 
551  const faceZone& fz = mesh_.faceZones()[zoneID_];
552  const indirectPrimitivePatch patch
553  (
554  IndirectList<face>(mesh_.faces(), fz),
555  mesh_.points()
556  );
557 
558  // Set the blocked faces, i.e. where alpha > alpha threshold value
559  boolList blockedFaces(fz.size(), false);
560  setBlockedFaces(alphaf, fz, blockedFaces);
561 
562  // Split the faceZone according to the blockedFaces
563  // - Returns a list of (disconnected) region index per face zone face
564  regionSplit2D regionFaceIDs(mesh_, patch, blockedFaces);
565 
566  // Global number of regions
567  const label nRegionsNew = regionFaceIDs.nRegions();
568 
569  // Calculate the addressing between the old and new region information
570  // Also collects particles that have traversed the faceZone
571  // - Note: may also update regionFaceIDs
572  calculateAddressing
573  (
574  nRegionsNew,
575  mesh_.time().value(),
576  regionFaceIDs
577  );
578 
579  // Process latest region information
580  tmp<surfaceScalarField> tphi = phiU();
581  accumulateParticleInfo(alphaf, tphi(), regionFaceIDs, fz);
582 
583  Log << " Collected particles : " << nCollectedParticles_ << nl
584  << " Collected volume : " << collectedVolume_ << nl
585  << " Discarded particles : " << nDiscardedParticles_ << nl
586  << " Discarded volume : " << discardedVolume_ << nl
587  << " Particles in progress : " << particles_.size() << nl
588  << endl;
589 
590  return true;
591 }
