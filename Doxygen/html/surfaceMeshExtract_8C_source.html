33 Description
34  Extract patch or faceZone surfaces from a polyMesh.
35  Depending on output surface format triangulates faces.
36 
37  Region numbers on faces no guaranteed to be the same as the patch indices.
38 
39  Optionally only extracts named patches.
40 
41  If run in parallel, processor patches get filtered out by default and
42  the mesh is merged (based on topology).

63 labelList getSelectedPatches
64 (
65  const polyBoundaryMesh& patches,
66  const wordRes& whitelist,
67  const wordRes& blacklist
68 )
69 {
70  DynamicList<label> patchIDs(patches.size());
71 
72  for (const polyPatch& pp : patches)
73  {
74  if (isType<emptyPolyPatch>(pp))
75  {
76  continue;
77  }
78  else if (Pstream::parRun() && bool(isA<processorPolyPatch>(pp)))
79  {
80  break; // No processor patches for parallel output
81  }
82 
83  const word& patchName = pp.name();
84 
85  bool accept = false;
86 
87  if (whitelist.size())
88  {
89  const auto matched = whitelist.matched(patchName);
90 
91  accept =
92  (
93  matched == wordRe::LITERAL
94  ? true
95  : (matched == wordRe::REGEX && !blacklist.match(patchName))
96  );
97  }
98  else
99  {
100  accept = !blacklist.match(patchName);
101  }
102 
103  if (accept)
104  {
105  patchIDs.append(pp.index());
106  }
107  }
108 
109  return patchIDs.shrink();
110 }

281  HashTable<label> compactZoneID(1024);

288  // processor patches)

312  forAllConstIters(patchSize, iter)
313  {
314  compactZoneID.insert(iter.key(), compactZoneID.size());
315  }
316

318  {
319  compactZoneID.insert(iter.key(), compactZoneID.size());
320  }

332  if (patchi != -1)
333  {
334  patchToCompactZone[patchi] = iter.val();
335  }
336  else
337  {
338  label zoneI = fzm.findZoneID(iter.key());
339  faceZoneToCompactZone[zoneI] = iter.val();

341  }

345  compactZones.setCapacity(nFaces);
346

349  {
350  const polyPatch& pp = bMesh[patchi];

354  compactZones.append(patchToCompactZone[pp.index()]);
355  }
356  }

359  {

361  forAll(pp, i)
362  {

366  }
367  }

372  (

374  mesh.points()
375  );
376 
377 
378  // Find correspondence to master points
379  labelList pointToGlobal;
380  labelList uniqueMeshPoints;
381  autoPtr<globalIndex> globalNumbers = mesh.globalData().mergePoints
382  (
383  allBoundary.meshPoints(),
384  allBoundary.meshPointMap(),
385  pointToGlobal,
386  uniqueMeshPoints
387  );
388 
389  // Gather all unique points on master
390  List<pointField> gatheredPoints(Pstream::nProcs());
391  gatheredPoints[Pstream::myProcNo()] = pointField
392  (
393  mesh.points(),
394  uniqueMeshPoints
395  );
396  Pstream::gatherList(gatheredPoints);
397 
398  // Gather all faces
399  List<faceList> gatheredFaces(Pstream::nProcs());
400  gatheredFaces[Pstream::myProcNo()] = allBoundary.localFaces();
401  forAll(gatheredFaces[Pstream::myProcNo()], i)
402  {
403  inplaceRenumber

405  pointToGlobal,

407  );
408  }
409  Pstream::gatherList(gatheredFaces);
410 
411  // Gather all ZoneIDs
412  List<labelList> gatheredZones(Pstream::nProcs());
413  gatheredZones[Pstream::myProcNo()].transfer(compactZones);
414  Pstream::gatherList(gatheredZones);
415 
416  // On master combine all points, faces, zones
417  if (Pstream::master())
418  {
419  pointField allPoints = ListListOps::combine<pointField>
420  (
421  gatheredPoints,
422  accessOp<pointField>()
423  );
424  gatheredPoints.clear();
425 
426  faceList allFaces = ListListOps::combine<faceList>
427  (
428  gatheredFaces,
429  accessOp<faceList>()
430  );
431  gatheredFaces.clear();
432 
433  labelList allZones = ListListOps::combine<labelList>
434  (
435  gatheredZones,
436  accessOp<labelList>()
437  );
438  gatheredZones.clear();
439 
440 
441  // Zones
442  surfZoneIdentifierList surfZones(compactZoneID.size());
443  forAllConstIters(compactZoneID, iter)
444  {
445  surfZones[*iter] = surfZoneIdentifier(iter.key(), *iter);

447  << " : " << surfZones[*iter].name()
448  << endl;
449  }
450 
451  UnsortedMeshedSurface<face> unsortedFace
452  (
453  std::move(allPoints),
454  std::move(allFaces),
455  std::move(allZones),
456  surfZones
457  );
458 
459 
460  MeshedSurface<face> sortedFace(unsortedFace);
461 
462  fileName globalCasePath
463  (
464  outFileName.isAbsolute()
465  ? outFileName
466  : (
467  runTime.processorCase()

470  )
471  );
472 
473  Info<< "Writing merged surface to " << globalCasePath << endl;
474 
475  sortedFace.write(globalCasePath);
476  }
477  }
478 
479  Info<< "End\n" << endl;
480 
481  return 0;
