33 Description
34  Converts all IOobjects associated with a case into the format specified
35  in the controlDict.
36 
37  Mainly used to convert binary mesh/field files to ASCII.
38 
39  Problem: any zero-size List written binary gets written as '0'. When
40  reading the file as a dictionary this is interpreted as a label. This
41  is (usually) not a problem when doing patch fields since these get the
42  'uniform', 'nonuniform' prefix. However zone contents are labelLists
43  not labelFields and these go wrong. For now hacked a solution where
44  we detect the keywords in zones and redo the dictionary entries
45  to be labelLists.
46 
47 Usage
48 
49  - foamFormatConvert [OPTION]
50 
51  \param -noConstant \n
52  Exclude the constant/ directory from the times list
53 
54  \param -enableFunctionEntries \n
55  By default all dictionary preprocessing of fields is disabled

93 bool writeZones(const word& name, const fileName& meshDir, Time& runTime)
94 {
95  IOobject io
96  (
97  name,
98  runTime.timeName(),
99  meshDir,
100  runTime,
101  IOobject::MUST_READ,
102  IOobject::NO_WRITE,
103  false
104  );
105 
106  bool writeOk = false;
107 
108  if (io.typeHeaderOk<cellZoneMesh>(false))
109  {
110  Info<< " Reading " << io.headerClassName()
111  << " : " << name << endl;
112 
113  // Switch off type checking (for reading e.g. faceZones as
114  // generic list of dictionaries).
115  const word oldTypeName = IOPtrList<entry>::typeName;
116  const_cast<word&>(IOPtrList<entry>::typeName) = word::null;
117 
118  IOPtrList<entry> meshObject(io);
119 
120  forAll(meshObject, i)
121  {
122  if (meshObject[i].isDict())
123  {
124  dictionary& d = meshObject[i].dict();
125 
126  if (d.found("faceLabels"))
127  {
128  d.set("faceLabels", labelList(d.lookup("faceLabels")));
129  }
130 
131  if (d.found("flipMap"))
132  {
133  d.set("flipMap", boolList(d.lookup("flipMap")));
134  }
135 
136  if (d.found("cellLabels"))
137  {
138  d.set("cellLabels", labelList(d.lookup("cellLabels")));
139  }
140 
141  if (d.found("pointLabels"))
142  {
143  d.set("pointLabels", labelList(d.lookup("pointLabels")));
144  }
145  }
146  }
147 
148  const_cast<word&>(IOPtrList<entry>::typeName) = oldTypeName;
149  // Fake type back to what was in field
150  const_cast<word&>(meshObject.type()) = io.headerClassName();
151 
152  Info<< " Writing " << name << endl;
153 
154  // Force writing as ascii
155  writeOk = meshObject.regIOobject::writeObject
156  (
157  IOstream::ASCII,
158  IOstream::currentVersion,
159  runTime.writeCompression(),
160  true
161  );
162  }
163 
164  return writeOk;
165 }

168 // Reduction for non-empty strings.
169 template<class StringType>
170 struct uniqueEqOp
171 {
172  void operator()(List<StringType>& x, const List<StringType>& y) const
173  {
174  List<StringType> newX(x.size()+y.size());
175  label n = 0;
176  forAll(x, i)
177  {
178  if (!x[i].empty())
179  {
180  newX[n++] = x[i];
181  }
182  }
183  forAll(y, i)
184  {
185  if (!y[i].empty() && !x.found(y[i]))
186  {
187  newX[n++] = y[i];
188  }
189  }
190  newX.setSize(n);
191  x.transfer(newX);
192  }
193 };
