111 void Foam::kLowReWallFunctionFvPatchScalarField::updateCoeffs()
112 {
113  if (updated())
114  {
115  return;
116  }
117 
118  const label patchi = patch().index();
119 
120  const turbulenceModel& turbModel = db().lookupObject<turbulenceModel>
121  (
122  IOobject::groupName
123  (
124  turbulenceModel::propertiesName,
125  internalField().group()
126  )
127  );
128 
129  const nutWallFunctionFvPatchScalarField& nutw =
130  nutWallFunctionFvPatchScalarField::nutw(turbModel, patchi);
131 
132  const scalarField& y = turbModel.y()[patchi];
133 
134  const tmp<scalarField> tnuw = turbModel.nu(patchi);
135  const scalarField& nuw = tnuw();
136 
137  const tmp<volScalarField> tk = turbModel.k();
138  const volScalarField& k = tk();
139 
140  const scalar Cmu25 = pow025(nutw.Cmu());
141 
142  scalarField& kw = *this;
143 
144  // Set k wall values
145  forAll(kw, facei)
146  {
147  const label celli = patch().faceCells()[facei];
148 
149  const scalar uTau = Cmu25*sqrt(k[celli]);
150 
151  const scalar yPlus = uTau*y[facei]/nuw[facei];
152 
153  if (nutw.yPlusLam() < yPlus)
154  {
155  kw[facei] = Ck_/nutw.kappa()*log(yPlus) + Bk_;
156  }
157  else
158  {
159  const scalar Cf =
160  1.0/sqr(yPlus + C_) + 2.0*yPlus/pow3(C_) - 1.0/sqr(C_);
161  kw[facei] = 2400.0/sqr(Ceps2_)*Cf;
162  }
163 
164  kw[facei] *= sqr(uTau);
165  }
166 
167  // Limit kw to avoid failure of the turbulence model due to division by kw
168  kw = max(kw, SMALL);
169 
170  fixedValueFvPatchField<scalar>::updateCoeffs();
171 
172  // TODO: perform averaging for cells sharing more than one boundary face
173 }
