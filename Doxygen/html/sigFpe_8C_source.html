66 // Can turn on/off via env variable containing a bool (true|false|on|off ...)

188 void Foam::sigFpe::set(bool verbose)
189 {
190  if (!sigActive_ && requested())
191  {
192  #if (defined(__linux__) && defined(__GNUC__)) || defined(__APPLE__)
193 
194  feenableexcept
195  (
196  FE_DIVBYZERO
197  | FE_INVALID
198  | FE_OVERFLOW
199  );
200 
201  setHandler("SIGFPE", SIGFPE, sigHandler);
202 
203  sigActive_ = true;
204  #endif
205 
206  if (verbose)
207  {
208  Info<< "trapFpe: Floating point exception trapping ";
209 
210  if (sigActive_)
211  {
212  Info<< "enabled (FOAM_SIGFPE)." << endl;
213  }
214  else
215  {
216  Info<< "- not supported on this platform" << endl;
217  }
218  }
219  }
220 
221 
222  nanActive_ = false;
223  if (isTrue("FOAM_SETNAN", switchNan_))
224  {
225  #ifdef __linux__
226  nanActive_ = true;
227  #endif
228 
229  if (verbose)
230  {
231  Info<< "setNaN : Initialise allocated memory to NaN ";
232 
233  if (nanActive_)
234  {
235  Info<< "enabled (FOAM_SETNAN)." << endl;
236  }
237  else
238  {
239  Info<< " - not supported on this platform" << endl;
240  }
241  }
242  }

246 void Foam::sigFpe::unset(bool verbose)
247 {
248  #if (defined(__linux__) && defined(__GNUC__)) || defined(__APPLE__)
249  if (sigActive_)
250  {
251  if (verbose)
252  {
253  Info<< "sigFpe : Disabling floating point exception trapping"
254  << endl;
255  }
256 
257  resetHandler("SIGFPE", SIGFPE);
258 
259  // Reset exception raising
260  const int oldExcept = fedisableexcept
261  (
262  FE_DIVBYZERO
263  | FE_INVALID
264  | FE_OVERFLOW
265  );
266 
267  if (oldExcept == -1)
268  {
269  FatalErrorInFunction
270  << "Cannot reset SIGFPE trapping"
271  << abort(FatalError);
272  }
273 
274  sigActive_ = false;
275  }
276  #endif
277 
278  nanActive_ = false;
279 }
