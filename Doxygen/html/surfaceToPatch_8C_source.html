51 // Adds empty patch if not yet there. Returns patchID.
52 label addPatch(polyMesh& mesh, const word& patchName)
53 {
54  label patchi = mesh.boundaryMesh().findPatchID(patchName);
55 
56  if (patchi == -1)
57  {
58  const polyBoundaryMesh& patches = mesh.boundaryMesh();
59 
60  List<polyPatch*> newPatches(patches.size() + 1);
61 
62  patchi = 0;
63 
64  // Copy all old patches
65  forAll(patches, i)
66  {
67  const polyPatch& pp = patches[i];
68 
69  newPatches[patchi] =
70  pp.clone
71  (
72  patches,
73  patchi,
74  pp.size(),
75  pp.start()
76  ).ptr();
77 
78  patchi++;
79  }
80 
81  // Add zero-sized patch
82  newPatches[patchi] =
83  new polyPatch
84  (
85  patchName,
86  0,
87  mesh.nFaces(),
88  patchi,
89  patches,
90  polyPatch::typeName
91  );
92 
93  mesh.removeBoundary();
94  mesh.addPatches(newPatches);
95 
96  Pout<< "Created patch " << patchName << " at " << patchi << endl;
97  }
98  else
99  {
100  Pout<< "Reusing patch " << patchName << " at " << patchi << endl;
101  }
102 
103  return patchi;
104 }

107 // Repatch single face. Return true if patch changed.
108 bool repatchFace
109 (
110  const polyMesh& mesh,
111  const boundaryMesh& bMesh,
112  const labelList& nearest,
113  const labelList& surfToMeshPatch,
114  const label facei,
115  polyTopoChange& meshMod
116 )
117 {
118  bool changed = false;
119 
120  label bFacei = facei - mesh.nInternalFaces();
121 
122  if (nearest[bFacei] != -1)
123  {
124  // Use boundary mesh one.
125  label bMeshPatchID = bMesh.whichPatch(nearest[bFacei]);
126 
127  label patchID = surfToMeshPatch[bMeshPatchID];
128 
129  if (patchID != mesh.boundaryMesh().whichPatch(facei))
130  {
131  label own = mesh.faceOwner()[facei];
132 
133  label zoneID = mesh.faceZones().whichZone(facei);
134 
135  bool zoneFlip = false;
136 
137  if (zoneID >= 0)
138  {
139  const faceZone& fZone = mesh.faceZones()[zoneID];
140 
141  zoneFlip = fZone.flipMap()[fZone.whichFace(facei)];
142  }
143 
144  meshMod.setAction
145  (
146  polyModifyFace
147  (
148  mesh.faces()[facei],// modified face
149  facei, // label of face being modified
150  own, // owner
151  -1, // neighbour
152  false, // face flip
153  patchID, // patch for face
154  false, // remove from zone
155  zoneID, // zone for face
156  zoneFlip // face flip in zone
157  )
158  );
159 
160  changed = true;
161  }
162  }
163  else
164  {
165  changed = false;
166  }
167  return changed;
168 }

172 int main(int argc, char *argv[])
173 {
174  argList::addNote
175  (
176  "Reads surface and applies surface regioning to a mesh"
177  );
178 
179  argList::noParallel();
180  argList::addArgument("surfaceFile");
181  argList::addOption

183  "faceSet",
184  "name",
185  "Only repatch the faces in specified faceSet"

192  );

197

199

201 
202  word setName;
203  const bool readSet = args.readIfPresent("faceSet", setName);
204 
205  if (readSet)
206  {
207  Info<< "Repatching only the faces in faceSet " << setName
208  << " according to nearest surface triangle ..." << endl;
209  }
210  else
211  {
212  Info<< "Patching all boundary faces according to nearest surface"
213  << " triangle ..." << endl;
214  }
215 
216  const scalar searchTol = args.get<scalar>("tol", 1e-3);
217 
218  // Get search box. Anything not within this box will not be considered.
219  const boundBox& meshBb = mesh.bounds();
220  const vector searchSpan = searchTol * meshBb.span();
221 
222  Info<< "All boundary faces further away than " << searchTol
223  << " of mesh bounding box " << meshBb
224  << " will keep their patch label ..." << endl;
225 
226 
227  Info<< "Before patching:" << nl
228  << " patch\tsize" << endl;
229 
230  forAll(mesh.boundaryMesh(), patchi)
231  {
232  Info<< " " << mesh.boundaryMesh()[patchi].name() << '\t'
233  << mesh.boundaryMesh()[patchi].size() << nl;
234  }
235  Info<< endl;
236 
237 
238  boundaryMesh bMesh;
239 
240  // Load in the surface.
241  bMesh.readTriSurface(surfName);
242 
243  // Add all the boundaryMesh patches to the mesh.
244  const PtrList<boundaryPatch>& bPatches = bMesh.patches();
245 
246  // Map from surface patch ( = boundaryMesh patch) to polyMesh patch
247  labelList patchMap(bPatches.size());
248 
249  forAll(bPatches, i)
250  {
251  patchMap[i] = addPatch(mesh, bPatches[i].name());
252  }
253 
254  // Obtain nearest face in bMesh for each boundary face in mesh that
255  // is within search span.
256  // Note: should only determine for faceSet if working with that.
257  labelList nearest(bMesh.getNearest(mesh, searchSpan));
258 
259  {
260  // Dump unmatched faces to faceSet for debugging.
261  faceSet unmatchedFaces(mesh, "unmatchedFaces", nearest.size()/100);
262 
263  forAll(nearest, bFacei)
264  {
265  if (nearest[bFacei] == -1)
266  {
267  unmatchedFaces.insert(mesh.nInternalFaces() + bFacei);
268  }
269  }
270 
271  Pout<< "Writing all " << unmatchedFaces.size()
272  << " unmatched faces to faceSet "
273  << unmatchedFaces.name()
274  << endl;
275 
276  unmatchedFaces.write();
277  }
278 
279 
280  polyTopoChange meshMod(mesh);
281 
282  label nChanged = 0;
283 
284  if (readSet)
285  {
286  faceSet faceLabels(mesh, setName);
287  Info<< "Read " << faceLabels.size() << " faces to repatch ..." << endl;
288 
289  for (const label facei : faceLabels)
290  {
291  if (repatchFace(mesh, bMesh, nearest, patchMap, facei, meshMod))
292  {
293  nChanged++;
294  }
295  }
296  }
297  else
298  {
299  forAll(nearest, bFacei)
300  {
301  const label facei = mesh.nInternalFaces() + bFacei;
302 
303  if (repatchFace(mesh, bMesh, nearest, patchMap, facei, meshMod))
304  {
305  ++nChanged;
306  }
307  }
308  }
309 
310  Pout<< "Changed " << nChanged << " boundary faces." << nl << endl;
311 
312  if (nChanged > 0)
313  {
314  meshMod.changeMesh(mesh, false);
315 
316  Info<< "After patching:" << nl
317  << " patch\tsize" << endl;
318 
319  forAll(mesh.boundaryMesh(), patchi)
320  {
321  Info<< " " << mesh.boundaryMesh()[patchi].name() << '\t'
322  << mesh.boundaryMesh()[patchi].size() << endl;
323  }
324  Info<< endl;
325 
326 
327  ++runTime;
328 
329  // Write resulting mesh
330  Info<< "Writing modified mesh to time " << runTime.value() << endl;
331  mesh.write();
332  }
333 
334 
335  Info<< "End\n" << endl;
336 
337  return 0;
338 }
