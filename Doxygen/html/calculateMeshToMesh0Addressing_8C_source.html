42 void Foam::meshToMesh0::calcAddressing()
43 {
44  if (debug)
45  {
46  InfoInFunction
47  << "Calculating mesh-to-mesh cell addressing" << endl;
48  }
49 
50  // set reference to cells
51  const cellList& fromCells = fromMesh_.cells();
52  const pointField& fromPoints = fromMesh_.points();
53 
54  // In an attempt to preserve the efficiency of linear search and prevent
55  // failure, a RESCUE mechanism will be set up. Here, we shall mark all
56  // cells next to the solid boundaries. If such a cell is found as the
57  // closest, the relationship between the origin and cell will be examined.
58  // If the origin is outside the cell, a global n-squared search is
59  // triggered.
60 
61  // SETTING UP RESCUE
62 
63  // visit all boundaries and mark the cell next to the boundary.
64 
65  if (debug)
66  {
67  InfoInFunction << "Setting up rescue" << endl;
68  }
69 
70  List<bool> boundaryCell(fromCells.size(), false);
71 
72  // set reference to boundary
73  const polyPatchList& patchesFrom = fromMesh_.boundaryMesh();
74 
75  forAll(patchesFrom, patchi)
76  {
77  // get reference to cells next to the boundary
78  const labelUList& bCells = patchesFrom[patchi].faceCells();
79 
80  forAll(bCells, facei)
81  {
82  boundaryCell[bCells[facei]] = true;
83  }
84  }
85 
86  treeBoundBox meshBb(fromPoints);
87 
88  scalar typDim = meshBb.avgDim()/(2.0*cbrt(scalar(fromCells.size())));
89 
90  treeBoundBox shiftedBb
91  (
92  meshBb.min(),
93  meshBb.max() + vector(typDim, typDim, typDim)
94  );
95 
96  if (debug)
97  {
98  Info<< "\nMesh" << endl;
99  Info<< " bounding box : " << meshBb << endl;
100  Info<< " bounding box (shifted) : " << shiftedBb << endl;
101  Info<< " typical dimension :" << shiftedBb.typDim() << endl;
102  }
103 
104  indexedOctree<treeDataCell> oc
105  (
106  treeDataCell(false, fromMesh_, polyMesh::CELL_TETS),
107  shiftedBb, // overall bounding box
108  8, // maxLevel
109  10, // leafsize
110  6.0 // duplicity
111  );
112 
113  if (debug)
114  {
115  oc.print(Pout, false, 0);
116  }
117 
118  cellAddresses
119  (
120  cellAddressing_,
121  toMesh_.cellCentres(),
122  fromMesh_,
123  boundaryCell,
124  oc
125  );
126 
127  forAll(toMesh_.boundaryMesh(), patchi)
128  {
129  const polyPatch& toPatch = toMesh_.boundaryMesh()[patchi];
130 
131  if (cuttingPatches_.found(toPatch.name()))
132  {
133  boundaryAddressing_[patchi].setSize(toPatch.size());
134 
135  cellAddresses
136  (
137  boundaryAddressing_[patchi],
138  toPatch.faceCentres(),
139  fromMesh_,
140  boundaryCell,
141  oc
142  );
143  }
144  else if
145  (
146  patchMap_.found(toPatch.name())
147  && fromMeshPatches_.found(patchMap_.find(toPatch.name())())
148  )
149  {
150  const polyPatch& fromPatch = fromMesh_.boundaryMesh()
151  [
152  fromMeshPatches_.find(patchMap_.find(toPatch.name())())()
153  ];
154 
155  if (fromPatch.empty())
156  {
157  WarningInFunction
158  << "Source patch " << fromPatch.name()
159  << " has no faces. Not performing mapping for it."
160  << endl;
161  boundaryAddressing_[patchi].setSize(toPatch.size());
162  boundaryAddressing_[patchi] = -1;
163  }
164  else
165  {
166  treeBoundBox wallBb(fromPatch.localPoints());
167  scalar typDim =
168  wallBb.avgDim()/(2.0*sqrt(scalar(fromPatch.size())));
169 
170  treeBoundBox shiftedBb
171  (
172  wallBb.min(),
173  wallBb.max() + vector(typDim, typDim, typDim)
174  );
175 
176  // Note: allow more levels than in meshSearch. Assume patch
177  // is not as big as all boundary faces
178  indexedOctree<treeDataFace> oc
179  (
180  treeDataFace(false, fromPatch),
181  shiftedBb, // overall search domain
182  12, // maxLevel
183  10, // leafsize
184  6.0 // duplicity
185  );
186 
187  const vectorField::subField centresToBoundary =
188  toPatch.faceCentres();
189 
190  boundaryAddressing_[patchi].setSize(toPatch.size());
191 
192  scalar distSqr = sqr(wallBb.mag());
193 
194  forAll(toPatch, toi)
195  {
196  boundaryAddressing_[patchi][toi] = oc.findNearest
197  (
198  centresToBoundary[toi],
199  distSqr
200  ).index();
201  }
202  }
203  }
204  }
205 
206  if (debug)
207  {
208  InfoInFunction
209  << "Finished calculating mesh-to-mesh cell addressing" << endl;
210  }
211 }

214 void Foam::meshToMesh0::cellAddresses
215 (
216  labelList& cellAddressing_,
217  const pointField& points,
218  const fvMesh& fromMesh,
219  const List<bool>& boundaryCell,
220  const indexedOctree<treeDataCell>& oc
221 ) const
222 {
223  // the implemented search method is a simple neighbour array search.
224  // It starts from a cell zero, searches its neighbours and finds one
225  // which is nearer to the target point than the current position.
226  // The location of the "current position" is reset to that cell and
227  // search through the neighbours continues. The search is finished
228  // when all the neighbours of the cell are farther from the target
229  // point than the current cell
230 
231  // set curCell label to zero (start)
232  label curCell = 0;
233 
234  // set reference to cell to cell addressing
235  const vectorField& centresFrom = fromMesh.cellCentres();
236  const labelListList& cc = fromMesh.cellCells();
237 
238  forAll(points, toI)
239  {
240  // pick up target position
241  const vector& p = points[toI];
242 
243  // set the sqr-distance
244  scalar distSqr = magSqr(p - centresFrom[curCell]);
245 
246  bool closer;
247 
248  do
249  {
250  closer = false;
251 
252  // set the current list of neighbouring cells
253  const labelList& neighbours = cc[curCell];
254 
255  forAll(neighbours, nI)
256  {
257  scalar curDistSqr =
258  magSqr(p - centresFrom[neighbours[nI]]);
259 
260  // search through all the neighbours.
261  // If the cell is closer, reset current cell and distance
262  if (curDistSqr < (1 - SMALL)*distSqr)
263  {
264  curCell = neighbours[nI];
265  distSqr = curDistSqr;
266  closer = true; // a closer neighbour has been found
267  }
268  }
269  } while (closer);
270 
271  cellAddressing_[toI] = -1;
272 
273  // Check point is actually in the nearest cell
274  if (fromMesh.pointInCell(p, curCell))
275  {
276  cellAddressing_[toI] = curCell;
277  }
278  else
279  {
280  // If curCell is a boundary cell then the point maybe either outside
281  // the domain or in an other region of the doamin, either way use
282  // the octree search to find it.
283  if (boundaryCell[curCell])
284  {
285  cellAddressing_[toI] = oc.findInside(p);
286  }
287  else
288  {
289  // If not on the boundary search the neighbours
290  bool found = false;
291 
292  // set the current list of neighbouring cells
293  const labelList& neighbours = cc[curCell];
294 
295  forAll(neighbours, nI)
296  {
297  // search through all the neighbours.
298  // If point is in neighbour reset current cell
299  if (fromMesh.pointInCell(p, neighbours[nI]))
300  {
301  cellAddressing_[toI] = neighbours[nI];
302  found = true;
303  break;
304  }
305  }
306 
307  if (!found)
308  {
309  // If still not found search the neighbour-neighbours
310 
311  // set the current list of neighbouring cells
312  const labelList& neighbours = cc[curCell];
313 
314  forAll(neighbours, nI)
315  {
316  // set the current list of neighbour-neighbouring cells
317  const labelList& nn = cc[neighbours[nI]];
318 
319  forAll(nn, nI)
320  {
321  // search through all the neighbours.
322  // If point is in neighbour reset current cell
323  if (fromMesh.pointInCell(p, nn[nI]))
324  {
325  cellAddressing_[toI] = nn[nI];
326  found = true;
327  break;
328  }
329  }
330  if (found) break;
331  }
332  }
333 
334  if (!found)
335  {
336  // Still not found so us the octree
337  cellAddressing_[toI] = oc.findInside(p);
338  }
339  }
340  }
341  }
342 }
