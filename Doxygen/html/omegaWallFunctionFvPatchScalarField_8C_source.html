183 void Foam::omegaWallFunctionFvPatchScalarField::calculate
184 (
185  const turbulenceModel& turbModel,
186  const List<scalar>& cornerWeights,
187  const fvPatch& patch,
188  scalarField& G0,
189  scalarField& omega0
190 )
191 {
192  const label patchi = patch.index();
193 
194  const nutWallFunctionFvPatchScalarField& nutw =
195  nutWallFunctionFvPatchScalarField::nutw(turbModel, patchi);
196 
197  const scalarField& y = turbModel.y()[patchi];
198 
199  const tmp<scalarField> tnuw = turbModel.nu(patchi);
200  const scalarField& nuw = tnuw();
201 
202  const tmp<volScalarField> tk = turbModel.k();
203  const volScalarField& k = tk();
204 
205  const fvPatchVectorField& Uw = turbModel.U().boundaryField()[patchi];
206 
207  const scalarField magGradUw(mag(Uw.snGrad()));
208 
209  const scalar Cmu25 = pow025(nutw.Cmu());
210 
211  // Set omega and G
212  forAll(nutw, facei)
213  {
214  const label celli = patch.faceCells()[facei];
215 
216  const scalar yPlus = Cmu25*y[facei]*sqrt(k[celli])/nuw[facei];
217 
218  const scalar w = cornerWeights[facei];
219 
220  const scalar omegaVis = 6*nuw[facei]/(beta1_*sqr(y[facei]));
221  const scalar omegaLog = sqrt(k[celli])/(Cmu25*nutw.kappa()*y[facei]);
222 
223  // Switching between the laminar sub-layer and the log-region rather
224  // than blending has been found to provide more accurate results over a
225  // range of near-wall y+.
226  //
227  // For backward-compatibility the blending method is provided as an
228  // option
229 
230  // Generation contribution is included using the blended option, or
231  // when using the switching option if operating in the laminar
232  // sub-layer
233  bool includeG = true;
234  if (blended_)
235  {
236  omega0[celli] += w*sqrt(sqr(omegaVis) + sqr(omegaLog));
237  }
238  else
239  {
240  if (nutw.yPlusLam() < yPlus)
241  {
242  omega0[celli] += w*omegaLog;
243  }
244  else
245  {
246  omega0[celli] += w*omegaVis;
247  includeG = false;
248  }
249  }
250 
251  if (includeG)
252  {
253  G0[celli] +=
254  w
255  *(nutw[facei] + nuw[facei])
256  *magGradUw[facei]
257  *Cmu25*sqrt(k[celli])
258  /(nutw.kappa()*y[facei]);
259  }
260  }
261 }
