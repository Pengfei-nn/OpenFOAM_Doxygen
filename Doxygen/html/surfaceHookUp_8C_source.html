208 class findNearestOpSubset
209 {
210  const indexedOctree<treeDataEdge>& tree_;
211 
212  DynamicList<label>& shapeMask_;
213 
214 public:
215 
216  findNearestOpSubset
217  (
218  const indexedOctree<treeDataEdge>& tree,
219  DynamicList<label>& shapeMask

222  tree_(tree),
223  shapeMask_(shapeMask)

225 
226  void operator()
227  (
228  const labelUList& indices,
229  const point& sample,
230 
231  scalar& nearestDistSqr,
232  label& minIndex,
233  point& nearestPoint
234  ) const
235  {
236  const treeDataEdge& shape = tree_.shapes();
237 
238  for (const label index : indices)
239  {
240  const label edgeIndex = shape.edgeLabels()[index];
241 
242  if (shapeMask_.found(edgeIndex))
243  {
244  continue;
245  }
246 
247  const edge& e = shape.edges()[edgeIndex];
248 
249  pointHit nearHit = e.line(shape.points()).nearestDist(sample);
250 
251  // Only register hit if closest point is not an edge point
252  if (nearHit.hit())
253  {
254  const scalar distSqr = sqr(nearHit.distance());
255 
256  if (distSqr < nearestDistSqr)
257  {
258  nearestDistSqr = distSqr;
259  minIndex = index;
260  nearestPoint = nearHit.rawPoint();
261  }
262  }
263  }
264  }
265 };
