187  }

199

201

205 
206  extruder.setRefinement(meshMod);
207

209 
210  pMesh.updateMesh(morphMap());
211  }
212  }
213

215  {
216  ++runTime;
217  }
218  else
219  {
220  pMesh.setInstance("constant");
221  }
222 
223  pMesh.write();
224 
225  Info<< "Finished extruding in = "
226  << runTime.cpuTimeIncrement() << " s." << endl;
227 
228  Info<< "\nEnd\n" << endl;
229 
230  return 0;
