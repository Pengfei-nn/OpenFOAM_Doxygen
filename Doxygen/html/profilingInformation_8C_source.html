88 Foam::Ostream& Foam::profilingInformation::write
89 (
90  Ostream& os,
91  const bool offset,
92  const scalar elapsedTime,
93  const scalar childTimes
94 ) const
95 {
96  // write in dictionary format
97 
98  os.beginBlock(word("trigger" + Foam::name(id_)));
99 
100  os.writeEntry("id", id_);
101  os.writeEntryIfDifferent("parentId", id_, parent().id());
102  os.writeEntry("description", description());
103  os.writeEntry("calls", calls() + (offset ? 1 : 0));
104  os.writeEntry("totalTime", totalTime() + elapsedTime);
105  os.writeEntry("childTime", childTime() + childTimes);
106  os.writeEntryIfDifferent<int>("maxMem", 0, maxMem_);
107  os.writeEntry("active", Switch(active()));
108 
109  os.endBlock();
110 
111  return os;
112 }
