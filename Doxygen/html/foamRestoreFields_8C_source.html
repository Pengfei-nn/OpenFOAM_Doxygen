32 Description
33  Restore field names by removing the ending.
34  The fields are selected automatically or can be specified as optional
35  command arguments.
36 
37  The operation 'mean' renames files ending with 'Mean' and makes
38  a backup of existing names, using the '.orig' ending.
39 
40  The operation 'orig' renames files ending with '.orig'.
41 
42 Usage
43  \b foamRestoreFields [OPTION]
44 
45  Options:
46  - \par -method mean | orig
47  The renaming method.
48 
49  - \par -processor
50  Use processor directories, taking information from processor0/
51 
52  - \par -dry-run
53  Test without actually moving/renaming files.
54 
55  - \par -verbose
56  Additional verbosity.

106 // Use predefined method to walk the directory and rename the files.
107 //
108 // If no target names are specified, the existing files are scanned for
109 // candidates.
110 label restoreFields
111 (
112  const restoreMethod method,
113  const fileName& dirName,
114  const wordHashSet& existingFiles,
115  const wordList& targetNames
116 )
117 {
118  // The file ending to search for.
119  const word ending(methodEndings[method]);
120 
121  // The backup ending for existing (if any)
122  word bak;
123 
124  switch (method)
125  {
126  case restoreMethod::MEAN:
127  bak = methodEndings[restoreMethod::ORIG];
128  break;
129 
130  default:
131  break;
132  }
133 
134  wordHashSet targets(targetNames);
135 
136  if (targets.empty())
137  {
138  // No target names specified - scan existing files for candidates.
139 
140  for (word f : existingFiles) // Operate on a copy
141  {
142  // Eg, check for "UMean" and save as "U"
143  if (f.removeEnd(ending) && f.size())
144  {
145  targets.insert(f);
146  }
147  }
148  }
149 
150  if (verbose)
151  {
152  Info<< "directory " << dirName.name() << nl;
153  }
154 
155  // Count of files moved, including backups
156  label count = 0;
157 
158  for (const word& dst : targets)
159  {
160  const word src(dst + ending);
161 
162  if (!existingFiles.found(src))
163  {
164  continue;
165  }
166 
167  if (bak.size() && existingFiles.found(dst))
168  {
169  if (dryrun || Foam::mv(dirName/dst, dirName/dst + bak))
170  {
171  Info<< " mv " << dst << " " << word(dst + bak) << nl;
172  ++count;
173  }
174  }
175 
176  if (dryrun || Foam::mv(dirName/src, dirName/dst))
177  {
178  Info<< " mv " << src << " " << dst << nl;
179  ++count;
180  }
181  }
182 
183  return count;
184 }

327  );
328  }
329

331 
332  if (timeDirs.empty())
333  {
334  Info<< "no times selected" << nl;
335  }
336 
337  for (const instant& t : timeDirs)
338  {
339  const word& timeName = t.name();
340 
341  Info<< "\nTime = " << timeName << nl;
342 
343  label count = 0;
344 
345  if (nProcs)
346  {
347  const wordHashSet files
348  (
349  getFiles(args.path()/"processor0", timeName)
350  );
351 
352  for (label proci=0; proci < nProcs; ++proci)
353  {
354  count += restoreFields
355  (
356  method,
357  args.path()/("processor" + Foam::name(proci))/timeName,
358  files,
359  targetNames
360  );
361  }
362  }
363  else
364  {
365  wordList files;

370  Pstream::scatter(files);
371 
372  count += restoreFields

376  wordHashSet(files),
377  targetNames
378  );
379  }
380 
381  if (dryrun)
382  {
383  Info<< "dry-run: ";

385  Info<< "moved " << count << " files" << nl;
386  }
387 
388  Info<< "\nEnd\n" << endl;
