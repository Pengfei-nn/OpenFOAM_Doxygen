33 inline void Foam::labelledTri::assign(const labelUList& lst)
34 {
35  const label sz = lst.size();
36 
37  // checkSize
38  if (sz < 3 || sz > 4)
39  {
40  FatalErrorInFunction
41  << "size " << sz << " != (3 or 4)"
42  << abort(FatalError);
43  }
44 
45  for (label i=0; i<3; ++i)
46  {
47  operator[](i) = lst[i];
48  }
49 
50  region_ = (sz > 3 ? lst[3] : 0);
51 }
