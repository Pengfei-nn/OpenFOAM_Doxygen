57 Foam::StandardWallInteraction<CloudType>::StandardWallInteraction
58 (
59  const dictionary& dict,
60  CloudType& cloud
61 )
62 :
63  PatchInteractionModel<CloudType>(dict, cloud, typeName),
64  mesh_(cloud.mesh()),
65  interactionType_
66  (
67  this->wordToInteractionType(this->coeffDict().getWord("type"))

70  mu_(0.0),

74  massStick_(nEscape_.size()),
75  injIdToIndex_()

78  this->coeffDict().lookupOrDefault("outputByInjectorId", false);
79 
80  switch (interactionType_)
81  {
82  case PatchInteractionModel<CloudType>::itOther:
83  {
84  const word interactionTypeName(this->coeffDict().getWord("type"));
85 
86  FatalErrorInFunction
87  << "Unknown interaction result type "
88  << interactionTypeName
89  << ". Valid selections are:" << this->interactionTypeNames_
90  << endl << exit(FatalError);
91 
92  break;
93  }
94  case PatchInteractionModel<CloudType>::itRebound:
95  {
96  e_ = this->coeffDict().lookupOrDefault("e", 1.0);
97  mu_ = this->coeffDict().lookupOrDefault("mu", 0.0);
98  break;
99  }
100  default:
101  {}
102  }
103 
104  // Determine the number of injectors and the injector mapping
105  label nInjectors = 0;
106  if (outputByInjectorId)
107  {
108  for (const auto& inj : cloud.injectors())
109  {
110  injIdToIndex_.insert(inj.injectorID(), nInjectors++);
111  }
112  }
113 
114  // The normal case, and safety if injector mapping was somehow null.
115  if (injIdToIndex_.empty())
116  {
117  nInjectors = 1;
118  }
119 
120  forAll(nEscape_, patchi)
121  {
122  nEscape_[patchi].setSize(nInjectors, Zero);
123  massEscape_[patchi].setSize(nInjectors, Zero);
124  nStick_[patchi].setSize(nInjectors, Zero);
125  massStick_[patchi].setSize(nInjectors, Zero);

151 template<class CloudType>
152 bool Foam::StandardWallInteraction<CloudType>::correct
153 (
154  typename CloudType::parcelType& p,
155  const polyPatch& pp,
156  bool& keepParticle
157 )
158 {
159  vector& U = p.U();
160 
161  if (isA<wallPolyPatch>(pp))
162  {
163  // Location for storing the stats.
164  const label idx =
165  (
166  injIdToIndex_.size()
167  ? injIdToIndex_.lookup(p.typeId(), 0)
168  : 0
169  );
170 
171  switch (interactionType_)
172  {
173  case PatchInteractionModel<CloudType>::itNone:
174  {
175  return false;
176  }
177  case PatchInteractionModel<CloudType>::itEscape:
178  {
179  keepParticle = false;
180  p.active(false);
181  U = Zero;
182 
183  const scalar dm = p.nParticle()*p.mass();
184 
185  nEscape_[pp.index()][idx]++;
186  massEscape_[pp.index()][idx] += dm;
187  break;
188  }
189  case PatchInteractionModel<CloudType>::itStick:
190  {
191  keepParticle = true;
192  p.active(false);
193  U = Zero;
194 
195  const scalar dm = p.nParticle()*p.mass();
196 
197  nStick_[pp.index()][idx]++;
198  massStick_[pp.index()][idx] += dm;
199  break;
200  }
201  case PatchInteractionModel<CloudType>::itRebound:
202  {
203  keepParticle = true;
204  p.active(true);
205 
206  vector nw;
207  vector Up;
208 
209  this->owner().patchData(p, pp, nw, Up);
210 
211  // Calculate motion relative to patch velocity
212  U -= Up;
213 
214  scalar Un = U & nw;
215  vector Ut = U - Un*nw;
216 
217  if (Un > 0)
218  {
219  U -= (1.0 + e_)*Un*nw;
220  }
221 
222  U -= mu_*Ut;
223 
224  // Return velocity to global space
225  U += Up;
226 
227  break;
228  }
229  default:
230  {
231  FatalErrorInFunction
232  << "Unknown interaction type "
233  << this->interactionTypeToWord(interactionType_)
234  << "(" << interactionType_ << ")" << endl
235  << abort(FatalError);
236  }
237  }
238 
239  return true;
240  }
241 
242  return false;
243 }
