32 Description
33  Utility to clean surfaces.
34 
35  Current functionality
36  - removes baffles
37  - collapses small edges, removing triangles.
38  - converts sliver triangles into split edges by projecting point onto
39  base of triangle.

83  const scalar minLen = args.get<scalar>(2);
84  const scalar minQuality = args.get<scalar>(3);
85  const fileName outFileName = args[4];
86 
87  Info<< "Reading surface " << inFileName << nl
88  << "Collapsing all triangles with" << nl

92 
93 
94  Info<< "Reading surface from " << inFileName << " ..." << nl << endl;
95 
96  triSurface surf
97  (
98  inFileName,
99  args.get<scalar>("scale", -1)
100  );
101  surf.writeStats(Info);
102 
103  if (!args.found("noClean"))
104  {
105  Info<< "Removing duplicate and illegal triangles ..." << nl << endl;
106  surf.cleanup(true);
107  }
108 
109  Info<< "Collapsing triangles to edges ..." << nl << endl;
110 
111  while (true)
112  {
113  const label nEdgeCollapse = collapseEdge(surf, minLen);
114 
115  if (nEdgeCollapse == 0)
116  {
117  break;
118  }
119  }
120  while (true)
121  {
122  const label nSplitEdge = collapseBase(surf, minLen, minQuality);
123 
124  if (nSplitEdge == 0)
125  {
126  break;
127  }
128  }
129 
130  Info<< nl
131  << "Resulting surface:" << endl;
132  surf.writeStats(Info);
133 
134  Info<< nl
135  << "Writing refined surface to " << outFileName << " ..." << endl;
136  surf.write(outFileName);
137 
138  Info<< "\nEnd\n" << endl;
139 
140  return 0;
