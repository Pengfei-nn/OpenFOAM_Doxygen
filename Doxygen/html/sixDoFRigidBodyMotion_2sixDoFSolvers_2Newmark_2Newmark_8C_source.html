72 void Foam::sixDoFSolvers::Newmark::solve
73 (
74  bool firstIter,
75  const vector& fGlobal,
76  const vector& tauGlobal,
77  scalar deltaT,
78  scalar deltaT0
79 )
80 {
81  // Update the linear acceleration and torque
82  updateAcceleration(fGlobal, tauGlobal);
83 
84  // Correct linear velocity
85  v() =
86  tConstraints()
87  & (v0() + aDamp()*deltaT*(gamma_*a() + (1 - gamma_)*a0()));
88 
89  // Correct angular momentum
90  pi() =
91  rConstraints()
92  & (pi0() + aDamp()*deltaT*(gamma_*tau() + (1 - gamma_)*tau0()));
93 
94  // Correct position
95  centreOfRotation() =
96  centreOfRotation0()
97  + (
98  tConstraints()
99  & (
100  deltaT*v0()
101  + aDamp()*sqr(deltaT)*(beta_*a() + (0.5 - beta_)*a0())
102  )
103  );
104 
105  // Correct orientation
106  vector piDeltaT =
107  rConstraints()
108  & (
109  deltaT*pi0()
110  + aDamp()*sqr(deltaT)*(beta_*tau() + (0.5 - beta_)*tau0())
111  );
112  Tuple2<tensor, vector> Qpi = rotate(Q0(), piDeltaT, 1);
113  Q() = Qpi.first();
114 }
