70 void Foam::isoCutCell::calcSubCellCentreAndVolume()
71 {
72  if (cellStatus_ == 0)
73  {
74  subCellCentre_ = vector::zero;
75  subCellVolume_ = 0.0;
76 
77  // Estimate the approximate cell centre as the average of face centres
78  label nCellFaces(1 + isoCutFaceCentres_.size() + fullySubFaces_.size());
79  vector cEst = isoFaceCentre_ + sum(isoCutFaceCentres_);
80  forAll(fullySubFaces_, facei)
81  {
82  cEst += mesh_.faceCentres()[fullySubFaces_[facei]];
83  }
84  cEst /= scalar(nCellFaces);
85 
86 
87  // Contribution to subcell centre and volume from isoface
88  const scalar pyr3Vol0 =
89  max(mag(isoFaceArea_ & (isoFaceCentre_ - cEst)), VSMALL);
90 
91  // Calculate face-pyramid centre
92  const vector pc0 = 0.75*isoFaceCentre_ + 0.25*cEst;
93 
94  // Accumulate volume-weighted face-pyramid centre
95  subCellCentre_ += pyr3Vol0*pc0;
96 
97  // Accumulate face-pyramid volume
98  subCellVolume_ += pyr3Vol0;
99 
100  // Contribution to subcell centre and volume from cut faces
101  forAll(isoCutFaceCentres_, facei)
102  {
103  // Calculate 3*face-pyramid volume
104  scalar pyr3Vol =
105  max
106  (
107  mag
108  (
109  isoCutFaceAreas_[facei]
110  & (isoCutFaceCentres_[facei] - cEst)
111  ),
112  VSMALL
113  );
114 
115  // Calculate face-pyramid centre
116  vector pc = 0.75*isoCutFaceCentres_[facei] + 0.25*cEst;
117 
118  // Accumulate volume-weighted face-pyramid centre
119  subCellCentre_ += pyr3Vol*pc;
120 
121  // Accumulate face-pyramid volume
122  subCellVolume_ += pyr3Vol;
123  }
124 
125  // Contribution to subcell centre and volume from fully submerged faces
126  forAll(fullySubFaces_, i)
127  {
128  const label facei = fullySubFaces_[i];
129  const point& fCentre = mesh_.faceCentres()[facei];
130  const vector& fArea = mesh_.faceAreas()[facei];
131 
132  // Calculate 3*face-pyramid volume
133  scalar pyr3Vol = max(mag(fArea & (fCentre - cEst)), VSMALL);
134 
135  // Calculate face-pyramid centre
136  vector pc = 0.75*fCentre + 0.25*cEst;
137 
138  // Accumulate volume-weighted face-pyramid centre
139  subCellCentre_ += pyr3Vol*pc;
140 
141  // Accumulate face-pyramid volume
142  subCellVolume_ += pyr3Vol;
143  }
144 
145  subCellCentre_ /= subCellVolume_;
146  subCellVolume_ /= scalar(3);
147  VOF_ = subCellVolume_/mesh_.cellVolumes()[cellI_];
148 
149  subCellCentreAndVolumeCalculated_ = true;
150 
151  if (debug)
152  {
153  vector sumSf = isoFaceArea_;
154  scalar sumMagSf = mag(isoFaceArea_);
155  forAll(isoCutFaceCentres_, facei)
156  {
157  sumSf += isoCutFaceAreas_[facei];
158  sumMagSf += mag(isoCutFaceAreas_[facei]);
159  }
160  forAll(fullySubFaces_, facei)
161  {
162  sumSf += mesh_.faceAreas()[fullySubFaces_[facei]];
163  sumMagSf += mag(isoCutFaceAreas_[facei]);
164  }
165  if (mag(sumSf) > 1e-10)
166  {
167  Pout<< "Warninig: mag(sumSf)/magSumSf = "
168  << mag(sumSf)/sumMagSf << " for surface cell"
169  << cellI_ << endl;
170  }
171  }
172  }
173  else if (cellStatus_ == 1)
174  {
175  // Cell fully above isosurface
176  subCellCentre_ = vector::zero;
177  subCellVolume_ = 0;
178  VOF_ = 0;
179  }
180  else if (cellStatus_ == -1)
181  {
182  // Cell fully below isosurface
183  subCellCentre_ = mesh_.cellCentres()[cellI_];
184  subCellVolume_ = mesh_.cellVolumes()[cellI_];
185  VOF_ = 1;
186  }
187 }

190 void Foam::isoCutCell::calcIsoFaceCentreAndArea()
191 {
192  // Initial guess of face centre from edge points
193  point fCentre(Zero);
194  label nEdgePoints(0);
195  forAll(isoFaceEdges_, ei)
196  {
197  DynamicList<point>& edgePoints = isoFaceEdges_[ei];
198  forAll(edgePoints, pi)
199  {
200  fCentre += edgePoints[pi];
201  nEdgePoints++;
202  }
203  }
204 
205  if (nEdgePoints > 0)
206  {
207  fCentre /= nEdgePoints;
208  }
209  else
210  {
211  DebugPout << "Warning: nEdgePoints = 0 for cell " << cellI_ << endl;
212  }
213 
214  vector sumN(Zero);
215  scalar sumA(0);
216  vector sumAc(Zero);
217 
218  forAll(isoFaceEdges_, ei)
219  {
220  const DynamicList<point>& edgePoints = isoFaceEdges_[ei];
221  const label nPoints = edgePoints.size();
222  for (label pi = 0; pi < nPoints-1; pi++)
223  {
224  const point& nextPoint = edgePoints[pi + 1];
225 
226  vector c = edgePoints[pi] + nextPoint + fCentre;
227  vector n = (nextPoint - edgePoints[pi])^(fCentre - edgePoints[pi]);
228  scalar a = mag(n);
229 
230  // Edges may have different orientation
231  sumN += Foam::sign(n & sumN)*n;
232  sumA += a;
233  sumAc += a*c;
234  }
235  }
236 
237  // This is to deal with zero-area faces. Mark very small faces
238  // to be detected in e.g., processorPolyPatch.
239  if (sumA < ROOTVSMALL)
240  {
241  isoFaceCentre_ = fCentre;
242  isoFaceArea_ = vector::zero;
243  }
244  else
245  {
246  isoFaceCentre_ = sumAc/sumA/scalar(3);
247  isoFaceArea_ = 0.5*sumN;
248  }
249 
250 
251  // Check isoFaceArea_ direction and change if not pointing out of subcell
252  if ((isoFaceArea_ & (isoFaceCentre_ - subCellCentre())) < 0)
253  {
254  isoFaceArea_ *= (-1);
255  }
256 
257  isoFaceCentreAndAreaCalculated_ = true;
258 }

261 void Foam::isoCutCell::calcIsoFacePointsFromEdges()
262 {
263  DebugPout
264  << "Enter calcIsoFacePointsFromEdges() with isoFaceArea_ = "
265  << isoFaceArea_ << " and isoFaceCentre_ = " << isoFaceCentre_
266  << " and isoFaceEdges_ = " << isoFaceEdges_ << endl;
267 
268  // Defining local coordinates with zhat along isoface normal and xhat from
269  // isoface centre to first point in isoFaceEdges_
270  const vector zhat = normalised(isoFaceArea_);
271  vector xhat = isoFaceEdges_[0][0] - isoFaceCentre_;
272  xhat = (xhat - (xhat & zhat)*zhat);
273  xhat.normalise();
274  vector yhat = normalised(zhat ^ xhat);
275 
276  DebugPout << "Calculated local coordinates" << endl;
277 
278  // Calculating isoface point angles in local coordinates
279  DynamicList<point> unsortedIsoFacePoints(3*isoFaceEdges_.size());
280  DynamicList<scalar> unsortedIsoFacePointAngles(3*isoFaceEdges_.size());
281  forAll(isoFaceEdges_, ei)
282  {
283  const DynamicList<point>& edgePoints = isoFaceEdges_[ei];
284  forAll(edgePoints, pi)
285  {
286  const point& p = edgePoints[pi];
287  unsortedIsoFacePoints.append(p);
288  unsortedIsoFacePointAngles.append
289  (
290  Foam::atan2
291  (
292  ((p - isoFaceCentre_) & yhat),
293  ((p - isoFaceCentre_) & xhat)
294  )
295  );
296  }
297  }
298 
299  DebugPout<< "Calculated isoFace point angles" << endl;
300 
301  // Sorting isoface points by angle and inserting into isoFacePoints_
302  labelList order(sortedOrder(unsortedIsoFacePointAngles));
303  isoFacePoints_.append(unsortedIsoFacePoints[order[0]]);
304  for (label pi = 1; pi < order.size(); pi++)
305  {
306  if
307  (
308  mag
309  (
310  unsortedIsoFacePointAngles[order[pi]]
311  -unsortedIsoFacePointAngles[order[pi-1]]
312  ) > 1e-8
313  )
314  {
315  isoFacePoints_.append(unsortedIsoFacePoints[order[pi]]);
316  }
317  }
318 
319  DebugPout<< "Sorted isoface points by angle" << endl;
320 }

323 Foam::label Foam::isoCutCell::calcSubCell
324 (
325  const label celli,
326  const scalar isoValue
327 )
328 {
329  // Populate isoCutFaces_, isoCutFacePoints_, fullySubFaces_, isoFaceCentre_
330  // and isoFaceArea_.
331 
332  clearStorage();
333  cellI_ = celli;
334  isoValue_ = isoValue;
335  const cell& c = mesh_.cells()[celli];
336 
337  forAll(c, fi)
338  {
339  const label facei = c[fi];
340 
341  const label faceStatus = isoCutFace_.calcSubFace(facei, isoValue_);
342 
343  if (faceStatus == 0)
344  {
345  // Face is cut
346  isoCutFacePoints_.append(isoCutFace_.subFacePoints());
347  isoCutFaceCentres_.append(isoCutFace_.subFaceCentre());
348  isoCutFaceAreas_.append(isoCutFace_.subFaceArea());
349  isoFaceEdges_.append(isoCutFace_.surfacePoints());
350  }
351  else if (faceStatus == -1)
352  {
353  // Face fully below
354  fullySubFaces_.append(facei);
355  }
356  }
357 
358  if (isoCutFacePoints_.size())
359  {
360  // Cell cut at least at one face
361  cellStatus_ = 0;
362  calcIsoFaceCentreAndArea();
363 
364  // In the rare but occuring cases where a cell is only touched at a
365  // point or a line the isoFaceArea_ will have zero length and here the
366  // cell should be treated as either completely empty or full.
367  if (mag(isoFaceArea_) < 10*SMALL)
368  {
369  if (fullySubFaces_.empty())
370  {
371  // Cell fully above isosurface
372  cellStatus_ = 1;
373  }
374  else
375  {
376  // Cell fully below isosurface
377  cellStatus_ = -1;
378  }
379  }
380  }
381  else if (fullySubFaces_.empty())
382  {
383  // Cell fully above isosurface
384  cellStatus_ = 1;
385  }
386  else
387  {
388  // Cell fully below isosurface
389  cellStatus_ = -1;
390  }
391 
392  return cellStatus_;
393 }

491 Foam::label Foam::isoCutCell::vofCutCell
492 (
493  const label celli,
494  const scalar alpha1,
495  const scalar tol,
496  const label maxIter
497 )
498 {
499  DebugInFunction
500  << "vofCutCell for cell " << celli << " with alpha1 = "
501  << alpha1 << " ------" << endl;
502 
503  // Finding cell vertex extremum values
504  const labelList& pLabels = mesh_.cellPoints(celli);
505  scalarField fvert(pLabels.size());
506  forAll(pLabels, pi)
507  {
508  fvert[pi] = f_[pLabels[pi]];
509  }
510  labelList order(sortedOrder(fvert));
511  scalar f1 = fvert[order.first()];
512  scalar f2 = fvert[order.last()];
513 
514  DebugPout << "fvert = " << fvert << ", and order = " << order << endl;
515 
516  // Handling special case where method is handed an almost full/empty cell
517  if (alpha1 < tol)
518  {
519  return calcSubCell(celli, f2);
520  }
521  else if (1 - alpha1 < tol)
522  {
523  return calcSubCell(celli, f1);
524  }
525 
526  // Finding the two vertices inbetween which the isovalue giving alpha1 lies
527  label L1 = 0;
528  label L2 = fvert.size() - 1;
529  scalar a1 = 1;
530  scalar a2 = 0;
531  scalar L3, f3, a3;
532 
533  while (L2 - L1 > 1)
534  {
535  L3 = round(0.5*(L1 + L2));
536  f3 = fvert[order[L3]];
537  calcSubCell(celli, f3);
538  a3 = volumeOfFluid();
539  if (a3 > alpha1)
540  {
541  L1 = L3; f1 = f3; a1 = a3;
542  }
543  else if (a3 < alpha1)
544  {
545  L2 = L3; f2 = f3; a2 = a3;
546  }
547  }
548 
549  if (mag(f1 - f2) < 10*SMALL)
550  {
551  DebugPout<< "Warning: mag(f1 - f2) < 10*SMALL" << endl;
552  return calcSubCell(celli, f1);
553  }
554 
555  if (mag(a1 - a2) < tol)
556  {
557  DebugPout<< "Warning: mag(a1 - a2) < tol for cell " << celli << endl;
558  return calcSubCell(celli, 0.5*(f1 + f2));
559  }
560 
561  // Now we know that a(f) = alpha1 is to be found on the f interval
562  // [f1, f2], i.e. alpha1 will be in the interval [a2,a1]
563  DebugPout
564  << "L1 = " << L1 << ", f1 = " << f1 << ", a1 = " << a1 << nl
565  << "L2 = " << L2 << ", f2 = " << f2 << ", a2 = " << a2 << endl;
566 
567 
568  // Finding coefficients in 3 deg polynomial alpha(f) from 4 solutions
569 
570  // Finding 2 additional points on 3 deg polynomial
571  f3 = f1 + (f2 - f1)/scalar(3);
572  calcSubCell(celli, f3);
573  a3 = volumeOfFluid();
574 
575  scalar f4 = f1 + (f2 - f1)*scalar(2)/scalar(3);
576  calcSubCell(celli, f4);
577  scalar a4 = volumeOfFluid();
578 
579  // Building and solving Vandermonde matrix equation
580  scalarField a(4), f(4), C(4);
581  {
582  a[0] = a1, f[0] = 0;
583  a[1] = a3, f[1] = (f3 - f1)/(f2 - f1);
584  a[2] = a4, f[2] = (f4 - f1)/(f2 - f1);
585  a[3] = a2, f[3] = 1;
586  scalarSquareMatrix M(4);
587  forAll(f, i)
588  {
589  forAll(f, j)
590  {
591  M[i][j] = pow(f[i], 3 - j);
592  }
593  }
594 
595  // C holds the 4 polynomial coefficients
596  C = a;
597  LUsolve(M, C);
598  }
599 
600  // Finding root with Newton method
601  f3 = f[1]; a3 = a[1];
602  label nIter = 0;
603  scalar res = mag(a3 - alpha1);
604  while (res > tol && nIter < 10*maxIter)
605  {
606  f3 -=
607  (C[0]*pow3(f3) + C[1]*sqr(f3) + C[2]*f3 + C[3] - alpha1)
608  /(3*C[0]*sqr(f3) + 2*C[1]*f3 + C[2]);
609  a3 = C[0]*pow3(f3) + C[1]*sqr(f3) + C[2]*f3 + C[3];
610  res = mag(a3 - alpha1);
611  nIter++;
612  }
613  // Scaling back to original range
614  f3 = f3*(f2 - f1) + f1;
615 
616  // Check result
617  calcSubCell(celli, f3);
618  const scalar VOF = volumeOfFluid();
619  res = mag(VOF - alpha1);
620 
621  if (res > tol)
622  {
623  DebugPout
624  << "Newton obtained f3 = " << f3 << " and a3 = " << a3
625  << " with mag(a3-alpha1) = " << mag(a3-alpha1)
626  << " but calcSubCell(celli,f3) gives VOF = " << VOF << nl
627  << "M(f)*C = a with " << nl
628  << "f_scaled = " << f << nl
629  << "f = " << f*(f2 - f1) + f1 << nl
630  << "a = " << a << nl
631  << "C = " << C << endl;
632  }
633  else
634  {
635  DebugPout<< "Newton did the job" << endl;
636  return cellStatus_;
637  }
638 
639  // If tolerance not met use the secant method with f3 as a hopefully very
640  // good initial guess to crank res the last piece down below tol
641  // Note: This is expensive because subcell is recalculated every iteration
642  scalar x2 = f3;
643  scalar g2 = VOF - alpha1;
644  scalar x1 = max(1e-3*(f2 - f1), 100*SMALL);
645  x1 = min(max(x1, f1), f2);
646  calcSubCell(celli, x1);
647  scalar g1 = volumeOfFluid() - alpha1;
648 
649  nIter = 0;
650  scalar g0(0), x0(0);
651  while (res > tol && nIter < maxIter && g1 != g2)
652  {
653  x0 = (x2*g1 - x1*g2)/(g1 - g2);
654  calcSubCell(celli, x0);
655  g0 = volumeOfFluid() - alpha1;
656  res = mag(g0);
657  x2 = x1; g2 = g1;
658  x1 = x0; g1 = g0;
659  nIter++;
660  }
661 
662  if (debug)
663  {
664  if (res < tol)
665  {
666  Pout<< "Bisection finished the job in " << nIter << " iterations."
667  << endl;
668  }
669  else
670  {
671  Pout<< "Warning: Bisection not converged " << endl;
672  Pout<< "Leaving vofCutCell with f3 = " << f3 << " giving a3 = "
673  << a3 << " so alpha1 - a3 = " << alpha1 - a3 << endl;
674  }
675  }
676 
677  return cellStatus_;
678 }

681 void Foam::isoCutCell::volumeOfFluid
682 (
683  volScalarField& alpha1,
684  const scalar f0
685 )
686 {
687  // Setting internal field
688  scalarField& alphaIn = alpha1;
689  forAll(alphaIn, celli)
690  {
691  const label cellStatus = calcSubCell(celli, f0);
692  if (cellStatus != 1)
693  {
694  // If cell not entirely above isosurface
695  alphaIn[celli] = volumeOfFluid();
696  }
697  }
698 
699  // Setting boundary alpha1 values
700  forAll(mesh_.boundary(), patchi)
701  {
702  if (mesh_.boundary()[patchi].size() > 0)
703  {
704  const label start = mesh_.boundary()[patchi].patch().start();
705  scalarField& alphap = alpha1.boundaryFieldRef()[patchi];
706  const scalarField& magSfp = mesh_.magSf().boundaryField()[patchi];
707 
708  forAll(alphap, patchFacei)
709  {
710  const label facei = patchFacei + start;
711  const label faceStatus = isoCutFace_.calcSubFace(facei, f0);
712 
713  if (faceStatus != 1)
714  {
715  // Face not entirely above isosurface
716  alphap[patchFacei] =
717  mag(isoCutFace_.subFaceArea())/magSfp[patchFacei];
718  }
719  }
720  }
721  }
722 }
