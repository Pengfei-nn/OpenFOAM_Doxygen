123 bool Foam::functionObjects::yPlus::execute()
124 {
125  volScalarField& yPlus =
126  lookupObjectRef<volScalarField>(typeName);
127 
128  if (foundObject<turbulenceModel>(turbulenceModel::propertiesName))
129  {
130  volScalarField::Boundary& yPlusBf = yPlus.boundaryFieldRef();
131 
132  const turbulenceModel& model =
133  lookupObject<turbulenceModel>
134  (
135  turbulenceModel::propertiesName
136  );
137 
138  const nearWallDist nwd(mesh_);
139  const volScalarField::Boundary& d = nwd.y();
140 
141  // nut needed for wall function patches
142  tmp<volScalarField> tnut = model.nut();
143  const volScalarField::Boundary& nutBf = tnut().boundaryField();
144 
145  // U needed for plain wall patches
146  const volVectorField::Boundary& UBf = model.U().boundaryField();
147 
148  const fvPatchList& patches = mesh_.boundary();
149 
150  forAll(patches, patchi)
151  {
152  const fvPatch& patch = patches[patchi];
153 
154  if (isA<nutWallFunctionFvPatchScalarField>(nutBf[patchi]))
155  {
156  const nutWallFunctionFvPatchScalarField& nutPf =
157  dynamic_cast<const nutWallFunctionFvPatchScalarField&>
158  (
159  nutBf[patchi]
160  );
161 
162  yPlusBf[patchi] = nutPf.yPlus();
163  }
164  else if (isA<wallFvPatch>(patch))
165  {
166  yPlusBf[patchi] =
167  d[patchi]
168  *sqrt
169  (
170  model.nuEff(patchi)
171  *mag(UBf[patchi].snGrad())
172  )/model.nu(patchi);
173  }
174  }
175  }
176  else
177  {
178  WarningInFunction
179  << "Unable to find turbulence model in the "
180  << "database: yPlus will not be calculated" << endl;
181  return false;
182  }
183 
184  return true;
185 }

188 bool Foam::functionObjects::yPlus::write()
189 {
190  const volScalarField& yPlus =
191  obr_.lookupObject<volScalarField>(type());
192 
193  Log << type() << " " << name() << " write:" << nl
194  << " writing field " << yPlus.name() << endl;
195 
196  yPlus.write();
197 
198  const volScalarField::Boundary& yPlusBf = yPlus.boundaryField();
199  const fvPatchList& patches = mesh_.boundary();
200 
201  forAll(patches, patchi)
202  {
203  const fvPatch& patch = patches[patchi];
204 
205  if (isA<wallFvPatch>(patch))
206  {
207  const scalarField& yPlusp = yPlusBf[patchi];
208 
209  const scalar minYplus = gMin(yPlusp);
210  const scalar maxYplus = gMax(yPlusp);
211  const scalar avgYplus = gAverage(yPlusp);
212 
213  if (Pstream::master())
214  {
215  Log << " patch " << patch.name()
216  << " y+ : min = " << minYplus << ", max = " << maxYplus
217  << ", average = " << avgYplus << nl;
218 
219  writeCurrentTime(file());
220  file()
221  << token::TAB << patch.name()
222  << token::TAB << minYplus
223  << token::TAB << maxYplus
224  << token::TAB << avgYplus
225  << endl;
226  }
227  }
228  }
229 
230  return true;
231 }
