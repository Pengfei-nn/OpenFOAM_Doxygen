65 void thermalBaffleModel::init()
66 {
67  if (active_)
68  {
69  const polyBoundaryMesh& rbm = regionMesh().boundaryMesh();
70 
71  // Check if region mesh in 1-D
72  label nTotalEdges = 0;
73  const label patchi = intCoupledPatchIDs_[0];
74  nTotalEdges = 2*nLayers_*rbm[patchi].nInternalEdges();
75  nTotalEdges +=
76  nLayers_*(rbm[patchi].nEdges() - rbm[patchi].nInternalEdges());
77 
78  reduce(nTotalEdges, sumOp<label>());
79 
80  label nFaces = 0;
81  forAll(rbm, patchi)
82  {
83  if (
84  rbm[patchi].size()
85  &&
86  (
87  isA<wedgePolyPatch>(rbm[patchi])
88  || isA<emptyPolyPatch>(rbm[patchi])
89  )
90  )
91  {
92  nFaces += rbm[patchi].size();
93  }
94  }
95  reduce(nFaces, sumOp<label>());
96 
97  if (nTotalEdges == nFaces)
98  {
99  oneD_ = true;
100  Info << "\nThe thermal baffle is 1D \n" << endl;
101  }
102  else
103  {
104  Info << "\nThe thermal baffle is 3D \n" << endl;
105  }
106 
107  forAll(intCoupledPatchIDs_, i)
108  {
109  const label patchi = intCoupledPatchIDs_[i];
110  const polyPatch& pp = rbm[patchi];
111 
112  if
113  (
114  !isA<mappedVariableThicknessWallPolyPatch>(pp)
115  && oneD_
116  && !constantThickness_
117  )
118  {
119  FatalErrorInFunction
120  << "' not type '"
121  << mappedVariableThicknessWallPolyPatch::typeName
122  << "'. This is necessary for 1D solution "
123  << " and variable thickness"
124  << "\n for patch. " << pp.name()
125  << exit(FatalError);
126  }
127  else if (!isA<mappedWallPolyPatch>(pp))
128  {
129  FatalErrorInFunction
130  << "' not type '"
131  << mappedWallPolyPatch::typeName
132  << "'. This is necessary for 3D solution"
133  << "\n for patch. " << pp.name()
134  << exit(FatalError);
135  }
136  }
137 
138  if (oneD_ && !constantThickness_)
139  {
140  const label patchi = intCoupledPatchIDs_[0];
141  const polyPatch& pp = rbm[patchi];
142  const mappedVariableThicknessWallPolyPatch& ppCoupled =
143  refCast
144  <
145  const mappedVariableThicknessWallPolyPatch
146  >(pp);
147 
148  thickness_ = ppCoupled.thickness();
149 
150  // Check that thickness has the right size
151  if (thickness_.size() != pp.size())
152  {
153  FatalErrorInFunction
154  << " coupled patches in thermalBaffle are " << nl
155  << " different sizes from list thickness" << nl
156  << exit(FatalError);
157  }
158 
159  // Calculate thickness of the baffle on the first face only.
160  if (delta_.value() == 0.0)
161  {
162  forAll(ppCoupled, localFacei)
163  {
164  label facei = ppCoupled.start() + localFacei;
165 
166  label faceO =
167  boundaryFaceOppositeFace_[localFacei];
168 
169  delta_.value() = mag
170  (
171  regionMesh().faceCentres()[facei]
172  - regionMesh().faceCentres()[faceO]
173  );
174  break;
175  }
176  }
177  }
178  }
179 }
