93 void Foam::diameterModels::coalescenceModels::PrinceBlanch::
94 addToCoalescenceRate
95 (
96  volScalarField& coalescenceRate,
97  const label i,
98  const label j
99 )
100 {
101  const phaseModel& continuousPhase = popBal_.continuousPhase();
102  const sizeGroup& fi = popBal_.sizeGroups()[i];
103  const sizeGroup& fj = popBal_.sizeGroups()[j];
104  const uniformDimensionedVectorField& g =
105  popBal_.mesh().time().lookupObject<uniformDimensionedVectorField>("g");
106 
107  const dimensionedScalar rij(1.0/(1.0/fi.d() + 1.0/fj.d()));
108 
109  const volScalarField collisionEfficiency
110  (
111  exp
112  (
113  - sqrt
114  (
115  pow3(rij)*continuousPhase.rho()
116  /(16.0*popBal_.sigmaWithContinuousPhase(fi.phase()))
117  )
118  *log(h0_/hf_)
119  *cbrt(popBal_.continuousTurbulence().epsilon())/pow(rij, 2.0/3.0)
120  )
121  );
122 
123  if (turbulence_)
124  {
125  coalescenceRate +=
126  (
127  C1_*pi*sqr(fi.d() + fj.d())
128  *cbrt(popBal_.continuousTurbulence().epsilon())
129  *sqrt(pow(fi.d(), 2.0/3.0) + pow(fj.d(), 2.0/3.0))
130  )
131  *collisionEfficiency;
132  }
133 
134  if (buoyancy_)
135  {
136  const dimensionedScalar Sij(pi/4.0*sqr(fi.d() + fj.d()));
137 
138  coalescenceRate +=
139  (
140  Sij
141  *mag
142  (
143  sqrt
144  (
145  2.14*popBal_.sigmaWithContinuousPhase(fi.phase())
146  /(continuousPhase.rho()*fi.d()) + 0.505*mag(g)*fi.d()
147  )
148  - sqrt
149  (
150  2.14*popBal_.sigmaWithContinuousPhase(fi.phase())
151  /(continuousPhase.rho()*fj.d()) + 0.505*mag(g)*fj.d()
152  )
153  )
154  )
155  *collisionEfficiency;
156  }
157 
158  if (laminarShear_)
159  {
160  FatalErrorInFunction
161  << "Laminar shear collision contribution not implemented for "
162  << this->type() << " coalescence model."
163  << exit(FatalError);
164  }
165 }
