168 static inline scalar averageSurrounding
169 (

171  const label i,
172  const label j
173 )
174 {
175  return
176  (
177  mat(i,j) + mat(i,j+1) + mat(i,j+2)
178  + mat(i+1,j) /* centre */ + mat(i+1,j+2)
179  + mat(i+2,j) + mat(i+2,j+1) + mat(i+2,j+2)
180  ) / 8.0; // Average
181 }

1008 void calc_drag_etc
1009 (
1010  double brs, double brr, bool blocked,
1011  double surr_br, double surr_dr,
1012  scalar* drags_p, scalar* dragr_p,
1013  double count,
1014  scalar* cbdi_p,
1015  double cell_vol
1016 )
1017 {
1018  // Total blockage ratio
1019  scalar br = brr + brs;
1020 
1021  // Idealise obstacle arrangement as sqrt(count) rows.
1022  // Make br the blockage ratio for each row.
1023  if (count > 1.0) { br /= std::sqrt(count); }
1024 
1025  const scalar alpha =
1026  (
1027  br < 0.99
1028  ? (1.0 - 0.5 * br) / (1.0 - br) / (1.0 - br)
1029  : GREAT
1030  );
1031 
1032  // For the moment keep separate the two contributions to the blockage-corrected drag
1033  /* An isolated long obstcale will have two of the surronding eight cells with the same blockage,
1034  so surr_br would be br/4. In this case no correction. Rising to full correction when
1035  all surrounding cells have the same blockage. */
1036  const scalar expon =
1037  (
1038  br > 0.0
1039  ? min(max((surr_br / br - 0.25) * 4.0 / 3.0, scalar(0)), scalar(1))
1040  : 0.0
1041  );
1042 
1043  const scalar alpha_r = ::pow(alpha, 0.5 + 0.5 * expon);
1044  const scalar alpha_s = ::pow(alpha, expon);
1045 
1046  *dragr_p *= alpha_r;
1047  *drags_p *= ::pow(alpha_s, 1.09);
1048  *cbdi_p = ( pars.cb_r * pars.cd_r * *dragr_p + pars.cb_s * pars.cd_s * *drags_p );
1049  if ( *cbdi_p < 0.0 ) { *cbdi_p = 0.0; }
1050 
1051  // Finally sum the drag.
1052  *drags_p = ( *drags_p * pars.cd_s + *dragr_p * pars.cd_r );
1053  if ( *drags_p < 0.0 ) { *drags_p = 0.0; }
1054  /* If well-blocked cells are surrounded by empty cells, the flow just goes round
1055  and the drag parameters have little effect. So, for any cells much more empty
1056  than the surrounding cells, we put some CR in there as well. */
1057  if ( (surr_dr * 0.25) > *drags_p )
1058  {
1059  *drags_p = surr_dr * 0.25;
1060  *cbdi_p = *drags_p * (pars.cb_r + pars.cb_s ) * 0.5;
1061  // Don't know whether surr. stuff was round or sharp; use average of cb factors
1062  }
1063  if ( blocked ) { *cbdi_p = 0.0; *drags_p = 0.0; *dragr_p = 0.0; }
1064 }
