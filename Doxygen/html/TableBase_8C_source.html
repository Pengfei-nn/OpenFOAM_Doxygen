110 template<class Type>
111 void Foam::Function1Types::TableBase<Type>::check() const
112 {
113  if (!table_.size())
114  {
115  FatalErrorInFunction
116  << "Table for entry " << this->name_ << " is invalid (empty)"
117  << nl << exit(FatalError);
118  }
119 
120  scalar prevValue(0);
121 
122  label i = 0;
123  for (const auto& item : table_)
124  {
125  const scalar& currValue = item.first();
126 
127  // Avoid duplicate values (divide-by-zero error)
128  if (i && currValue <= prevValue)
129  {
130  FatalErrorInFunction
131  << "out-of-order value: "
132  << currValue << " at index " << i << nl
133  << exit(FatalError);
134  }
135  prevValue = currValue;
136  ++i;
137  }
138 }

141 template<class Type>
142 bool Foam::Function1Types::TableBase<Type>::checkMinBounds
143 (
144  const scalar x,
145  scalar& xDash
146 ) const
147 {
148  const scalar minLimit = table_.first().first();
149  const scalar maxLimit = table_.last().first();
150 
151  if (x < minLimit)
152  {
153  switch (bounding_)
154  {
155  case bounds::repeatableBounding::ERROR:
156  {
157  FatalErrorInFunction
158  << "value (" << x << ") less than lower "
159  << "bound (" << minLimit << ")" << nl
160  << exit(FatalError);
161  break;
162  }
163  case bounds::repeatableBounding::WARN:
164  {
165  WarningInFunction
166  << "value (" << x << ") less than lower "
167  << "bound (" << minLimit << ")" << nl
168  << " Continuing with the first entry" << endl;
169 
170  // Behaviour as per CLAMP
171  xDash = minLimit;
172  return true;
173  break;
174  }
175  case bounds::repeatableBounding::CLAMP:
176  {
177  xDash = minLimit;
178  return true;
179  break;
180  }
181  case bounds::repeatableBounding::REPEAT:
182  {
183  // Adjust x to >= minX
184  const scalar span = maxLimit - minLimit;
185  xDash = fmod(x - minLimit, span) + minLimit;
186  break;
187  }
188  }
189  }
190  else
191  {
192  xDash = x;
193  }
194 
195  return false;
196 }

272 {
