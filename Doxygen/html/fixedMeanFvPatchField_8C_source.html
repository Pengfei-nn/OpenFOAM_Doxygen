100 template<class Type>
101 void Foam::fixedMeanFvPatchField<Type>::updateCoeffs()
102 {
103  if (this->updated())
104  {
105  return;
106  }
107 
108  const scalar t = this->db().time().timeOutputValue();
109  Type meanValue = meanValue_->value(t);
110 
111  Field<Type> newValues(this->patchInternalField());
112 
113  Type meanValuePsi =
114  gSum(this->patch().magSf()*newValues)
115  /gSum(this->patch().magSf());
116 
117  if (mag(meanValue) > SMALL && mag(meanValuePsi)/mag(meanValue) > 0.5)
118  {
119  newValues *= mag(meanValue)/mag(meanValuePsi);
120  }
121  else
122  {
123  newValues += (meanValue - meanValuePsi);
124  }
125 
126  this->operator==(newValues);
127 
128  fixedValueFvPatchField<Type>::updateCoeffs();
129 }
