136 
137  forAll(zones, zonei)

139  const label colourI = zonei % 8;

141 
142  os << "MATERIAL \"" << zones[zonei].name() << "Mat\" rgb "
143  << colourMap[colourCompI] << ' ' << colourMap[colourCompI+1]
144  << ' ' << colourMap[colourCompI+2]

148  }
149 
150  os << "OBJECT world" << nl
