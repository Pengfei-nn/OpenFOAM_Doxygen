351 void Foam::twoPhaseSystem::solve()
352 {
353  const Time& runTime = mesh_.time();
354 
355  volScalarField& alpha1 = phase1_;
356  volScalarField& alpha2 = phase2_;
357 
358  const surfaceScalarField& phi1 = phase1_.phi();
359  const surfaceScalarField& phi2 = phase2_.phi();
360 
361  const dictionary& alphaControls = mesh_.solverDict
362  (
363  alpha1.name()
364  );
365 
366  label nAlphaSubCycles(alphaControls.get<label>("nAlphaSubCycles"));
367  label nAlphaCorr(alphaControls.get<label>("nAlphaCorr"));
368 
369  word alphaScheme("div(phi," + alpha1.name() + ')');
370  word alpharScheme("div(phir," + alpha1.name() + ')');
371 
372  alpha1.correctBoundaryConditions();
373 
374 
375  surfaceScalarField phic("phic", phi_);
376  surfaceScalarField phir("phir", phi1 - phi2);
377 
378  tmp<surfaceScalarField> alpha1alpha2f;
379 
380  if (pPrimeByA_.valid())
381  {
382  alpha1alpha2f =
383  fvc::interpolate(max(alpha1, scalar(0)))
384  *fvc::interpolate(max(alpha2, scalar(0)));
385 
386  surfaceScalarField phiP
387  (
388  pPrimeByA_()*fvc::snGrad(alpha1, "bounded")*mesh_.magSf()
389  );
390 
391  phir += phiP;
392  }
393 
394  for (int acorr=0; acorr<nAlphaCorr; acorr++)
395  {
396  volScalarField::Internal Sp
397  (
398  IOobject
399  (
400  "Sp",
401  runTime.timeName(),
402  mesh_
403  ),
404  mesh_,
405  dimensionedScalar("Sp", dgdt_.dimensions(), 0.0)
406  );
407 
408  volScalarField::Internal Su
409  (
410  IOobject
411  (
412  "Su",
413  runTime.timeName(),
414  mesh_
415  ),
416  // Divergence term is handled explicitly to be
417  // consistent with the explicit transport solution
418  fvc::div(phi_)*min(alpha1, scalar(1))
419  );
420 
421  forAll(dgdt_, celli)
422  {
423  if (dgdt_[celli] > 0.0)
424  {
425  Sp[celli] -= dgdt_[celli]/max(1.0 - alpha1[celli], 1e-4);
426  Su[celli] += dgdt_[celli]/max(1.0 - alpha1[celli], 1e-4);
427  }
428  else if (dgdt_[celli] < 0.0)
429  {
430  Sp[celli] += dgdt_[celli]/max(alpha1[celli], 1e-4);
431  }
432  }
433 
434  surfaceScalarField alphaPhic1
435  (
436  fvc::flux
437  (
438  phic,
439  alpha1,
440  alphaScheme
441  )
442  + fvc::flux
443  (
444  -fvc::flux(-phir, scalar(1) - alpha1, alpharScheme),
445  alpha1,
446  alpharScheme
447  )
448  );
449 
450  phase1_.correctInflowOutflow(alphaPhic1);
451 
452  if (nAlphaSubCycles > 1)
453  {
454  for
455  (
456  subCycle<volScalarField> alphaSubCycle(alpha1, nAlphaSubCycles);
457  !(++alphaSubCycle).end();
458  )
459  {
460  surfaceScalarField alphaPhic10(alphaPhic1);
461 
462  MULES::explicitSolve
463  (
464  geometricOneField(),
465  alpha1,
466  phi_,
467  alphaPhic10,
468  (alphaSubCycle.index()*Sp)(),
469  (Su - (alphaSubCycle.index() - 1)*Sp*alpha1)(),
470  UniformField<scalar>(phase1_.alphaMax()),
471  zeroField()
472  );
473 
474  if (alphaSubCycle.index() == 1)
475  {
476  phase1_.alphaPhi() = alphaPhic10;
477  }
478  else
479  {
480  phase1_.alphaPhi() += alphaPhic10;
481  }
482  }
483 
484  phase1_.alphaPhi() /= nAlphaSubCycles;
485  }
486  else
487  {
488  MULES::explicitSolve
489  (
490  geometricOneField(),
491  alpha1,
492  phi_,
493  alphaPhic1,
494  Sp,
495  Su,
496  UniformField<scalar>(phase1_.alphaMax()),
497  zeroField()
498  );
499 
500  phase1_.alphaPhi() = alphaPhic1;
501  }
502 
503  if (pPrimeByA_.valid())
504  {
505  fvScalarMatrix alpha1Eqn
506  (
507  fvm::ddt(alpha1) - fvc::ddt(alpha1)
508  - fvm::laplacian(alpha1alpha2f()*pPrimeByA_(), alpha1, "bounded")
509  );
510 
511  alpha1Eqn.relax();
512  alpha1Eqn.solve();
513 
514  phase1_.alphaPhi() += alpha1Eqn.flux();
515  }
516 
517  phase1_.alphaRhoPhi() =
518  fvc::interpolate(phase1_.rho())*phase1_.alphaPhi();
519 
520  phase2_.alphaPhi() = phi_ - phase1_.alphaPhi();
521  phase2_.correctInflowOutflow(phase2_.alphaPhi());
522  phase2_.alphaRhoPhi() =
523  fvc::interpolate(phase2_.rho())*phase2_.alphaPhi();
524 
525  Info<< alpha1.name() << " volume fraction = "
526  << alpha1.weightedAverage(mesh_.V()).value()
527  << " Min(" << alpha1.name() << ") = " << min(alpha1).value()
528  << " Max(" << alpha1.name() << ") = " << max(alpha1).value()
529  << endl;
530 
531  // Ensure the phase-fractions are bounded
532  alpha1.max(0);
533  alpha1.min(1);
534 
535  alpha2 = scalar(1) - alpha1;
536  }
537 }
