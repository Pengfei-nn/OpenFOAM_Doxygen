171 //- Return the rigid-body inertia of the combined body
172 inline rigidBodyInertia operator+
173 (
174  const rigidBodyInertia& rbi1,
175  const rigidBodyInertia& rbi2
176 )
177 {
178  const scalar m12 = rbi1.m() + rbi2.m();
179  const vector c12 = (rbi1.m()*rbi1.c() + rbi2.m()*rbi2.c())/m12;
180 
181  return rigidBodyInertia
182  (
183  m12,
184  c12,
185  rbi1.Ic() + rbi1.Icc(c12) + rbi2.Ic() + rbi2.Icc(c12)
186  );
187 }

191 inline spatialVector operator&
192 (
