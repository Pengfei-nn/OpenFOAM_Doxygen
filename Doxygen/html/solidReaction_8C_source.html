149  OStringStream& reaction
150 ) const
151 {
152  for (label i = 0; i < glhs().size(); ++i)
153  {
154  if (i > 0)

156  reaction << " + ";

158  if (mag(glhs()[i].stoichCoeff - 1) > SMALL)
