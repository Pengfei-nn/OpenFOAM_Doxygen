112 bool Foam::functionObjects::runTimePostPro::scalarBar::add
113 (
114  const vector& textColour,
115  vtkRenderer* renderer,
116  vtkLookupTable* lut
117 ) const
118 {
119  // Add scalar bar legend
120 
121  if (!visible_ || !renderer)
122  {
123  return false;
124  }
125 
126  const label fontSizeValue = (fontSize_ ? fontSize_ : defaultFontSize);
127 
128  auto sbar = vtkSmartPointer<vtkScalarBarActor>::New();
129  sbar->SetLookupTable(lut);
130  sbar->SetNumberOfLabels(nLabels_);
131  sbar->SetLabelFormat(labelFormat_.c_str());
132 
134 
135  // Work-around to supply our own scalarbar title
136  // - Default scalar bar title text is scales by the scalar bar box
137  // dimensions so if the title is a long string, the text is shrunk to fit
138  // Instead, suppress title and set the title using a vtkTextActor
139 
140  vtkSmartPointer<vtkTextActor> titleActor;
141  vtkTextProperty* titleProp;
142 
143  if (titleHack_)
144  {
145  // Place the scalar bar title ourselves
146  sbar->SetUnconstrainedFontSize(true);
147 
148  titleActor = vtkSmartPointer<vtkTextActor>::New();
149  titleActor->SetInput(title_.c_str());
150 
151  titleProp = titleActor->GetTextProperty();
152  titleProp->SetJustificationToCentered();
153  }
154  else
155  {
156  // Use the standard scalar bar title
157  sbar->SetUnconstrainedFontSize(fontSize_ != 0);
158  sbar->SetTitle(title_.c_str());
159  titleProp = sbar->GetTitleTextProperty();
160  }
161 
162  titleProp->SetFontFamilyToArial();
163 
164  // Title size was supplied by user (absolute size)
165  // or use preset factor (3) of label font size
166 
167  if (titleSize_)
168  {
169  titleProp->SetFontSize(titleSize_);
170  }
171  else
172  {
173  // Auto
174  titleProp->SetFontSize(defaultTitleSizeFactor*fontSizeValue);
175 
176  // Or this??
177  // if (!titleHack_) titleProp->SetFontSize(fontSizeValue);
178  }
179 
180  titleProp->SetJustificationToCentered();
181  titleProp->SetVerticalJustificationToBottom();
182  titleProp->SetBold(bold_);
183  titleProp->SetItalic(italic_);
184  titleProp->SetShadow(shadow_);
185 
186  titleProp->SetColor(textColour[0], textColour[1], textColour[2]);
187 
188  auto labProp = sbar->GetLabelTextProperty();
189 
190  labProp->SetColor(textColour[0], textColour[1], textColour[2]);
191 
192  if (titleHack_ || fontSize_)
193  {
194  labProp->SetFontSize(fontSizeValue);
195  }
196  labProp->ShadowOff();
197  labProp->BoldOff(); // or: labProp->SetBold(bold_);
198  labProp->ItalicOff();
199 
200  // Positioning
201  {
202  vtkCoordinate* coord = sbar->GetPositionCoordinate();
203 
204  coord->SetCoordinateSystemToNormalizedViewport();
205  coord->SetValue(position_.first(), position_.second());
206  }
207 
208  if (vertical_)
209  {
210  sbar->SetOrientationToVertical();
211  sbar->SetTextPositionToSucceedScalarBar();
212  sbar->SetWidth(size_.first());
213  sbar->SetHeight(size_.second());
214  // Standard is sbar->SetBarRatio(0.375);
215  }
216  else
217  {
218  sbar->SetOrientationToHorizontal();
219  sbar->SetTextPositionToPrecedeScalarBar();
220 
221  // Adjustments since not using scalarbar title property
222  sbar->SetWidth(size_.first());
223  sbar->SetHeight(size_.second());
224  // sbar->SetBarRatio(0.5);
225  // Standard is sbar->SetBarRatio(0.375);
226  // sbar->SetTitleRatio(0.01);
227  }
228 
229  if (titleActor)
230  {
231  vtkCoordinate* coord = titleActor->GetPositionCoordinate();
232 
233  coord->SetCoordinateSystemToNormalizedViewport();
234 
235  coord->SetValue
236  (
237  position_.first() + (0.5 * sbar->GetWidth()),
238  position_.second() + (1.01 * sbar->GetHeight())
239  );
240  }
241 
242  // sbar->DrawFrameOn();
243  // sbar->DrawBackgroundOn();
244  // sbar->UseOpacityOff();
245  // sbar->VisibilityOff();
246  sbar->VisibilityOn();
247 
248  renderer->AddActor(sbar);
249 
250  if (titleActor)
251  {
252  renderer->AddActor2D(titleActor);
253  }
254 
255  return true;
256 }
