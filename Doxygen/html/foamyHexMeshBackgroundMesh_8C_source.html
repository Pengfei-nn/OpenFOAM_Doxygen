302 // Determine for every point a signed distance to the nearest surface
303 // (outside is positive)
304 tmp<scalarField> signedDistance
305 (
306  const scalarField& distSqr,
307  const pointField& points,
308  const searchableSurfaces& geometry,
309  const labelList& surfaces
310 )
311 {
312  tmp<scalarField> tfld(new scalarField(points.size(), Foam::sqr(GREAT)));
313  scalarField& fld = tfld();
314 
315  // Find nearest
316  List<pointIndexHit> nearest;
317  labelList nearestSurfaces;
318  searchableSurfacesQueries::findNearest
319  (
320  geometry,
321  surfaces,
322  points,
323  scalarField(points.size(), Foam::sqr(GREAT)),//distSqr
324  nearestSurfaces,
325  nearest
326  );
327 
328  // Determine sign of nearest. Sort by surface to do this.
329  DynamicField<point> surfPoints(points.size());
330  DynamicList<label> surfIndices(points.size());
331 
332  forAll(surfaces, surfI)
333  {
334  // Extract points on this surface
335  surfPoints.clear();
336  surfIndices.clear();
337  forAll(nearestSurfaces, i)
338  {
339  if (nearestSurfaces[i] == surfI)
340  {
341  surfPoints.append(points[i]);
342  surfIndices.append(i);
343  }
344  }
345 
346  // Calculate sideness of these surface points
347  label geomI = surfaces[surfI];
348  List<volumeType> volType;
349  geometry[geomI].getVolumeType(surfPoints, volType);
350 
351  // Push back to original
352  forAll(volType, i)
353  {
354  label pointi = surfIndices[i];
355  scalar dist = mag(points[pointi] - nearest[pointi].hitPoint());
356 
357  volumeType vT = volType[i];
358 
359  if (vT == volumeType::OUTSIDE)
360  {
361  fld[pointi] = dist;
362  }
363  else if (vT == volumeType::INSIDE)
364  {
365  fld[i] = -dist;
366  }
367  else
368  {
369  FatalErrorInFunction
370  << "getVolumeType failure, neither INSIDE or OUTSIDE"
371  << exit(FatalError);
372  }
373  }
374  }
375 
376  return tfld;
377 }
