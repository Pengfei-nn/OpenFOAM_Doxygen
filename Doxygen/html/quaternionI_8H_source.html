210 inline Foam::quaternion::quaternion
211 (
212  const tensor& rotationTensor
213 )
214 {
215  scalar trace =
216  rotationTensor.xx()
217  + rotationTensor.yy()
218  + rotationTensor.zz();
219 
220  if (trace > 0)
221  {
222  scalar s = 0.5/Foam::sqrt(trace + 1.0);
223 
224  w_ = 0.25/s;
225  v_[0] = (rotationTensor.zy() - rotationTensor.yz())*s;
226  v_[1] = (rotationTensor.xz() - rotationTensor.zx())*s;
227  v_[2] = (rotationTensor.yx() - rotationTensor.xy())*s;
228  }
229  else
230  {
231  if
232  (
233  rotationTensor.xx() > rotationTensor.yy()
234  && rotationTensor.xx() > rotationTensor.zz()
235  )
236  {
237  const scalar s = 2.0*Foam::sqrt
238  (
239  1.0
240  + rotationTensor.xx()
241  - rotationTensor.yy()
242  - rotationTensor.zz()
243  );
244 
245  w_ = (rotationTensor.zy() - rotationTensor.yz())/s;
246  v_[0] = 0.25*s;
247  v_[1] = (rotationTensor.xy() + rotationTensor.yx())/s;
248  v_[2] = (rotationTensor.xz() + rotationTensor.zx())/s;
249  }
250  else if
251  (
252  rotationTensor.yy() > rotationTensor.zz()
253  )
254  {
255  const scalar s = 2.0*Foam::sqrt
256  (
257  1.0
258  + rotationTensor.yy()
259  - rotationTensor.xx()
260  - rotationTensor.zz()
261  );
262 
263  w_ = (rotationTensor.xz() - rotationTensor.zx())/s;
264  v_[0] = (rotationTensor.xy() + rotationTensor.yx())/s;
265  v_[1] = 0.25*s;
266  v_[2] = (rotationTensor.yz() + rotationTensor.zy())/s;
267  }
268  else
269  {
270  const scalar s = 2.0*Foam::sqrt
271  (
272  1.0
273  + rotationTensor.zz()
274  - rotationTensor.xx()
275  - rotationTensor.yy()
276  );
277 
278  w_ = (rotationTensor.yx() - rotationTensor.xy())/s;
279  v_[0] = (rotationTensor.xz() + rotationTensor.zx())/s;
280  v_[1] = (rotationTensor.yz() + rotationTensor.zy())/s;
281  v_[2] = 0.25*s;
282  }
283  }
284 }

358 inline Foam::tensor Foam::quaternion::R() const
359 {
360  const scalar w2 = sqr(w());
361  const scalar x2 = sqr(v().x());
362  const scalar y2 = sqr(v().y());
363  const scalar z2 = sqr(v().z());
364 
365  const scalar txy = 2*v().x()*v().y();
366  const scalar twz = 2*w()*v().z();
367  const scalar txz = 2*v().x()*v().z();
368  const scalar twy = 2*w()*v().y();
369  const scalar tyz = 2*v().y()*v().z();
370  const scalar twx = 2*w()*v().x();
371 
372  return tensor
373  (
374  w2 + x2 - y2 - z2, txy - twz, txz + twy,
375  txy + twz, w2 - x2 + y2 - z2, tyz - twx,
376  txz - twy, tyz + twx, w2 - x2 - y2 + z2
377  );
378 }
