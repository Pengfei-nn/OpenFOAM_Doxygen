61 inline uint32_t Foam::endian::swap32(uint32_t u)
62 {
63 #ifdef USE_BUILTIN_BYTESWAP
64  return __builtin_bswap32(u);
65 #else
66  return
67  (
68  (((u) & 0xff000000) >> 24) // 3 -> 0
69  | (((u) & 0x00ff0000) >> 8) // 2 -> 1
70  | (((u) & 0x0000ff00) << 8) // 2 <- 1
71  | (((u) & 0x000000ff) << 24) // 3 <- 0
72  );
73
