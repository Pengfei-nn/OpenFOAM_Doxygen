346 void Foam::RASModels::kineticTheoryModel::correct()
347 {
348  // Local references
349  volScalarField alpha(max(alpha_, scalar(0)));
350  const volScalarField& rho = phase_.rho();
351  const surfaceScalarField& alphaRhoPhi = alphaRhoPhi_;
352  const volVectorField& U = U_;
353  const volVectorField& Uc_ =
354  refCast<const twoPhaseSystem>(phase_.fluid()).otherPhase(phase_).U();
355 
356  const scalar sqrtPi = sqrt(constant::mathematical::pi);
357  dimensionedScalar ThetaSmall("ThetaSmall", Theta_.dimensions(), 1e-6);
358  dimensionedScalar ThetaSmallSqrt(sqrt(ThetaSmall));
359 
360  tmp<volScalarField> tda(phase_.d());
361  const volScalarField& da = tda();
362 
363  tmp<volTensorField> tgradU(fvc::grad(U_));
364  const volTensorField& gradU(tgradU());
365  volSymmTensorField D(symm(gradU));
366 
367  // Calculating the radial distribution function
368  gs0_ = radialModel_->g0(alpha, alphaMinFriction_, alphaMax_);
369 
370  if (!equilibrium_)
371  {
372  // Particle viscosity (Table 3.2, p.47)
373  nut_ = viscosityModel_->nu(alpha, Theta_, gs0_, rho, da, e_);
374 
375  volScalarField ThetaSqrt("sqrtTheta", sqrt(Theta_));
376 
377  // Bulk viscosity p. 45 (Lun et al. 1984).
378  lambda_ = (4.0/3.0)*sqr(alpha)*da*gs0_*(1 + e_)*ThetaSqrt/sqrtPi;
379 
380  // Stress tensor, Definitions, Table 3.1, p. 43
381  volSymmTensorField tau
382  (
383  rho*(2*nut_*D + (lambda_ - (2.0/3.0)*nut_)*tr(D)*I)
384  );
385 
386  // Dissipation (Eq. 3.24, p.50)
387  volScalarField gammaCoeff
388  (
389  "gammaCoeff",
390  12*(1 - sqr(e_))
391  *max(sqr(alpha), residualAlpha_)
392  *rho*gs0_*(1.0/da)*ThetaSqrt/sqrtPi
393  );
394 
395  // Drag
396  volScalarField beta
397  (
398  refCast<const twoPhaseSystem>(phase_.fluid()).Kd()
399  );
400 
401  // Eq. 3.25, p. 50 Js = J1 - J2
402  volScalarField J1("J1", 3*beta);
403  volScalarField J2
404  (
405  "J2",
406  0.25*sqr(beta)*da*magSqr(U - Uc_)
407  /(
408  max(alpha, residualAlpha_)*rho
409  *sqrtPi*(ThetaSqrt + ThetaSmallSqrt)
410  )
411  );
412 
413  // particle pressure - coefficient in front of Theta (Eq. 3.22, p. 45)
414  volScalarField PsCoeff
415  (
416  granularPressureModel_->granularPressureCoeff
417  (
418  alpha,
419  gs0_,
420  rho,
421  e_
422  )
423  );
424 
425  // 'thermal' conductivity (Table 3.3, p. 49)
426  kappa_ = conductivityModel_->kappa(alpha, Theta_, gs0_, rho, da, e_);
427 
428  fv::options& fvOptions(fv::options::New(mesh_));
429 
430  // Construct the granular temperature equation (Eq. 3.20, p. 44)
431  // NB. note that there are two typos in Eq. 3.20:
432  // Ps should be without grad
433  // the laplacian has the wrong sign
434  fvScalarMatrix ThetaEqn
435  (
436  1.5*
437  (
438  fvm::ddt(alpha, rho, Theta_)
439  + fvm::div(alphaRhoPhi, Theta_)
440  - fvc::Sp(fvc::ddt(alpha, rho) + fvc::div(alphaRhoPhi), Theta_)
441  )
442  - fvm::laplacian(kappa_, Theta_, "laplacian(kappa,Theta)")
443  ==
444  - fvm::SuSp((PsCoeff*I) && gradU, Theta_)
445  + (tau && gradU)
446  + fvm::Sp(-gammaCoeff, Theta_)
447  + fvm::Sp(-J1, Theta_)
448  + fvm::Sp(J2/(Theta_ + ThetaSmall), Theta_)
449  + fvOptions(alpha, rho, Theta_)
450  );
451 
452  ThetaEqn.relax();
453  fvOptions.constrain(ThetaEqn);
454  ThetaEqn.solve();
455  fvOptions.correct(Theta_);
456  }
457  else
458  {
459  // Equilibrium => dissipation == production
460  // Eq. 4.14, p.82
461  volScalarField K1("K1", 2*(1 + e_)*rho*gs0_);
462  volScalarField K3
463  (
464  "K3",
465  0.5*da*rho*
466  (
467  (sqrtPi/(3*(3.0 - e_)))
468  *(1 + 0.4*(1 + e_)*(3*e_ - 1)*alpha*gs0_)
469  +1.6*alpha*gs0_*(1 + e_)/sqrtPi
470  )
471  );
472 
473  volScalarField K2
474  (
475  "K2",
476  4*da*rho*(1 + e_)*alpha*gs0_/(3*sqrtPi) - 2*K3/3.0
477  );
478 
479  volScalarField K4("K4", 12*(1 - sqr(e_))*rho*gs0_/(da*sqrtPi));
480 
481  volScalarField trD
482  (
483  "trD",
484  alpha/(alpha + residualAlpha_)
485  *fvc::div(phi_)
486  );
487  volScalarField tr2D("tr2D", sqr(trD));
488  volScalarField trD2("trD2", tr(D & D));
489 
490  volScalarField t1("t1", K1*alpha + rho);
491  volScalarField l1("l1", -t1*trD);
492  volScalarField l2("l2", sqr(t1)*tr2D);
493  volScalarField l3
494  (
495  "l3",
496  4.0
497  *K4
498  *alpha
499  *(2*K3*trD2 + K2*tr2D)
500  );
501 
502  Theta_ = sqr
503  (
504  (l1 + sqrt(l2 + l3))
505  /(2*max(alpha, residualAlpha_)*K4)
506  );
507 
508  kappa_ = conductivityModel_->kappa(alpha, Theta_, gs0_, rho, da, e_);
509  }
510 
511  Theta_.max(0);
512  Theta_.min(100);
513 
514  {
515  // particle viscosity (Table 3.2, p.47)
516  nut_ = viscosityModel_->nu(alpha, Theta_, gs0_, rho, da, e_);
517 
518  volScalarField ThetaSqrt("sqrtTheta", sqrt(Theta_));
519 
520  // Bulk viscosity p. 45 (Lun et al. 1984).
521  lambda_ = (4.0/3.0)*sqr(alpha)*da*gs0_*(1 + e_)*ThetaSqrt/sqrtPi;
522 
523  // Frictional pressure
524  volScalarField pf
525  (
526  frictionalStressModel_->frictionalPressure
527  (
528  phase_,
529  alphaMinFriction_,
530  alphaMax_
531  )
532  );
533 
534  nuFric_ = frictionalStressModel_->nu
535  (
536  phase_,
537  alphaMinFriction_,
538  alphaMax_,
539  pf/rho,
540  D
541  );
542 
543  // Limit viscosity and add frictional viscosity
544  nut_.min(maxNut_);
545  nuFric_ = min(nuFric_, maxNut_ - nut_);
546  nut_ += nuFric_;
547  }
548 
549  if (debug)
550  {
551  Info<< typeName << ':' << nl
552  << " max(Theta) = " << max(Theta_).value() << nl
553  << " max(nut) = " << max(nut_).value() << endl;
554  }
555 }
