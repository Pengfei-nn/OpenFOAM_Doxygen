98 void fWallFunctionFvPatchScalarField::updateCoeffs()
99 {
100  if (updated())
101  {
102  return;
103  }
104 
105  const label patchi = patch().index();
106 
107  const turbulenceModel& turbModel = db().lookupObject<turbulenceModel>
108  (
109  IOobject::groupName
110  (
111  turbulenceModel::propertiesName,
112  internalField().group()
113  )
114  );
115 
116  const nutWallFunctionFvPatchScalarField& nutw =
117  nutWallFunctionFvPatchScalarField::nutw(turbModel, patchi);
118 
119  const scalarField& y = turbModel.y()[patchi];
120 
121  const tmp<scalarField> tnuw = turbModel.nu(patchi);
122  const scalarField& nuw = tnuw();
123 
124  scalarField& f = *this;
125 
126  if (isA<v2fBase>(turbModel))
127  {
128  const v2fBase& v2fModel = refCast<const v2fBase>(turbModel);
129 
130  const tmp<volScalarField> tk = turbModel.k();
131  const volScalarField& k = tk();
132 
133  const tmp<volScalarField> tepsilon = turbModel.epsilon();
134  const volScalarField& epsilon = tepsilon();
135 
136  const tmp<volScalarField> tv2 = v2fModel.v2();
137  const volScalarField& v2 = tv2();
138 
139  const scalar Cmu25 = pow025(nutw.Cmu());
140  const scalar N = 6.0;
141 
142  // Set f wall values
143  forAll(f, facei)
144  {
145  const label celli = patch().faceCells()[facei];
146 
147  const scalar uTau = Cmu25*sqrt(k[celli]);
148 
149  const scalar yPlus = uTau*y[facei]/nuw[facei];
150 
151  if (nutw.yPlusLam() < yPlus)
152  {
153  const scalar v2c = v2[celli];
154  const scalar epsc = epsilon[celli];
155  const scalar kc = k[celli];
156 
157  f[facei] = N*v2c*epsc/(sqr(kc) + ROOTVSMALL);
158  f[facei] /= sqr(uTau) + ROOTVSMALL;
159  }
160  else
161  {
162  f[facei] = 0.0;
163  }
164  }
165  }
166  else if (isA<kEpsilonPhitFBase>(turbModel))
167  {
168  // (LUU:p. 176)
169  f = 0.0;
170  }
171  else
172  {
173  FatalErrorInFunction
174  << "The RAS model is neither the v2f nor kEpsilonPhitF model. "
175  << "Therefore, fWallFunction is not usable." << nl
176  << exit(FatalError);
177  }
178 
179  fixedValueFvPatchField<scalar>::updateCoeffs();
180 
181  // TODO: perform averaging for cells sharing more than one boundary face
182 }
183
