124 
125  //
126  // pick up faces connected to "localPoints"
127  //
128

193  facesToSubset[curFaces[i]] = true;
194  }
195  }
196  }
197 
198 
199  //
200  // pick up faces with centre inside "zone"
201  //
202 
203  if (markedZone.size() == 2)
204  {
205  const point& min = markedZone[0];
206  const point& max = markedZone[1];

237

239 
240  const bool outside(surfDict.get<bool>("outside"));
241 
242  if (outside)

267  forAll(facesToSubset, facei)
268  {
269  if (!facesToSubset[facei])

284  }

286  }
287 
288 
289  if (meshSubsetDict.found("plane"))
290  {

292

294  const scalar distance(planeDict.get<scalar>("distance"));
295  const scalar cosAngle(planeDict.get<scalar>("cosAngle"));
296 
297  // Select all triangles that are close to the plane and

299 
300  forAll(surf1.faceCentres(), facei)
301  {
302  const point& fc = surf1.faceCentres()[facei];
303  const point& nf = surf1.faceNormals()[facei];
304 
305  if (pl.distance(fc) < distance && mag(pl.normal() & nf) > cosAngle)
306  {
307  facesToSubset[facei] = true;
308  }
309  }
310  }
311

320 
321  if (markedFaces.size())
322  {
323  Info<< "Found " << markedFaces.size() << " marked face(s)." << endl;
324 
325  // Check and mark faces to pick up
326  forAll(markedFaces, facei)
327  {
328  if
329  (

332  )
333  {

339  }
340

342  facesToSubset[markedFaces[facei]] = true;
343

345  if (addFaceNeighbours)
346  {

348  surf1.faceFaces()[markedFaces[facei]];
349 
350  forAll(curFaces, i)
351  {
352  label facei = curFaces[i];
353 
354  if (!facesToSubset[facei])
355  {
356  facesToSubset[facei] = true;
357  nFaceNeighbours++;
358  }
359  }
360  }
361  }
362  }
363 
364  if (addFaceNeighbours)
365  {
366  Info<< "Added " << nFaceNeighbours

368  }
369 
370

372  {
373  Info<< "Inverting selection." << endl;
374 
375  forAll(facesToSubset, i)
376  {
377  facesToSubset[i] = facesToSubset[i] ? false : true;
378  }
379  }
380 
381 
382  // Create subsetted surface
383  triSurface surf2(surf1.subsetMesh(facesToSubset));
384
