43 void Foam::conformationSurfaces::hasBoundedVolume
44 (
45  List<volumeType>& referenceVolumeTypes
46 ) const
47 {
48  vector sum(Zero);
49  label totalTriangles = 0;
50 
51  forAll(surfaces_, s)
52  {
53  const searchableSurface& surface(allGeometry_[surfaces_[s]]);
54 
55  if
56  (
57  surface.hasVolumeType()
58  && (
59  normalVolumeTypes_[regionOffset_[s]]
60  != extendedFeatureEdgeMesh::BOTH
61  )
62  )
63  {
64  pointField pts(1, locationInMesh_);
65 
66  List<volumeType> vTypes
67  (
68  pts.size(),
69  volumeType::UNKNOWN
70  );
71 
72  surface.getVolumeType(pts, vTypes);
73 
74  referenceVolumeTypes[s] = vTypes[0];
75 
76  Info<< " is " << referenceVolumeTypes[s].str()
77  << " surface " << surface.name()
78  << endl;
79  }
80 
81  if (isA<triSurface>(surface))
82  {
83  const triSurface& triSurf = refCast<const triSurface>(surface);
84 
85  const pointField& surfPts = triSurf.points();
86 
87  Info<< " Checking " << surface.name() << endl;
88 
89  label nBaffles = 0;
90 
91  Info<< " Index = " << surfaces_[s] << endl;
92  Info<< " Offset = " << regionOffset_[s] << endl;
93 
94  for (const labelledTri& f : triSurf)
95  {
96  const label patchID = f.region() + regionOffset_[s];
97 
98  // Don't include baffle surfaces in the calculation
99  if
100  (
101  normalVolumeTypes_[patchID]
102  != extendedFeatureEdgeMesh::BOTH
103  )
104  {
105  sum += f.areaNormal(surfPts);
106  }
107  else
108  {
109  ++nBaffles;
110  }
111  }
112  Info<< " has " << nBaffles << " baffles out of "
113  << triSurf.size() << " triangles" << nl;
114 
115  totalTriangles += triSurf.size();
116  }
117  }
118 
119  Info<< " Sum of all the surface normals (if near zero, surface is"
120  << " probably closed):" << nl
121  << " Note: Does not include baffle surfaces in calculation" << nl
122  << " Sum = " << sum/(totalTriangles + SMALL) << nl
123  << " mag(Sum) = " << mag(sum)/(totalTriangles + SMALL)
124  << endl;
125 }

345

386  WarningInFunction

388  << surface.name()

392  }
393 
394  // Load patch info
395  if (dict.found("patchInfo"))
396  {
397  globalPatchInfo.set
398  (
399  surfI,
400  dict.subDict("patchInfo").clone()
401  );
402  }
403 
404  readFeatures
405  (
406  surfI,
407  dict,
408  geomName,
409  featureI
410  );
411 
412  const wordList& rNames = surface.regions();
413 
414  if (dict.found("regions"))
415  {
416  const dictionary& regionsDict = dict.subDict("regions");
417 
418  forAll(rNames, regionI)
419  {
420  const word& regionName = rNames[regionI];
421 
422  if (regionsDict.found(regionName))
423  {
424  Info<< " region " << regionName << endl;
425 
426  // Get the dictionary for region
427  const dictionary& regionDict = regionsDict.subDict
428  (
429  regionName
430  );
431

433  {
434  regionPatchInfo[surfI].insert
435  (
436  regionI,
437  regionDict.subDict("patchInfo").clone()
438  );
439  }
440 
441  regionVolumeTypes[surfI].insert
442  (
443  regionI,
444  extendedFeatureEdgeMesh::sideVolumeTypeNames_
445  [
446  regionDict.lookupOrDefault<word>
447  (
448  "meshableSide",
449  extendedFeatureEdgeMesh::
450  sideVolumeTypeNames_
451  [
452  globalVolumeTypes[surfI]
453  ]
454  )
455  ]
456  );
457 
458  readFeatures(regionDict, regionName, featureI);
459  }
460  }
461  }
462 
463  surfI++;
464  }
465  }
466 
467 
468  if (unmatchedKeys.size() > 0)

472  << " The following entries were not used : "

474  << endl;
475  }

478  // Calculate local to global region offset
479  label nRegions = 0;
480

496

526 
527  patchInfo_.set(globalRegionI, iter()().clone());

529  }

531

534  {
535  Info<< nl << "Reading additionalFeatures" << endl;
536  }
537

543  Info<< nl << " " << featureName << endl;
544 
545  readFeatures(featureSubDict, featureName, featureI);

549  features_.setSize(featureI);
550

555

562  globalBounds_.min() -= newSpan;

568 
569  referenceVolumeTypes_.setSize
570  (
571  surfaces_.size(),
572  volumeType::UNKNOWN
573  );
574 
575  Info<< endl
576  << "Testing for locationInMesh " << locationInMesh_ << endl;
577 
578  hasBoundedVolume(referenceVolumeTypes_);

593  }

650 Foam::Field<bool> Foam::conformationSurfaces::wellInOutSide
651 (
652  const pointField& samplePts,
653  const scalarField& testDistSqr,
654  const bool testForInside
655 ) const
656 {
657  List<List<volumeType>> surfaceVolumeTests
658  (
659  surfaces_.size(),
660  List<volumeType>
661  (
662  samplePts.size(),
663  volumeType::UNKNOWN
664  )
665  );
666 
667  // Get lists for the volumeTypes for each sample wrt each surface
668  forAll(surfaces_, s)
669  {
670  const searchableSurface& surface(allGeometry_[surfaces_[s]]);
671 
672  const label regionI = regionOffset_[s];
673 
674  if (normalVolumeTypes_[regionI] != extendedFeatureEdgeMesh::BOTH)
675  {
676  surface.getVolumeType(samplePts, surfaceVolumeTests[s]);
677  }
678  }
679 
680  // Compare the volumeType result for each point wrt to each surface with the
681  // reference value and if the points are inside the surface by a given
682  // distanceSquared
683 
684  // Assume that the point is wellInside until demonstrated otherwise.
685  Field<bool> insideOutsidePoint(samplePts.size(), testForInside);
686 
687  //Check if the points are inside the surface by the given distance squared
688 
689  labelList hitSurfaces;
690  List<pointIndexHit> hitInfo;
691  searchableSurfacesQueries::findNearest
692  (
693  allGeometry_,
694  surfaces_,
695  samplePts,
696  testDistSqr,
697  hitSurfaces,
698  hitInfo
699  );
700 
701  forAll(samplePts, i)
702  {
703  const pointIndexHit& pHit = hitInfo[i];
704 
705  if (pHit.hit())
706  {
707  // If the point is within range of the surface, then it can't be
708  // well (in|out)side
709  insideOutsidePoint[i] = false;
710 
711  continue;
712  }
713 
714  forAll(surfaces_, s)
715  {
716  const label regionI = regionOffset_[s];
717 
718  if (normalVolumeTypes_[regionI] == extendedFeatureEdgeMesh::BOTH)
719  {
720  continue;
721  }
722 
723  const searchableSurface& surface(allGeometry_[surfaces_[s]]);
724 
725  if
726  (
727  !surface.hasVolumeType()
728  //&& surfaceVolumeTests[s][i] == volumeType::UNKNOWN
729  )
730  {
731  pointField sample(1, samplePts[i]);
732  scalarField nearestDistSqr(1, GREAT);
733  List<pointIndexHit> info;
734 
735  surface.findNearest(sample, nearestDistSqr, info);
736 
737  const vector hitDir =
738  normalised
739  (
740  info[0].rawPoint() - samplePts[i]
741  );
742 
743  pointIndexHit surfHit;
744  label hitSurface;
745 
746  findSurfaceNearestIntersection
747  (
748  samplePts[i],
749  info[0].rawPoint() - 1e-3*mag(hitDir)*hitDir,
750  surfHit,
751  hitSurface
752  );
753 
754  if (surfHit.hit() && hitSurface != surfaces_[s])
755  {
756  continue;
757  }
758  }
759 
760  if (surfaceVolumeTests[s][i] == volumeType::OUTSIDE)
761  {
762  if
763  (
764  normalVolumeTypes_[regionI]
765  == extendedFeatureEdgeMesh::INSIDE
766  )
767  {
768  insideOutsidePoint[i] = !testForInside;
769  break;
770  }
771  }
772  else if (surfaceVolumeTests[s][i] == volumeType::INSIDE)
773  {
774  if
775  (
776  normalVolumeTypes_[regionI]
777  == extendedFeatureEdgeMesh::OUTSIDE
778  )
779  {
780  insideOutsidePoint[i] = !testForInside;
781  break;
782  }
783  }
784  }
785  }
786 
787  return insideOutsidePoint;
788 }
