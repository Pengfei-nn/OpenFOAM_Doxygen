38 template<class CloudType>
39 Foam::ConeInjection<CloudType>::ConeInjection
40 (
41  const dictionary& dict,
42  CloudType& owner,
43  const word& modelName
44 )
45 :
46  InjectionModel<CloudType>(dict, owner, modelName, typeName),
47  positionAxis_(this->coeffDict().lookup("positionAxis")),
48  injectorCells_(positionAxis_.size()),
49  injectorTetFaces_(positionAxis_.size()),
50  injectorTetPts_(positionAxis_.size()),
51  duration_(this->coeffDict().getScalar("duration")),
52  parcelsPerInjector_
53  (
54  this->coeffDict().getScalar("parcelsPerInjector")
55  ),
56  flowRateProfile_
57  (
58  TimeFunction1<scalar>
59  (
60  owner.db().time(),
61  "flowRateProfile",
62  this->coeffDict()
63  )
64  ),
65  Umag_
66  (
67  TimeFunction1<scalar>
68  (
69  owner.db().time(),
70  "Umag",
71  this->coeffDict()
72  )
73  ),
74  thetaInner_
75  (
76  TimeFunction1<scalar>
77  (
78  owner.db().time(),
79  "thetaInner",
80  this->coeffDict()
81  )
82  ),
83  thetaOuter_
84  (
85  TimeFunction1<scalar>
86  (
87  owner.db().time(),
88  "thetaOuter",
89  this->coeffDict()
90  )
91  ),
92  sizeDistribution_
93  (
94  distributionModel::New
95  (
96  this->coeffDict().subDict("sizeDistribution"), owner.rndGen()
97  )
98  ),
99  nInjected_(this->parcelsAddedTotal()),
100  tanVec1_(positionAxis_.size()),
101  tanVec2_(positionAxis_.size())
102 {
103  duration_ = owner.db().time().userTimeToTime(duration_);
104 
105  // Normalise direction vector and determine direction vectors
106  // tangential to injector axis direction
107  forAll(positionAxis_, i)
108  {
109  vector& axis = positionAxis_[i].second();
110  axis.normalise();
111 
112  vector tangent = Zero;
113  scalar magTangent = 0.0;
114 
115  Random& rnd = this->owner().rndGen();
116  while (magTangent < SMALL)
117  {
118  vector v = rnd.sample01<vector>();
119 
120  tangent = v - (v & axis)*axis;
121  magTangent = mag(tangent);
122  }
123 
124  tanVec1_[i] = tangent/magTangent;
125  tanVec2_[i] = axis^tanVec1_[i];
126  }
127 
128  // Set total volume to inject
129  this->volumeTotal_ = flowRateProfile_.integrate(0.0, duration_);
130 
131  updateMesh();
132 }

192 template<class CloudType>
193 Foam::label Foam::ConeInjection<CloudType>::parcelsToInject
194 (
195  const scalar time0,
196  const scalar time1
197 )
198 {
199  if ((time0 >= 0.0) && (time0 < duration_))
200  {
201  const scalar targetVolume = flowRateProfile_.integrate(0, time1);
202 
203  const scalar volumeFraction = targetVolume/this->volumeTotal_;
204 
205  const label targetParcels =
206  ceil(positionAxis_.size()*parcelsPerInjector_*volumeFraction);
207 
208  return targetParcels - nInjected_;
209  }
210 
211  return 0;
212 }

252 template<class CloudType>
253 void Foam::ConeInjection<CloudType>::setProperties
254 (
255  const label parcelI,
256  const label,
257  const scalar time,
258  typename CloudType::parcelType& parcel
259 )
260 {
261  Random& rnd = this->owner().rndGen();
262 
263  // Set particle velocity
264  const label i = parcelI % positionAxis_.size();
265 
266  scalar t = time - this->SOI_;
267  scalar ti = thetaInner_.value(t);
268  scalar to = thetaOuter_.value(t);
269  scalar coneAngle = degToRad(rnd.position<scalar>(ti, to));
270 
271  scalar alpha = sin(coneAngle);
272  scalar dcorr = cos(coneAngle);
273  scalar beta = twoPi*rnd.sample01<scalar>();
274 
275  vector normal = alpha*(tanVec1_[i]*cos(beta) + tanVec2_[i]*sin(beta));
276  vector dirVec = dcorr*positionAxis_[i].second();
277  dirVec += normal;
278  dirVec.normalise();
279 
280  parcel.U() = Umag_.value(t)*dirVec;
281 
282  // Set particle diameter
283  parcel.d() = sizeDistribution_().sample();
284 
285  // Increment number of particles injected
286  nInjected_++;
287 }
