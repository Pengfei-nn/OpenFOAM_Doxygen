101 inline Foam::SHA1Digest Foam::SHA1::digest() const
102 {
103  SHA1Digest dig;
104 
105  if (finalized_)
106  {
107  calcDigest(dig);
108  }
109  else
110  {
111  // Avoid disturbing current data - use a copy
112  SHA1 sha(*this);
113  if (sha.finalize())
114  {
115  sha.calcDigest(dig);
116  }
117  }
118 
119  return dig;
120 }
