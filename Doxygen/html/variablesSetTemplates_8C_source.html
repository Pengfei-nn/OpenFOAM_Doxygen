190 void variablesSet::swapAndRename
191 (
192  autoPtr<GeometricField<Type, PatchField, GeoMesh>>& p1,
193  autoPtr<GeometricField<Type, PatchField, GeoMesh>>& p2
194 )
195 {
196  // Swaping pointers is OK for the mean flow fields known by the
197  // variablesSet (and, in essence, by the solver).
198  // The problem is that turbulence models know references to U and phi
199  // which cannot be swapped.
200  /*
201  const word name1 = p1().name();
202  const word name2 = p2().name();
203  p1.swap(p2);
204 
205  p2().rename("temp");
206  p1().rename(name1);
207  p2().rename(name2);
208  */
209 
210  // Copy back-up fields to original instead. Slower but there seems to be
211  // no other way
212  GeometricField<Type, PatchField, GeoMesh> temp("temp", p1());
213  p1() == p2();
214  p2() == temp;
215 }
