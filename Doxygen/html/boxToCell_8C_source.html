69 void Foam::boxToCell::combine(topoSet& set, const bool add) const
70 {
71  const pointField& ctrs = mesh_.cellCentres();
72 
73  forAll(ctrs, elemi)
74  {
75  for (const auto& bb : bbs_)
76  {
77  if (bb.contains(ctrs[elemi]))

79  addOrDelete(set, elemi, add);

81  }
