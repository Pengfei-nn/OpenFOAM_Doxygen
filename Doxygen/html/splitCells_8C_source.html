32 Description
33  Utility to split cells with flat faces.
34 
35  Uses a geometric cut with a plane dividing the edge angle into two so
36  might produce funny cells. For hexes it will use by default a cut from
37  edge onto opposite edge (i.e. purely topological).
38 
39  Options:
40  - split cells from cellSet only
41  - use geometric cut for hexes as well
42 
43  The angle is the angle between two faces sharing an edge as seen from
44  inside each cell. So a cube will have all angles 90. If you want
45  to split cells with cell centre outside use e.g. angle 200
46

107 // Given sin and cos of max angle between normals calculate whether f0 and f1
108 // on celli make larger angle. Uses sinAngle only for quadrant detection.
109 bool largerAngle
110 (
111  const primitiveMesh& mesh,
112  const scalar cosAngle,
113  const scalar sinAngle,
114 
115  const label celli,
116  const label f0, // face label
117  const label f1,
118 
119  const vector& n0, // normal at f0
120  const vector& n1
121 )
122 {
123  const labelList& own = mesh.faceOwner();
124 
125  bool sameFaceOrder = !((own[f0] == celli) ^ (own[f1] == celli));
126 
127  // Get cos between faceArea vectors. Correct so flat angle (180 degrees)
128  // gives -1.
129  scalar normalCosAngle = n0 & n1;
130 
131  if (sameFaceOrder)
132  {
133  normalCosAngle = -normalCosAngle;
134  }
135 
136 
137  // Get cos between faceCentre and normal vector to determine in
138  // which quadrant angle is. (Is correct for unwarped faces only!)
139  // Correct for non-outwards pointing normal.
140  const vector c1c0 =
141  normalised
142  (
143  mesh.faceCentres()[f1] - mesh.faceCentres()[f0]
144  );
145 
146  scalar fcCosAngle = n0 & c1c0;
147 
148  if (own[f0] != celli)
149  {
150  fcCosAngle = -fcCosAngle;
151  }
152 
153  if (sinAngle < 0.0)
154  {
155  // Looking for concave angles (quadrant 3 or 4)
156  if (fcCosAngle <= 0)
157  {
158  // Angle is convex so smaller.
159  return false;
160  }
161  else
162  {
163  if (normalCosAngle < cosAngle)
164  {
165  return false;
166  }
167  else
168  {
169  return true;
170  }
171  }
172  }
173  else
174  {
175  // Looking for convex angles (quadrant 1 or 2)
176  if (fcCosAngle > 0)
177  {
178  // Concave angle
179  return true;
180  }
181  else
182  {
183  // Convex. Check cos of normal vectors.
184  if (normalCosAngle > cosAngle)
185  {
186  return false;
187  }
188  else
189  {
190  return true;
191  }
192  }
193  }
194 }

197 // Split hex (and hex only) along edgeI creating two prisms
198 bool splitHex
199 (
200  const polyMesh& mesh,
201  const label celli,
202  const label edgeI,
203 
204  DynamicList<label>& cutCells,
205  DynamicList<labelList>& cellLoops,
206  DynamicList<scalarField>& cellEdgeWeights
207 )
208 {
209  // cut handling functions
210  edgeVertex ev(mesh);
211 
212  const edgeList& edges = mesh.edges();
213  const faceList& faces = mesh.faces();
214 
215  const edge& e = edges[edgeI];
216 
217  // Get faces on the side, i.e. faces not using edge but still using one of
218  // the edge endpoints.
219 
220  label leftI = -1;
221  label rightI = -1;
222  label leftFp = -1;
223  label rightFp = -1;
224 
225  const cell& cFaces = mesh.cells()[celli];
226 
227  for (const label facei : cFaces)
228  {
229  const face& f = faces[facei];
230 
231  label fp0 = f.find(e[0]);
232  label fp1 = f.find(e[1]);
233 
234  if (fp0 == -1)
235  {
236  if (fp1 != -1)
237  {
238  // Face uses e[1] but not e[0]
239  rightI = facei;
240  rightFp = fp1;
241 
242  if (leftI != -1)
243  {
244  // Have both faces so exit
245  break;
246  }
247  }
248  }
249  else
250  {
251  if (fp1 != -1)
252  {
253  // Face uses both e[1] and e[0]
254  }
255  else
256  {
257  leftI = facei;
258  leftFp = fp0;
259 
260  if (rightI != -1)
261  {
262  break;
263  }
264  }
265  }
266  }
267 
268  if (leftI == -1 || rightI == -1)
269  {
270  FatalErrorInFunction
271  << " rightI:" << rightI << abort(FatalError);
272  }
273 
274  // Walk two vertices further on faces.
275 
276  const face& leftF = faces[leftI];
277 
278  label leftV = leftF[(leftFp + 2) % leftF.size()];
279 
280  const face& rightF = faces[rightI];
281 
282  label rightV = rightF[(rightFp + 2) % rightF.size()];
283 
284  labelList loop(4);
285  loop[0] = ev.vertToEVert(e[0]);
286  loop[1] = ev.vertToEVert(leftV);
287  loop[2] = ev.vertToEVert(rightV);
288  loop[3] = ev.vertToEVert(e[1]);
289 
290  scalarField loopWeights(4);
291  loopWeights[0] = -GREAT;
292  loopWeights[1] = -GREAT;
293  loopWeights[2] = -GREAT;
294  loopWeights[3] = -GREAT;
295 
296  cutCells.append(celli);
297  cellLoops.append(loop);
298  cellEdgeWeights.append(loopWeights);
299 
300  return true;
301 }

304 // Split celli along edgeI with a plane along halfNorm direction.
305 bool splitCell
306 (
307  const polyMesh& mesh,
308  const geomCellLooper& cellCutter,
309 
310  const label celli,
311  const label edgeI,
312  const vector& halfNorm,
313 
314  const boolList& vertIsCut,
315  const boolList& edgeIsCut,
316  const scalarField& edgeWeight,
317 
318  DynamicList<label>& cutCells,
319  DynamicList<labelList>& cellLoops,
320  DynamicList<scalarField>& cellEdgeWeights
321 )
322 {
323  const edge& e = mesh.edges()[edgeI];
324 
325  const vector eVec = e.unitVec(mesh.points());
326 
327  vector planeN = eVec ^ halfNorm;
328 
329  // Slightly tilt plane to make it not cut edges exactly
330  // halfway on fully regular meshes (since we want cuts
331  // to be snapped to vertices)
332  planeN += 0.01*halfNorm;
333  planeN.normalise();
334 
335  // Define plane through edge
336  plane cutPlane(mesh.points()[e.start()], planeN);
337 
338  labelList loop;
339  scalarField loopWeights;
340 
341  if
342  (
343  cellCutter.cut
344  (
345  cutPlane,
346  celli,
347  vertIsCut,
348  edgeIsCut,
349  edgeWeight,
350  loop,
351  loopWeights
352  )
353  )
354  {
355  // Did manage to cut cell. Copy into overall list.
356  cutCells.append(celli);
357  cellLoops.append(loop);
358  cellEdgeWeights.append(loopWeights);
359 
360  return true;
361  }
362 
363  return false;
364 }

367 // Collects cuts for all cells in cellSet
368 void collectCuts
369 (
370  const polyMesh& mesh,
371  const geomCellLooper& cellCutter,
372  const bool geometry,
373  const scalar minCos,
374  const scalar minSin,
375  const cellSet& cellsToCut,
376 
377  DynamicList<label>& cutCells,
378  DynamicList<labelList>& cellLoops,
379  DynamicList<scalarField>& cellEdgeWeights
380 )
381 {
382  // Get data from mesh
383  const cellShapeList& cellShapes = mesh.cellShapes();
384  const labelList& own = mesh.faceOwner();
385  const labelListList& cellEdges = mesh.cellEdges();
386  const vectorField& faceAreas = mesh.faceAreas();
387 
388  // Hex shape
389  const cellModel& hex = cellModel::ref(cellModel::HEX);
390 
391  // cut handling functions
392  edgeVertex ev(mesh);
393 
394 
395  // Cut information per mesh entity
396  boolList vertIsCut(mesh.nPoints(), false);
397  boolList edgeIsCut(mesh.nEdges(), false);
398  scalarField edgeWeight(mesh.nEdges(), -GREAT);
399 
400  for (const label celli : cellsToCut)
401  {
402  const labelList& cEdges = cellEdges[celli];
403 
404  for (const label edgeI : cEdges)
405  {
406  label f0, f1;
407  meshTools::getEdgeFaces(mesh, celli, edgeI, f0, f1);
408 
409  const vector n0 = normalised(faceAreas[f0]);
410  const vector n1 = normalised(faceAreas[f1]);
411 
412  if
413  (
414  largerAngle
415  (
416  mesh,
417  minCos,
418  minSin,
419 
420  celli,
421  f0,
422  f1,
423  n0,
424  n1
425  )
426  )
427  {
428  bool splitOk = false;
429 
430  if (!geometry && cellShapes[celli].model() == hex)
431  {
432  splitOk =
433  splitHex
434  (
435  mesh,
436  celli,
437  edgeI,
438 
439  cutCells,
440  cellLoops,
441  cellEdgeWeights
442  );
443  }
444  else
445  {
446  vector halfNorm;
447 
448  if ((own[f0] == celli) ^ (own[f1] == celli))
449  {
450  // Opposite owner orientation
451  halfNorm = 0.5*(n0 - n1);
452  }
453  else
454  {
455  // Faces have same owner or same neighbour so
456  // normals point in same direction
457  halfNorm = 0.5*(n0 + n1);
458  }
459 
460  splitOk =
461  splitCell
462  (
463  mesh,
464  cellCutter,
465  celli,
466  edgeI,
467  halfNorm,
468 
469  vertIsCut,
470  edgeIsCut,
471  edgeWeight,
472 
473  cutCells,
474  cellLoops,
475  cellEdgeWeights
476  );
477  }
478 
479 
480  if (splitOk)
481  {
482  // Update cell/edge/vertex wise info.
483  label index = cellLoops.size() - 1;
484  const labelList& loop = cellLoops[index];
485  const scalarField& loopWeights = cellEdgeWeights[index];
486 
487  forAll(loop, i)
488  {
489  label cut = loop[i];
490 
491  if (ev.isEdge(cut))
492  {
493  edgeIsCut[ev.getEdge(cut)] = true;
494  edgeWeight[ev.getEdge(cut)] = loopWeights[i];
495  }
496  else
497  {
498  vertIsCut[ev.getVertex(cut)] = true;
499  }
500  }
501 
502  // Stop checking edges for this cell.
503  break;
504  }
505  }
506  }
507  }
508 
509  cutCells.shrink();
510  cellLoops.shrink();
511  cellEdgeWeights.shrink();
512 }
