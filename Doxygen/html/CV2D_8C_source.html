238 {
239  Info<< "insertInitialGrid: ";
240 
241  startOfInternalPoints_ = number_of_vertices();
242  label nVert = startOfInternalPoints_;
243 
244  scalar x0 = qSurf_.globalBounds().min().x();
245  scalar xR = qSurf_.globalBounds().max().x() - x0;
246  int ni = int(xR/meshControls().minCellSize()) + 1;
247  scalar deltax = xR/ni;
248 
249  scalar y0 = qSurf_.globalBounds().min().y();
250  scalar yR = qSurf_.globalBounds().max().y() - y0;
251  int nj = int(yR/meshControls().minCellSize()) + 1;
252  scalar deltay = yR/nj;
253 
254  Random rndGen(1321);
255  scalar pert = meshControls().randomPerturbation()*min(deltax, deltay);
256 
257  for (int i=0; i<ni; i++)
258  {
259  for (int j=0; j<nj; j++)
260  {
261  Foam::point p(x0 + i*deltax, y0 + j*deltay, 0);
262 
263  if (meshControls().randomiseInitialGrid())
264  {
265  p.x() += pert*(rndGen.sample01<scalar>() - 0.5);
266  p.y() += pert*(rndGen.sample01<scalar>() - 0.5);
267  }
268 
269  if (qSurf_.wellInside(p, 0.5*meshControls().minCellSize2()))
270  {
271  insert(Point(p.x(), p.y()))->index() = nVert++;
272  }

276  Info<< nVert << " vertices inserted" << endl;

414  || fit->vertex(2)->internalOrBoundaryPoint()

416  {
417  fit->faceIndex() = dualVerti;
418 
419  dualVertices[dualVerti] = toPoint2D(circumcenter(fit));
420 
421  dualVerti++;
422  }
423  }
424

426

428  (
429  startOfSurfacePointPairs_,
430  vector2D::zero
431  );
432 
433  // Calculate target size and alignment for vertices
434  scalarField sizes
435  (
436  number_of_vertices(),
437  meshControls().minCellSize()
438  );
439 
440  Field<vector2D> alignments
441  (
442  number_of_vertices(),
443  vector2D(1, 0)
444  );
445 
446  for

448  Triangulation::Finite_vertices_iterator vit = finite_vertices_begin();
449  vit != finite_vertices_end();
450  ++vit
451  )
452  {
453  if (vit->internalOrBoundaryPoint())
454  {
455  point2D vert = toPoint2D(vit->point());
456 
457  // alignment and size determination
458  pointIndexHit pHit;
459  label hitSurface = -1;
460

462  (
463  toPoint3D(vert),
464  meshControls().span2(),
465  pHit,
466  hitSurface
467  );
468 
469  if (pHit.hit())
470  {
471  vectorField norm(1);
472  allGeometry_[hitSurface].getNormal
473  (
474  List<pointIndexHit>(1, pHit),
475  norm
476  );
477

479 
480  sizes[vit->index()] =
481  cellSizeControl_.cellSize
482  (
483  toPoint3D(vit->point())
484  );
485  }
486  }
487  }
488 
489  // Info<< "Calculated alignments" << endl;
490

495  scalar l = 0.7;

514  }

517  const point2D& dualV2 =
518  dualVertices[eit->first->neighbor(eit->second)->faceIndex()];
519 
520  scalar dualEdgeLength = mag(dualV1 - dualV2);

523  point2D dVB = toPoint2D(vB->point());

528  alignmentDirsA[1] = vector2D

540  alignmentDirsB[0].x()

544

546  {
547  const vector2D& a(alignmentDirsA[aA]);
548 
549  scalar maxDotProduct = 0.0;
550 
551  forAll(alignmentDirsB, aB)
552  {
553  const vector2D& b(alignmentDirsB[aB]);
554 
555  scalar dotProduct = a & b;
556 
557  if (mag(dotProduct) > maxDotProduct)
558  {
559  maxDotProduct = mag(dotProduct);
560 
561  alignmentDirs[aA] = a + sign(dotProduct)*b;
562 
563  alignmentDirs[aA].normalise();

565  }

568  vector2D rAB = dVA - dVB;

570  scalar rABMag = mag(rAB);

582

584

588  0.5*(sizes[vA->index()] + sizes[vB->index()]);

591  // alignment is nearest surface normal)

663

665  {
666  displacementAccumulator[vB->index()] += -delta;
667  }
668  }
669  }
670  }
671  }

677  displacementAccumulator *= relaxation;

679  label numberOfNewPoints = pointsToInsert.size();
680 
681  for
682  (
683  Triangulation::Finite_vertices_iterator vit = finite_vertices_begin();
684  vit != finite_vertices_end();
685  ++vit
686  )
687  {
688  if (vit->internalPoint())
689  {
690  if (pointToBeRetained.test(vit->index()))
691  {
692  pointsToInsert.push_front
693  (
694  toPoint
695  (
696  toPoint2D(vit->point())

698  )
699  );
700  }
701  }
702  }
703

705  // This is faster than removing and moving points.
706  this->clear();
707 
708  insertBoundingBox();
709 
710  reinsertFeaturePoints();

713 
714  label nVert = startOfInternalPoints_;
715 
716  Info<< "Inserting " << numberOfNewPoints << " new points" << endl;
717

732  )
733  {
734  vit->index() = nVert++;
