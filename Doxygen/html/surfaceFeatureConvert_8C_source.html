48 int main(int argc, char *argv[])
49 {
50  argList::addNote
51  (
52  "Convert between edgeMesh formats"
53  );
54  argList::noParallel();
55  argList::addArgument("input", "The input edge file");
56  argList::addArgument("output", "The output edge file");
57  argList::addOption
58  (
59  "scale",
60  "factor",
61  "Geometry scaling factor - default is 1"
62  );
63 
64  argList args(argc, argv);
65  Time runTime(args.rootPath(), args.caseName());
66 
67  const fileName importName = args[1];
68  const fileName exportName = args[2];
69 
70  // Disable inplace editing
71  if (importName == exportName)
72  {
73  FatalErrorInFunction
74  << "Output file " << exportName << " would overwrite input file."
75  << exit(FatalError);
76  }
77 
78  // Check that reading/writing is supported
79  if
80  (
81  !edgeMesh::canReadType(importName.ext(), true)
82  || !edgeMesh::canWriteType(exportName.ext(), true)
83  )
84  {
85  return 1;
86  }
87 
88  edgeMesh mesh(importName);
89 
90  Info<< "\nRead edgeMesh " << importName << nl;
91  mesh.writeStats(Info);
92  Info<< nl
93  << "\nwriting " << exportName;
94 
95  scalar scaleFactor = 0;
96  if (args.readIfPresent("scale", scaleFactor) && scaleFactor > 0)
97  {
98  Info<< " with scaling " << scaleFactor << endl;
99  mesh.scalePoints(scaleFactor);
100  }
101  else
102  {
103  Info<< " without scaling" << endl;
104  }
105 
106  mesh.write(exportName);
107  mesh.writeStats(Info);
108 
109  Info<< "\nEnd\n" << endl;
110 
111  return 0;
112 }
